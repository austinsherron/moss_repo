window4 = tkinter.Tk()
c4 = tkinter.Canvas(window4, width = 1000, height = 750) # new canvas for solar
c4.pack()
c4.create_oval(470,470,530,530, fill='yellow') # sun
for i in range(1, 9):    # for loop for the orbitals of planets
    x1 = 490 - ((i * 25) + (2**i))
    x2 = 510 + ((i * 25) + (2**i))
    c4.create_oval(x1,x1,x2,x2)
    if(i==3):
        c4.create_oval(x1 - 5,500,x1 + 5, 510, fill='blue') # Earth (coolest)
    elif (i==5):
        c4.create_oval(x1 - 20,485,x1 + 20, 525, fill='orange') # Jupiter
    elif(i==6):
        c4.create_oval(x1 - 10,495,x1 + 10, 515, fill='yellow') # Saturn
    else:
        c4.create_oval(x1 - 5,500,x1 + 5, 510, fill='black') # less cool planets

tkinter.mainloop()          # Combine all the elements and display the window
