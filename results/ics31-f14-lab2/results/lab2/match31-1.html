import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window
my_canvas_square_diamond = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas_house_eye = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on


my_canvas_square_diamond.pack()            # Put the canvas into the window
my_canvas_house_eye.pack()

'''Draws a orange square'''
my_canvas_square_diamond.create_line(5, 5, 205, 5, fill='orange')
my_canvas_square_diamond.create_line(205, 5, 205, 205, fill='orange')
my_canvas_square_diamond.create_line(205, 205, 5, 205, fill='orange')
my_canvas_square_diamond.create_line(5, 205, 5, 5, fill='orange')


'''Draws a diamond'''
my_canvas_square_diamond.create_line(212, 146, 354, 5, fill='orange')
my_canvas_square_diamond.create_line(354, 5, 495, 146, fill='orange')
my_canvas_square_diamond.create_line(495, 146, 354, 288, fill='orange')
my_canvas_square_diamond.create_line(354, 288, 212, 146, fill='orange')

'''Draws a house'''
my_canvas_house_eye.create_polygon(395, 0, 295, 100, 495, 100, fill='yellow')
my_canvas_house_eye.create_rectangle(295, 100, 495, 275, fill='blue')
my_canvas_house_eye.create_rectangle(345, 200, 395, 275, fill='green')
my_canvas_house_eye.create_rectangle(450, 125, 485, 150, fill='purple')

'''Draws an eye'''
my_canvas_house_eye.create_oval(5, 10, 205, 90)
my_canvas_house_eye.create_oval(80, 10, 130, 90, fill='black')
my_canvas_house_eye.create_oval(55, 10, 155, 90)

tkinter.mainloop()          # Combine all the elements and display the window
