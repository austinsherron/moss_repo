#D
KRONE_PER_EURO = 7.46
KRONE_PER_POUND = 8.81
KRONE_PER_DOLLAR = 5.50

print("Hello! Welcome to the Copenhagen Chamber of Commerce!")
print("Please enter this information:")
business = str(input("Business Name: "))
euros = int(input("Number of Euros: "))
pounds = int(input("Number of Pounds: "))
dollars = int(input("Number of Dollars: "))
print("")



Krone_Euro = euros * KRONE_PER_EURO
Krone_Pound = pounds * KRONE_PER_POUND
Krone_Dollar = dollars * KRONE_PER_DOLLAR

print("Copenhagen Chamber of Commerce")
print("Business Name: ", business)
print(euros, "Euros is", Krone_Euro , "Krone.")
print(pounds, "Pounds is", Krone_Pound , "Krone.")
print(dollars, "Dollars is", Krone_Dollar , "Krone.")
print('')
print('Total Krone: ', Krone_Euro + Krone_Pound + Krone_Dollar)
print('')
