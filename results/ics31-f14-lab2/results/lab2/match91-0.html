    my_canvas_house.create_line(50,490,450,490)
    my_canvas_house.create_line(450,490,450,250)

def draw_house_roof ():

def draw_window (x: int, y:int) -> 'draws window':
    'Draws a window starting at (x, y) in top left corner'
    my_canvas_house.create_line(x, y, x, y+80)
    my_canvas_house.create_line(x, y+80, x+80, y+80)
    my_canvas_house.create_line(x+80, y+80, x+80, y)
    my_canvas_house.create_line(x+80, y, x, y)

    my_canvas_house.create_line(x, y+40, x+80, y+40) # Horizontal Line through the window
