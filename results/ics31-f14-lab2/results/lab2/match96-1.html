KRONE_PER_EURO = 7.46
KRONE_PER_POUND = 8.81
KRONE_PER_DOLLAR = 5.50
name = input('Business name: ')
euros = float(input('Number of euros: '))
pounds = float(input('Number of pounds: '))
dollars = float(input('Number of dollars: '))
print('\nCopenhagen Chamber of Commerce')
print('Business name: ', name)
krone_to_euro = euros * KRONE_PER_EURO
krone_to_pound = pounds * KRONE_PER_POUND
krone_to_dollar = dollars * KRONE_PER_DOLLAR
print(euros, 'euros is', krone_to_euro, ' krone')
print(pounds, 'pounds is', krone_to_pound, ' krone')
print(dollars, 'dollars is', krone_to_dollar, ' krone')
total_krone = krone_to_euro + krone_to_pound + krone_to_dollar
print('\nTotal krone: ', total_krone)
