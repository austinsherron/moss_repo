def convert_to_krones():
    KRONE_PER_EURO = 7.46
    KRONE_PER_POUND = 8.81
    KRONE_PER_DOLLAR = 5.50
    name = input("what is your business name?")
    euros = int(input("How many euros does your company have?"))
    pounds = int(input("How many pounds does your company have?"))            
    dollars = int(input("How many dollars does your company have?"))
    euro_to_krone = euros * KRONE_PER_EURO
    pound_to_krone = pounds * KRONE_PER_POUND
    dollar_to_krone = dollars * KRONE_PER_DOLLAR
    print("Copenhagen Chamber of Commerce")
    print("Business name:", name)
    print(euros, "euros is", euro_to_krone, "krone")
    print(pounds, "pounds is", pound_to_krone, "krone")
    print(dollars, "dollars is",dollar_to_krone, "krone")
    print("Total krone:", euro_to_krone + pound_to_krone + dollar_to_krone)
convert_to_krones()
