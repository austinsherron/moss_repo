import math # import math to use sqrt function
TRANSFORM_X = 200 # Values use to move the whole diamond as whole
TRANSFORM_Y = 20
side_length = 200

# Next four line draw a red diamond.
#   creates a line at a 45deg angle with the side length.
#   Trigonometry for x-coordinate/y-coordinates -> sqrt(2)/2 * sidelength

my_canvas2.create_line(0 + TRANSFORM_X, 0 + TRANSFORM_Y,
                      (math.sqrt(2)/2)*side_length + TRANSFORM_X, (math.sqrt(2)/2)*side_length + TRANSFORM_Y, fill = 'purple2')

                       0 + TRANSFORM_X, (math.sqrt(2)/2)*side_length*2 + TRANSFORM_Y, fill = 'purple2')
