# Exchange rate constants
KRONE_PER_EURO = 7.46
KRONE_PER_POUND = 8.81
KRONE_PER_US = 5.50

print('Please provide this information:')
business_name = input('Business name: ')
euros = float(input('Number of euros: '))
pounds = float(input('Number of pounds: '))
dollars = float(input('Number of dollars: '))

#Convert currency
converted_euros = euros * KRONE_PER_EURO
converted_pounds = pounds * KRONE_PER_POUND
converted_dollars = dollars * KRONE_PER_US
total_krone = converted_euros + converted_pounds + converted_dollars

#Print new values
print('\nCopenhagen Chamber of Commerce')
print('Business name: ', business_name)
print(euros, "euros is", converted_euros, 'krone')
print(pounds, "pounds is", converted_pounds, 'krone')
print(dollars, "dollars is", converted_dollars, 'krone')


print('\nTotal krone: ', total_krone)
print('\n')
