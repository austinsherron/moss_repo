# d
EURO_TO_KRONE   = 7.46
POUND_TO_KRONE  = 8.81
DOLLAR_TO_KRONE = 5.50

print('Please provide this information:')

name        = input('Business name: ')
num_euros   = int(input('Number of euros: '))
num_pounds  = int(input('Number of pounds: '))
num_dollars = int(input('Number of dollars: '))

num_euros_to_krone = num_euros * EURO_TO_KRONE
num_pounds_to_krone = num_pounds * POUND_TO_KRONE
num_dollars_to_krone = num_dollars * DOLLAR_TO_KRONE

total_krone = num_euros_to_krone + num_pounds_to_krone + num_dollars_to_krone


print('\nCopenhagen Chamber of Commerce')
print('Business name: ', name)
print(num_euros, 'euros is', num_euros_to_krone, 'krone')
print(num_pounds, 'pounds is', num_pounds_to_krone, 'krone')
print(num_dollars, 'dollars is', num_dollars_to_krone, 'krone', '\n')
print('Total krone:', total_krone, '\n')
