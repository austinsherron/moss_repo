# d
KRONE_PER_EURO = 7.46
KRONE_PER_POUND = 8.81
KRONE_PER_DOLLAR = 5.50

print('Please provide this information:')
business_name = input('Business name: ')
euros = int(input('Number of euros: '))
pounds = int(input('Number of pounds: '))
dollars = int(input('Number of dollars: '))

e_krone = euros * KRONE_PER_EURO
p_krone = pounds * KRONE_PER_POUND
d_krone = dollars * KRONE_PER_DOLLAR
total_krone = e_krone + p_krone + d_krone

print('\nCopenhagen Chamber of Commerce',
      '\nBusiness name: ', business_name,
      '\n', euros, 'euros is ', e_krone, ' krone',
      '\n', pounds, 'pounds is ', p_krone, ' krone',
      '\n', dollars, 'dollars is ', d_krone, ' krone')
print('\nTotal krone: ', total_krone)
