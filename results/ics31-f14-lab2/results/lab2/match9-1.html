#Problem 2 - A diamond with sides of 200 pixels long
"""
Each side of the diamond can be seen as the hypotenuse of a right triangle with angles of 45°, 45°, and 90°.
Therefore, if the hypotenuse is designated to be 200 pixels long, the other two sides of the triangle can be solved
for using the Pythagorean theorem and are both the same length. They represent the the vertical and horizontal distance
the four corners are spaced apart from each other (designed in the code as amount_to_move).
"""
from math import sqrt
length_of_side = 200
amount_to_move = sqrt(((length_of_side) ** 2) / 2) #Solved for using the Pythagorean theorem

#The coordinates of the four corners of the diamond
point1_x = 200
point1_y = 50
point2_x = 200 - amount_to_move
point2_y = 50 + amount_to_move
point3_x = 200
point3_y = 50 + 2 * amount_to_move
point4_x = 200 + amount_to_move
point4_y = 50 + amount_to_move
my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

#Diamond created using lines connecting the four corners represented by the coordinates above
my_canvas.create_line(point1_x, point1_y, point2_x, point2_y, fill='red')  # Draw red line
my_canvas.create_line(point2_x, point2_y, point3_x, point3_y, fill='blue')   # Draw blue line
my_canvas.create_line(point3_x, point3_y, point4_x, point4_y, fill='green')   # Draw green line
my_canvas.create_line(point4_x, point4_y, point1_x, point1_y, fill='yellow')   # Draw yellow line

tkinter.mainloop()          # Combine all the elements and display the window
