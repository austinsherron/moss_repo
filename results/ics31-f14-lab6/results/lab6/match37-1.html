def initials(s:str) -> str:
    ''' Returns initials of the name in all capital letters '''


    return ''.join(s[0] for s in s.split()).upper()

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

alphabet = 'abcdefghijklmnopqrstuvwxyz'


def Caesar_encrypt(s:str, n:int) -> str:
    ''' Return encrypted string '''
    a = alphabet[n:]+alphabet[0:n]
    table = s.maketrans(alphabet, a)
    return s.translate(table)

assert Caesar_encrypt('hello', 5) == 'mjqqt'

    

def Caesar_decrypt(s:str, n:int) -> str:
    ''' Decrypts the string '''
    a = alphabet[n:]+alphabet[0:n]
    table = s.maketrans(a, alphabet)
    return s.translate(table)

assert Caesar_decrypt('mjqqt', 5) == 'hello'

punctuation = '()?:;,.!/"'
def replace_punctuation(L:str)->list:
    ''' Changes punctuation in list to white space '''
    table = L.maketrans(punctuation, '          ')
    L = L.translate(table)
    return L.split()
    

def list_of_words(L:list) -> list:
    ''' Returns list of words with white space and punctuation (except
    apostrophes) removed '''
    result = []
    for i in range(len(L)):
        result.append(replace_punctuation(L[i]))
    return result
