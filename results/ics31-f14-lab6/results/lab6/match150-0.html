def punctuation_blank(a: str) -> str:
    " returns str w/o punctuation "
    result = ''
    for b in a:
        if b in 'AaBbCcDdEeFfGgHhIiJjKkLlMmNnOoPpQqRrSsTtUuVvWwXxYyZz 1234567890':
            result += b
    return result
def count_words(c: str) -> int:
    " counts the # of words within a string "
    e = 1
    for d in c:
        if d==' ':
            e = e + 1
    return e
assert count_words('Can you help me?') == 4
print(count_words('Hello, how are you today?'))
def sentence_stats(z: str) -> str:
    " returns a specified set of sentence stats "
    print('Characters:', len(punctuation_blank(z)))
    print('Words:', count_words(z))
    print('Average word length:', len(punctuation_blank(z))/count_words(z))

def Caesar_encrypt(j:str, k: int) -> str:
    "Returns the ciphertext"
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    alphabet2 = ''
    x = (k%26)
    alphabet2 = alphabet2.join(alphabet[x:].split() + alphabet[0:x].split())
    j = j.translate(str.maketrans(alphabet, alphabet2))
    return j
print(Caesar_encrypt('please work', 1))
print(Caesar_encrypt('please work', 29))

def Caesar_decrypt(m:str, n: int) -> str:
    "Returns the plaintext"
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    alphabet2 = ''
    x = (n%26)
    alphabet2 = alphabet2.join(alphabet[x:].split() + alphabet[0:x].split())
    m = m.translate(str.maketrans(alphabet2, alphabet))
    return m
print(Caesar_decrypt('qmfbtf xpsl', 1))
print(Caesar_decrypt('qmfbtf xpsl', 27))
