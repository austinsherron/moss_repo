def sentence_stats(a:str)->str:
        '''Prints out statistics about the string'''
        print('Characters: ',character_count(a),'\n'+
          'Words: ',words_count(a),'\n'+
          'Average word lengths: ',average_word_length(a))

def character_count(a:str)->int:
    return len(a)

def count_without_space(a:str)->int:
    space=' '
    no_space=''
    for i in a:
        if i not in space:
           no_space=no_space+i
    return len(no_space)
    
def non_punctuation(a:str)->str:
    punctuations = '''!()-[]{};:'"\,<>./?@#$%^&*_~'''
    non_punc=''
    for i in a:

        if i not in punctuations:
            non_punc=non_punc+i
    return non_punc

def words_count(a:tuple)->int:
    return len(non_punctuation(a).split())
    
def average_word_length(a:str)->int:
    return count_without_space(non_punctuation(a))/(words_count(a))

def initials(a:str)->str:
        word_list=a.split()
        word_str=''

        freq_list=[]

        star_list=[]

alphabets='abcdefghijklmnopqrstuvwxyz'
def rotation(n:int)->str:
        '''rotate the alphabet'''
        newstring=alphabets
        for i in range(n):
                newstring+=newstring[i]
        return newstring[n:]


def Caesar_encrypt(s:str,n:int)->str:
        '''Encrypt s'''
        s=s.lower()


        if n>25:
                n=(n%25)-1
        trantab=s.maketrans(alphabets,rotation(n))
        return s.translate(trantab)
        
print(Caesar_encrypt('hello',2))
print(Caesar_encrypt("ILOVEYOU",2))
print(Caesar_encrypt('hello',28)) # returns the same result as when n=2

def Caesar_decrypt(s:str,n:int)->str:
        s=s.lower()
        if n>25:
                n=(n%25)-1
        reverse=s.maketrans(rotation(n),alphabets)
        return s.translate(reverse)

print(Caesar_decrypt('jgnnq',2))
print(Caesar_decrypt('knqxgaqw',2))
print(Caesar_decrypt('jgnnq',28)) # returns the same result as when n=2

def remove_punctuations(s: str)->str:
        punctuations=',./:;"`!@#$%^&*(){}|\''
        for i in punctuations:
                if i in s:
                        s=s.replace(i,'')
        return s
