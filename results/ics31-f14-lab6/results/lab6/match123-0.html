def blanks(sent:str) -> str:
    #Function turns a string with all whitespaces removed.
    result=('')
    for i in sent:
        if i == (' '):
            result+=('')
        else:
            result+=i
    return result


def punctuation(sent:str) -> str:
    #Function translates all punctuation marks into white spaces
    p = str.maketrans('.,?!:*',
                      '      ')
    return sent.translate(p)

def initials(name:str) -> str:
    #Returns the first letter of each word capitalized all in one string
    n = name.upper() #Makes the entire string capitalized
    result=('')
    x = str.split(n) #Splits the string into separate objects
    for i in x:
        result+=i[0] #Adds the first letter of each object to a blank string and returns it
    return result

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def Caesar_encrypt (x:str, y:int)->str:
    #Encrypts a string using y as a code for a Caesar cipher
    #This makes it so that if y is greater than 25 it will loop back around
    if y > 25:
        y = y%26 #You use mod to get the remainder of y/26 because if you're moving 26 spaces you end back at the spot it originally was
    #Changes all the letters in string ALPHABET above to an ALPHABET string that starts at y and ends at y
    change = str.maketrans(ALPHABET,
                           ALPHABET[y:] + ALPHABET[:y])
    #Changes the capital letters in the string by using the .upper function applied to the ALPHABET string
    change2 = str.maketrans(ALPHABET.upper(),
                            ALPHABET.upper()[y:] + ALPHABET.upper()[:y])
    c = x.translate(change)
    #After encrypting the lower case letters, this will encrypt the upper case letters and return the entire string
    return c.translate(change2)
print(Caesar_encrypt('Hi Andy Z. Le',1))
def Caesar_decrypt(x:str, y:int)->str:
    if y > 25:
        y = y%26
    #Reverse of the above function to decrypt the message
    change = str.maketrans(ALPHABET[y:] + ALPHABET[:y],
                           ALPHABET)
    change2 = str.maketrans(ALPHABET.upper()[y:] + ALPHABET.upper()[:y],
                            ALPHABET.upper())
    c = x.translate(change)
    return c.translate(change2)
print(Caesar_decrypt('Ij Boez A. Mf', 1))

def stats(x:list)->str:
    #Prints out a list of stats about a list of strings
    result = 0
    result_empty = 0
    result_nonempty = 0
    result_nonempty_l = 0
    for i in x:
        if i.isspace() == True: #checks if a line is just empty white space, if it is add 1 to a variable that will be used to determine the number of empty lines
            result_empty += 1
        else: #else add 1 to a variable that will be used to determine the number of non empty lines and add the length of the string to a variable that determines the total characters in all the nonempty lines.
            result_nonempty += 1
            result_nonempty_l += len(i)
    for i in x:
        result += len(i)
    print(len(x),'   lines in list')
    print(result_empty, '   empty lines')
    print(result/len(x), 'average characters per line')
    print(result_nonempty_l/result_nonempty, 'average characters per non-empty line')

#Return a list of individual words from an existing list of strings with all punctuation and white space removed
def list_of_words(x:list)->list:
    result=[]
    for i in x:
        #Reusing the punctuation function from c.2 to turn all punctuation to white space which will be unincluded when using split()
        result += str.split(punctuation(i))
    return result
