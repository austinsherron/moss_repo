    '''returns the initials of the name in all capital letters'''

def Ceaser_encrypt (x:str, y:int) -> str:
    '''Takes a string and an interger and ceaser cipher the string by the indicated key'''
    lowercase = 'abcdefghijklmnopqrstuvwxyz'
    upperx = x.upper()
    if y > 26:
        y = y % 26
    z = lowercase[y:] + lowercase[0:y]
    table = upperx.maketrans('ABCDEFGHIJKLMNOPQRSTUVWXYZ', z)
    print(upperx.translate(table))

def Ceaser_decrypt (x:str, y:int) -> str:
    '''Takes a string and an interger and ceaser cipher the string by the indicated key'''
    lowercase = 'abcdefghijklmnopqrstuvwxyz'
    upperx = x.upper()
    z = lowercase[-y:] + lowercase[0:-y]
    table = upperx.maketrans('ABCDEFGHIJKLMNOPQRSTUVWXYZ', z)
    print(upperx.translate(table))

def list_of_words(x:list) -> list:
    '''Removes all white space and puctuation in a list and just returns a list of words'''
    punctuation = str.maketrans('.,?!:;" ','        ')


    result = []
    for i in x:
        s = 0
        i = i.translate(punctuation)
        for j in range(len(i)):
            
            if i[j] == ' ':
                if not j == len(i) - 1:
                    if not i[j+1] == ' ':
                        s = j+1
            else:
                if j == len(i) - 1:
                    result.append(i[s:])
                elif i[j+1] == ' ':
                    result.append(i[s:j+1])
    return result
