import string

def punctuation(sentence:str)->str:
    '''Replaces all characters in a string that are not letters with spaces'''
    for i in sentence:
        if i not in string.ascii_letters:
          sentence = sentence.replace(i,' ')
    return sentence

def sentence_stats(sentence:str) -> str:
    '''Prints the # of characters, # of words, and average word length of a string'''
    sentence = punctuation(sentence)
    word_list = sentence.split()
    num = 0
    for x in word_list:
        num += len(x)
    num /= len(word_list)
    print (word_list)
    print('Characters:',len(sentence),'\nWords',len(word_list),'\nAverage word length:',num)

def initials (name:str) -> str:
    '''Returns the initials of a string of a name'''
    name_list = name.split()
    name_initials = ('')
    for x in name_list:
        name_initials += x[0]
    name_initials = name_initials.upper()
    return name_initials

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def alphabet_rotate(num:int) -> str:
    '''Takes a number and rotates the characters in the string alphabet by that number'''
    if num > 26:
            num = num%26
    alphabet1 = ''
    alphabet1 += alphabet[num:] + alphabet[:num]
    return alphabet1


def Caesar_encrypt(msg:str,key:int) -> str:
    '''Takes a string and encrypts it according to the key'''
    rotate = str.maketrans(alphabet,alphabet_rotate(key))
    return msg.translate(rotate)

assert Caesar_encrypt("hi there",3) == 'kl wkhuh'

print(Caesar_encrypt('cat',3))
print(Caesar_encrypt('cat',29))


def Caesar_decrypt(msg:str,key:int) -> str:
    '''Takes an encrypted string and decrypts it according to the key'''
    rotate = str.maketrans(alphabet_rotate(key),alphabet)
    return msg.translate(rotate)

assert Caesar_decrypt('kl wkhuh',3) == 'hi there'

def stats (strlist:list):
    '''Takes a list of string and prints the number of lines, the number of empty lines, the averagee number of characters for all lines, and the average number of characters for non-empty lines'''
    lines_in_list = len(strlist)
    empty_lines = 0
    total_chars_string = 0

    for x in strlist:
        #counts the empty lines in the string and adds the number to empty_lines
        if x == '':
            empty_lines += 1    
            
    for y in strlist:
        #counts the total number of characters
        total_chars_string += len(y)
        
    print('{:5}   lines in the list\n{:5}   empty lines\n{:7.1f} average characters per line\n{:7.1f} average characters per non-empty line'.format(lines_in_list,empty_lines,total_chars_string/lines_in_list,total_chars_string/(lines_in_list-empty_lines)))

def list_of_words(strlist:list) -> list:

        result = []
