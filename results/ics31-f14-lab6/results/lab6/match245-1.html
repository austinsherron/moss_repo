def convert(x:str) -> str:
    'Translates all punctuation marks to blanks'
    y = ['.','!','?','*',':']
    j = ''
    for i in x:
        for k in y:
            if i == k:
                i = ' '
        j = j + i
    return j
        
def sentence_stats(x:str) -> str:
    'Takes in a string as a parameter and prints out statistics'
    z = 0
    j = convert(x)
    y = j.split()
    for i in range(len(y)):
        z += len(y[i])
        k = z/len(y)
    print('Characters: ', len(x))
    print('Words: ', len(y))
    print('Average word length: ', k)

def initials(x:str) -> str:
    'Takes a string representing a full name and returns the initials in all capital letters'
    z = ''
    y = x.split()
    for i in y:
        z = z + i[0]
    return z.upper()


assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def Caesar_encrypt(x:str,y:int) -> str:
    'Encrypts a str with Caesar cipher with a key y'
    k = []
    t = ''
    if y > 26:
        y = y % 26
    for i in x:
        if i != ' ':
            i = chr(ord(i)+y)
            k.append(i)
        if i == ' ':
            k.append(i)
    for q in k:
        t = t + q
    return t

def Caesar_decrypt(x:str,y:int) -> str:
    'Decrypts a str with Caesar cipher with a key y'
    k = []
    t = ''
    if y > 26:
        y = y % 26
    for i in x:
        if i != ' ':
            i = chr(ord(i)-y)
            k.append(i)
        if i == ' ':
            k.append(i)
    for q in k:
        t = t + q
    return t

def list_of_words(L:list) -> list:
    'Takes a list of strings and returns a list of individual words with all white space and punctuation removed'
    k = []
    for i in L:
        f = i.split()
        for q in f:
            j = q.replace(' ','').replace('\t','').replace('\s','').replace(',','').replace('.','').replace('?','').replace('!','')
            k.append(j)
    return k
