def intitals(name):
    char_length = len(name)
    initial = name[0:1]
    for i in range(char_length):
        if name[i] == ' ':
            initial_location = name[i]
            initial += ' ' + initial_location 
    return initial.upper()

def stats(l: list) -> None:
    '''prints statistics of a list.'''
    N = 0
    P = 0
    for i in l:
        N += len(i)
        if len(i) == 0:
            P += 1
    print("{:2}{}{:2}{}{}{}{}{}"
          .format(len(l),
            '    Lines in the list \n', P, '    Total empty \n', N/len(l),
            '  Average characters per line \n', N/(len(l)-P),
            '  Average characters per non-empty line', sep=""))

import string
def list_of_words(l: list) -> str:
    '''returns a list of individual
    words with all white space and
    punctuation removed'''
    result = []
    for x in l:
        result.extend(x.split())

    p = string.punctuation.replace("'", "")
    for word in range(len(result)):
        new = ""
        for c in result[word]:
            if c not in p:
                new += c
        result[word] = new

    return result
print(list_of_words(list1))
