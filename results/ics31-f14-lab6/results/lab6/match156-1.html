             'm','n','o','p','q','r','s','t','u','v','w','x','y','z']

def set_the_table1() ->str:
    '''set the first for .translate'''
    charoriginal = []
    charoriginal.extend(alphabet)
    charsoriginal = ''
    for i in charoriginal:
        charsoriginal = charsoriginal+i
    return charsoriginal

def set_the_table2(stt2:int) ->str:
    '''set the second for .translate'''
    char1 = []
    char1.extend(alphabet)
    char2 = []
    char3 = []
    chars3 = ''
    if stt2 > 26:
        stt2 = stt2 % 26
               
    for i in range(stt2):
        char2.extend(char1[0])
        char1.remove(char1[0])
            
    char3.extend(char1)
    char3.extend(char2)

    for i in char3:
        chars3 = chars3+i
    return chars3

def Caesar_encrypt(ces:str,cek:int) ->str:
    '''encrypt to ciphertext'''
    table = str.maketrans(set_the_table1(),set_the_table2(cek))
    return ces.translate(table)
    

def Caesar_decrypt(cds:str,cdk:int) ->str:                                       
    '''decrypt to plaintext'''            


    table = str.maketrans(set_the_table2(cdk),set_the_table1())
    return cds.translate(table)

def list_of_words(low:list) ->list:
    '''return a list of words without space and punctuation'''


    nlist = []
    for i in low:
        nlist.extend(i.split(' '))
    while nlist.count('') != 0:
        for i in nlist:
            if i == '':
                nlist.remove(i)
    for i in range(len(nlist)):
        nlist[i]=nlist[i].strip('.')
        nlist[i]=nlist[i].strip(',')
    return nlist
