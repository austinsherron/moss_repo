def trans_punct_blanks(s: str) -> str:
    b = ''


    for i in s:
        if i in letters:
            b = b + i
        else:
            b = b + ' '
    return b

'''Takes a string and returns index[0] of substirngs'''
def initials(a: str)-> str:


    l = []
    l = a.split()
    p = ''
    for i in l:
        p = p + i[0].upper()
    return p
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'
