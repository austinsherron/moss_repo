def punc_to_blank(s:str) -> str:
    '''Takes a string as a input and translate all punctuation to blanks'''
    table = str.maketrans('`~!@#$%^&*()-_=+[{]}\|;:,<.>/?',
                          '                              ')
    return s.translate(table)
def sentence_stats(s:str):
    '''takes in a string as a parameter and prints out these statistics
    about the string: its length in characters, its length in words, and
    the average length of each word.'''
    print('Characters: ', len(s))
    wordlist = punc_to_blank(s).split()
    print('Wods: ', len(wordlist))
    s1 = ''.join(wordlist)
    print('Average word length: ', len(s1)/len(wordlist))

def initials(s:str) -> str:
    '''takes as input a string representing a full name (e.g.,  Robert B. Qwerty)
    and returns the initials of the name in all capital letters (e.g., RBQ).'''
    initial = ''
    s = s.upper()
    wordlist = s.split()
    for w in wordlist:
        initial += w[0]
    return initial

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def distribution_of_rolls(n:int):
    '''Takes one number—the number of times to roll two dice—and
    prints thedistribution of the values of those rolls'''
    print('Distribution of dice rolls','\n')
    s = ''
    formatStr = '{0:2}:{1:6} ( {2:4}% ) '
    for i in range(n):
        s += str(roll2dice())
    for i in range(2,13):
        num = s.count(str(i))
        print(formatStr.format(i, num, num * 100 / n), '*' * num)
    print('-----------------------------')
    print('       ', n , 'rolls           ')
    return

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def roated(n:int) -> str:
    '''Takes a number and produces a "rotated" alphabet with the specified
    number of characters taken off the front and added on to the end of the
    string.'''
    return ALPHABET[n:] + ALPHABET[:n]

def Caesar_encrypt(s:str, key:int) -> str:
    '''Takes two arguments: a string containing the message and an int
    for the key,indicating how far down the alphabet to find each substitute
    letter and returns the ciphertext'''
    table = str.maketrans(ALPHABET,roated(key%26))
    return s.translate(table)
assert Caesar_encrypt('hi there',29) == 'kl wkhuh'
assert Caesar_encrypt('hi', 26) == 'hi'

def Caesar_decrypt(s:str, key:int) -> str:
    '''Takes two arguments: a string containing the message and an int for
    the key, indicating how far down the alphabet to find each substitute
    letter and returns the plaintext.'''
    table = str.maketrans(roated(key%26), ALPHABET)
    return s.translate(table)
assert Caesar_decrypt('kl wkhuh',3) == 'hi there'

def stats(L:[str]):
    '''prints the statistics of the list of strings, including # of lines,
       # of empty lines, average characters per line, and average characters
       per non-empty line'''
    formatStr = '{0:6.1f}  {1}'
    print(formatStr.format(len(L),'lines in the list'))
    SL = L[:]
    TotalofNonEmpty = 0
    num_of_empty = 0
    for i in range(len(SL)):
        SL[i] = SL[i].strip()
        if SL[i] == '':
            num_of_empty += 1
    print(formatStr.format(num_of_empty,'empty lines'))
    totalCharacters=0
    for s in L:
        totalCharacters+= len(s)
    print(formatStr.format(totalCharacters/ len(L), 'average characters per line'))
    for s in SL:
        TotalofNonEmpty += len(s)
    print(formatStr.format(TotalofNonEmpty/ (len(L) - num_of_empty),'average characters per non-empty line'))

def list_of_words(SL:[str]) -> 'list of str':
    '''takes a list of strings and returns a list of individual words with all
       white space and punctuation removed except for apostrophes and single quotes'''

    result = []
    table1 = str.maketrans('`~!@#$%^&*()-_=+[{]}\|;:,<.>/?',
                           '                              ')
    for s in SL:
        s = s.translate(table1)
        result.extend(s.split())
    return result

print(list_of_words(['hello there!', 'bye-bye.','$5']))
print(list_of_words(['apostrophe\'s car is new']))
