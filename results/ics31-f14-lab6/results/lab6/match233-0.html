def sentence_stats(s: str):
    '''Prints length of string, amount of words, and average word length'''
    print('Characters: ', len(s))
    s = s.translate(str.maketrans(punctuation, ' '*len(punctuation))).split()
    print('Words: ', len(s))
    avg = 0
    for n in s:
      avg +=(len(n))
    print('Average word length: ', avg/len(s))
    
sentence_stats('Hello World!')
sentence_stats('I would like some yogurt, please')

def initials(s: str)-> str:
  '''Takes a string/name and returns the first initial of each word together'''
  s = s.upper().split()
  Initials = ''
  for word in s:
    Initials += word[0]
  return Initials
assert initials('Roshan Suasin') == 'RS'
assert initials('Bill Gates') == 'BG'
assert initials('David g kay') == 'DGK'

  result = [0 for num in range(12)]

  for num in range(2,13):
    print('{0:2}: {1:5} ({2:4.1f}%) {3:>}'.format(num, result[num-2], result[num-2]* n/100, '*' * result[num-2]))

  print('{:12} rolls'.format(n))

alpha = 'abcdefghijklmnopqrstuvwxyz'


def rotate(n: int)-> str:
  '''Rotates the alphabet by the given number'''
  return alpha[n%26:] + alpha[0:n%26]



def Caesar_encrypt(plainText: str, key: int)-> str:
  '''Encrypts a Text using a Caesar Cipher with # key'''
  return plainText.lower().translate(str.maketrans(alpha, rotate(key)))
assert Caesar_encrypt('hello world',6) == 'nkrru cuxrj'
assert Caesar_encrypt('Hello My Servile', 6) == 'nkrru se ykxbork'

def Caesar_decrypt(cipheredText: str, key: int)-> str:
  '''Decrypts a ciphered Text using a Caesar Cipher with # key'''
  return cipheredText.lower().translate(str.maketrans(rotate(key),alpha))

def stats(textList: list):
  print('{:7} lines in the list'.format(len(textList)))
  
  result = 0
  for i in textList:
    if i == '':
      result += 1
  print('{:7} empty lines'.format(result))
  
  avg = 0
  for n in textList:
    avg +=(len(n))
  print('{:7.1f} average characters per line'.format(avg/len(textList)))
  
  nonEmpty = 0
  for n in textList:
    if n != '':
      nonEmpty += 1
  print('{:7.1f} average characters per non-empty line'.format(avg/nonEmpty))

stats(Text)

def list_of_words(l: list)-> list:
  '''Takes a list of strings and returns a list of individial words of the strings
  without whitespace or punctuations (except apostrophes/single quotes)'''
  wordList = []
  for sentence in l:
    newSentence = sentence.strip(punctuation).split()
    wordList.extend(newSentence)
  return wordList
