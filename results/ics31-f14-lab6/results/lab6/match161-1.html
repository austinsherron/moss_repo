def replace_punct(s: str) -> str:
    "Replaces punctuation with blank space"
    table = str.maketrans('.,!>:;"?','        ')
    return s.translate(table)

    "Creates initials of name entered as parameter"

def rotate(num:int) -> str:
    "Rotates the beginning of the alphabet by num"
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    if num > 25:
        num = num%26
    new_alphabet = alphabet[num:] + alphabet[:num]
    return new_alphabet
def Caesar_encrypt(S: str, num: int) -> str:
    "Creates encryption"
    new_alpha = rotate(num)
    alphabet = str.maketrans('abcdefghijklmnopqrstuvwxyz',new_alpha)
    S = S.lower()
    S = S.translate(alphabet)
    return S
assert Caesar_encrypt('aaaa',2) == 'cccc'
assert Caesar_encrypt('This is a test statement', 23) == 'qefp fp x qbpq pqxqbjbkq'


def Caesar_decrypt(S: str, num: int) -> str:
    "Decodes encryption"
    new_alpha = rotate(-num)
    alphabet = str.maketrans('abcdefghijklmnopqrstuvwxyz',new_alpha)
    S = S.lower()
    S = S.translate(alphabet)
    return S
assert Caesar_decrypt('qefp fp x qbpq pqxqbjbkq',23) == 'this is a test statement'

def remove_punctuation(lst: [str]) -> [str]:
    "Takes a list of strings and returns the list without the punctuation"
    result = []
    for i in lst:
        check = ''
        for x in i:
            if x.isalpha() == True:
                check += x
        result.append(check)
    return result
          
def list_of_words(lst: [str]) -> [str]:
    "Returns a list of words from the given list of strings"
    result = []
    for i in lst:
        i = i.split()
        for x in i:
            result.append(x)
    return remove_punctuation(result)
