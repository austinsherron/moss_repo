def character_stats(c:str)->float:
    "Return the length in characters of the string"
    return len(c)
def word_stats(w:str)->float:
    "Return the length in words of the string"
    table= str.maketrans('~!@#$%^&*()-_+=[]{}`;:",./?<>\'',' '*30 )
    word=w.translate(table).split()
    return len(word)      
def sentence_stats(s:str)->float:
    "Return the length in characters, the length in words and the average length of each word in the string"
    table= str.maketrans('~!@#$%^&*()-_+=[]{}`;:",./?<>\'',' '*30 )
    total_word_only= len(s.translate(table).replace(' ',''))
    average_word_length= total_word_only/word_stats(s)
    print('\nCharacters:', character_stats(s),'\nWords:', word_stats(s),'\nAverage word length:', average_word_length,'\n')

def initials(n:str)->str:

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def Caesar_encrypt(s:str, key:int)->str:
    "Return the ciphertext"
    s=s.lower()
    if key > 25:
        key= key % 26
    cipher=ALPHABET[key:]+ALPHABET[:key]
    table= str.maketrans(ALPHABET, cipher)
    encryption = s.translate(table)
    return encryption
assert Caesar_encrypt('Anteater', 2) == 'cpvgcvgt'
assert Caesar_encrypt('I need to sleep', 29) == 'l qhhg wr vohhs'


def Caesar_decrypt(a:str, key:int)->str:
    "Return the plaintext"
    a=a.lower()
    if key > 25:
        key= key % 26
    cipher= ALPHABET[key:]+ALPHABET[:key]
    table=str.maketrans(cipher,ALPHABET)
    decryption= a.translate(table)
    return decryption
assert Caesar_decrypt('gva gva gva', 7) == 'zot zot zot'
assert Caesar_decrypt('am pfowbg vojs pssb sohsb pm ncapwsg', 40) == 'my brains have been eaten by zombies'

def stats(l:list)->int:
    average_per_line = []
    average_per_nline = []
    count_of_nline = 0
    for i in l:
        average_per_line.append(len(i))
    for i in l:
        x = i.strip(' ')
        if len(x) != 0:
            average_per_nline.append(len(i))
        else:
            count_of_nline +=1
    lines = len(l)
    empty_lines = count_of_nline
    average_character = sum(average_per_line)/len(l)
    average_character2 = sum(average_per_nline)/(len(l)-count_of_nline)
    print("{:3d}{:2} lines in the list".format(lines,' '))
    print("{:3d}{:2} empty lines".format(empty_lines,' '))
    print("{:5.1f} average characters per line".format(average_character))
    print("{:5.1f} average characters per non-empty line".format(average_character2))
