def sentence_stats(sentence:str)->float:
    '''Takes a string and prints the length of the chars, length of the words, and the av
       average length of each word'''
    count=0
    print("Characters: ", len(sentence))
    result=sentence.split()
    print("Words: ", len(result))
    for i in result:
        count=count+len(i)
    print("Average word length: ", count/len(result))

def initials(full_name:str)->str:
    '''Takes the string and returns the initials of the full name'''        
    new=''
    Capital_letters= full_name.upper()
    result=Capital_letters.split()

        new+=i[0]

def rotate_alphabet(x:int)->str:
    '''sets starting point of alphabet and then goes all the way around back to that letter'''
    A='abcdefghijklmnopqrstuvwxyz'
    if x>25:
        x=x%26
    result=A[x:]+A[:x]
    return result
          
def Caesar_encrypt(word:str, num:int)->str:
    '''returns the cyphertext'''
    NL=rotate_alphabet(num)
    A=str.maketrans('abcdefghijklmnopqrstuvwxyz',NL)
    word=word.lower()
    word=word.translate(A)
    return word
print(Caesar_encrypt('Hi, I am a fool',7))



def Caesar_decrypt(word:str,num:int)->str:
    '''decrypts the message'''
    NL=rotate_alphabet(-num)
    A=str.maketrans('abcdefghijklmnopqrstuvwxyz', NL)
    word=word.lower()
    word=word.translate(A)
    return word
print(Caesar_decrypt('op, p ht h mvvs', 7))
