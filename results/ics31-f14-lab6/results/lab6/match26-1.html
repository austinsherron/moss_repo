def initials(name: str)->str:
    result = name.split()
    total = ''
    for i in result:
        x = i[0].capitalize()
        total += x
    return total

def distribution_of_rolls(num: int):
    result = []     #empty result list
    for i in range(num):    #for every number in the input num (ex: 200) it runs through the for loop
        result.append(roll2dice())  #calls the roll2dice function that returns the number of each dice added together (200 times)
    print('Distribution of dice rolls') 
    for i in range(2, 13):  #This loop starts at i = 2 and goes to i = 3,...., and ends at i = 12
        result_i = result.count(i)  #for each time it goes through the loop (i=2,3,4,etc.) result.count(i) checks for all the i in the list
                                    #Ex: for the first loop i = 2 and result.count(i) counts all the 2's in the reuslt list.  
        print('{:>3}, {}, {:>3}, {:>7f}, {:}'.format(i, '\t', result_i, ((result_i/num)*100)), '\t', (result_i*'*'))   #now it prints the formated information for each i in the loop
    print('------------------------------')
    print('\t', num, 'rolls')

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotated(num: int)->str:
    alphabet = ''
    alphabet += ALPHABET[num:]
    alphabet += ALPHABET[0:num]
    return alphabet

print(ALPHABET)
print(rotated(3))



def Caesar_encrypt(plaintext: str, num: int)->str:
    table = plaintext.maketrans('abcdefghijklmnopqrstuvwxyz', rotated(num))
    return plaintext.translate(table)

print(Caesar_encrypt('hi there', 3))

def Caesar_decrypt(ciphertext: str, num: int)->str:
    table = ciphertext.maketrans('abcdefghijklmnopqrstuvwxyz', rotated(-num))
    return ciphertext.translate(table)

print(Caesar_decrypt('kl wkhuh', 3))

def stats(list_of_str: list)->None:
    total_length = 0
    for line in list_of_str:
        length = len(line)
        total_length += length
        average_line = total_length/len(list_of_str)
        average_non_empty_line =  total_length/(len(list_of_str) - list_of_str.count(''))      
    print('{:>5}{:>24}{}{:>5}{:>18}{}{:>7.1f}{:>32}{}{:>7.1f}{:>42}'.format(len(list_of_str), 'lines in the list','\n',  list_of_str.count(''), 'empty lines', '\n', average_line, 'average characters per line', '\n', average_non_empty_line, 'average characters per non-empty line'))
