def no_punctuation(s: str) -> str:
    '''
    *********************************************
    NO_PUNCTUATION
    ---------------------------------------------
    Description:
    Translates the string so there is no punctuation

    PARAMETERS:
    s: str - string to modify

    RETURNS:
    str


    *********************************************
    '''
    result = ''
    for i in s:
        if not i in string.punctuation:
            result += i
    return result
assert no_punctuation('asdf123!!') == 'asdf123'

def average_word(s: str) -> float:
    '''
    *********************************************
    AVERAGE_WORD
    ---------------------------------------------
    Description:
    Returns the average length of each word in the
    string s

    PARAMETERS:
    s: str - string to count

    RETURNS:
    float
    *********************************************
    '''
    result = 0
    s = no_punctuation(s).split()
    if len(s) == 0:
        return result
    
    for i in s:
        result += len(i)

    return result/len(s)
assert average_word('I Love UCIe') == 3

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def Caesar_encrypt(s: str, key: int) -> str:
    '''
    *********************************************
    CAESAR_ENCRYPT
    ---------------------------------------------
    Description:


    Encypts the strings, s, with a given key

    PARAMETERS:
    s: str - string to encrypt
    key: int - key to use

    RETURNS:
    str
    *********************************************
    '''
    return s.lower().translate( str.maketrans(ALPHABET, ALPHABET[key % len(ALPHABET):] + ALPHABET[:key % len(ALPHABET)]))
assert Caesar_encrypt('Cat', 27) == 'dbu'


assert Caesar_encrypt('abc', 5) == 'fgh' 

def Caesar_decrypt(s: str, key: int) -> str:
    '''
    *********************************************
    CAESAR_DECRYPT
    ---------------------------------------------
    Description:
    Decypts the strings, s, with a given key

    PARAMETERS:
    s: str - string to decrypt
    key: int - key to use

    RETURNS:


    str
    *********************************************
    '''
    return Caesar_encrypt(s, -1 * key)
assert Caesar_decrypt('dbu', 1) == 'cat'
assert Caesar_decrypt('fgh', 5) == 'abc'

def empty_lines(s: [str]) -> int:
    '''
    *********************************************
    EMPTY_LINES
    ---------------------------------------------
    Description:
    Counts the number of empty lines in a list
    of strings

    PARAMETERS:
    s: [str] - list of strings

    RETURNS:
    int
    *********************************************
    '''
    result = 0
    for i in s:
        if len(i.split()) == 0:
            result += 1
    return result
assert empty_lines( ['', '        ', '          efwef  ']) == 2

def average_characters_per_line(s: [str]) -> float:
    '''
    *********************************************
    AVERAGE_CHARACTERS_PER_LINE
    ---------------------------------------------
    Description:
    Returns the average characters per line
    includes punctuation and blank lines

    PARAMETERS:
    s: [str] - list of strings

    RETURNS:
    int
    *********************************************
    '''
    total = 0
    if len(s)==0:
        return 0
    for i in s:
        total = total + len(i)
    return total/len(s)

def average_characters_per_non_empty_line(s: [str]) -> float:
    '''
    *********************************************
    AVERAGE_CHARACTERS_PER_LINE
    ---------------------------------------------
    Description:
    Returns the average characters per line
    includes punctuation but not empty lines

    PARAMETERS:
    s: [str] - list of strings

    RETURNS:
    int
    *********************************************
    '''
    total = 0
    empty = 0
    if len(s)==0:
        return 0
    for i in s:
        if not len(i.split()) == 0:
            total = total + len(i)
        else:
            empty +=1
    return total/(len(s)-empty)

def list_of_words(s: [str])->[str]:
    '''
    *********************************************
    LIST_OF_WORDS
    ---------------------------------------------
    Description:
    Returns a list of strings with all punctuation
    removed except an apostrophe

    PARAMETERS:
    s: [str] - list of strings

    RETURNS:
    [str]


    *********************************************
    '''
    result = []
    punc = list(string.punctuation)
    punc.remove("'")
    for i in s:
        i = ''.join(ch for ch in i if ch not in punc)
        result.extend(i.split())
    return result
