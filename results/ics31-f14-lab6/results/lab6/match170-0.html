def translate_punctuation(sentc: str) -> str:
    '''replaces punctuations with blanks'''
    table = str.maketrans("""!?'.:;*@#$%^&"()-_=+][{}\/>,<`~|""", "                                ")
    return sentc.translate(table)

def sentence_stats(sentence: str):
    '''takes in a string as a parameter and prints out these statistics about the string:
    its length in characters, its length in words, and the average length of each word.'''
    print('Characters: ', len(sentence))
    trans_sentence = translate_punctuation(sentence)
    word_list = trans_sentence.split()
    print('Words: ', len(word_list))

    count = 0
    for w in word_list:
        count += len(w)
        
    print('Average word length: ', count/len(word_list))

def initials(full_name: str) -> str:
    '''takes as input a string representing a full name (e.g., Robert B. Qwerty)
    and returns the initials of the name in all capital letters (e.g., RBQ).'''
    sep_name = full_name.split()
    initial = ""
    for n in sep_name:
        initial += n[0].upper()
    return initial

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def roll2dice() -> int:

def distribution_of_rolls(n:int):
    '''takes one number—the number of times to roll two dice—and
    prints the distribution of the values of those rolls in the form shown below.'''
    sum = []
    for i in range(n):
        sum.append(roll2dice())
    
    print("Distibrution of dice rolls: ")
    for d in range(2,13):
        counter = sum.count(d)
        print('{0:4} {1:5} {2:7.2f}% {3:}'. format(str(d), str(counter), (counter/n)*100 , counter * '*'))
        
    print("-" * 60)
    print("            ", n, "rolls")

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'



def rotate_alphabet(num: int)-> str:
    rotated=''
    rotated+=ALPHABET[num:]+ALPHABET[:num]
    return rotated

def Caesar_encrypt(message: str, key: int) ->str:
    '''takes two arguments: a string containing the message the plaintext for encryption
    and an int for the key, indicating how far down the alphabet to find each substitute letter.
    The encryption function returns the ciphertext'''
    encrypted = ''
    message = message.lower()
    table = str.maketrans(ALPHABET, rotate_alphabet(key))
    encrypted = message.translate(table)
    return encrypted
assert Caesar_encrypt('cat',3)== 'fdw'
assert Caesar_encrypt('cat',29)=='cat'

def Caesar_decrypt(message: str, key: int) -> str:
    '''takes two arguments: a string containing the message the the ciphertext for decryption
    and an int for the key, indicating how far down the alphabet to find each substitute letter.
    the decryption function returns the plaintext.'''
    decrypted = ''
    message = message.lower()
    table = str.maketrans(rotate_alphabet(key), ALPHABET)
    decrypted = message.translate(table)
    return decrypted
assert Caesar_decrypt('fdw',3)=='cat'

    483   empty lines
    53.7 average characters per line
    65.9 average characters per non-empty line'''

    print('{0:4}  '.format(len(L)), 'lines in the list')
    

    for i in L:
        empty_count = 0
        if i == '':
            empty_count += 1
            
    print('{0:4}  '.format(empty_count), 'empty lines')
    for i in L:
        total_characters=0
        total_characters+=len(i)

    
    print('{0:6.1f}'.format((total_characters)/(len(L))), 'average characters per line')

    non_empty_lines = len(L) - empty_count
    print('{0:6.1f}'.format(total_characters/non_empty_lines), 'average characters per non-empty line')

def list_of_words(L: list) -> list:
    '''takes a list of strings as above and returns a list of individual words with all white space
    and punctuation removed (except for apostrophes/single quotes).'''
    table = str.maketrans("!?.:;*@#$%^&()-_=+][{}\/>,<`~|", "                              ")
    word_list = []
    for i in L:
        trans_sentence = i.translate(table)
        word_list += trans_sentence.split()
    return word_list
