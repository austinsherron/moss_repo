def no_punc(sentence: str)->str:
    new = ""
    for i in sentence:
        if i == ' ' or (i>= 'A' and i<= 'Z') or (i>= 'a' and i<= 'z'):
            new = new + i
    return new

def sentence_stats(sentence: str):
    
    a = len(sentence)

    b = len(sentence.split())

def initials(name:str)->str:
    result = ''
    for i in name.split():
        result = result + i[0].upper()
    return result
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'

def stats(text: list):
    count = 0
    all_character = 0
    character = 0
    for i in text:
        all_character = all_character + len(i)
        character = all_character
        if i.replace(' ','') == '':
            count = count + 1
            character = all_character - len(i)
    print ("{:6}   lines in the list\n{:6}   empty lines\n{:8.1f} average characters per line\n{:8.1f} average characters per non-empty lines".format(len(text), count, all_character/len(text), character/(len(text)-count)))
