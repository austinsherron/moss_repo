def sentence_stats(first: str) -> str:
    tempString = "";
    
    for y in range(len(first)):
        if (check_punctuation(first[y]) == False):
            tempString += first[y];
            
    tempString = " ".join(tempString.split())
    spaces = count_spaces(tempString);
    
    print("\""+tempString+"\" statistics:");
    print("\tCharacters: " + str(len(first)));
    print("\tWords: " + str((spaces + 1)));
    print("\tAverage word length: " + str((len(tempString) - spaces)/(spaces + 1)));

def count_spaces(first: str) -> int:
    # Counts the spaces between words
    spaces = 0;
    for x in first:
        if (x == " "):
            spaces += 1;
    return spaces;
def check_punctuation(first: str) -> bool:
    # Checks for punctuation
    nonos = ["'", "?", ":", "!", ".", ",", "&", "*"];
    for x in nonos:
        if (first == x):
            return True;
    return False;

    txt = "";
    first = first.title();
    for x in first:
        for y in abc:

alphabet = "abcdefghijklmnopqrstuvwxyz";
def Caesar_encrypt(first: str, second: int) -> str:
    if second > 26:
        second = second % 26;
    return first.lower().translate(str.maketrans(alphabet, alphabet_rotate(second)));
    
def Caesar_decrypt(first: str, second: int) -> str:
    if second > 26:
        second = second % 26; 


    return first.lower().translate(str.maketrans(alphabet_rotate(second), alphabet));
def alphabet_rotate(move: int) -> str:
    return alphabet[move:] + alphabet[:move];

def stats(strings: list) -> str:
    empty = 0;
    lines = 0;
    sumCharacters = 0;
    for x in strings:
        lines += 1;
        if len(x) == 0:
            empty += 1;
        else:
            sumCharacters += len(x);
    print('{:6.0f}     {:}'.format((lines), 'lines in the list'));
    print('{:6.0f}     {:}'.format((empty), 'empty lines'));
    print('{:8.1f}   {:}'.format((sumCharacters/lines),'average characters per line'));
    print('{:8.1f}   {:}'.format((sumCharacters/(lines-empty)),'average characters per non empty line'));

    
stats(text2);
