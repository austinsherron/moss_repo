def no_punctuation(a:str)->str:
    ''' Removing the punctuation in a str '''
    a = a.replace('*', '')
    a = a.replace('?', '')


    a = a.replace('!', '')
    a = a.replace(':', '')
    a = a.replace('  ', ' ')
    a = a.replace('.', '')
    return a

def average_word_length(a:str, b:int)->float:
    ''' Return the average word length '''


    new = []
    x = 0
    y = 0
    length = 0
    while (y != -1):
         y = a.find(' ',x + 1)
         if(y == -1):
             new.append(a[x:].strip())
         else:
             new.append(a[x:y].strip())
         x = y     
    for i in range(b):
        length += len(new[i])
    answer = length/b
    return answer

def sentence_stats(a:str):

    length = word.count(" ") + 1
    print('Characters: ' + str(len(a)))
    print('Words: ' + str(length))

def initials(a:str)->str:
    ''' Return the initials of someone's name '''
    result = ''
    for x in a.split():
        result += x[0]
    return result.upper()

alphabet = 'abcdefghijklmnopqrstuvwxyz'

    'Return a Caesar cipher of the string'
    change = str.maketrans(alphabet, alphabet[y:26] + alphabet[0:y])
    return x.translate(change)

def character(a:list)->list:
    ''' Return just the characters '''


    result = [ ]
    for i in range(len(a)):
        result.append(a[i].replace(' ',''))
        result[i] = a[i].replace('\t','') and no_punctuation(a[i].strip())
    return result

def stats(a:list)->str:
    ''' Print this: lines in the list, empty lines, average characters per line, average characters per non-empty line '''
    whole_list = ' '
    number_of_lines = len(a)
    empty_lines = 0


    my_list = character(a)
    for i in range(len(a)):
        if a[i].isspace():
            empty_lines += 1
        whole_list += my_list[i]
    print('{0:5} lines in the list'.format(number_of_lines))
    print('{0:5} empty lines'.format(empty_lines))
    print('{0:5} average characters per line'.format(len(whole_list) / number_of_lines + empty_lines))
    print('{0:5} average characters per non-empty line'.format(len(whole_list) / number_of_lines))

stats(quote)
