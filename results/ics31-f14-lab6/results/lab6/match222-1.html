def contains (n: str, s: str) -> bool:
    '''takes as input two strings. The function checks if the second
    string occurs in the first string. If the second string does occur,
    the function returns True; if not, it returns False.'''

        return True
    else:
        return False

def sentence_stats (n:str) -> str:
    '''takes in a string as a parameter and prints out these statistics
    about the string: its length in characters, its length in words, and
    the average length of each word.'''
    g = 0
    for b in n.split():
        g = g + len(b)
    print('Characters: ', len(n), '\n',
          'Words: ', len(n.split()), '\n',
          'Average word length: ', g/len(n.split()))

def initials(x:str) -> str:
    '''takes as input a string representing a full name (e.g., Robert B.
    Qwerty) and returns the initials of the name in all capital letters
    (e.g., RBQ).'''
    namelist = x.split()
    initial = ''
    for i in namelist:
        initial = initial + i[0]
    return initial.upper()

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def newalphabet(n:int) -> str:
    'rearranges alphabet by shifting to the right n times'
    ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
    ALPHABET = ALPHABET[n:]+ALPHABET[:n]
    return ALPHABET


def Caesar_encrypt (s: str, n: int) -> str:
    'takes a plain text and encrypts it into code by shifting to right of alphabet'
    t = s.lower()
    table = str.maketrans(ALPHABET[0:26],newalphabet(n % 26))
    return t.translate(table)
                          
print(Caesar_encrypt('Tyler',5))

def Caesar_decrpyt(s: str, n:int) -> str:
    'takes the ciphertext and turns it back into plaintext'
    table = str.maketrans(ALPHABET[0:26],newalphabet(-n % 26))
    return s.translate(table)

def list_of_words(stringlist:list) -> list:
    '''takes a list of strings as above and returns a list of individual
    words with all white space and punctuation removed (except for
    apostrophes/single quotes).'''
    newlist = []
    for string in stringlist:
        splitted = string.split()
        for word in splitted:
            newlist.append(word)
    finallist = []
    for word2 in newlist:
        toappend = ''
        for character in word2:
            if character not in '. , ? !':
                toappend += character
        finallist.append(toappend)     
    return(finallist)
