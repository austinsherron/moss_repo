MENU = """
Restaurant Collection Program --- Choose one
 a:  Add a new restaurant to the collection
 r:  Remove a restaurant from the collection
 s:  Search the collection for selected restaurants
 v:  Search the collection for restaurants with selected cuisine
 d:  Search the collection for restaurants with selected dish
 p:  Print all the restaurants
 m:  Create a menu of dishes
 c:  Change prices for the dishes served
 q:  Quit
"""

def handle_commands(C: list) -> list:
    """ Display menu, accept and process commands.
    """
    while True:
        response = input(MENU)
        if response=="q":
            return C
        elif response=='a':
            r = Restaurant_get_info()
            C = Collection_add(C, r)
        elif response=='r':
            n = input("Please enter the name of the restaurant to remove:  ")
            C = Collection_remove_by_name(C, n)
        elif response=='p':
            print(Collection_str(C))
        elif response=='s':
            n = input("Please enter the name of the restaurant to search for:  ")
            for r in Collection_search_by_name(C, n):
                print(Restaurant_str(r))
        elif response=='v':
            n = input("Please enter the name of the cuisine to search for:  ")
            Result_search_by_cuisine(C, n)
        elif response=='d':
            n = input("Please enter the name of the dish to search for: ")
            Result_search_by_dish(C, n)
        elif response=='m':
            Menu_str(Menu_enter())
        elif response=="c":
            n = int(input("Please enter the percent to change the prices by:  "))
            C = Restaurants_change_price(C, n)
        else:
            invalid_command(response)

    ''' Return a string representing the collection
    '''
    s = ""

### DISH
Dish = namedtuple('Dish', 'name price calories')
# Constructor:   r1 = Dish('Escargot', 15.99, 400)

def dish_str(Dish : object) -> str:
    '''Returns given dish fields in readable format'''
    result = (Dish.name + ' ($' + str(Dish.price) + '): ' + str(Dish.calories) + ' cal')
    return result

def dish_get_info() -> Dish:
    '''Prompt user for fields of Dish; create and return.
    '''
    return Dish(
        input("Please enter the dish's name:  "),
        float(input("Please enter the price of the dish:  ")),
        int(input("Please enter the calories of the dish:  ")))

### MENU
# A menu is a list of dishes

def menu_change_price(dl: list, n: float) -> list:
    """Changes the price of all the Dishes in the Menu
    """
    percent_change = (100 + n)/100
    result = []
    for d in dl:
        changed_dish = Dish(d.name, d.price * percent_change, d.calories)
        result.append(changed_dish)
    return result

def Menu_enter() -> list:
    '''Prompts the user if they would like to add a Dish to a Menu of
       dishes until they say no'''
    answer = input('Would you like to add a dish to the menu? (Yes/No) ')
    new_Dishes = []
    check = True
    while check == True:
        if (answer == 'yes') or (answer == 'Yes'):
            new_Dishes.append(dish_get_info())
            check = True
            answer = input('Would you like to add a dish to the menu? (Yes/No) ')
        if (answer == 'no') or (answer == 'No'):
            check = False
            return new_Dishes

def Menu_str(dl: list) -> str:
    '''Converts a list of Dishes into readable format'''
    format_str = ''
    for d in dl:
        format_str =  format_str + ' -- ' + dish_str(d)
    return(format_str + ' --')

def Menu_price_total(R: Restaurant) -> float:
    total_price = 0
    for d in R.menu:
        total_price = total_price + d.price
    return total_price

def Menu_calories_total(R: Restaurant) -> float:
    total_calories = 0
    for d in R.menu:
        total_calories = total_calories + d.calories
    return total_calories

def Menu_return_averages(R: Restaurant) -> str:
    Menu_price_average = Menu_price_total(R) / len(R.menu)
    Menu_calories_average = Menu_calories_total(R) / len(R.menu)
    averages = "Average price:  $" + str(Menu_price_average) + '.  ' + 'Average calories:  ' + str(Menu_calories_average)
    return averages
