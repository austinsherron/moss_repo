def contains(s1:str,s2:str)->bool:
    '''Takes two strings as parameters
    and returns True if the second string occurs in the first.'''
    if s2 in s1:
        return True
    else:
        return False

def initials(s1:str)->str:
    result=''
    newlist=s1.split()
    for x in newlist:
        result = result+x[0]
    return result.upper()

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

ALPHABET = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
alphabet = 'abcdefghijklmnopqrstuvwxyz'


punctuation = ''',./?'";:!@#$%^&*'''

def Caesar_encrypt(word: str, shift: int) -> str:
    word = word.lower()
    newword = []
    for i in range(len(word)):
        for j in range(len(alphabet)):
            if word[i] == alphabet[j]:
                if ((j + shift) > 25):
                    shift = shift - 26
                else:
                    newword.append(alphabet[j+shift])
        for k in range(len(punctuation)):
            if word[i] == punctuation[k]:
                newword.append(word[i])
        
    return "".join(newword)

print(Caesar_encrypt("Hello school", 1))

def Caesar_decrypt(word: str, shift: int) -> str:
    newword = []
    for i in range(len(word)):
        for j in range(len(alphabet)):
            if word[i] == alphabet[j]:
                if ((j - shift) < 0):
                    shift = shift + 26
                else:
                    newword.append(alphabet[j-shift])
        for k in range(len(punctuation)):
            if word[i] == punctuation[k]:
                newword.append(word[i])
                               
    return "".join(newword)

print(Caesar_decrypt(Caesar_encrypt("Hello school", 1), 1))

def stats(wordlist: list):
    length = len(wordlist)
    empty = 0
    total = 0
    non_empty_total = 0
    for i in wordlist:
        if i == "":
            empty += 1
            total += 1
        total += len(i)
        non_empty_total += len(i)
    print("{:6} {:}".format((length),"lines in the list"))
    print("{:6} {:}".format((empty), "empty lines"))
    print("{:6.1f} {:}".format((total/length), "average characters per line"))
    print("{:6.1f} {:}".format((non_empty_total), "average characters per non-empty line"))

stats(["I like to play",
       "basketball and videogames",
       "but also eat",
       "it's getting really cold",
       "what"])

def spaces(phrase: str) -> str:
    table = str.maketrans('.,!?:;"/*',
                          '         ')
    return phrase.translate(table)



def list_of_words(words: list) -> list:
    for i in range(len(words)):
        words[i] = spaces(words[i])
    new = " ".join(words)    
    newlist = new.split()
    return newlist

print(list_of_words(["I like to play",
                     "basketball and videogames",
                     "but also eat",
                     "it's getting really cold",
                     "what"]))
