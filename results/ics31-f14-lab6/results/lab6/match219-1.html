def remove_punctuation(s:str) -> list:
    '''Takes string and converts words in string to list'''
    translation = str.maketrans("!@#$%^&*(),.<>/?';:-_=+", '                       ')
    return(s.translate(translation).split())

def sentence_stats(s:str) -> None:
    '''Takes a string and prints out stats'''
    print('Characters: ' , len(s))
    print('Words: ', len(remove_punctuation(s)))
    total = 0
    for i in remove_punctuation(s):
        total += len(i)
    print('Average word length: ', total/len(remove_punctuation(s)))

def initials(s:str)->str:
    '''Takes a name and returns the initials of the name in all caps'''
    p = s.upper() # ALANTURING

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def shift_table(key:int):
    ''' Takes a integer and produces a shifted alphabet based on that integer '''
    shift_alphabet = alphabet[key:26] + alphabet[0:key]
    return shift_alphabet



def Caesar_encrypt(str1:str, key:int)->str:
    ''' Takes a string and a integer (key) and creates a Caeser cipher '''
    str2 = str1.lower()
    table = str.maketrans(alphabet,shift_table(key))
    return str2.translate(table)
    
assert Caesar_encrypt('Apple', 1) == 'bqqmf'
assert Caesar_encrypt("Mouse", 5) == 'rtzxj'




def Caesar_decrypt(s:str, key:int) -> str:
    ''' Takes an encrypted string and the key and deciphers it'''
    s2 = s.lower()
    table = str.maketrans(shift_table(key),alphabet)
    return s2.translate(table)

assert Caesar_decrypt('bqqmf', 1) == 'apple'
assert Caesar_decrypt("rtzxj", 5) == 'mouse'

def stats(L:list)->None:
    ''' Takes a list of strings and prints its stats '''
    lines_in_list = len(L)
    empty_lines = 0
    total_characters = 0
    characters = "1234567890ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz!@#$%^&*(),.<>/?';:-_=+"
    translation = str.maketrans(characters, len(characters)*'h' )
    for i in range(len(L)):
        if 'h' not in  L[i].translate(translation):
            empty_lines += 1
    for i in range(len(L)):
        total_characters += len(L[i])
    average_characters = total_characters / len(L)
    non_empty = total_characters/ (len(L) - empty_lines)
    f = '{:5}   lines in the list \n {:4}   empty lines\n {:6.1f} average characters per line \n {:6.1f} average characters per non-empty line'
    print(f.format(lines_in_list, empty_lines, average_characters, non_empty))
    
print(stats(["Apple", "Orange", "Potato", "Carrots", "Peaches", "", "    "]))

def list_of_words(L:list)->list:
    ''' Takes a list of strings and returns a list of individual words with all white space and punctuation
removed'''
    translation = str.maketrans("!@#$%^&*(),.<>/?;:-_=+", '                      ')
    result = []
    for i in L:
        result.extend(((i.translate(translation)).strip()).split())
    return result
