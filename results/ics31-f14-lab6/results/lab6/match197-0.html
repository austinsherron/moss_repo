def punctuation_blank(sentence: str)-> 'Sentence':
    '''Takes all the punctuation in the sentence and changes it to a space
    '''
    punct = "!?.,:;*\""
    for x in sentence:
        if x in punct:
            sentence = sentence.replace(x, ' ')
    return sentence
assert punctuation_blank('Zot?.') == 'Zot  '


def sentence_stats(sentence: str) ->'Statistics about the sentence':
    '''Takes a sentence and returns the character length, word length,
    and average length of each word'''
    word_count = (punctuation_blank(sentence).split())
    word_list = []
    for x in word_count:
        word_list.append(len(x))
    print('Characters: ', len(sentence))
    print('Words: ', len(word_count))
    print('Average word length: ', sum(word_list)/len(word_count))

def initials(name: str) -> str:
    '''Takes a name and returns the initials based of the first letter of each
    word in the name'''
    int_str = ''
    word_list = name.split()
    for x in word_list:
        int_str += x[0].capitalize()
    return int_str
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def rotate_alpha(number:int) ->str:
    '''Takes the first letters of the alphabet specified by the number and
    moves them to the end'''
    newalpha = alphabet
    for i in range(0, number):
        newalpha = alphabet.replace(alphabet[0:i + 1], '')
        newalpha = newalpha + alphabet[0:i + 1]
    return newalpha
assert rotate_alpha(3) == 'defghijklmnopqrstuvwxyzabc'


def Caesar_encrypt(text: str, num:int) -> str:
    '''Takes a text and changes the letter according to the key'''
    text = text.lower()
    table = str.maketrans(alphabet, rotate_alpha(num))
    text = text.translate(table)
    return text
assert(Caesar_encrypt('Hello world', 4)) ==  "lipps asvph"


def Caesar_decrypt(text: str, num: int) -> str:
    '''Takes an encoded string and decrypts based on the key'''
    text = text.lower()
    table = str.maketrans(rotate_alpha(num), alphabet)
    text = text.translate(table)
    return text
assert Caesar_decrypt("Lipps asvph", 4) == 'hello world'

def find_empty_line(strlist:list)->int:
    '''takes a list of strings and finds the number of empty lines'''
    emptystr = []
    for x in strlist:
        if x == '':
            emptystr.append(x)
    return len(emptystr)

def count_characters(strlist:list)->int:
    '''finds the amount of characters in the list'''
    characterlen = []
    for x in strlist:
       characterlen.append(len(x))
    return sum(characterlen)
