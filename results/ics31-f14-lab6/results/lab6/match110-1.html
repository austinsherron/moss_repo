    '''correct the range of the key'''
    c = b%26

def rotate_alphabet(a:int)->str:
    '''rotate alphabet by a number'''
    result = ''
    result = ALPHABET[a:] + ALPHABET[0:a]
    return result


def Caesar_encrypt(a:str,b:int)->str:
    '''encrypt a string by a key'''
    b = key_correct_range(b)
    ALPHABET_change = ''
    ALPHABET_change = rotate_alphabet(b)
    table = str.maketrans(ALPHABET, ALPHABET_change)
    result = ''
    result = a.translate(table)
    return result
