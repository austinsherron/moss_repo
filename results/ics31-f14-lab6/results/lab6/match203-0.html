def initials(s: str)->str:


    """takes as input a string representing a full name (e.g., Robert B. Qwerty)
    and returns the initials of the name in all capital letters (e.g., RBQ)."""
    z = s.title()
    l = z.split()
    x = ''
    for i in l:
        x += i[0]
    return(x)
        
assert initials('dsk jl hvgf') == 'DJH'

alphabet = 'abcdefghijklmnopqrstuvwxyz'
def Caesar_encrypt(s: str, i:int):
    '''takes two arguments: a string containing the message and an int for the key,
    indicating how far down the alphabet to find each substitute letter.'''
    alphabet2 = alphabet
    alphabet2 += alphabet[0:i%26]
    return s.translate(str.maketrans(alphabet, alphabet2[i%26:]))
print(Caesar_encrypt('hello', 37))

def Caesar_decrypt(s:str, i:int):
    '''returns plaintext of encryption'''
    decrypt = ''
    decrypt +=alphabet[i%26:26]
    decrypt +=alphabet[0:i%26]
    return s.translate(str.maketrans(decrypt, alphabet))
print(Caesar_decrypt('spwwz', 37))

def stats(l: list):
    '''takes a list of strings and prints statistics as follows:
 16824  lines in the list
   483  empty lines
  53.7  average characters per line
  65.9  average characters per non-empty line'''
    emp=0
    strlength=0
    print('{0:>5}  lines in the list'.format(len(l)))
    for i in l:
        strlength+=len(i)
        if i == '':
            emp += 1
    print('{0:>5}  empty lines'.format(emp))
    print('{0:>5}  average characters per line'.format(strlength/len(l)))
    print('{0:>5}  average characters per non-empty line'.format(strlength/(len(l)-emp)))
