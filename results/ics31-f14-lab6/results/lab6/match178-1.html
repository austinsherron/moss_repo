def clear_punctuation(x: str) -> str:
    empty = ''
    for p in x:
        if p in 'abcdefghijklmnopqrstuvxyzABCDEFGHIJKLMNOPQRSTUVXYZ':
            empty += p 
    return empty

def sentence_starts(x: str) -> int:
    Character = len(x)
    Word = len(x.split())
    Average = float(len(clear_punctuation(x))/Word)
    print('Characters:', Character)
    print('Words:', Word)
    print('Average:', Average)

def initials(x: str) -> str:
    empty = x.split()
    Var = ''
    for x in empty:
        Var  = Var + x[0].upper()
    return Var
        

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate(n: int) -> str:
    return ALPHABET[n:] + ALPHABET[:n]
    
def Caesar_encrypt(plaintext: str, n: int) -> str:
    encrypt = str.maketrans(ALPHABET, rotate(n))
    return plaintext.lower().translate(encrypt)

def Caesar_decrypt(ciphertext: str, n: int) -> str:
    decrypt = str.maketrans(ALPHABET, rotate(n))
    return ciphertext.lower().translate(decrypt)
    
assert Caesar_encrypt("hi there", 3) == 'kl wkhuh'
assert Caesar_decrypt('kl wkhuh', -3) == 'hi there'

def list_of_words(L:list) -> list:
    EL = []
    var = ''
    for i in L:
        for x in i:
            if x not in ['\n', '\r\n', ' ', '']:
                var = var + x
            elif var != '':
                EL.append(var)
                var = ''
    return EL
