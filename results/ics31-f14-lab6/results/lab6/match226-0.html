def punctuation_deleter(s:str) -> str:
    '''Deletes any punctuation marks'''
    table = str.maketrans('.,!?*:"',
                          '       ')
    return s.translate(table)

def sentence_stats(x:str):
    '''Prints out the statistics of each string'''
    print('Characters:', len(x))
    final_sentence = punctuation_deleter(x)
    num_of_words = []
    num_of_words = final_sentence.split()
    print('Words:', len(num_of_words))
    total_words = 0
    for x in num_of_words:
        total_words += len(x)
    print('Average word length:', total_words/len(num_of_words))

def initials(x:str) -> str:
    '''Returns the initials of a name'''
    initlist = []
    y = x.split()
    for i in range(len(y)):
        initlist.append(y[i][0].upper())
    return ''.join(initlist)
    
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def distribution_of_rolls(n:int)-> str:

    print('Distribution of dice rolls')
    rolls_list = []

def Caesar_encrpyt(s:str,y:int) -> str:
    '''Takes a string and an integer and returns a ciphertext'''
    letters = "abcdefghijklmnopqrstuvwxyz"
    letters2 = []
    for i in range(len(letters)):
        letters2.append(letters[(i+y)%26])
    letters2 = ''.join(letters2)

    table = str.maketrans(letters, letters2)
    return s.lower().translate(table)

def Caesar_decrpyt(s:str,y:int) -> str:
    '''Takes a string and an integer and returns the plaintext'''
    letters = "abcdefghijklmnopqrstuvwxyz"
    letters2 = []
    for i in range(len(letters)):
        letters2.append(letters[(i-y)%26])
    letters2 = ''.join(letters2)

    table = str.maketrans(letters, letters2)
    return s.lower().translate(table)

def stats(y:list) -> str:
    '''Prints statistics of the strings'''
    numbers = []
    numbers2 = []

    print('{:0.0f}     {}'.format(len(y),'lines in the list'))
    print('{:2.0f}    {}'.format(y.count(''),'empty lines'))
    for i in range(len(y)):
        numbers.append(len(y[i]))
        if (y[i] != ''):
            numbers2.append(len(y[i]))

    average = sum(numbers)/len(numbers)
    average_second = sum(numbers2)/len(numbers2)
    print('{:5.1f} {}'.format(average,'average characters per line'))
    print('{:5.1f} {}'.format(average_second,'average characters per non-empty line'))

def list_of_words(n:list) -> str:
    '''Returns list of individual words with whitespace and punctuation removed'''
    words = []
    punctuation = '.,+=()%#*@&^!?">\</$[]{}-_'
    for i in range(len(n)):
        words.extend(n[i].split())
    for i in range(len(words)):
        table = str.maketrans(punctuation, '                          ')
        words[i] = words[i].translate(table)
    return ' '.join(list(words))
print(list_of_words(speech))
