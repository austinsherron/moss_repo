ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotateAlphabet(n:str,m:int)->str:
    if m > 26:
        m = m % 26
    p = ''
    p = n[m:] + n[:m]
    
    return p
    
assert rotateAlphabet(ALPHABET, 1) == 'bcdefghijklmnopqrstuvwxyza'
def Caesar_encrypt(n:str,m:int)->str:
    toTranslate = n.maketrans(ALPHABET,rotateAlphabet(ALPHABET,m))
    
    return n.translate(toTranslate)

assert Caesar_encrypt('kenny',1) == 'lfooz'
assert Caesar_encrypt('cat',1) == 'dbu'

def Caesar_decrypt(n:str,m:int)->str:
  


    toTranslate = n.maketrans(rotateAlphabet(ALPHABET,m),ALPHABET)
    return n.translate(toTranslate)
plaintext = str(input('input a plaintext: '))
key = int(input('input a key: '))
print('\n','encrypted document: ')
print(Caesar_encrypt(plaintext,key),'\n')
print('decrypted document: ')
print(Caesar_decrypt(Caesar_encrypt(plaintext,key),key))

def list_of_words(n:list)->list:


    k = []
    for i in n:
        k.extend(i.split(' '))
    while k.count('') != 0:
        for i in k:
            if i == '':
                k.remove(i)
    for i in range(len(k)):
        k[i]=k[i].strip('.')
        k[i]=k[i].strip(',')
        k[i]=k[i].strip('\t\t')
    return k
print(list_of_words(Gettysburg))
