Alphabet ='abcdefghijklmnopqrstuvwsyz'

def Caesar_encrypt(string:'contain the original message',key:int):
    """Return the ciphertext"""
    string = string.lower()
    key1 = key%26
    new_order = Alphabet[key1:]+Alphabet[:key1]
    table = str.maketrans(Alphabet,new_order)
    string = string.translate(table)
    string =string.capitalize()
    return string
print(Caesar_encrypt('I love UCI', 1))
assert  Caesar_encrypt('I love UCI', 1) == 'J mpwf vdj'


def Caesar_decrypt(string:'contain the message',key:int):
    """Returns the plaintext"""
    string = string.lower()
    key2 = key%26
    new_order1 = Alphabet[(-key2):]+Alphabet[:(-key2)]
    table = str.maketrans(Alphabet,new_order1)
    string= string.translate(table)
    string= string.capitalize()
    return string
print (Caesar_decrypt('j mpwf vdj', 1))


assert Caesar_decrypt('j mpwf vdj', 1)=='I love uci'

    lines_in_the_list = len(list_of_strings)
    empty_lines = 0
    for i in list_of_strings:
        if i == '':
            empty_lines+=1
    m = 0
    for i in list_of_strings:
        m +=len(i)
    average_characters_per_line = m/len(list_of_strings)
    average_chracters_per_non_empty_line = m/(lines_in_the_list - empty_lines )
    print (str(lines_in_the_list)+'    '+'lines in the list\n'+

def list_of_words(list_of_strings:list)->list:
    """Return a list of individual words with all white space and punctuation
removed"""
    table = str.maketrans('!@#$%^&*()~?','            ')
    list2 = []
    for i in list_of_strings:
        list2+=i.translate(table).split()
    return list2
