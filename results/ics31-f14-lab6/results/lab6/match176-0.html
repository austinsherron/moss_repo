def contains(string1, string2: str) -> bool:
    '''The function checks if the second string occurs in the first string and if the
    second string does occur than the function returns True if not, it returns False.'''
    return string1.find(string2) >= 0

assert contains('banana', 'ana')
assert not contains('racecar', 'ck')
assert contains('apple', 'ap')

def translate_punctuation(string: str) -> str:
    '''Replaces non alphabetical characters with spaces'''  
    result = string
    for c in string:
        if not c.isalpha():
            result = result.replace(c, ' ')
    return result

def sentence_stats(string: str) -> None:
    '''Takes in a string as a parameter and prints out these statistics about the string: its
    length in characters, its length in words, and the average length of each word.'''
    string = translate_punctuation(string)
    print('Characters: ', len(string))
    print('Words: ', len(string.split()))
    total = 0
    for w in string.split():
        total += len(w)
    average = total / len(string.split())
    print('Average Word Length: ', average)
    print(string.split())
    print()

def initials(name: str) -> str:
    '''Takes as input a string representing a full name and returns the
    initials of the name in all capital letters.'''
    result = ''
    words = name.split()
    for w in words:
        result += w[0].upper()
    return result



assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotated_alphabet(regular: str, num: int) -> str:
    '''Takes two arguments: a string containing the message and an int for the key which indicates how far
    down the alphabet to find each substitute letter.'''
    return regular[num % 26:] + regular[:num % 26]

def Caesar_encrypt(plaintext: str, key: int) -> str:


    '''Takes two arguments: a string containing the message and an int for the key which indicates how far
    down the alphabet to find each substitute letter. The encryption function returns the ciphertext.'''
    trantab = str.maketrans(ALPHABET, rotated_alphabet(ALPHABET, key))
    return plaintext.lower().translate(trantab)

def Caesar_decrypt(ciphertext: str, key: int) -> str:


    '''Takes two arguments: a string containing the message and an int for the key which indicates how far
    down the alphabet to find each substitute letter.The decryption function returns the plaintext.'''
    trantab = str.maketrans(rotated_alphabet(ALPHABET, key), ALPHABET)
    return ciphertext.lower().translate(trantab)

print(Caesar_encrypt('hi there', 3))
print(Caesar_decrypt('kl wkhuh', 3))
print(Caesar_encrypt('I hate pie', 7))
print(Caesar_decrypt('p ohal wpl', 7))
print(Caesar_encrypt('hi there', 29))
print(Caesar_decrypt('kl wkhuh', 29))

def stats(SL: list) -> None:
    ''' Takes a list of strings and prints statistics in different lines.'''
    lines = 1
    for i in SL:
        if '\n' in i:
            lines += 1
    empty = 0
    for i in SL:
        if i == '\n':
            empty += 1
    totalChar = 0
    for i in SL:
        totalChar += len(i)
    print("{:6d}   lines in the list".format(lines))
    print("{:6d}   empty lines in the list".format(empty))
    print("{:8.1f} average character per line".format(totalChar/lines))
    print("{:8.1f} average character per non-empty line".format(totalChar/(lines - empty)))

stats(SL)

def translate_punctuation_exception(string: str) -> str:
    '''Replaces non alphabetical characters with spaces except apostrophies'''  
    result = string
    for c in string:
        if not c.isalpha() and not c == "'":
            result = result.replace(c, ' ')
    return result

def list_of_words(SL: list) -> list:
    ''' Takes a list of strings and returns a list of individual words with all white space and punctuation removed.''' 
    result = ''
    for s in SL:
        result += translate_punctuation_exception(s) + " "
    return result.split()

print(list_of_words(SL))
