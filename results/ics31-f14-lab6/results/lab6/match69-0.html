def change_to_spaces(phrase: str) -> str:
    table = str.maketrans('.,!?:;"/*', '         ')
    return phrase.translate(table)


def sentence_stats(word: str):

def initials(name: str) -> str:
    names = name.split()
    initials = []
    for i in names:
        initials.append(i[0])
    new = "".join(initials)
    return new

ALPHABET = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
alphabet = ALPHABET.lower()


punctuation = ''',./?'";:!@#$%^&* '''

def Caesar_encrypt(word: str, shift: int) -> str:
    word = word.lower()
    newword = []
    for i in range(len(word)):
        for j in range(len(alphabet)):
            if word[i] == alphabet[j]:
                if ((j + shift) > 25):
                    shift = shift - 26
                else:
                    newword.append(alphabet[j+shift])
        for k in range(len(punctuation)):
            if word[i] == punctuation[k]:
                newword.append(word[i])
        
    return "".join(newword)

print(Caesar_encrypt("Aaron Ching", 1))

def Caesar_decrypt(word: str, shift: int) -> str:
    newword = []
    for i in range(len(word)):
        for j in range(len(alphabet)):
            if word[i] == alphabet[j]:
                if ((j - shift) < 0):
                    shift = shift + 26
                else:
                    newword.append(alphabet[j-shift])
        for k in range(len(punctuation)):
            if word[i] == punctuation[k]:
                newword.append(word[i])
                               
    return "".join(newword)

print(Caesar_decrypt(Caesar_encrypt("Aaron Ching", 1), 1))

def stats(wordlist: list):
    length = len(wordlist)
    empty = 0
    total = 0
    non_empty_total = 0
    for i in wordlist:
        if i == "":
            empty += 1
            total += 1
        total += len(i)
        non_empty_total += len(i)
    print("{:6} {:}".format((length),"lines in the list"))
    print("{:6} {:}".format((empty), "empty lines"))
    print("{:6.1f} {:}".format((total/length), "average characters per line"))
    print("{:6.1f} {:}".format((non_empty_total), "average characters per non-empty line"))

                    "that is being done in",
                    "an ICS 31 class",

def list_of_words(wordlist: list) -> list:

        wordlist[i] = change_to_spaces(wordlist[i])
