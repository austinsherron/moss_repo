def sentence_stats(s:str):
    '''Takes in a string as a parameter and prints out these statistics
       about the string: its length in characters, its length in words,
       and the average length of each word.'''
    table = str.maketrans('.,!?:;"*', '        ')
    s2 = s.translate(table)
    L = s2.split()
    sum1 = 0
    print("Characters: ",len(s))
    print("Words: ",len(L))
    for i in L:
        sum1 += len(i)
    print("Average word length: ",sum1/len(L))

def initials(s:str) -> str:
    '''Takes as input a string representing a full name
       and returns the initials of the name in all capital letters '''
    L = s.split()
    intls = ''
    for i in L:
        intls += i[0]
    return intls.upper()
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotate_alphabet(s:str, num:int) -> str:
    return s[num:27] + s[0:num]
print(rotate_alphabet(ALPHABET,3))


def Caesar_encrypt(s: str, k: int) -> str:
    '''Takes two arguments: a string containing the message
      (the plaintext for encryption) and an int for the key,
      indicating how far down the alphabet to find each substitute letter.
      The encryption function returns the ciphertext.'''
    s1 = s.lower()
    table1 = str.maketrans(ALPHABET,rotate_alphabet(ALPHABET, k))
    return s1.translate(table1)
print(Caesar_encrypt('Cat in the hat', 1))
    
    




def Caesar_decrypt(s: str, k: int) -> str:
    '''Takes two arguments: a string containing the message
      (the ciphertext for decryption) and an int for the key,
      indicating how far down the alphabet to find each substitute letter.
      The decryption function returns the plaintext.'''
    s2 = s.lower()
    table2 = str.maketrans(rotate_alphabet(ALPHABET, k), ALPHABET)
    return s2.translate(table2)
print(Caesar_decrypt('dbu jo uif ibu',1))

def stats(L:list)->None:
    '''Takes a list of strings and prints statistics.'''
    print('{0:5.1f} {1}'.format(len(L),'lines in the list'))
    num_empty_lines = 0
    for i in L:
        if i == len(i)*' ':
            num_empty_lines +=1
    print('{0:5.1f} {1}'.format(num_empty_lines, 'empty lines'))
    sum_of_chars = 0
    for x in L:
        s = x.split()
        for n in s:
            sum_of_chars += len(n)
    print('{0:5.1f} {1}'.format(sum_of_chars/len(L), 'average characters per line'))
    print('{0:5.1f} {1}'.format(sum_of_chars/(len(L)-num_empty_lines),
                                'average characters per non-empty line'))
    
stats(L1)

def list_of_words(L:list)->list:
    '''Takes a list of strings as above and returns a list of
      individual words with all white space and punctuation
      removed (except for apostrophes/single quotes).'''
    table = str.maketrans('.,!?:;"*', '        ')
    L2 = []
    for i in L:
        i2 = i.translate(table)
        L2.extend(i2.split())
    return L2
print(list_of_words(L1))
