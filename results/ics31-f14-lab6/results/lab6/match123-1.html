def punctuation( s: str ) -> str:
    punc = str.maketrans( '!.,?*:;',
                          '       ' )
    return s.translate( punc )

def sentence_stats( s: str ) -> str:
    pfix = punctuation( s )
    count = len( s )
    y = pfix.split()
    words = len( y )
    total = 0
    for x in y:
        total += len( x )
    return 'Characters: ' + str( count ) + '\n' +  'Words: ' + str( words ) + '\n' + 'Average word length: ' + str( total/words )

def initials( s: str ) -> str:
    '''Takes the string and returns the initials in uppercase form'''
    y = s.split()
    init = ''
    for x in y:
        init += x[0].upper() 
    return init

assert initials( 'Bill Cosby' ) == 'BC'
assert initials( 'Guido van Rossum' ) == 'GVR'
assert initials( 'alan turing' ) == 'AT'

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def Caesar_encrypt( message: str, k: int ) -> str:
    if k > 25:
        k = k%26
    change = str.maketrans( alphabet,
                            alphabet[k:] + alphabet[:k] )
    return message.lower().translate( change )


def Caesar_decrypt( message: str, k: int ) -> str:
    if k > 25:
        k = k%26
    change = str.maketrans( alphabet,
                            alphabet[-k:] + alphabet[:-k] )
    return message.lower().translate( change )

def stats( s: list ) -> str:
    empty = 0
    total = 0
    fulllength = 0
    full = 0
    for x in s:
        if x.isspace() == True:
            empty += 1
        else:
            full += 1
            fulllength += len( x )
    for a in s:
        total += len( a )
    print( len( s ), 'lines in the list' )
    print( empty, 'empty lines' )
    print( total/len( s ), 'average character per line' )
    print( fulllength/full, 'average characters per non-empty line' )

def list_of_words( s: list ) -> list:
