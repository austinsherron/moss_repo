def initials (s:str) -> str:
    '''Returns capitalized initials of name'''
    result = ''
    for i in s.split():
        result += i[0].upper()
    return result

def rotated(string: str, num: float):
    if num >= 26:
        num -= 26
    return string[num:] + string[:num]
def Caesar_encrypt(m:str, k:int) -> str:
    ''' entering plaintext returns cipher text'''
    real_alpha = 'abcdefghijklmnopqrstuvwxyz'
    real_ALPHA = real_alpha.upper()
    plus_upper_alpha = real_alpha + real_ALPHA
    new_alpha = rotated(plus_upper_alpha,k)
    trans = str.maketrans(plus_upper_alpha, new_alpha)
    new_string = ''
    for i in m:
        new_str = i.translate(trans)
        new_string += (new_str)  
    return new_string

def Caesar_decrypt(m:str, k:int) -> str:
    '''Entering ciphertext and returns plaintext'''
    real_alpha = 'abcdefghijklmnopqrstuvwxyz'
    real_ALPHA = real_alpha.upper()
    plus_upper_alpha = real_alpha + real_ALPHA
    new_alpha = rotated(plus_upper_alpha,k)
    trans = str.maketrans(new_alpha, plus_upper_alpha)
    new_string = ''
    for i in m:
        new_str = i.translate(trans)
        new_string += (new_str)  
    return new_string

def empty_line(s: list) -> int:
    '''Return number of empty lines in the function'''
    result = []
    for i in s:
        if i == '':
            result.append(i)
    return len(result)

def total_characters(s: list) -> int:
    '''Returns total number of characters in a line'''
    result = []
    for i in s:
        result.append(len(i))
    return sum(result)

def punct_removed(s: str) -> str:
    '''Removes punctuation and replaces it with blanks'''
    a = '*!@#$%^&()-_+=\|]}[{;:.<>?/"'
    result = ''
    for i in a:
        result = s.replace(i, '')
        s = result 
    return result



def list_of_words(l: list) -> list:
    '''returns a list of words with all white space and punctuation removed'''
    result = []
    for i in l:
        i = punct_removed(i)
        result.extend(i.split())
    return result
