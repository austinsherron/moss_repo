def sentence_stats(s:str):
    '''prints out len in char, len in words, and avg len of words for s'''
    print("Characters: ", len(s))
    s = s.translate(str.maketrans(string.punctuation,' '*len(string.punctuation))).split()
    print("Words: ", len(s))
    total = 0
    for i in s:
        total += len(i)
    print("Average word length: ", total/len(s))

def initials(s:str) -> str:
    '''returns the initials of the name in call caps'''
    s = s.upper().split()
    result = ''
    for i in s:
        result += i[0]
    return result
assert initials('Bill Gates') == 'BG'
assert initials('nathan Isaac wang') == 'NIW'
assert initials('BOB') == 'B'

def distribution_of_rolls(n:int) -> None:
    '''rolls 2 dice the number of times specified and returns the results'''
    result = [0 for x in range(12)]
    for i in range(n):
        roll = randrange(1,7) + randrange(1,7)
        result[roll-2] += 1

    print("Distribution of dice rolls")
    print()
    for x in range(2,13):
        print("{0:2}: {1:3} ({2:4.1f}%) {3:<}".format(x,result[x-2],result[x-2]/n*100,'*'*result[x-2]))

    print('-'*30)
    print("{0:7} rolls".format(n))
    
distribution_of_rolls(100)
print()
distribution_of_rolls(200)
print()
distribution_of_rolls(500)

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotate(x:int) -> str:
    '''rotates the alphabet by the number given'''
    return ALPHABET[x%26:] + ALPHABET[0:x%26]
assert rotate(29) == 'defghijklmnopqrstuvwxyzabc'
assert rotate(5) == 'fghijklmnopqrstuvwxyzabcde'
assert rotate(13) == 'nopqrstuvwxyzabcdefghijklm'




def Caesar_encrypt(m:str,k:int) -> str:
    '''encrypts a message by rotating the alphabet by the given number'''
    return m.lower().translate(str.maketrans(ALPHABET,rotate(k)))
assert Caesar_encrypt('hello there!',2) == 'jgnnq vjgtg!'
assert Caesar_encrypt('my name is nathan',6) == 'se tgsk oy tgzngt'


def Caesar_decrypt(m:str,k:int) -> str:
    '''decrypts a message by rotating the alphabet by the given number'''
    return m.lower().translate(str.maketrans(rotate(k),ALPHABET))
assert Caesar_decrypt('fytc y lgac byw',50) == 'have a nice day'
assert Caesar_decrypt('yfnj zk xfzex?',17) == 'hows it going?'

def stats(s:list) -> None:
    '''takes a list of strings and prints the lines in the list, empty lines, avg char per line, and avg char per non-empty line'''
    lines = len(s)
    
    empty = 0
    for i in s:
        if i == '':
            empty += 1
            
    total = 0
    for i in s:
        total += len(i)
    avg = total/lines

    non = 0
    for i in s:
        if i != '':
            non += 1
    avgnon = total/non

    print('{:6}   lines in the list'.format(lines))
    print('{:6}   empty line(s)'.format(empty))
    print('{:8.1f} average characters per line'.format(avg))
    print('{:8.1f} average characters per non-empty line'.format(avgnon))

def list_of_words(s:list) -> list:
    '''takes the list and returns a list of individual words without whitespace and punctuation'''
    new = []
    for i in s:
        new.extend(i.strip(string.punctuation).split())
