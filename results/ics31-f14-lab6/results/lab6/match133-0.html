def list_of_words(L:list) -> list:
    '''takes a string list and returns a list consist of individual words '''


    newList = []
    word = ''
    for i in L:
        for x in i:
            if x not in ['\n', '\r\n', ' ', '']:
                word = word + x
            elif word != '':
                newList.append(word)
                word = ''
    return newList
