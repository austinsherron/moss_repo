def punct(n: str) -> str:
    '''Takes a sentence and returns that sentence with punctuation turned into spaces'''

    return n.translate(table)

def initials(n: str) -> str:
    '''Takes a string and representing a full name (e.g., Montecristo)and returns the initials of the name in all capital letters (e.g., M).'''
    answer = ''
    for i in n.split():
        answer += i[0].upper()
    return answer

assert initials('Montecristo') == 'M'   
assert initials('Double Lift') == 'DL'

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def code_change(n:int) ->str:


    '''this will shift the alphabet letters by adding it to the end of the string'''
    return ALPHABET[n:] + ALPHABET[:n]
assert code_change(2) == 'cdefghijklmnopqrstuvwxyzab'

def Caesar_encrypt(n:str, s:int) -> str:
    '''Encrypts a Ceaser code by the specified key and message'''
    n = n.lower()


    table = n.maketrans(ALPHABET, code_change(s))
    return n.translate(table)
assert Caesar_encrypt('Cow', 1) == 'dpx'
assert Caesar_encrypt('do1g', 2) == 'fq1i'

def Caesar_decrypt(n:str, s:int) -> str:
    '''Decrypts an encrypted message back to the original message'''
    n = n.lower()
    table = n.maketrans(ALPHABET, code_change(s * -1))
    return n.translate(table)
assert Caesar_decrypt('dpx', 1) == 'cow'
assert Caesar_decrypt('fq1i', 2) == 'do1g'
