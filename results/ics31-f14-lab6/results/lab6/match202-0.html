def sentence_stats(s: str) -> None:
    ''' Prints out these statistics about the string: its length in characters, its length in words, and the average length of each word '''

    print('Words:', len(s.split()))
    a = 0
    for i in s.split():
        a = a + len(i)
    print('Average word length:', a / len(s.split()))

def initials (s: str) -> str:
    ''' Returns the initials of the name in all capital letters '''

def distribution_of_rolls(n: int) -> None:
    ''' Prints the distribution of the values of those rolls '''
    rolls = []
    print("Distribution of dice rolls")
    for i in range(n):
        rolls.append(roll2dice())
    for i in range(2,13):
        print(str(i) + ":     ", str(rolls.count(i)), "(", str(rolls.count(i) / len(rolls) * 100), "%)", rolls.count(i) * "*", sep = '')

    print("----------------------------- \n" +
        "      " + str(len(rolls)) + " rolls")

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate (key: int) -> str:
    ''' Rotates the alphabet based on the key '''
    return ALPHABET [key%26: ] + ALPHABET [0:key%26]





def Caesar_encrypt (s: str, key: int) -> str:
    ''' Encrypts a message based off of the key '''
    table = str.maketrans(ALPHABET, rotate(key))
    return s.lower().translate(table)

def Caesar_decrypt (s: str, key: int) -> str:
    ''' Decrypts a message based off of the key '''
    table = str.maketrans(rotate(key), ALPHABET)
    return s.lower().translate(table)

def stats(s: list) -> None:
    ''' Prints the statistics of a list of strings '''
    lines = 0
    empty_lines = 0
    characters = []
    for i in s:
        characters.append(len(i))
        if len(i) == 0:
            empty_lines = empty_lines + 1
        else:
            lines = lines + 1    
    print("{:4d}     lines in the list".format(lines))
    print("{:4d}     empty lines".format(empty_lines))
    print("{:6.1f}   average characters per line".format(sum(characters)/len(s)))
    print("{:6.1f}   average characters per non-empty line".format(sum(characters)/(lines - empty_lines)))

def list_of_words(s: list) -> list:
    ''' Returns a list of individual words with all white space and punctuation removed '''
    words = []
    for i in s:
        i = i.replace('.', '')
        i = i.replace(',', '')
        i = i.replace('!', '')
        i = i.replace('?', '')
        i = i.replace(':', '')
        i = i.replace(';', '')
        for r in i.split():
            words.append(r)
    return words
