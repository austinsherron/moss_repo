def change_char(h: str) ->str:
    b = "?!.,':;*"
    result = ''
    for char in h:
        if char in b:
            result += ' '
        else:
            result += char
    return result


def sentence_stats(a:str):
    '''Takes a string and prints out statistics regarding the string'''
    a = change_char(a)
    character_count = len(a) - a.count(' ')
    print("Characters: ", len(a))
    w = a.split()
    print("Words: ", len(w))
    average = character_count / len(w)
    print("Average word length: ", average)
    

sentence_stats("The red balloon")
sentence_stats('***The ?! quick brown fox:  jumps over the lazy dog.')

def initials (n: str,)-> str:
    result = ""
    new = n.upper().split()
    for x in new:
        result += x[0]
    return result
        
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def distribution_of_rolls(number: int):

def rotated_alpha(a: str, x:int) -> str:
    result = ''
    result += a[x:] + a[:x]
    return result

assert rotated_alpha('computer', 2) == 'mputerco'
assert rotated_alpha('cat', 2) == 'tca'

def Caesar_encrypt(message: str, x:int) -> str:
    '''Takes an unaltered message(plaintext) and returns a ciphertext'''
    table = str.maketrans(ALPHABET, rotated_alpha(ALPHABET, x))
    return message.translate(table)

def Caesar_decrypt(message: str, x:int) -> str:
    '''Takes a ciphertext and returns the plain text'''
    table = str.maketrans(rotated_alpha(ALPHABET, x), ALPHABET)

def stats(x: list):
    '''takes a list of strings and prints the statistics of each string'''
    lines = len(x)


    emptylines = 0
    characters = 0
    for s in x:
        if s == ' ':
            emptylines += 1
    for z in x:
        characters += len(z)
    print('{:6} lines in the list\n {:5} empty lines\n {:5.1f} average characters per line\n {:5.1f} average characters per non-empty line'.format(lines, emptylines, (characters / lines), (characters/(lines - emptylines))))

def list_of_words(x: list) -> list:
    '''Takes a list of strings and returns a list of individual words'''
    result = []
    for s in x:
        s = change_char(s)
        s = s.split()
        result += s
    return result
