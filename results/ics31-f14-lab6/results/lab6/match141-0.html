ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotation(n: int)-> str:
    '''rotates the alphabet'''
    newstring = ALPHABET
    i = 0
    for i in range(n):
        newstring += newstring[i]
    return newstring[n:]
print(rotation(3))

def Caesar_encrypt(a:str, b:int):
    '''encrypts an inputed string using a displacement integer'''
    if b >=25:
        b = (b%25) - 1
    a = a.lower()
    transtab = a.maketrans(ALPHABET, rotation(b))
    return a.translate(transtab)
print(Caesar_encrypt('abc',29))
def Caesar_decrypt(a:str, b:int):
    ''' decrypts an inputed string '''
    a = a.lower() 


    if b >=25:
        b = (b%25) - 1
    reverse = a.maketrans(rotation(b),ALPHABET)
    return a.translate(reverse)

def stats(strings: list):
    counter = 0
    for b in strings:
        if b == '':
            counter += 1
    print("{:6}    lines in the list".format(len(strings)))
    print("{:6}    empty lines in the list".format(counter))
    print("{:9.2f} average characters per line".format(sum(map(len, strings))/len(strings)))
    print("{:9.2f} average characters per non-empty line".format(sum(map(len, strings))/(len(strings)-counter)))

def remove_punctuation(s: str) -> str:
    '''removes punctuation'''
    punctuation = ',./;"`!@#$%^&*():?><{}-_=+'
    for i in punctuation:
        if i in s:
            s = s.replace(i,'')
    return s
print(remove_punctuation("endure."))
