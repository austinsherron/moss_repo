def remove_punct(s:str):
    #Translates all punctuation into spaces, removes punctuation
    table = str.maketrans(',./<>?;\'[]{}~!@#$%^&*()_+-=:' , '                            ')
    return s.translate(table)

def remove_spaces(s:str):
    #Removes unnecessary spaces. Keeps spaces between words at one space
    wordlist = s.split()
    return ' '.join(wordlist)


def sentence_stats(s: str):
    #Prints statistics about the inputted string
    print('Characters: '+str(len(s)))
    s = remove_punct(s)
    s = remove_spaces(s)
    wResult=1
    for i in s:
        if i == ' ':
            wResult += 1
    print('Words: '+str(wResult))
    sResult=0
    for i in s:
        if i != ' ':
            sResult += 1
    print('Average word length: '+str(sResult/wResult))
    
sentence_stats('I love UCI')
sentence_stats('***The ?! quick brown fox:  jumps over the lazy dog.')

def initials(s:str):
    #Returns the initials of the name within the parameter
    namelist = s.split()
    initialStr = ''
    for i in namelist:
        initialStr = initialStr+(i[0].upper())
    return initialStr

print('Bill Cosby:', initials('Bill Cosby'))
print('Guido van Rossum:', initials('Guido van Rossum'))
print('alan turing:', initials('alan turing'))

def Caesar_encrypt(s:str, n=int):
    #Encrypts a string by moving each letter n letters forward


    s = s.lower()
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    new = ''
    for i in s:
        if i in alphabet:
            new = new+alphabet[(alphabet.find(i)+n)%26]
        else:
            new = new+i
    return new

print(Caesar_encrypt('I love UCI', 1))

def Caesar_decrypt(s:str, n:int):
    #Decrypts string by moving n each letter n letters backward


    s = s.lower()
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    new = ''
    for i in s:
        if i in alphabet:
            new = new+alphabet[(alphabet.find(i)-n)%26]
        else:
            new = new+i
    return new

            strList.append(x[i])

def list_of_words (x:list) ->list:
    '''takes a list of strings as above and returns a list of individual words with all white space and punctuation removed '''
    final_list = []
    for i in x:
        a = remove_punct(i)
        word_list = a.split()
        final_list.extend(word_list)
