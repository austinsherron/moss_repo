alphabet='abcdefghijklmnopqrstuvwxyz'
def rotated_alphabet(S:str,n:int)->str:
    '''produces a "rotated" alphabet with the specified number of characters taken of the
    front and added on to the end of the string'''
    a=alphabet[n:]+ alphabet[:n]
    return a
    
def Caesar_encrypt(s:str, n:int)->str:
    "encrypt the plaintext"
    a=rotated_alphabet(s, n).lower()
    table2=str.maketrans(alphabet, a)
    return s.lower().translate(table2)
assert Caesar_encrypt('abc', 25)=='zab'
assert Caesar_encrypt('hIj', 2)=='jkl'
assert Caesar_encrypt('XYZ', 3)=='abc'

def Caesar_decrypt(s:str,n:int)->str:
    "decrypt the code"
    a=rotated_alphabet(s, n).lower()
    table=str.maketrans(a, alphabet)
    return s.lower().translate(table)
assert Caesar_decrypt('zab', 25)=='abc'
assert Caesar_decrypt('jkl', 2)=='hij'


assert Caesar_decrypt('abc', 3)=='xyz'

def list_of_words(l:list)->list:
    table = str.maketrans("!:\,.?",'      ')
    A=[]
    for i in l:
        A.append(i.translate(table))
    B=[]
    for i in A:
        B.append(i.replace(" ",""))
    return B
