def initials(x:str):
    y = x.split(' ') #Splits each word in the sentence and put it in a list
    result =[] #Creates a list for the first letters of the each word
    for word in y:
        result.append(word[0].upper())
    return ''.join(result)

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def Caesar_encrypt(x:str, y:int):


    if y>25:
        y = y%26 #If the key is greater than 25, the function wraps around
    changed = str.maketrans(alphabet,#Maketrans changes input to output
                            alphabet[y:] + alphabet[:y])#Alphabet is changed
    #to an output that is an alphabet that starts with y and ends at the y
    changed2 = str.maketrans(alphabet.upper(),
                             alphabet.upper()[y:] + alphabet.upper()[:y])
    #Translation for the uppercased letters
    #The string is translated
    lower_gone = x.translate(changed)
    #lower_gone is the function with all lower letters translated
    return lower_gone.translate(changed2) #translates the uppers and returns 

print(Caesar_encrypt("ZoGe",27))

def Caesar_decrypt(x:str, y:int): #Reverses the encryption


    if y>25:
        y = y%26
    changed = str.maketrans(alphabet[y:] + alphabet[:y],
                            alphabet)
    changed2 = str.maketrans(alphabet.upper()[y:] + alphabet.upper()[:y],
                             alphabet.upper())
    lower_gone = x.translate(changed)
    return lower_gone.translate(changed2)

print(Caesar_decrypt("ApHf",1))

def empty_line(x:S)->list:
    empty = 0
    for sentence in x:
        if sentence.isspace() == True:
            empty+=1
    return empty
def avg_characters (x:S) -> list:
    result = 0
    for sentence in x:
        result+=len(sentence)
    return result
def avg_characters_no_space(x:S) -> list:
    result = 0
    for sentence in x:
        if sentence.isspace() == False:
            result+=len(sentence)
    return result
