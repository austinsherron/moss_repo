def punctuation(x: str)->None:
    ''' Translates all punctuation marks to blank'''
    table2 = str.maketrans('.,!?;:"*@#$%^&(){}[]<>/',
                           '                       ')
    return x.translate(table2).split()


def sentence_stats(x: str):
    '''Prints out statistics about character, words, and average word length'''


    print('Characters:', len(x))  
    print('Words:', len(punctuation(x)))
    total_length = 0
    for i in punctuation(x):      
        total_length += len(i)
    print('Average length of words:', total_length/len(punctuation(x)))
    return

def initials(x:str)->str:
    '''Returns initials of name in all capital letters'''
    result = ''
    for i in x.split():
        result = result + i[0].upper()
    return result

alphabet =  'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz'

def rotated_alphabet(x:str,y:int)-> str:
    "Returns str changed by x"
    result = ''
    change = len(x)
    for i in range(change):
        if y + i < change:
            result += x[y + i]
        if y + i >= change:
            result += x[y + i - change]
    return result

def Caesar_encrypt(x:str, y:int)->str:
    '''Returns x in a ciphertext'''


    table = str.maketrans(alphabet, rotated_alphabet(alphabet,y))
    return x.translate(table).lower()

def Caesar_decrypt(x:str, y:int)->str:
    '''Returns x in a plaintext'''
    return Caesar_encrypt(x, -y)


assert Caesar_encrypt('pineapple', 2) == 'rkpgcrrng'
assert Caesar_decrypt('rkpgcrrng', 2) == 'pineapple'
print('Encrypt "Hello!" with key = 3:', Caesar_encrypt("Hello!", 3))
print('Decrypt Above:', Caesar_decrypt('khoor!', 3))

def stats(L:list)->None:
    '''Prints statistics about lines in list, empty lines,
       average characters per line, and
       average characters per non-empty line'''
    h = '{:6}'


    lines_in_list = len(L)
    characters_per_line = 0
    empty_lines = 0
    for i in L:
        if i == "":
            empty_lines += 1
        characters_per_line += len(i)
    average_per_line = characters_per_line / lines_in_list
    average_per_nonempty_line = characters_per_line / (lines_in_list - empty_lines)
    print(h.format(lines_in_list), "lines in the list")
    print(h.format(empty_lines), "empty lines")
    print(h.format(average_per_line), "average characters per line")
    print(h.format(average_per_nonempty_line), "average characters per non-empty line")

def list_of_words(L:list)->list:
    '''Returns L as individual words without space or punctuation'''
    result = []
    for i in L:
        for x in punctuation(i):
            result.append(x)
    return result
