def sentence_stats(string: str):
    '''prints info about the string: char length, word length, avg length of each word'''
    print('Characters: ', len(string))
    table = str.maketrans('*?!:.;,', '       ')
    string = string.translate(table)
    new_list = string.split()
    print('Words: ', len(new_list))
    total = 0
    for w in new_list:
        total += len(w)
    print('Average word length: ', total / len(new_list))
    return
print("I like sentence stats.")

def initials(name: str) -> str:
    '''returns initials of the name in all caps'''
    name = name.upper()
    name_in_list = name.split()
    initial_string = ''
    for w in name_in_list:
        initial_string += w[0]
    return initial_string
assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def distribution_of_rolls(number: int):
    '''prints stats about the number of rolls'''
    roll_list = []
    for i in range(number):
        roll_list.append(roll2dice())

    for t in range(2, 13):
        count = roll_list.count(t)
        print('{0:2d}: {1:5d} ({2:4.1f}%) {3:15s}'.format(t, count, (count / len(roll_list)) * 100, count * '*'))
    print('-----------------------------\n', '\t', number, 'rolls')
    return

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

print('-----e.1-----')
def rotate_alphabet(number: int) -> str:
    '''returns a rotated alphabet based on the key given'''
    if number > 26:
        number = number % 26 #(e.3) in this if statement
    return ALPHABET[number:] + ALPHABET[:number]



def Caesar_encrypt(message: str, key: int) -> str:
    '''returns ciphertext out of the message using the key'''
    message = message.lower()
    key_table = str.maketrans(ALPHABET, rotate_alphabet(key))
    return message.translate(key_table)
assert Caesar_encrypt('testing', 1) == 'uftujoh'
assert Caesar_encrypt('more assert', 0) == 'more assert'
    


def Caesar_decrypt(message: str, key: int) -> str:
    '''returns plaintext (OG message) using the key'''
    message = message.lower()
    key_table = str.maketrans(rotate_alphabet(key), ALPHABET)
    return message.translate(key_table)
assert Caesar_decrypt('uftujoh', 1) == 'testing'
assert Caesar_decrypt('more assert', 0) == 'more assert'

print("'Hello World!' encrypted with key = 7 is", Caesar_encrypt('Hello World!', 7))
print("'olssv dvysk!' decrypted with key = 7 is", Caesar_decrypt('olssv dvysk!', 7))

def stats(str_list: list):
    '''prints stats of the string'''
    print("{:2d} {}".format(len(str_list), 'lines in the list'))
    print("{:2d} {}".format(str_list.count(''), 'empty lines'))
    total = 0
    for i in str_list:
        total += len(i)
    print("{:2.0f} {}".format(total / len(str_list), 'average characters per line'))
    new_length = 0
    for i in str_list:
        if i != '':
            new_length += 1
    print("{:2.0f} {}".format(total / new_length, 'average characters per non-empty line'))       
    return
stats(list_of_strings)

def list_of_words(str_list: list) -> list:
    '''returns a list of individual words with whitespace and punctuation removed (except ' <- that)'''
