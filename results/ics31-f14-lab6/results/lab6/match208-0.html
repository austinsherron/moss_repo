def sentence_stats(n:str)->str:
    ''' print the length in character, length in words, and average length of
        each word in the first parameter'''
    key= str.maketrans('~!@#$%^&*()`;:",./?<>=+-_\[]{}', ' ' * 30)
    
    w= n.translate(key).split()
    
    words=len(w)
    
    words_after_punc= n.translate(key).replace(' ','')
    
    total_word_after_punc= len(words_after_punc)
    
    characters= len(n)
    
    average= total_word_after_punc/words
    
    print('\nCharacters:', characters,'\nWords:', words,
          '\nAverage word length:', average,'\n')
    
sentence_stats('I love UCI')
sentence_stats('***The ?! quick brown fox:  jumps over the lazy dog.')

def initials(n:str)->str:
    ''' Return the initial of each word in the string'''
    result=''
    x=n.upper().split()
    for i in x:
        result= result+ i[0]
    return result
assert initials('Bill Cosby')=='BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def Caesar_encrypt(m:str,key: int)-> int:
    ''' Translate the first parameter according to the second parameter'''
    
    if key>25:
        key= 25%26
        
    n= str.maketrans('abcdefghijklmnopqrstuvwxyz', 'abcdefghijklmnopqrstuvwxyz'[key: ] + 'abcdefghijklmnopqrstuvwxyz'[:key])
    
    translation=m.lower().translate(n)
    
    return translation

def stats(n:[str])-> str:
    ''' print the statistic of the string'''

    total_characters=0
    for each_str in n:
        total_characters+= len(each_str)
        
    empty_lines=0
    for i in n:
        if i.strip()=='':
            empty_lines+= 1
        
    print("{:4d}{:2} lines in the list".format(len(n),' '))
    print("{:4d}{:2} empty lines".format(empty_lines,' '))
          
    print("{:6.1f} average characters per line".format(total_characters/len(n), ' '))
    print("{:6.1f} average characters per non-empty line".format(total_characters/(len(n)-empty_lines),' '))
    

stats(a)

print('-------------------------')
