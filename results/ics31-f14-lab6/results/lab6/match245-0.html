def remove_punctuation(string:str) -> str:
    ''' Removes punctuation from a string. '''
    table1 = str.maketrans('*.,!?;:"','        ')
    return string.translate(table1)

def sentence_stats(string:str):
    ''' Takes a string. Returns stats about its length in characters,
         its length in words, and the average length of each word '''
    words = len(remove_punctuation(string).split())
    characters = len(string)
    letters = len(remove_punctuation(string).replace(" ", ""))
    print("Characters:", characters)
    print("Words:", words)
    print("Average word length:", letters / words)

def initials(string:str) -> str:
    ''' Return first character of each work in string '''

ALPHABET = "abcdefghijklmnopqrstuvwxyz"
def Caesar_encrypt(message:str, key:int) -> str:
    ''' Shift characters in message by key '''
    key = key % 26
    shifted_alphabet = ALPHABET[key:] + ALPHABET[:key]
    table = str.maketrans(ALPHABET, shifted_alphabet)
    return message.translate(table)

def Caesar_decrypt(message:str, key:int) -> str:
    ''' Undo shift characters in message by key '''
    key = -key % -26
    shifted_alphabet = ALPHABET[key:] + ALPHABET[:key]
    table = str.maketrans(ALPHABET, shifted_alphabet)
    return message.translate(table)

print(Caesar_decrypt("dbu", 1))
print(Caesar_encrypt("cat", 29))
print(Caesar_encrypt("cat", 3))
print(Caesar_decrypt(Caesar_encrypt("cat", 3),3))

assert Caesar_decrypt("dbu", 1)  == 'cat'
assert Caesar_encrypt("cat", 29) == 'fdw'
assert Caesar_encrypt("cat", 3)  == 'fdw'
assert Caesar_decrypt(Caesar_encrypt("cat", 3),3) == 'cat'
