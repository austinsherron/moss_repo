def sentence_stats(x:str):
    '''Takes in a string as a parameter and prints out:its length in characters,
    its length in words, and average length of each word.'''
    characters=len(x)
    print('Characters:',characters)
    table=x.maketrans('"./,:[];?\!','           ')
    z=x.translate(table)
    y=z.split()
    words=len(y)
    print('Words:',words)
    a=0
    for char in y:
        a+=len(char)
    print('Average word length:',a/words)

def initials(name:str)->str:
    '''takes as input a string representing a full name
    and returns the initials of the name in all capital letters  '''
    first=''
    ini=name.title().split()
    for n in range(len((ini))):
        first= first+(ini[n][0])
    return first

ALPHABET='abcdefghijklmnopqrstuvwxyz'

def new_alpha(x:str,n:int)->str:


    return x[n:27]+ x[0:n]


def Caesar_encrypt(plaintext:str,n:int)->str:
    '''takes two arguments: a string containing the message
    and an int for the key and returns the ciphertext'''
    x=plaintext.lower()
    code=x.maketrans(ALPHABET,new_alpha(ALPHABET,n))
    return x.translate(code)
print(Caesar_encrypt('Hello',2))

def Caesar_decrypt(ciphertext:str,n:int)->str:
    '''takes two arguments: a string containing the message
    and an int for the key and returns the plaintext'''
    x=ciphertext.lower()
    code=x.maketrans(new_alpha(ALPHABET,n),ALPHABET)
    return x.translate(code)
print(Caesar_decrypt('Jgnnq.',2))

def stats(s):
    '''Takes a list and prints the statistics.'''
    empty_lines=0
    for num in s:
        if num == len(num)*'  ':
            empty_lines+=1
    average_characters=0
    for n in s:
        m=n.split()
        for m in s:
            average_characters += len(m)
    print('{0:5.1f} {1}'.format(len(s),'lines in the list'))
    print('{0:5.1f} {1}'.format(empty_lines,'empty lines'))
    print('{0:5.1f} {1}'.format(average_characters/len(s),'average characters per line'))
    print('{0:5.1f} {1}'.format(average_characters/len(s)-empty_lines,'average characters per non-empty line'))

def list_of_words(x:list):
    ''' takes a list of strings as above and
    returns a list of individual words with all white space and punctuation removed'''
    words=[]
    table=str.maketrans('"./,:[];?\!','           ')
    for word in x:
        words.extend(word.split())
        word.translate(table)
    return words
