def Caesar_encrypt(x:str,y:int)->str:
    '''encrypts x with cipher substitute y away from x of alphabet'''
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    alphabet2 = []
    for i in range(len(alphabet)):
        alphabet2.append(alphabet[(i+y)%26])
    alphabet2 = ''.join(alphabet2)
    table = str.maketrans(alphabet, alphabet2)
    return x.lower().translate(table)

print(Caesar_encrypt('cats',0))
print(Caesar_encrypt('cats',1))



def Caesar_decrypt(x:str,y:int)->str:
    ''' decrypts x with cipher key y '''
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    alphabet2 = []
    for i in range(len(alphabet)):
        alphabet2.append(alphabet[(i-y)%26])
    alphabet2 = ''.join(alphabet2)
    table = str.maketrans(alphabet, alphabet2)
    return x.lower().translate(table)

print(Caesar_decrypt('cats',0))
print(Caesar_decrypt('dbut',1))

def list_of_words(x:list)->str:
    ''' takes list of words, removes punctuation and puts it together'''
    wordlist = []
    punc = ',./\][}{+=-_()*&^%$#@!<>?"'


    for i in range(len(x)):
        wordlist.extend(x[i].split())
    for i in range(len(wordlist)):
        table = str.maketrans(punc, ' '*len(punc))     
        wordlist[i] = wordlist[i].translate(table)
    return ' '.join(list(wordlist))
print(list_of_words(stuff))
