def initials(name:str)->str:
    'Return initial of the name'
    ini = [name[0]]
    ini_str =''
    for i in range(len(name)):
       if name[i-1] ==' ':
           ini.append(name[i])
    for j in ini:
        ini_str +=j
    
    return ini_str.upper()

assert initials('Bill Cosby') == 'BC'
assert initials('Guido van Rossum') == 'GVR'
assert initials('alan turing') == 'AT'

def roll2dice():
    d1 = randrange(1,7)
    d2 = randrange(1,7)

def Ceasar_encrypt(string:str,key:int)->str:
    string = string.lower()
    Alphabet = 'abcdefghijklmnopqrstuvwxyz'
    encrypted_text = ''
    if key > 25:
        key = key%25
    for i in range(len(string)):
        if string[i] in Alphabet:
            position = Alphabet.find(string[i])
            if position+key > 26:
                out_of_range_position = (position+key)%26
                encrypted_text += Alphabet[out_of_range_position]
            else:
                encrypted_text+=Alphabet[position+key]
                            
        else:
            encrypted_text +=string[i]
    return encrypted_text


assert Ceasar_encrypt('a',1) =='b'
assert Ceasar_encrypt('abcd',2) =='cdef'
assert Ceasar_encrypt('cat',29) == Ceasar_encrypt('cat',4)
assert Ceasar_encrypt('I love dogs',25) == 'h knud cnfr'



def Ceasar_decrypt(crypt:str,key:int)->str:
    Alphabet = 'abcdefghijklmnopqrstuvwxyz'
    crypt = crypt.lower()
    decrypted_text = ''
    if key > 25:
        key = key%25
    for i in range(len(crypt)):
        if crypt[i] in Alphabet:
            position = Alphabet.find(crypt[i])
            decrypted_text += Alphabet[position-key]
        else:
            decrypted_text += crypt[i]
    return decrypted_text

assert Ceasar_decrypt('cdef',2) == 'abcd'
assert Ceasar_decrypt('h knud cnfr',25) == 'i love dogs'

def stats(list_of_strings:list):
    number_of_lines = len(list_of_strings)
    number_of_empty = 0
    total_characters = 0
    for string in list_of_strings:
        if len(string) == 0:
            number_of_empty +=1
        
    for string in list_of_strings:
        total_characters += len(string)
    average_characters = total_characters/number_of_lines
    average_non_empty = total_characters/(number_of_lines - number_of_empty)

    print('{:5s} {:} {:}'.format(str(number_of_lines), '   ' ,'lines in the list'))
    print('{:5s} {:} {:}'.format(str(number_of_empty) ,'   ' , 'empty lines'))
    print('{:5.1f} {:} {:}'.format(average_characters , '   ' ,'average characters per line'))
    print('{:5.1f} {:} {:}'.format(average_non_empty , '   ' , 'average characters per non empty line'))

def list_of_words(str_list:list)->list:
    words_list = []
    table = str.maketrans(',.:;!?','      ')
    for string in str_list:
        string = string.translate(table)
        short_list = string.split()
        for word in short_list:
            words_list.append(word)
    return words_list
