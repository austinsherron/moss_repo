def calculate_shipping (w: float) -> float:
    'Returns the price for shipping of a package given the weight of the package'
    if w < 2:
        return 2.00
    elif w < 10:
        return 5.00
    else:
        return 5.00 + 1.5 * (w - 10)
assert calculate_shipping(1.5) == 2
assert calculate_shipping(7) == 5
assert calculate_shipping(15) == 12.5

print('We are printing shipping prices given the weights 1.5, 7, and 15:')
print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))

def num_inventory(bl: 'list of books') -> int:
    'Takes a list of Books and returns the total number of books in stock of that list'
    total  = 0
    for i in range(len(bl)):
        total += bl[i].inStock
    return total
        
if num_inventory(before_2000) > num_inventory(after_2000):
    print('More titles before 2000 (', num_inventory(before_2000), 'vs', num_inventory(after_2000), ')')
elif num_inventory(before_2000) < num_inventory(after_2000):
    print('More titles after 2000 (', num_inventory(after_2000), 'vs', num_inventory(before_2000), ')')
else:
    print("It's a tie")

def inventory_value(bi: 'book in a list') -> float:
    'Takes a Book object and returns the total value of the inventory of the Book'
    total_value = bi.inStock * bi.price
    return total_value

def top_value(bl: 'book list') -> Book:
    'Takes a book list and returns the Book object with the highest total value' 
    Best_Book = Book('Ethan', 'Memoirs', 'Autobiographical', 0, -1, 1)
    for i in range(len(bl)):
        if inventory_value(Best_Book) < inventory_value(bl[i]):
            Best_Book = bl[i]
    return Best_Book

print('The highest value book is', top_value(BSI).title, 'by', top_value(BSI).author, 'at a value of $', inventory_value(top_value(BSI)), ".")

def draw_face(my_window1: 'Window', x: int, y: int, scale: float, hair_color: str, eye_color: str, expression: str) -> None:
    'Draws a face using the top left coordinate, hair color, eye color, and expression (smile/frown/surprised)'
    my_window1 = tkinter.Tk()    # Create the graphics window
    my_canvas1 = tkinter.Canvas(my_window1, width=700, height=700)  # Create a 700x700 canvas to draw on A
    my_canvas1.pack()    # Put the canvas into the window
    #CALL HAIR FUNCTION
    draw_hair(my_canvas1, x, y, hair_color)
    #CALL HEAD FUNCTION
    draw_head(my_canvas1, x, y)
    #CALL NOSE FUNCTION
    draw_nose(my_canvas1, x, y)
    #CALL EYE FUNCTION
    draw_eye(my_canvas1, x, y, eye_color)
    draw_eye(my_canvas1, x+50, y, eye_color)
    #CALL EYEBROW FUNCTION
    draw_eyebrows(my_canvas1, x, y, hair_color, expression)
    draw_eyebrows(my_canvas1, x + 50, y, hair_color, expression)
    #CALL MOUTH FUNCTION
    draw_mouth(my_canvas1, x, y, expression)

    A.create_line(50 + x, 50 + y, 7 + x, 22 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)

    A.create_line(50 + x, 50 + y, 25 + x, 2 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)
    A.create_line(50 + x, 50 + y, 37 + x, -2 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)
    A.create_line(50 + x, 50 + y, 50 + x, -3 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)
    A.create_line(50 + x, 50 + y, 63 + x, -2 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)
    A.create_line(50 + x, 50 + y, 74 + x, 2 + y, smooth = True, capstyle="projecting", width=3, fill = hair_color)

def draw_head(A: "Canvas", x: int, y: int):
    #HEAD
    A.create_oval(0 + x, 0 + y, 100 + x, 100 + y, fill = 'Wheat') #head

def draw_nose(A: "Canvas", x: int, y: int) -> None:
    #NOSE
    A.create_line(50 + x, 33 + y, 10 + x, 70 + y, 60 + x, 66 + y, smooth = True) #nose
    A.create_line(53 + x, 62 + y, 60 + x, 48 + y, 67 + x, 62 + y, smooth = True) #nostril
    A.create_oval(57 + x, 60 + y, 63 + x, 66 + y, fill = 'dim grey')

def draw_eye(A: "Canvas", x: int, y: int, eye_color: 'str') -> None:
    #EYEBALL
    A.create_line(20 + x, 30 + y, 30 + x, 30 + y, fill='black', width=12, capstyle="round")
    A.create_line(20 + x, 30 + y, 30 + x, 30 + y, fill='white', width=10, capstyle="round")
    A.create_oval(20 + x, 25 + y, 30 + x, 35 + y, fill=eye_color)
    A.create_oval(23 + x, 28 + y, 28 + x, 33 + y, fill='black')

def draw_eyebrows(A: "Canvas", x: int, y: int, hair_color: str, expression: 'str for smile/frown/surprised') -> None:
    if expression == 'smile' or expression == 'surprised':
        #RAISED EYEBROWS
        A.create_line(15 + x, (23 - 3) + y, 35 + x, (23 - 3) + y, width=3, fill = hair_color)
    elif expression == 'frown':
        #EYEBROWS
        A.create_line(15 + x, 23 + y, 35 + x, 23 + y, width=4, fill = hair_color)

def draw_mouth(A: "Canvas", x: int, y: int, expression: 'str for smile/frown/surprised') -> None:
    if expression == 'smile':
        #MOUTH SMILE
        A.create_line(25 + x, 75 + y, 50 + x, 95 + y, 75 + x, 75 + y, smooth="True", width=5, capstyle="round", fill="firebrick")
        A.create_line(25 + x, 75 + y, 50 + x, 95 + y, 75 + x, 75 + y, smooth="True", width=2, capstyle="round")
    elif expression == 'frown':
        #MOUTH FROWN
        A.create_line(25 + x, 85 + y, 50 + x, 65 + y, 75 + x, 85 + y, smooth="True", width=5, capstyle="round", fill="firebrick")
        A.create_line(25 + x, 85 + y, 50 + x, 65 + y, 75 + x, 85 + y, smooth="True", width=2, capstyle="round")
    elif expression == 'surprised':
        #MOUTH SURPRISE
        A.create_oval(38 + x, 71 + y, 62 + x, 95 + y, outline="firebrick", fill="firebrick")
        A.create_oval(40 + x, 73 + y, 60 + x, 93 + y, fill="dim grey")
