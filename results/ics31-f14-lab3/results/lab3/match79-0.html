def calculate_shipping (weight:int) -> float:
    ''' Returns price of shipping according to input weight '''
    if weight < 2:
        return 2.00
    elif 2 <= weight & weight < 10:
        return 5.00
    elif 10 <= weight:
        return 5.00 + 1.50 * (weight - 10)
assert calculate_shipping(1) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(4) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value (b:Book) -> float:
    ''' Returns the value of the inventory of the input book '''
    return b.price * b.instock

def top_value (lb:[Book]) -> Book:
    highest_value_book = Book('','','',0,0,0)  #an empty book variable to store the book with the highest value by far
    for b in lb:
        if inventory_value(highest_value_book) < inventory_value(b):
            highest_value_book =b       #replace the value of highest_value_book
    return highest_value_book

hv_book = top_value(BIS)

      'at a value of $', inventory_value(hv_book))
