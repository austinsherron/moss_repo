def calculate_shipping(Pound: int) -> float:
    "Returns the cost of shipping a certain item based on its weight."
    if Pound < 2:
        return 2.00
    elif Pound >= 2 and Pound <= 10:
        return 5.00
    else:
        return 5.00 + (1.50 * (Pound - 10))

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(1))
print(calculate_shipping(13))

def inventory_value(BList: Book) -> float:
    "Returns the price of the total stock of a book."
    return BList.price * BList.instock

def top_value(BList: Book) -> Book:
    "Returns the top valued book from a list of books."
    Highest_Value = inventory_value(BList[0])
    for i in range(len(BList)):
        if inventory_value(BList[i]) > Highest_Value:
            Highest_Value = inventory_value(BList[i])


    for i in range(len(BList)):
        if Highest_Value == inventory_value(BList[i]):
            return BList[i]

print('The highest-value book is ', top_value(BSI).title ,
      ' by ', top_value(BSI).author, ' Leo at a value of $ ',
      inventory_value(top_value(BSI)), sep = '')
