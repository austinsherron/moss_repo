def calculate_shipping(w: float) -> float:
    "returns the price based on the weight"
    if w < 2.00:
        return 2.00
    elif w >= 2.00 and w < 10.00:
        return 5.00
    elif w >= 10.00:
        return 5.00 + (w-10.00)*1.50

assert calculate_shipping(1.50) == 2.00
assert calculate_shipping(7.00) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2.00) == 5.00
assert calculate_shipping(10.00) == 5.00

def inventory_value(b: Book) -> float:
    "returns the inventory value of a book (stock*price)"
    return b.instock * b.price

assert inventory_value(BSI[-1]) == .25 * 1.1 * 1411
assert inventory_value(BSI[2]) == 56 * 1.1 * 25

def top_value(bl: list) -> Book:
    "returns the Book with the highest inventory value in a given list"
    sortedbl = sorted(bl, key = inventory_value)
    return sortedbl[-1]

TB = top_value(BSI)
print("The highest-value book is", TB.title, 'by', TB.author,
      "at a value of", inventory_value(TB))

def draw_nose(x: float, y: float, l:float):
    my_canvas.create_polygon(x, y,
                             x - (l/2), y + l,
                             x + (l/2), y + l,
                             x, y)

def draw_mouth(x: float, y: float, l: float):
    my_canvas.create_line(x, y, x + l, y)

def draw_face(x: float, y: float, l: float, w: float):
    draw_eye(x, y, l, w)
    draw_eye(x + 2*w, y, l, w)
    draw_nose(2*w, 3*l, w)
    draw_mouth(w, 8*l, 2*w)
    
    
draw_face(60, 60, 60, 110)
draw_face(50, 50, 50, 100)
tkinter.mainloop()
