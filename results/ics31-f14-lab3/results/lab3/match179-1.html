def print_even_numbers(integers: list) -> None:
    """Takes a list of integers and prints out even numbers by only printing out
       the numbers who have a remainder of zero after being divided by two"""
    for x in range(len(integers)):
        if integers[x]%2==0:
            print(integers[x])

def calculate_shipping(weight: float) -> float:
    "Takes the weight of the packaged to be shipped and returns its shipping price"
    if weight < 2:
        return 2.00
    if weight >= 2 and weight < 10:
        return 5.00
    if weight >= 10:
        return (5.00 + 1.5 *(weight-10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
