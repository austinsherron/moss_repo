def calculate_shipping (value: float) -> float:
    '''Calulates shipping by giving different slots for weight'''
    if value < 2.0:
        result = 2.00
    elif 2.0 <= value < 10.0:
        result = 5.00
    else:
        result = (5.00 + (1.50 * (value - 10)))
    return result

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))
