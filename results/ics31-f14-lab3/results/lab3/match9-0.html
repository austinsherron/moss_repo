def calculate_shipping(weight) -> float:
    "Will compute how much it costs to ship"
    if weight < 2.00:
        return 2.00

    if 2.00 < weight < 10.00:
        return 5.00

    if weight >= 10.00:
        return 5.00 + ((weight-10)*1.5)



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(4) == 5.00
assert calculate_shipping(15) == 12.50
