def calculate_shipping (n:float)->float:
   
    if n < 2:
        the_rate = 2
    elif 2 <= n < 10: 
        the_rate = 5
    else :
        the_rate = 5+(n-10)*1.5

import tkinter
my_window = tkinter. Tk ()
my_canvas = tkinter.Canvas (my_window, width = 600, height = 600)
my_canvas.pack()

def create_eye (x:float, h: float, v:float):
    # v help the eye move vertical, h help the eye move horizonal 
    my_canvas.create_oval (x+h,x+v-100,x+h-100,x+v-40,fill= 'white')
    my_canvas.create_oval (x+h-25,x+v-95,x+h-75,x+v-45,fill='brown')
    my_canvas.create_oval (x+h-40,x+v-80,x+h-60,x+v-60,fill= 'black')

def create_nose (x:float, h: float, v:float):
    my_canvas.create_line (x+h+10,x+v-35,x+h+10,x+v-5)

def create_mouth (x:float, h: float, v:float):
    my_canvas.create_oval (x+h-10,x+v-100,x+h-80,x+v-10,fill= 'red')

def create_face (x:float, h:float, v:float):
    my_canvas.create_oval (x+h+300,x+v+20,x+h+18,x+v+300)
    create_eye (120+x,24+h,40+v)
    create_eye (120+x,150+h,40+v)
    create_nose(140+x,5+h,30+v)
    create_mouth(130+x,70+h,150+v)
    tkinter.mainloop()

create_face(50,50,0)
