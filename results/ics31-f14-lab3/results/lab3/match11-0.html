def calculate_shipping(x: (float)):
    '''takes weight and returns price for shipping'''
    if x < 2:
        return (2.00)
    if x >= 2 and x < 10:
        return (5.00)
    return (5.00 + (1.50 * (x - 10)))

assert calculate_shipping(7) == 5.00
