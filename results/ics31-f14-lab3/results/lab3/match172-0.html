def calculate_shipping(weight : float) -> float:
    'Calculates the shipping cost based on weight'
    if weight < 2:
        price = 2.00
    elif weight >= 2 and weight < 10:
        price = 5.00
    else:
        price = 5.00 + 1.50 * (weight - 10)
    return price

assert calculate_shipping(1) == 2.00
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(4) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book : Book) -> float:
    '''Returns the value of inventory'''
    return book.price * book.instock

def top_value(bookList : list) -> Book:
    '''Returns the book with highest valued inventory'''
    book = bookList[0]
    for count in range(1, len(bookList)):
        if inventory_value(bookList[count]) > inventory_value(book):
            book = bookList[count]
    return book

my_window = tkinter.Tk()

def draw_eye (x : int, y : int):
    '''Draw an eye'''

    my_canvas.create_oval(x, y , x + 100, y + 50, fill="white") #retina
    my_canvas.create_oval(x + 30, y + 10, x + 70, y + 40, fill="brown") #iris
    my_canvas.create_oval(x + 40, y + 15, x + 60, y + 35, fill="black") #pupil


def draw_nose (x : int, y : int):
    '''Draw a nose'''
    my_canvas.create_line(x, y, x + 70, y - 90, fill='orange')
    my_canvas.create_line(x + 70, y - 90, x + 140, y, fill='orange')
    my_canvas.create_line(x + 140, y, x, y, fill='orange')

def draw_mouth (x : int, y : int):
    '''Draw a mouth'''
    my_canvas.create_oval(x, y , x + 100, y + 50, fill='red')
    
def draw_face(x : int, y : int):
    '''Draw a face'''
