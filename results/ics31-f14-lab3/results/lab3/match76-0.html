def calculate_shipping(w: float) -> float:
    '''Returns the price of shipping package depending on its weight'''
    if w < 2:
        result = 2.00
    elif w >= 2 and w < 10:
        result = 5.00
    elif w >= 10:
        result = 5.00 + ((w - 10) * 1.50)
    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))

def inventory_value(Book: Book) -> float:
    '''Returns the value of a book's inventory'''
    return Book.price * Book.instock
assert inventory_value(BSI[0]) == 4000.00
print(inventory_value(BSI[0]))
print()

def top_value(L: list) -> Book:
    '''Returns the book with the highest inventory value from a list of books'''
    top = 0
    top_book = L[0]
    for i in range(len(L)):
        if L[i].price * L[i].instock > top:
            top = L[i].price * L[i].instock
            top_book = L[i]
    return top_book
assert top_value(BSI) == BSI[1]
print(top_value(BSI))
print()

print("The highest-value book is " + top_value(BSI).title + " by " +
      top_value(BSI).author + " at a value of $",
      top_value(BSI).price * top_value(BSI).instock, ".", sep="")

def draw_eye(x, y, x1, y1):
    '''Returns a drawing of an eye given the coordinates of the top left corner and bottom
    right corner of a cicle'''
    my_canvas.create_oval(x, y, x1, y1)
    my_canvas.create_oval(x * 1.05, y * 1.05, x1 * .95, y1 * .95, fill="yellow")
    my_canvas.create_oval(x * 1.07, y * 1.07, x1 * .93, y1 * .93, fill="black")

def draw_nose(x, y, x1, y1):
    '''Returns a pointy nose given the (x,y) coordinates of the beginning
    and end of a line'''
    my_canvas.create_line(x, y, x1, y1)
    my_canvas.create_line(x1, y1, x, y + (y1 - y))
    
def draw_mouth(x, y, x1, y1):
    '''Returns an arc of -180 degrees given the (x,y) coordinates of the rectangle
    it's inscribed in''' 
    my_canvas.create_arc(x, y, x1, y1, extent=-180, fill="black")

def draw_face(x, y, x1, y1):
    '''Returns a face given the coordinates of the top left corner and bottom
    right corner of a cicle'''
    my_canvas.create_oval(x, y, x1, y1)
    draw_eye(x * 1.10, y * 1.25, x1 * .65, y1 * .75)
    draw_eye(x * 1.6675, y * 1.25, x1 * .95, y1 * .75)
    draw_nose(x * 1.45, y * 1.45, x1 * .80, y1 * .80)
    draw_mouth(x * 1.35, y * 1.45 , x1 * .85, y1 * .95)
