def calculate_shipping(weight:float) -> None:
    ''' returns price for shipping the package '''
    if weight < 2:
        return(2.00)
    elif weight < 10:
        return(5.00)
    else:
        return(5.00 + 1.50*(weight-10))
        
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter  
my_window = tkinter.Tk()   
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()  

def draw_eye(x1:int, y1:int, x2:int, y2:int) -> None:
    '''return one eye'''        
    my_canvas.create_oval(x1, y1, x2, y2, fill = 'yellow')
    x3 = x1 + 75
    x4 = x2 - 75
    my_canvas.create_oval(x3, y1, x4, y2, fill = 'green')
    x5 = x1 + 125
    y3 = y1 + 50
    x6 = x2 - 125
    y4 = y2 - 50
    my_canvas.create_oval(x5, y3, x6, y4, fill = 'black')



def draw_nose(x1:int, y1:int, length_x: int, length_y:int) -> None:                 
    x2 = x1 + length_x
    y2 = y1 + length_y
    my_canvas.create_rectangle(x1, y1, x2, y2, fill = 'pink')


def draw_mouth(x1:int, y1:int, diameter_x:int, diameter_y:int)-> None:
    x2 = x1 + diameter_x
    y2 = y1 + diameter_y
    my_canvas.create_oval(x1,y1,x2,y2, fill='red')



def draw_face_shape(x1:int,y1:int,diameter:int)-> None:
    x2 = x1 + diameter
    y2 = y1 + diameter
    my_canvas.create_oval(x1, y1, x2, y2,)




def draw_face():         
    draw_eye(100, 200, 400, 350)
    draw_eye(600, 200, 900, 350)
    draw_nose(450, 275, 100, 200)
    draw_mouth(300,550, 400, 200)
    draw_face_shape(15,30,975)
    tkinter.mainloop()
