def find_area_circle(radius) -> float:
    """ Calculate area of a circle """
    area = 3.14159 * radius ** 2
    return area
    assert find_area_circle(1) == 3.14159
    assert find_area_circle(5) == 78.53975

def calculate_shipping(package: int) -> float:
    " Calculate the price for shipping the package "
    if package < 2:
        r = 2.00
    elif 2 <= package < 10:
        r = 5.00
    elif package >= 10:
        x = package - 10
        x = x * 1.50
        r = 5.00 + x
    return r

    assert calculate_shipping(1.5) == 2.00
    assert calculate_shipping(7) == 5.00
    assert calculate_shipping(15) == 12.50

import tkinter
my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()
    
def draw_eye(x = int, y = int):
    my_canvas.create_oval(x,y,x+75,y+50, fill = 'white')
    my_canvas.create_oval(x+35,y+10,x+70,y+40, fill = 'blue')
    my_canvas.create_oval(x+40,y+15,x+65,y+35, fill = 'black')
    
def draw_nose(x = int, y = int, w = int):
    my_canvas.create_polygon(x, y, x - (.5*w), y + 70, x + w, y + 50, x, y, fill = 'brown')

def draw_mouth(x = int, y = int, w = int):
    my_canvas.create_oval(x, y + 100, x + w, y + 25)

def draw_head(x = int, y = int, w = int):


    my_canvas.create_oval(x, y , x + 2.5*w, y + 4*w)
def draw_face(x = int, y = int, w = int):
    draw_eye(x+50,y+100)
    draw_eye(x+150,y+100)
    draw_nose(x+150, y+200, .35*w)
    draw_mouth(x+105, y+300, 1.2*w)
    draw_head(x+25, y+25, w)

draw_face(0,0,100)
draw_face(500,500,100)

tkinter.mainloop()
