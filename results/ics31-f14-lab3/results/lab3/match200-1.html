def draw_eye(x: int, y: int, canvas):
    'contains code for eye'
    canvas.create_oval(x, y, x + 100, y + 50, fill = 'red')
    canvas.create_oval(x + 33, y, x + 66, y + 50, fill = 'yellow')
    canvas.create_oval(x + 45, y + 20, x + 55, y + 30, fill = 'black')

def draw_mouth(x: int, y:int, canvas):
    'contains code for mouth'
    canvas.create_oval(x, y, x + 200, y + 75, fill = 'yellow')

def draw_nose(x: int, y: int, canvas):
    'contains code for nose'
    canvas.create_polygon(x, y, x - 50, y + 50, x + 50, y + 50,fill = 'yellow', outline = 'black')

def draw_face():
    my_window = tkinter.Tk()
    canvas = tkinter.Canvas(my_window, width=500, height=500)
    canvas.pack()
    canvas.create_oval(3, 25, 500, 475, fill = 'midnight blue')
    draw_eye(100, 100, canvas)
    draw_eye(300, 100, canvas)
    draw_mouth(150, 300, canvas)
    draw_nose(250, 200, canvas)
    tkinter.mainloop()
    
draw_face()
