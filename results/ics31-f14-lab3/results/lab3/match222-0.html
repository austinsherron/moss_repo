def calculate_shipping (x):
    ''' Returns the price for shipping the package.  '''
    if x < 2:
        return 2.00
    elif (x < 10 or x == 2):
        return 5.00
    else:
        return 5.00 + (( x - 10)*1.5)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00

import tkinter

my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1800, height=1800)
my_canvas.pack()

def draw_eye(x: int , y: int):
    ''' Takes the starting point,(x,y), and draw an eye. '''
        
    my_canvas.create_oval(x + 125, y + 300 , x + 375 , y + 150 , fill = 'white')
    my_canvas.create_oval(x + 175, y + 300 , x + 325 , y + 150, fill = 'blue')
    my_canvas.create_oval(x + 200, y + 275 , x + 300 , y + 175 , fill = 'black')

def draw_nose(x: int , y: int):
    ''' Takes the starting point,(x,y), and draw an nose. '''
    
    my_canvas.create_line( x + 375 , y + 350 , x + 350 , y + 425 )
    my_canvas.create_line( x + 350 , y + 425 , x + 450 , y + 425 )
    my_canvas.create_line( x + 450 , y + 425 , x + 420 , y + 350 )

def draw_mouth(x: int , y: int):
    ''' Takes the starting point,(x,y), and draw an mouth. '''
    my_canvas.create_line(x + 300, y + 600 , x + 500 , y + 600)

    
def draw_face(x: 'int or float', y: 'int or float'):
    ''' Takes the center point,(x,y), of a face and draw it. '''
    my_canvas.create_oval(x , y, x + 800 , y + 800 , fill = 'white')    
              
    draw_eye(x,y)
    draw_eye(x + 300,y)
    draw_nose(x,y)
    draw_mouth(x,y)
        
draw_face(10,10)
draw_face(850,10)
        
tkinter.mainloop()
