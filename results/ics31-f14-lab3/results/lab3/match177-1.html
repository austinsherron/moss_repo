    '''Returns the price of shipping'''

        return 2.00
    elif w < 10:
        return 5.00
    else:
        return 5 + (w-10)*1.5

def inventory_value (p:tuple) -> float:
    '''Returns the value of the total inventory of books'''
    return (p.price * p.instock)

def top_value (k:list):
    '''Returns the item in the list with the highest inventory value'''
    for y in range(len(k)):
        inventory_value(k[y])
    return sorted(k,key=inventory_value)[-1]

import tkinter
    
my_window1 = tkinter.Tk()
my_canvas1 = tkinter.Canvas(my_window1,width=500,height=500)
my_canvas1.pack()


def draw_eye(x,y,l: int):
    my_canvas1.create_oval(x,y,x+l,y+l)
    my_canvas1.create_oval(x+(l/6),y+(l/6),(x+l)-(l/6),(y+l)-(l/6),fill='blue')
    my_canvas1.create_oval(x+(l/3),y+(l/3),(x+l)-(l/3),(y+l)-(l/3),fill='black')

def draw_nose(x2,y2,l2: int):
    my_canvas1.create_line(x2,y2,x2+l2,y2+l2/4)
    my_canvas1.create_line(x2+l2,y2+l2/4,x2,y2-l2-4)

def draw_mouth(x3,y3,l3:int):
    my_canvas1.create_line(x3,y3,x3+l3,y3)

def draw_face():
    draw_eye(100,200,70)
    draw_eye(200,200,70)
    draw_nose(190,325,30)
    draw_mouth(100,370,200)
draw_face()
