def calculate_shipping(n:float)-> float:
    '''takes weight and calculates shipping cost'''
    if (n < 2):
        return 2.00


    elif (n < 10):
        return 5.00
    else:
        return 5 + 1.5*(n-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15)==12.50

my_window2 = tkinter.Tk()    

my_canvas2 = tkinter.Canvas(my_window2, width=500, height=500)

def draw_eye(x,y):
    '''creates an eye given the upper left coordinate'''
    my_canvas2.create_oval(x, y, x+50 , y+30, outline = 'black')
    my_canvas2.create_oval(x+15, y+5, x+30 , y+25, fill = 'light blue')
    my_canvas2.create_oval(x+20, y+10, x+25 , y+20, fill = 'black')

def draw_nose(x,y):
    '''creates a nose starting from the middle top'''
    my_canvas2.create_line(x,y,x+25,y+35, fill='black')
    my_canvas2.create_line(x+25,y+35,x,y+35, fill='black')
def draw_mouth(x,y):
    '''creates a mouth'''
    my_canvas2.create_polygon(x, y, x+25, y+50, x+ 50 ,y, fill='red')

def draw_face(x,y):
    '''creates a face given the upper left coordinate '''
    my_canvas2.create_oval(x, y, x+200, y+300, outline = 'black')
    draw_eye(x+15, y+100)
    draw_eye(x+135, y+100)
    draw_mouth(x+80,y+200)
    draw_nose(x+100,y+140)

draw_face(0,50)
