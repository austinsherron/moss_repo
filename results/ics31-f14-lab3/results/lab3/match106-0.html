def calculate_shipping(weight : float) -> float:
    #
    if weight < 2:
        return 2.00
    if weight < 10:
        return 5.00
    return 5.00 + 1.50 * (weight - 10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def create_square(x : int, y: int, l: int):
    # creates rectangle on tkinter at x,y with sides l

    my_window = tkinter.Tk()    # Create the graphics window

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window

    my_canvas.create_rectangle(x, y, x + l, y + l)

def inventory_value(book : Book) -> int:
    #finds inventory val by instock * book price
    return book.instock * book.price

def top_value(books : list) -> Book:
    #top value of book list
    temp = books[0]
    for b in books:
        if inventory_value(temp) < inventory_value(b):
            temp = b
    return temp

expensive_book = top_value(BSI)

print('The highest-value book is', expensive_book.title, 'by', expensive_book.author, 'at a value of $', inventory_value(expensive_book),'.')

    draw_eye(x+50, y+50 , my_canvas)

    draw_nose(x+225, y+125, my_canvas)

    tkinter.mainloop()             

def draw_eye(x : int, y: int , canvas):
    #draws eye
    canvas.create_oval(x,y,x+150,y+100, fill='orange') #pupil
    canvas.create_oval(x+55,y+30,x+95,y+70, fill='orange') #iris

def draw_nose(x : int, y: int, canvas):
    #draws nose
    canvas.create_polygon(x,y, x-50,y+100, x, y + 130 , x+50, y+100)

def draw_mouth(x :int, y: int, canvas):
    #draws mouth
    canvas.create_polygon(x,y, x + 150 , y -50, x , y + 50, x - 150 , y -50)
