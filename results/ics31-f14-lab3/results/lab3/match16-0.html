def abbreviate (n: str) -> str:
    '''Return a three-letter abbreviation'''
    return n[0]+n[1]+n[2]

assert abbreviate('abril') == 'abr'


def find_area_square (n: int) -> int:
    ''' Return a string with the area of the square '''
    return n ** 2

assert find_area_square(1) == 1
assert find_area_square(5) == 25


def find_area_circle (n: int) -> int:
    ''' Return a string with the area of the circle '''
    return n ** 2 * 3.14159

assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def caculate_shipping (n: int) -> str:
    '''Return a string with the price for shipping the package'''
    if n < 2:
        return 2.00
    if 2 <= n <= 10:
        return 5.00
    if n > 10:
        return 5 + 1.5 * (n-10)

assert caculate_shipping(1.5) == 2.00
assert caculate_shipping(7) == 5.00
assert caculate_shipping(15) == 12.50

my_window2 = tkinter.Tk()


my_canvas2 = tkinter.Canvas(my_window2, width = 800, height = 800)
my_canvas2.pack()

def draw_eye(x, y, z):
    my_canvas2.create_oval(x, y, x+1.5*z, y+1.25*z)


    my_canvas2.create_oval(x+0.5*z, y, x+1.5*z, y+z, fill = 'brown')
    my_canvas2.create_oval(x+z, y+0.5*z, x+1.5*z, y+z, fill = 'black')

def draw_nose(x, y, z):
    my_canvas2.create_polygon(x, y, x+z, y+z, x-z, y+z)

def draw_mouth(x, y, z):
    my_canvas2.create_oval(x+0.5*z, y+0.5*z, 0.5*x+z, 0.5*y+z, fill = 'red')
    


def draw_face(x, y, z):
    my_canvas2.create_oval(x, y, 2*(x+z), 2*(y+z))
    draw_eye(x+z, y+z, 0.5*z)
    draw_eye(x+2*z, y+z, 0.5*z)
    draw_nose(2*x, 2*y, 0.5*z)
    draw_mouth(1.5*x+0.3*z, 2.2*y+0.5*z, 0.5*x+z)

draw_face(200,200,100)
