def calculate_shipping(wt:float) -> float:
    if wt < 2:
       return 2.00
    elif wt < 10: 
        return 5.00
    elif wt >= 10:
        return 5 + (wt - 10) * 1.50

import tkinter              

my_window = tkinter.Tk()    

my_canvas = tkinter.Canvas(my_window, width=10000, height=10000)  
my_canvas.pack()

def draw_eye(a,b,c,d):
    my_canvas.create_oval(a,b,c,d)
    my_canvas.create_oval(a+100,b-75,c-50,d+75,fill='green')
    my_canvas.create_oval(a+115,b-60,c-65,d+60,fill='black')

def draw_nose(a,b,c,d):
    my_canvas.create_line(a,b,c,d)
    my_canvas.create_line(a-15,b+100,c+40,d)



def draw_mouth(a,b,c,d):
    my_canvas.create_arc(a,b,c,d)



def draw_face(a,b,c,d):
    my_canvas.create_oval(a,b,c,d)
    draw_eye(a+50,b+200,c-350,d-450)
    draw_eye(a+350,b+200,c-50,d-450)
    draw_nose(a+275,b+150,c-340,d-300)
    draw_mouth(a,b+350,c-150,d-100)
    
draw_face(50,100,650,650)
draw_face(800,100,1400,650)


tkinter.mainloop()
