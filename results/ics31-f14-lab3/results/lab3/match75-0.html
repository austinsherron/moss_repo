#calculate_shipping weight:float -> float
def calculate_shipping (weight:float) -> float:
    "Returns the shipping price depending on the parameter weight."
    if weight < 2:
        return 2.00
    if weight >= 2 and weight < 10:
        return 5.00
    if weight >= 10:
        return 5.00 + (weight - 10)*1.5

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))
