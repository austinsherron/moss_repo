def calculate_shipping(weight: float) -> float:
    'Calculates the price for shipping a package with a certain weight'
    if weight < 2:
        return 2.00
    if weight < 10:
        return 5.00
    return 5.00 + 1.50*(weight-10)

assert calculate_shipping(1.5)==2.00

def create_square(x: int, y: int, side: int) -> None:
    'Draws a square with the lenght of side, and the upper-left corner'
    'having coordinates (x, y)'

    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    LOWER_RIGHT_COORDINATE_X = x + side
    LOWER_RIGHT_COORDINATE_Y = y + side

    my_canvas.create_rectangle(x, y, LOWER_RIGHT_COORDINATE_X,
                               LOWER_RIGHT_COORDINATE_Y)

    tkinter.mainloop()

def draw_eyes (x1: int, y1: int, f:str, b: bool) -> None:
    """Takes in parameters to draw an eye in any location"""
    my_canvas.create_oval(x1,y1,x1+100,y1+200,fill='white') #Draws outer eye
    my_canvas.create_oval(x1+30,y1+100,x1+70,y1+150, fill=f) #Draws iris
    my_canvas.create_oval(x1+40,y1+120,x1+60,y1+140,fill='white', outline='white') #Draws pupil
    if b==True:
        draw_eyes(x1+200,y1,f, False)
