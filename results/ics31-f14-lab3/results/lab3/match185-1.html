def calculate_shipping(weight: int) -> int:
        " Returns price from a given weight"
        if weight < 2:
            return(2.00)
        else:
            if weight < 10 and weight >= 2:
                 return(5.00)
            else:
                if weight >= 10:
                        price = (weight - 10) *1.50 + 5
                        return price
                
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print("Shipping rate of a 1.5 pound package is", calculate_shipping(1.5))
print("Shipping rate of a 7 pound package is", calculate_shipping(7))
print("Shipping rate of a 15 pound package is", calculate_shipping(15))
