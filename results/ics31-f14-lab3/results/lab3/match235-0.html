def calculate_shipping(weight:float) -> float:
    if ( weight < 2.00):
        return 2.00
    elif (weight < 10.00):
        return 5.00
    else:
        return 5.00 + (1.50 * (weight - 10))



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_face(x1: int, y1: int):
    my_canvas.create_oval(x1+5, y1+5, x1+500, y1+500, fill='white')
    draw_eye(x1, y1)
    draw_eye(x1+250, y1)
    draw_nose(x1, y1)
    draw_mouth(x1, y1)    
    return


#draw_eye
def draw_eye(x1, y1):
    my_canvas.create_oval(50+x1, 130+y1, 220+x1, 205+y1, fill='white')
    my_canvas.create_oval(100+x1, 130+y1, 170+x1, 205+y1, fill='lightblue')
    my_canvas.create_oval(125+x1, 150+y1, 150+x1, 175+y1, fill='black')
    return

def draw_nose(x1, y1):
    my_canvas.create_polygon(x1+250, y1+250, x1+300, y1+300, x1+200, y1+300, fill='brown')
    return

def draw_mouth(x1, y1):
    my_canvas.create_oval(150+x1, 350+y1, 350+x1, 450+y1, fill='red')
    return
