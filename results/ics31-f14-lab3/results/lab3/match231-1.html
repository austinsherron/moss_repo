def calculate_shipping(x:int)->int:
    """This function takes the weight of a package to be shipped, returns the price for shipping"""
    if x < 2:
        return 2.00
    elif x >= 2 and x < 10:
        return 5.00
    elif x >= 10:
        return 5 + (x-10)*1.5

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(x:Book)->int:
