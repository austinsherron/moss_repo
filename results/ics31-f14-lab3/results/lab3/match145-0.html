def calculate_shipping(weight: float) -> float:
    '''caculate the shipping fees given the weight of the shipment'''
    if(weight < 2):
        result = 2.00
    elif (weight == 2):
        result = 5.00
    elif (weight < 10):
        result = 5.00
    elif(weight >= 10):
        result = 5.00 + (weight - 10) * 1.5

    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye(n: list):
    my_canvas.create_oval(n[0]+100,n[1]+60,n[0]+300,n[1]+160) # most outer oval
    my_canvas.create_oval(n[0]+160,n[1]+60,n[0]+240,n[1]+160,fill='blue') # outer oval
    my_canvas.create_oval(n[0]+180,n[1]+90,n[0]+220,n[1]+130,fill='black') # inner oval
def draw_nose(n:list):
    my_canvas.create_oval(n[0]+300,n[1]+200,n[0]+400,n[1]+300,fill='red')
def draw_mouth(n:list):
    my_canvas.create_oval(n[0]+100,n[1]+400,n[0]+600,n[1]+500,fill='red')

def draw_face(n: list):
    my_canvas.create_oval(n[0],n[1],n[0]+700,n[1]+600,fill='white')
    draw_eye([n[0],n[1]])
    draw_eye([n[0]+300,n[1]])
    draw_nose([n[0],n[1]])
    draw_mouth([n[0],n[1]])

''' Draws a face by calling the draw_face function wich calls 3 other functions '''
