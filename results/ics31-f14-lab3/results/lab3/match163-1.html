#calculate_shipping: Takes the weight and returns a price
    #Example: 1.5 --> 2.00
def calculate_shipping(weight):
    if weight < 2:
        return 2.00
    elif 2 <= weight < 10:
        return 5.00
    elif weight >= 10:
        return 5.00 + 1.50*(weight-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book):
    for x in range(len(BSI)):

            return BSI[x].price*BSI[x].instock

def top_value(bookList):
    highestvalue = 0
    highestbook = 0
    for x in range(len(bookList)):
        if inventory_value(BSI[x].title) > highestvalue:
