def calculate_shipping (weight: float):
    if weight < 2.00:
        return 2.00
    if 2.00 <= weight < 10.00:
        return 5.00
    if weight >= 10.00:
        return 5.00 + 1.50 *(weight - 10.00)


assert calculate_shipping(1.9) == 2.00
assert calculate_shipping(5) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(10) == 5.00

def inventory_value(Book: object):
    value = Book.instock * Book.price
    return value
print(inventory_value(BSI[0])) #checks the value of the first book in BSI

def top_value(Booklist: list):
    topvalue = []
    for i in Booklist:
        topvalue.append(inventory_value(i))
    topvalue.sort()
    for i in Booklist:
        if i.price * i.instock == topvalue[-1]:
            return i
print('The highest-value book is,', top_value(BSI).title,'by', top_value(BSI).author, 'at a value of,', (top_value(BSI).instock * top_value(BSI).price))

def draw_eye(x: int,y: int):  #where x,y is the center of the eye
    my_canvas.create_oval(x-100, y-45, x+100, y+45) 
    my_canvas.create_oval(x-45, y-45, x+45, y+45, fill ='black')
    my_canvas.create_oval(x-27.5, y-27.5, x+27.5, y+27.5, fill ='white') 
def draw_nose(x: int,y: int): #where x,y is the starting coordinates of the tip of the nose
    my_canvas.create_line(x,y,x+25,y+50)
    my_canvas.create_line(x,y,x-25,y+50)
    my_canvas.create_line(x+25,y+50,x-25,y+50)

def draw_mouth(x: int,y:int): #where x,y is the top left coordinates
    my_canvas.create_rectangle(x,y,x+200,y+100, fill ="white")

def draw_face(x: int, y:int): #where x,y is the center of the face
    my_canvas.create_oval(x-300,y-300,x+300,y+300)
 
    draw_eye(x-150,y-150)
    draw_eye(x+150,y-150)
    draw_nose(x,y)
    draw_mouth(x-100,y+150)
