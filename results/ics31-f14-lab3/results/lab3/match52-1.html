def calculate_shipping (package:float) -> float:
    ''' Returns cost of package in different circumstances'''
    if package < 2:
        result = 2.00 
    elif package < 10:
        result = 5.00 
    else:
        result = 5.00 + 1.5*(package - 10)
    return result

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print('The cost to ship a 1.7 pound package is', calculate_shipping(1.7), 'dollars')
print ('The cost to ship a 8 pound package is', calculate_shipping(8), 'dollars')
print('The cost to ship a 13 pound package is',calculate_shipping(13), 'dollars')

def inventory_value(b:object)->int:
    '''Returns value of inventory of book'''
    return b.price * b.instock
def top_value(l:list)->object:
    '''Returns Book object that has highest value inventory'''
    largest_value = 0
    count = 0
    c = -1
    for o in l:
        if inventory_value(o)>largest_value:
            largest_value = inventory_value(o)
            count = c + 1
        c = c + 1
    return l[count]

import tkinter
my_window = tkinter.Tk()


my_canvas = tkinter.Canvas(my_window, width=800, height=800)
my_canvas.pack()
def draw_eye (x:int, y:int) -> int:
    '''Returns coordinates for eye'''
    my_canvas.create_oval(x, y, x-140, y-90, outline ='black', fill = 'red') #Eyeball
    my_canvas.create_oval(x-20,y-20, x-80, y-80, fill = 'deep sky blue') #Iris
    my_canvas.create_oval(x-33, y-34, x-62, y-65, fill = 'black') #Pupil
    return
def draw_nose(x: int, y: int) ->int:
    '''Returns coordinates for nose''' 


    my_canvas.create_polygon(x, y, x - 60, y-60, x - 120, y - 10, fill = 'medium spring green') #nose
    return
def draw_mouth(x: int, y: int) -> int:
    '''Returns coordinates for mouth'''
    my_canvas.create_arc(x, y, x+ 100, y + 100, fill = 'firebrick', start = 300)
    return


def draw_face(x:int, y:int) -> int:
    '''Returns coordinates for face''' 
    my_canvas.create_oval(x, y, x + 600, y+ 600, fill =  'lemon chiffon')
    draw_eye(x+240, y+240)
    draw_eye(x+490, y+240)
    draw_nose(x+365, y+290)
    draw_mouth(x+215, y+350)
    return
draw_face(10, 10)

tkinter.mainloop()
