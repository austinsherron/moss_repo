def abbreviate(s : str) -> str:
    ''' Returns the first 3 letters of a string'''
    return s[:3]

assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'

def find_area_square(x : int) -> int:
    '''Returns the area of a square with the sides of x length'''
    return x ** 2

assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(x : int) -> int:
    '''Returns the area of a circle with x diameter'''
    return x ** 2 * 3.14159

assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def print_even_numbers(X : list):

def calculate_shipping(i : int) -> float:
    '''Returns shipping cost based on weight of package'''
    if i < 2:
        return 2.00
    elif i >= 2 and i < 10:
        return 5.00
    else:
        return 5.00 + ( i - 10) * 1.50

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

    '''Returns prices of a restaurant namedtuple'''
    return r.price

def costliest(RL : list) -> str:
    '''Returns most expensive restaurant'''
    RL.sort(key=restaurant_price)
    return RL[-1].name

    return sorted(RL, key=restaurant_price, reverse=True)[0].name

def inventory_value(b : Book) -> float:
    return b.price * b.instock

def top_value(B : list) -> Book:
    biggest = 0
    x = Book('', '', '', 0, 0.0, 0)
    for i in B:
        if biggest < inventory_value(i):
            biggest = inventory_value(i)
            x = i
    return x

          top_value(BSI).author,'at a value of $',
          inventory_value(top_value(BSI)))

def draw_eyes(x : int, y : int, length : int, my_canvas):
    my_canvas.create_oval(x, y, x + length, y + length / 2, fill='white')
    my_canvas.create_oval(x + length / 4, y,
                          x + 3 * length / 4, y + length / 2, fill='brown')
    my_canvas.create_oval(x + 3 * length / 8 , y + length / 8,
                          x + 5 * length / 8, y + 3 * length / 8, fill='black')

def draw_nose(x : int, y : int, diameter : int, my_canvas):
    my_canvas.create_oval(x, y, x + diameter, y + diameter)
    
def draw_mouth(x : int, y : int, length : int, my_canvas):
    my_canvas.create_oval(x, y, x + length * 2, y + length / 3)
    my_canvas.create_line(x, y + length / 6, x + length * 2, y + length / 6)



def draw_face(x : int, y : int, nose : int, eye : int, mouth : int):
    my_window = tkinter.Tk()    # Create the graphics window

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()
    
    draw_eyes(x, y, eye, my_canvas)
    draw_eyes(x + 2 * eye, y, eye, my_canvas)
    draw_nose(x + eye * 1.4, y + eye, nose, my_canvas)
    draw_mouth(x + eye / 2, y + eye * 2, mouth, my_canvas)
    
draw_face(50, 50, 20, 50, 50)
draw_face(100, 50, 40, 120, 90)

tkinter.mainloop()
