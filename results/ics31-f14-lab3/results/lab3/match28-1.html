def calculate_shipping (n:float) ->float:
    ''' returns the price for shipping the package based on weight'''
    if n<2.0:
        return 2.00
    else:
        if n<10.0:
            return 5.00
        else:
            weight=n-10
            stacks=weight*1.5
            return stacks+5.0

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
    
print (calculate_shipping(1.5))
