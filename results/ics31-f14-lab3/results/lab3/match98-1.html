def calculate_shipping(weight: float)-> float:
    """ return the price for shipping packages wiht different weight"""
    if weight < 2:
        return 2.00
    if 10 > weight >= 2:
        return 5.00
    if 10 <= weight:
        return ((weight-10.00)*1.50) + 5.00

assert calculate_shipping(1.5)==2.00
assert calculate_shipping(7)==5.00
assert calculate_shipping(15)==12.50

assert calculate_shipping(2)==5.00
assert calculate_shipping(10)==5.00

def create_square(x: int, y: int, length: int):


    """ Creates a tkinter square"""
    my_window=tkinter.Tk()
    my_canvas=tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()
    my_canvas.create_rectangle(x,y,x+length,y+length)
    tkinter.mainloop()

def inventory_value(inbook: Book) -> float:
    """ return the value of the inventory """
    return inbook.price*inbook.instock

def top_value(booklist: list) -> Book:
    """ return the Book object that has the highest value inventory"""
    sortedbooks=sorted(booklist,key=inventory_value)
    return sortedbooks[-1]

def create_circle_incanvas(x: int, y: int, diameter: int, color=None):
    """ Creates a tkinter circle - expect existing canvas, named my_canvas"""
    my_canvas.create_oval(x,y,x+diameter,y+diameter,fill=color)
    


def draw_eye(x: int, y: int, length: int):
    """ draw a eye in a existing tkinter canvas"""
    create_circle_incanvas(x,y,length, color="white")
    create_circle_incanvas(x+(length/5), y+(length*(2/5)), (length*(3/5)), color="blue")
    create_circle_incanvas(x+(length*(2/5)), y+(length/2), (length/5), color="black")

def draw_nose(x: int, y: int, length: int):
    """ draw a nose in a existing tkinter canvas"""

    my_canvas.create_line(x,y,x,y+length)
    
def draw_mouth(x:int, y: int, length: int):
    """ draw a mouth in a existing tkinter canvas"""
    my_canvas.create_line(x,y,x+length,y)

def draw_face(x:int, y: int, diameter: int):
    """ draw a face in a existing tkinter canvas"""
    create_circle_incanvas(x,y,diameter)
    draw_eye(x+diameter*(1/6),y+diameter*(1/3),(diameter/6))
    draw_nose(x+(diameter*(1/2)),y+diameter*(1/2),diameter/4)
    draw_eye(x+(diameter*(2/3)),y+diameter*(1/3),diameter/6)
    draw_mouth(x+diameter*(1/3),y+diameter*(11/12),diameter/3)
    
my_window=tkinter.Tk()
my_canvas=tkinter.Canvas(my_window,width=500, height=500)
my_canvas.pack()
draw_face(5,5,400)
tkinter.mainloop()
