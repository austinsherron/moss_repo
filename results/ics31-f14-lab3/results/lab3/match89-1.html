def calculate_shipping(weight:float)->float:
    '''obtain weight and calculate shipping, return rate'''
    if weight < 2:
        rate = 2.0
    elif weight >= 2 and weight <10:
            rate = 5.0
    else:
            rate = 5 + 1.5 * (weight - 10)


    return rate

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
new_weight = 2.01
print('The shipping cost of your package is:')
print('$',calculate_shipping(new_weight))
