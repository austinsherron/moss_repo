def calculate_shipping(weight) -> float:

    '''Calculate and return the rate for different weight'''

    if weight < 2:

        rate = 2

        return rate

    elif weight >= 2 and weight < 10:

        rate = 5

        return rate

    elif weight >= 10:

        rate = 5 + ((weight - 10) * 1.5)

        return rate



assert calculate_shipping(1.3) == 2.00

assert calculate_shipping(6) == 5.00

assert calculate_shipping(15) == 12.50

results_weight = calculate_shipping(float(input("How much does your package weigh?")))

print(results_weight)

def inventory_value(Book: str) -> float:

        '''Calculates inventory value'''

        return Book.instock*Book.price

print('First book inventory value is: ',inventory_value(BSI[0]))

def top_value (Book: str) -> str:

        '''Returns the highest value book '''

        Best_Price = inventory_value(Book[0])

    

        for n in Book:

            if Best_Price<inventory_value(n):

                Best_Price= inventory_value(n)

                Best_Selling = n

        return Best_Selling
