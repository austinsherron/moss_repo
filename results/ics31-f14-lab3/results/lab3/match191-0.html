def calculate_shipping (weight: float) -> float:
    ' Returns shipping price depending on weight ' 
    if weight < 2 :
        result = float (2.00)
    elif weight < 10 :
        result = float (5.00)
    elif weight >= 10 :
        result = float (5.00 + 1.50 * (weight - 10))
    return result    
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print (calculate_shipping (1111))

def draw_face (x_face: float, y_face: float):
   my_window = tkinter.Tk()    
   my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
   my_canvas.pack()
   my_canvas.create_oval (0 + x_face, 0 + y_face, 300 + x_face, 350 + y_face, fill = 'bisque')
   draw_eye (80 + x_face, 100 + y_face, 50, my_canvas)
   draw_eye (160 + x_face, 100 + y_face, 50, my_canvas)
   draw_mouth (100 + x_face, 200 + y_face, 180 + x_face, 200 + y_face, my_canvas)
   draw_nose (140 + x_face, 150 + y_face, 155 + x_face, 170 + y_face, my_canvas)

def draw_eye (x_eye: float, y_eye: float, length_eye: float, canvas):
   canvas.create_oval (x_eye, y_eye, x_eye + length_eye, y_eye + length_eye, fill = 'white')
   canvas.create_oval (x_eye + 18, y_eye + 18, x_eye + length_eye - 18, y_eye + length_eye - 18, fill = 'blue')
   canvas.create_oval (x_eye + 23, y_eye + 23, x_eye + length_eye - 23, y_eye + length_eye - 23, fill = 'black')

def draw_mouth (x1_mouth: float, y1_mouth: float, x2_mouth, y2_mouth, canvas):
   canvas.create_line (x1_mouth, y1_mouth, x2_mouth, y2_mouth, smooth = 'true')

def draw_nose (x1_nose: float, y1_nose: float, x2_nose: float, y2_nose: float, canvas):
   canvas.create_line (x1_nose, y1_nose, x2_nose, y2_nose)
   canvas.create_oval (x2_nose - 15, y2_nose, x2_nose - 12, y2_nose) 
   canvas.create_line (x1_nose + 5, y2_nose, x2_nose, y2_nose)
