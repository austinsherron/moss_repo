def calculate_shipping(weight: float) -> float:
    if weight < 2:
        print(2.00)
        return 2.00
    elif weight < 10:
        print(5.00)
        return 5.00
    else:
        print(5.00 + ((weight - 10) * 1.5))
        return 5.00 + ((weight - 10) * 1.5)

def inventory_value(book :Book) -> float :
    return book.price * book.instock

def top_value(booklist : list) -> Book :
    highest = booklist[0]
    for book in booklist:
        if inventory_value(book) > inventory_value(highest) :
            highest = book
    return highest
