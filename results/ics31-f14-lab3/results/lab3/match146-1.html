#calculate_shipping: weight_of_package: float -> float
def calculate_shipping (weight_of_package: float) -> float:
    """prints out the calulated price, which can fall into 2 static prices or a third variable price depending on the weight"""
    if weight_of_package < 2:
        return 2
    elif weight_of_package >= 2 and weight_of_package < 10:
            return 5
    elif weight_of_package >= 10:
        return (5 + ((weight_of_package-10)*1.5))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00


assert calculate_shipping(10.5) == 5.75
assert calculate_shipping(11) == 6.5
assert calculate_shipping(15) == 12.50

#draw_eye: eye_x: int eye_y: int, eye_color: str, canvas: str
def draw_eye (eye_x: int, eye_y: int, eye_color: str, canvas: str):
    """draws an eye with the given color for the iris"""
    canvas.create_oval(eye_x, eye_y, eye_x+200, eye_y-100, fill="white")
    canvas.create_oval(eye_x+55, eye_y-5, eye_x+145, eye_y-95, fill=eye_color)
    canvas.create_oval(eye_x+75, eye_y-25, eye_x+125, eye_y-75, fill="black")
    canvas.create_oval(eye_x+100, eye_y-75, eye_x+125, eye_y-50, fill="white")

#draw_mouth: mouth_x: int, mouth_y: int, mouth_color : str, canvas: str
def draw_mouth (mouth_x: int, mouth_y: int, mouth_color : str, canvas: str):
    """draws a mouth with the given color lipstick"""
    canvas.create_oval(mouth_x, mouth_y, mouth_x+200, mouth_y-50, fill=mouth_color)
    canvas.create_line(mouth_x, mouth_y-25, mouth_x+200, mouth_y-25, fill="black")
    
#draw_nose: nose_x: int, nose_y: int, canvas: str
def draw_nose (nose_x: int, nose_y: int, canvas: str):
    """draws a nose"""
    canvas.create_line(nose_x, nose_y, nose_x+50, nose_y-100, fill="black")
    canvas.create_line(nose_x+50, nose_y-100, nose_x+100, nose_y, fill="black")


#draw_face: face_x: int, face_y: int, mouth_color: str, eye_color: str, skin_color: str
def draw_face (face_x: int, face_y: int, mouth_color: str, eye_color: str, skin_color: str):
    "draws a face with the chosen skin color, with two eyes with the color of iris given, a nose, and a mouth of the given lipstick color"""
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=900, height=900)
    my_canvas.pack()
    my_canvas.create_oval(face_x-75, face_y-175, face_x+550, face_y+500, fill=skin_color)
    draw_eye (face_x, face_y+60, eye_color, my_canvas)
    draw_eye (face_x+265, face_y+60, eye_color, my_canvas)
    draw_nose (face_x+180, face_y+200, my_canvas)
    draw_mouth(face_x+130, face_y+300, mouth_color, my_canvas)

draw_face (200, 200, "red", "green", "#FFCCCC")
