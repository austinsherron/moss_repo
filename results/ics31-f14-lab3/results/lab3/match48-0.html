def calculate_shipping(weight: float) -> float:
    if weight < 2:
        rate = 2.00
    elif weight >= 2 and weight < 10:
        rate = 5.00
    else:
        rate = 5.00 + (1.50 * (weight - 10))
    return(rate)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye(x_face_one: int, y_face_one: int, x_face_two: int, y_face_two: int) -> None:
    my_canvas.create_oval((x_face_one + 50), (y_face_one + 75), (x_face_two - 150), (y_face_two - 145), fill='white')
    my_canvas.create_oval((x_face_one + 60), (y_face_one + 75), (x_face_two - 160), (y_face_two - 145), fill='blue')
    my_canvas.create_oval((x_face_one + 70), (y_face_one + 85), (x_face_two - 170), (y_face_two - 155), fill='black')

def draw_nose(x_face_one: int, y_face_one: int, x_face_two: int, y_face_two: int) -> None:
    my_canvas.create_line((x_face_one + 125), (y_face_one + 100), (x_face_two - 105), (y_face_two - 100)) 
    my_canvas.create_line((x_face_two - 105), (y_face_two - 100), (x_face_one + 125), (y_face_two - 100))

def draw_mouth(x_face_one: int, y_face_one: int, x_face_two: int, y_face_two: int) -> None:
    my_canvas.create_line((x_face_one + 75), (y_face_one + 175), (x_face_one + 75), (y_face_two - 50))
    my_canvas.create_line((x_face_one + 75), (y_face_two - 50), (x_face_two - 75), (y_face_two - 50))
    my_canvas.create_line((x_face_two - 75), (y_face_two - 50), (x_face_two - 75), (y_face_one + 175))

def draw_face(x_face_one: int, y_face_one: int, x_face_two: int, y_face_two: int):
    my_canvas.create_oval(x_face_one, y_face_one, x_face_two, y_face_two)
    draw_eye(x_face_one, y_face_one, x_face_two, y_face_two)
    draw_eye((x_face_one + 100), y_face_one, (x_face_two + 100), y_face_two)
    draw_nose(x_face_one, y_face_one, x_face_two, y_face_two)
    draw_mouth(x_face_one, y_face_one, x_face_two, y_face_two)

#Print face one

#Print face two
