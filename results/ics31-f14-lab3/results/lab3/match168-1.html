def calculate_shipping (x:float) -> float:
    '''calculates shipping price'''
    if (x < 2):
        return 2
    elif (x>2 and x<10):
        return 5
    elif (x > 10):
        return ((1.5 * (x-10)) + 5)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter

        
def draw_eyes (x: int, y:int):
    """draws eyes"""
    my_canvas.create_oval(x+50,y+30,x+130, y+80, fill='white') #eye1
    my_canvas.create_oval(x+160, y+30, x+240, y+80, fill='white')#eye2
    my_canvas.create_oval(x+70, y+30, x+110, y+80, fill= 'black')#pupil1
    my_canvas.create_oval(x+180, y+30, x+220, y+80, fill='black')#pupil2


def draw_nose (x:int, y:int):
    """draws nose"""
    my_canvas.create_polygon(x+135,y+100,x+85,y+150,x+185,y+150, fill='black')

def draw_mouth (x:int, y:int):
    """draws mouth"""
    my_canvas.create_oval(x+135,y+180,x+185, y+230, fill='white')

def draw_face (x:int, y:int,):
        """draws face"""
        my_canvas.create_oval(x,y,x+300, y+300, fill='brown')
        draw_eyes(x,y)
        draw_nose(x,y)
        draw_mouth(x,y)
        tkinter.mainloop()
