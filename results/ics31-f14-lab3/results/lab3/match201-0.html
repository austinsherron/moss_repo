def calculate_shipping(f:float)->float:
    if f < 2.00:
        price=2.00
    elif 1.99 < f < 10:
        price =5.00
    elif f >9.99:
        price=5.00+((f-10)*1.5)
    return price
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter              
my_window = tkinter.Tk()

my_canvas.pack()        

def draw_eye():
    my_canvas.create_oval(30,50,70,70, fill='white')
    my_canvas.create_oval(40,50,60,70, fill='light blue')
    my_canvas.create_oval(90,50,130,70, fill='white')
    my_canvas.create_oval(100,50,120,70,fill='light blue')
    return
def draw_nose():
    my_canvas.create_polygon(80,75,95,85,65,85, fill='pink')
    return
def draw_mouth():
    my_canvas.create_arc(30,95,100,130, fill="black")
def draw_face():
    my_canvas.create_oval(20,1,150,140, fill='tan')
    draw_nose()
    draw_mouth()
    draw_eye()
    
draw_face()
