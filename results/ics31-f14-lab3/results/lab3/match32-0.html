# Define funcition 'calculate_shipping'
def calculate_shipping(weight: float) -> float:
    'Takes the weight of an object as input and calcualtes and returns the price of shipping for that object.'
    rate = '$'

    if(weight < 2):
        rate += str(2.00)
    elif(weight >= 10):
        rate += str(5.00 + ((weight - 10) * 1.50))
    else:
        rate += str(5.00)

    return rate

# Print return from function 'calculate_shipping'
print('The shipping price for an object of weight 1.5 is', calculate_shipping(1.5))
print('The shipping price for an object of weight 7 is', calculate_shipping(7))
print('The shipping price for an object of weight 15 is', calculate_shipping(15))

# Test function 'calculate_shipping'
assert calculate_shipping(1.5) == '$2.0'
assert calculate_shipping(7) == '$5.0'
assert calculate_shipping(15) == '$12.5'

    my_canvas.create_oval(x, y, x + 80, y + 30)
    my_canvas.create_oval(x + 25, y, x + 55, y + 30, fill='blue')
    my_canvas.create_oval(x + 30, y + 5, x + 50, y + 25, fill='black')

# Function to draw a nose
