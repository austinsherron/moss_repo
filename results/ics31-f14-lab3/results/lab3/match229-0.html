def calculate_shipping(weight: float) -> float:
    " Return price of shipping "
    if( weight < 2 ):
        cost = 2.00
    elif( 2 <= weight < 10 ):
        cost = 5.00
    else:
        cost = 5.00 + (1.50*(weight-10))
    return cost
assert calculate_shipping(1) == 2.00
assert calculate_shipping(5) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(11) == 6.50
assert calculate_shipping(2) == 5.00
