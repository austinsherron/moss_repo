def abbreviate(string: str) -> str:
    '''Returns the first three characters of given string'''
    if len(string) <= 3:
        return string
    else:
        return string[0] + string[1] + string[2]

assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'
assert abbreviate('Saideep') == 'Sai'
assert abbreviate('Henry') == 'Hen'

      abbreviate(input("\n\nEnter the month (or any string) : ")))

def calculate_shipping(weight: float) -> float:
    '''Returns the shipping cost of a package of given weight'''


    if weight < 2:
        return 2.0
    elif weight < 10:
        return 5.0
    else:
        return 5.0 + ((weight - 10) * 1.5)

assert calculate_shipping(2) == 5
assert calculate_shipping(10) == 5
assert calculate_shipping(11) == 6.5
assert calculate_shipping(1) == 2

print("The cost of shipping the package is : $",
      calculate_shipping(float(input("\n\nEnter the weight of the package : "))))

def inventory_value(book: Book) -> float:
    '''Returns the value of inventory of a given book'''
    return book.price * book.instock

def top_value(BSI_new: 'list of Books') -> Book:
    '''Returns the Book object with the highest inventory value in a
    given list of Books'''
    BSi_new = sorted(BSI_new, key = inventory_value)
    return BSI_new[-1]

highest = top_value(BSI)   #variable to hold what is returned by top_value
print("\n\nThe highest-value book is", highest.title, "by", highest.author,
      "at a value of $", inventory_value(highest))

import tkinter

def draw_mouth(my_canvas: 'canvas object'):
    '''Draw a mouth'''
    print("\n1. Happy Face\n2. Sad Face\n")
    choice = int(input("Enter your choice : "))
    if choice == 1:
        my_canvas.create_line(200, 320, 220, 340, 250, 345, 280, 340, 300, 320)
    else:
        my_canvas.create_line(200, 330, 300, 330)
    return


def draw_eye(my_canvas: 'canvas object', x: float, y: float, color: str):
    '''Draw an eye'''
    my_canvas.create_oval(x, y, x+70, y+70, fill = 'white')
    my_canvas.create_oval(x+10, y+20, x+60, y+70, fill = color)
    my_canvas.create_oval(x+25, y+30, x+45, y+70, fill = 'black')
    return
    
def draw_nose(my_canvas: 'canvas object'):
    '''Draw a nose'''
    color = input("\nWhat color to fill in nose ? ")
    my_canvas.create_polygon(210, 280, 290, 280, 250, 230, fill = color)
    return

    
def draw_face():
    '''Draws the face by calling functions to draw eyes, nose and mouth'''
    my_window = tkinter.Tk()
    
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    color = input("What color to fill in face ? ")
    my_canvas.create_oval(100, 100, 400, 400, fill = color)

    draw_nose(my_canvas)
    color = input("\nWhat color of eye : ")
    y = 160
    draw_eye(my_canvas, 160, y, color)
    draw_eye(my_canvas, 270, y, color)
    draw_mouth(my_canvas)

    tkinter.mainloop()
    return

    
draw_face()
