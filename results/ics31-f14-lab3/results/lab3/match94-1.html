def calculate_shipping (n: float) -> float:
    "Return price for shipping the package"
    if n < 2.00:
        return 2.00
    elif n < 10:
        return 5.00
    else:
        return 5.00 + 1.50*(n-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(1))
print(calculate_shipping(3))
print(calculate_shipping(20))

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()
def draw_nose (x: int, y: int, d: int):
   "Return a nose"
   my_canvas.create_oval(x,y,x+d,y+d)

def draw_mouth (x: int, y: int,x1: int,y1: int):
   "Return a mouth"
   my_canvas.create_oval(x,y,x1,y1)

def draw_eye (x: int, y: int, d:int):
   "Return an eye"
   my_canvas.create_oval(x,y,x+d,y+d, fill='ivory')
   my_canvas.create_oval(x+10,y+10,x+10+(d/2),y+10+(d/2), fill='cadet blue')
   my_canvas.create_oval(x+15,y+15,x+15+(d/4),y+15+(d/4), fill='black')

def draw_face (x: int, y: int, d: int):
   "Return a face"
   my_canvas.create_oval(x,y,x+d,y+d)
   draw_eye(90,110,60)
   draw_eye(240,110,60)
   draw_mouth(100,250,300,300)
   draw_nose(175,175,50)
draw_face(50,50,300)
tkinter.mainloop()
