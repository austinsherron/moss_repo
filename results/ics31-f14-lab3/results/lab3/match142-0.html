def calculating_shipping (weight: float):
    if weight < 2 :
        return 2.00
    elif weight < 10:
        return 5.00
    elif weight >= 10:
        return 5.00+ (weight-10)*1.5
assert calculating_shipping(1.5) == 2.00


assert calculating_shipping(7) == 5.00
assert calculating_shipping(15) == 12.50
