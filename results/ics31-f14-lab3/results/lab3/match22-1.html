def find_area_circle(radius: int) -> float:
    radius = 3.14159 * radius * radius
    return radius
assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def print_even_numbers(nums: list)-> str:


    if len(nums) == 1:
        if nums[0]%2==0:
            print(nums[0])
    else:
        for num in nums:
            if num%2 == 0:
                print(num)

print_even_numbers([3])
print_even_numbers([10])

def calculate_shipping(weight: int)-> float:


    float(weight)
    if weight < 2.00:
        return 2.00
    elif weight >= 2.00 and weight < 10.00:
        return 5.00
    else:
        return 5.00 + (weight - 10) * 1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter
def create_square(x: int, y: int, length: int):


    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width = x*2 + length, height = y*2 + length)
    my_canvas.pack()
    my_canvas.create_rectangle(x,y,x+length,y+length)
    my_canvas.mainloop()
