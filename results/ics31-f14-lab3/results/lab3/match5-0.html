def abbreviate(string: str) -> str:
    "Returns the first three letters of the string."
    return string[0:3]

assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'

def find_area_square(length: int) -> int:
    "Returns the area of the square."
    return length**2

assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(rad: int) -> float:
    "Returns the area of the circle of radius rad."
    return 3.14159*(rad**2)

assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def calculate_shipping(f: float)-> float:
    "Returns the price of shipping the package."
    if(f<2):
        return 2.00
    if(f>=10):
        return 5.00+1.5*(f-10)
    else: return 5.00

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5.00

    "finds total number of books in stock"
    x=0

def inventory_value(book: Book)->float:
    'returns the inventory value of the book'
    return book.price* book.instock

    'returns the book with the highest inventory value'

def draw_eye(x:int,y:int, my_canvas:'canvas', color:'color')->None:
    'draws an eye with upper-left corner (x,y)'
    my_canvas.create_oval(x, y, x+100, y+100, fill='white')
    my_canvas.create_oval(x+30, y+30, x+70, y+70, fill=color)
    my_canvas.create_oval(x+40, y+40, x+60, y+60, fill='black')
def draw_nose(x:int,y:int, my_canvas:'canvas', color:'color')->None:
    'draws an nose with upper-left corner (x,y)'
    my_canvas.create_line(x, y, x+50, y+50, fill=color)
    my_canvas.create_line(x, y+50, x+50, y+50, fill=color)
def draw_mouth(x:int,y:int, my_canvas:'canvas', color:'color')->None:
    'draws an mouth with upper-left corner (x,y)'
    my_canvas.create_line(x, y, x+15, y+15, fill=color)
    my_canvas.create_line(x+15, y+15, x+80, y+15, fill=color)
    my_canvas.create_line(x+80, y+15, x+95, y, fill=color)
def draw_face(x:int,y:int,eye_color:'',nose_color:'', mouth_color:'')->None:
    'draws an complete face with upper-left corner (x,y)'
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()
    my_canvas.create_oval(x, y, x+300, y+300)
    draw_eye(x+40, y+30, my_canvas, eye_color)
    draw_eye(x+150, y+30, my_canvas, eye_color)
    draw_nose(x+140, y+130, my_canvas, nose_color)
    draw_mouth(x+110, y+230, my_canvas, mouth_color)
    tkinter.mainloop()
draw_face(100,100,'blue', 'orange','red')
draw_face(150,110,'brown','green','purple')
