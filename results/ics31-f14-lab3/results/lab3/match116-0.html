def calculate_shipping(weight: float) -> float:
    if weight < 2:
         return 2.00
    elif 2 <= weight < 10:
         return 5.00
    elif weight >= 10:
         return 5.00 + ((weight - 10) * 1.50)

import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

def draw_eye(x, y, w, l:float) -> None:
    #Draws two pupils and irises
    my_canvas.create_oval(x, y, x+w, y+l, fill='white')
    my_canvas.create_oval(x+w*2, y, x+w*3, y+l, fill='white')
    my_canvas.create_oval(x+w*.32, y+l*.32, x+w/1.5, y+l/1.5, fill ='black')
    my_canvas.create_oval(x+w*2.32, y+l*.32, x+w*2.68, y+l/1.5, fill='black')
def draw_nose(x, y, w, l:float) -> None:
    #Draws a red clown nose
    my_canvas.create_oval(x+w, y+l, x+w*2, y+l*2, fill='red')
def draw_mouth(x, y, w, l:float) -> None:
    #Draws a gaping mouth
    my_canvas.create_oval(x, y+l*3.5, x+w*3, y+l*2.2, fill='black')
def draw_face(x, y, w, l:float) -> None:
    #Draws the entire face
    my_canvas.create_oval(x-w, y-l, x+w*4, y+l*4, fill='tan')
    draw_eye(x, y, w, l)
    draw_nose(x, y, w, l)
    draw_mouth(x, y, w, l)
        
draw_face(100,100,100,100)
draw_face(40,45,30,15)
draw_face(412,171,22,74)
