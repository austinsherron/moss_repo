def calculate_shipping(w:float)-> float:        #Functions 5
    '''calculates the shipping price based on given weight'''
    
    priceU2 = 2.00
    priceU10 = 5.00
    price_plus = 1.50
    
    if (w < 2):
        final_cost = priceU2
        print('Your shipping cost is $', final_cost)

    elif (2 <= w < 10):
        final_cost = priceU10
        print('Your shipping cost is $', final_cost)

    elif (w >=10):
        overW = (w - 10)
        final_cost = ((overW*price_plus) + priceU10)
        print('Your shipping cost is $', final_cost)

    return final_cost
