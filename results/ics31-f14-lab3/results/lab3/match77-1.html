def calculate_shipping(weight: int) ->int:
    if weight < 2:
        return  2
    elif 2 < weight < 10:
        return  5
    elif weight > 10:
        return 5 + (weight - 10) * 1.5



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter
def draw_nose(x,y,l):
    my_canvas.create_oval(x-l,y-l,x+l,y+l,fill='orange')
def draw_mouth(w,a,s,d):
    my_canvas.create_oval(w,a,s,d,fill='yellow')
def draw_eye(c,d,e):
    my_canvas.create_oval(c-e, d-(2*e), c+e, d+(2*e), fill='white')
    my_canvas.create_oval(c-e/2, d-e/2, c+e/2, d+e/2, fill='black')
    my_canvas.create_oval(c-e/4, d-e/4, c+e/4, d+e/4, fill='blue')

my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
my_canvas.pack()
