def calculate_shipping (weight:float) -> float:
    "Calculate rate based on weight"
    if weight < 2:
        return 2.00
    elif 2 < weight < 10:
        return 5.00
    else:
        return 5 + (1.50 * (weight - 10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye (x:int, y:int) -> int:
    my_canvas.create_oval(x, y + 20, x + 100, y + 70, fill='white')    
    my_canvas.create_oval(x + 25, y + 20, x + 75, y + 70, fill='blue')          


    my_canvas.create_oval(x + 35, y + 30, x + 65, y + 60, fill='black')      

def draw_nose (x:int, y:int) -> int:
    my_canvas.create_line(x, y, x + 35, y + 35)
    my_canvas.create_line(x + 35, y + 35, x, y + 35)

def draw_mouth(x:int, y:int) -> int:
    my_canvas.create_arc (x + 25, y, x + 250, y - 150, start=180,
                          extent=180, fill='red')
    
def draw_face (a, b, c, d, e, f, g, h, v, x, y, z) -> int:
    my_canvas.create_oval(a, b, c, d, fill='wheat')       #Face Outline
    draw_eye(e, f)                                        #Left Eye
    draw_eye(g, h)                                        #Right Eye
    draw_nose(v, x)
    draw_mouth(y,z)
