def calculate_shipping(weight: float) -> float:
    '''Return the price of the weighted shipping package'''
    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5 + (weight - 10) * 1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(Book: Book) -> float:
    '''Return the value of the inventory of a book'''
    return Book.price * Book.instock

assert inventory_value(BSI[1] )== 691.35

def top_value (list_of_books: list) -> str:
    '''Return the highest valued inventory from a list of books'''
    inventoryvalue=0
    top_valued_book= None
    for Book in list_of_books:
        if inventory_value(Book)>inventoryvalue:
            
            inventoryvalue = inventory_value(Book)
            top_valued_book = Book
    return top_valued_book
assert top_value(BSI).title == "All About Apple"

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1250, height=650)
my_canvas.pack()


def draw_eye(x, y):
    my_canvas.create_oval(x + 400, y + 100, x + 100, y + 200, fill = 'white')
    my_canvas.create_oval(x + 300, y + 100, x + 200, y + 200, fill = 'yellow')
    my_canvas.create_oval(x + 275, y + 125, x + 225, y + 175, fill = 'black')

def draw_nose(x, y):
    my_canvas.create_line(x + 100, y, x, y + 150)
    my_canvas.create_line(x, y + 150, x + 100, y + 200)
    
def draw_mouth(x, y):
    my_canvas.create_oval(x + 100, y, x, y + 100, fill = 'red')
    
def draw_face(x, y):
    my_canvas.create_oval(x + 1200, y, x, y + 600, fill = 'white')
    draw_eye(x + 100, y)
    draw_eye(x + 600, y)
    draw_mouth(x + 525, y + 450)
    draw_nose(x + 500, y + 200)

draw_face(10,10)


tkinter.mainloop()
