my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width = 1000, height = 600)
my_canvas.pack()

def draw_eye(x: int, y:int) -> None:
    my_canvas.create_oval (x, y, x + 200, y + 125, fill='white')
    my_canvas.create_oval (x + 50, y + 5, x + 150, y + 105, fill='brown')


    my_canvas.create_oval (x + 75, y + 20, x + 125, y + 80, fill='black')

def draw_nose(x: int, y: int) -> None:
    my_canvas.create_oval (x ,y, x + 100, y + 100, fill = 'red',
                                outline = 'black', width = 8)

def draw_mouth(x: int, y: int) -> None:
   my_canvas.create_oval(x, y, x + 200, y + 100, fill = 'red')


def draw_face_outline(x: int, y: int):
    my_canvas.create_oval (x, y, x + 500, y + 500, fill = 'tan')

def draw_face(x: int, y:int) -> None:
    draw_face_outline(x,y)
    draw_eye(x+25, y+100)
    draw_eye(x+250,y+100)
    draw_nose(x+200,y+250)
    draw_mouth(x+150,y+375)
draw_face(0,0)
draw_face(500,0)
tkinter.mainloop()
