def calculate_shipping (n:float) -> float:
    ''' Calculates price for shipping given weight of package '''
    if n < 2:
        return 2
    if n >= 10:
        return 5 + 1.50*(n-10)
    else:
        return 5
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window
my_canvas = tkinter.Canvas(my_window, width=1200, height=800)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

def draw_eye (x:int, y: int) -> int:
    ''' Draws an eye in tkinter canvas encased in a rectangle
    with given coordinates at top left corner (x, y) '''
    my_canvas.create_oval(x, y, (x+100), (y+80), fill='white') #sclera
    my_canvas.create_oval((x+25), (y+5), (x+75), (y+75), fill='green') #iris
    my_canvas.create_oval((x+45), (y+35), (x+55), (y+45), fill='black') #pupil
def draw_nose (x:int, y:int) -> int:
    ''' Draws a nose in tkinter canvas at given coordinates (x, y) '''
    my_canvas.create_line(x, y, x, (y+50), fill='black')
def draw_mouth (x:int, y:int) -> int:
    ''' Draws a mouth in tkinter canvas at given coordinates (x, y) '''
    my_canvas.create_oval(x, y, (x+200), (y+150), fill='black')
def draw_face (x:int, y:int) -> int:
    ''' Draws a face enclosed in a rectangle with the top left corner
    at given coordinates (x, y) '''
    my_canvas.create_oval(x, y, (x+500), (y+500), fill='yellow')
    draw_eye((x+100), (y+140))
    draw_eye((x+300), (y+140))
    draw_nose((x+250), (y+240))
    draw_mouth((x+150), (y+310))
draw_face(10,10)
