def calculate_shipping(weight: float) ->float:
    if weight < 2:
        rate = 2
    elif weight >= 2 and weight < 10:
        rate = 5
    elif weight >= 10:
        rate = 5 + ((weight-10)*1.5)
    return rate

my_window3 = tkinter.Tk()
my_canvas3 = tkinter.Canvas(my_window3, width=500, height=500)
my_canvas3.pack()
def draw_eye (x_cord: int, y_cord: int, x2_cord: int, y2_cord: int, start_point: int):
    x1_eye =(x2_cord-x_cord)/6
    y1_eye = (y2_cord-y_cord)/8
    x2_eye = (x2_cord-x_cord)/2
    y2_eye = (y2_cord-y_cord)/4
    
    my_canvas3.create_oval(x1_eye+start_point, y1_eye, x2_eye+start_point, y2_eye)
    my_canvas3.create_oval((x1_eye *7/4)+start_point, (y1_eye *5/4), (x2_eye *3/4)+start_point, (y2_eye *7/8))
    my_canvas3.create_oval((x1_eye *3/2)+start_point, y1_eye, (x2_eye *5/6)+start_point, y2_eye)

def draw_mouth(x_cord: int, y_cord: int, x2_cord: int, y2_cord: int):
    x1_mouth = (x2_cord-x_cord)/2-((x2_cord-x_cord)/5)+ x_cord
    y1_mouth = (y_cord) + (y2_cord - y_cord)*3/4
    x2_mouth = (x2_cord-x_cord)/2+((x2_cord-x_cord)/5) + x_cord
    y2_mouth = (y_cord) + (y2_cord - y_cord)*7/8
    my_canvas3.create_oval(x1_mouth, y1_mouth, x2_mouth, y2_mouth)
    my_canvas3.create_line(x1_mouth, y1_mouth+ (y2_mouth-y1_mouth)/2, x2_mouth, y1_mouth+(y2_mouth-y1_mouth)/2)

def draw_nose(x_cord: int, y_cord: int, x2_cord: int, y2_cord, start_point: int):
    x1_nose = (x2_cord-x_cord)/2-((x2_cord-x_cord)/8)
    y1_nose = (y2_cord-y_cord)/2+((y2_cord-y_cord)/6)
    x2_nose = (x2_cord-x_cord)/2+((x2_cord-x_cord)/8)
    y2_nose = (y2_cord-y_cord)/2+((y2_cord-y_cord)/6)
    
    my_canvas3.create_line(x1_nose+ start_point, y1_nose, x2_nose+start_point, y2_nose)
    my_canvas3.create_line(x1_nose+start_point, y1_nose, x2_nose-((x2_nose-x1_nose)/2)+start_point, y2_nose - (y2_nose*4/8))
    my_canvas3.create_line(x2_nose-((x2_nose-x1_nose)/2)+start_point, y2_nose - (y2_nose*4/8), x2_nose+start_point, y2_nose)
    
    
def draw_face (x_cord: int, y_cord: int, x2_cord: int, y2_cord: int):

    my_canvas3.create_oval(x_cord, y_cord, x2_cord, y2_cord)
    draw_eye(x_cord, y_cord, x2_cord, y2_cord, x_cord)
    draw_eye(x_cord, y_cord, x2_cord, y2_cord, x_cord+ (x2_cord - x_cord)/3)
    draw_mouth(x_cord, y_cord, x2_cord, y2_cord)
    draw_nose(x_cord, y_cord, x2_cord, y2_cord, x_cord)
    



    
draw_face(10, 10, 310, 410)
