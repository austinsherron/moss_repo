def abbreviate(name: str) -> str:
    "Abbreviate word down to first three letters"
    return name[0]+name[1]+name[2]
assert abbreviate('January') == 'Jan'

def find_area_square(side: int) -> int:
    "Use the given length of a square to find its area"


    return side ** 2
assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(radius: int) -> float:
    "Use the given radius of a circle to find its area"
    return math.pi * (radius ** 2)
assert find_area_circle(1) == math.pi
assert find_area_circle(5) == 25 * math.pi

def calculate_shipping(Pounds: int) -> int:
    "Calculates shipping, for packages under 2 pounds, 2 dollars. For packages 2 to 10 pounds, 5 dollars. For packages of 10 pounds or more, 5 dollars plus 1.5 per pound for each pound over 10."
    if Pounds < 2:
        return 2.00
    if Pounds >= 2 and Pounds < 10:
        return 5.00
    if Pounds >= 10:
        return 5 + 1.5 * (Pounds-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
