def calculate_shipping (n: int) -> int:
    if n < 2:
        return 2.00  ## Return shipping price under 2 lbs
    elif 2 <= n < 10:  ## Between 2 and 10
        return 5.00
    else:
        a = n - 10  
        return 5.00 + (a * 1.50 )
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(11))

def year_sort( L: list ) -> str:
    newBooks = []  ## Create empty arrays to fill
    oldBooks = []
    newTotal = 0  ## Total will add itself plus the instock value
    oldTotal = 0
    for i in range(len(L)):
        if L[i].year >= 2000:
            newBooks.append( L[i].title )
            newTotal = newTotal + L[i].instock  ## Cummulatively adds instock
        else:
            oldBooks.append( L[i].title )
            oldTotal = oldTotal + L[i].instock
    print( 'Books before 2000:', oldBooks )
    print( 'Books 2000 or after:', newBooks )
    if newTotal > oldTotal:
        print( 'More titles 2000 or later', '(', newTotal, 'vs', oldTotal, ')' )
    else:
        print( 'More titles 1999 or earlier', '(', oldTotal, 'vs', newTotal, ')' )
    
year_sort( BSI )
