def calculate_shipping(weight: float)->float:
    "Returns price of shipping for package"
    if weight < 2:
        price = 2
    elif weight < 10:
        price = 5
    else:
        price = 5 + (weight - 10)*1.5
    return (price)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print (calculate_shipping(15))

def inventory_value(qwer : Book) -> float:
    "Returns value of inventory for a book"
    return (qwer.instock * qwer.price)

def top_value(x: list)->Book:
    "Returns book object with highest value"
    high = x[0]
    for y in range(len(x)):
        if inventory_value(x[y]) > inventory_value(x[y-1]):
            high = x[y]
    return (high)

print('The hightest-value book is', top_value(BSI).title, 'by', top_value(BSI).author, 'at a value of $', inventory_value(top_value(BSI)))
