    ''' Take the name of a month and returns it's three letter abbreviation. '''

    ''' Take the length of one side of a square and returns the area of the square. '''

def print_even_numbers(integers: list):
    ''' Take a list of integers as input and prints each even number on the list. '''
    for i in range (0, len(integers)):
        if integers[i] % 2 == 0:
            print(integers[i])
    return

def calculate_shipping(weight: int) -> float:
    ''' Take a weight of a package and reutnr the price for shipping the package. '''
    if weight < 2:
        price = 2.00
    elif weight < 10:
        price = 5.00
    else:
        price = 12.50
    return price

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter
def create_square(x1: int, y1: int, length: (int)):
    ''' Take x and y coordinate of upper-left corner and the length of a side, and
        draw a square. '''
    window_width = 2 * x1 + length
    window_height = 2 * y1 + length
    x2 = x1 + length
    y2 = y1 + length

    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=window_width, height=window_height)
    my_canvas.pack() 
    my_canvas.create_rectangle(x1, y1, x2, y2)
    tkinter.mainloop()
    return
        
create_square(50, 60, 300)
