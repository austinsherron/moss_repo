def find_area_square(side:int) -> int:
    """Computes area of square using side * side"""
    area = side ** 2
    return (area)

def calculate_shipping(weight:int) -> float:
    """takes input weight and calculates price through booleans"""
    if weight < 2:
        return float(2)
    elif weight >= 2 and weight < 10:
        return float(5)
    else:
            return float(5 + ((weight - 10) * 1.5))

def draw_eye(x,y:int):
    my_canvas.create_oval(x, y, x + 100,y + 50, fill = 'white')
    my_canvas.create_oval(x + 30,y + 50, x + 80, y + 10, fill = 'green')
    my_canvas.create_oval(x + 50,y + 30, x + 60, y + 40, fill = 'black')
    
def draw_nose(x,y:int):
    my_canvas.create_polygon(x, y, x - 30, y + 40,x + 30,y + 40,fill = 'LightSalmon')
    
def draw_mouth(x,y:int):
    my_canvas.create_oval(x, y, x + 50, y + 50 , fill = 'FireBrick')
    
    
def draw_face(x,y,a,b,c,e,f,g,h:int):
    my_canvas.create_oval(x, y, x + 300, y + 300 , fill = 'Khaki')
