def calculate_shipping(pounds:float)->float:
    if pounds < 2:
        rate = 2.00
        return rate
    elif 2 <= pounds < 10:
        rate = 5.00
        return rate
    else:
        rate = 5.00 + 1.50*(pounds - 10)
        return rate
 
assert calculate_shipping(1.5) == 2.00   #Test if the function is correct or not.
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
