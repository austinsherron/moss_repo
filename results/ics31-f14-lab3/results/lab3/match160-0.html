def calculate_shipping(pound:float) -> float :
    'Takes in weight in pounds and returns rate.'
    if pound < 2 :
        return 2.00
    elif pound >= 2 and pound < 10 :
        return 5.00
    elif pound >= 10 :
        return 5.00 + 1.50*(pound-10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5

import tkinter

def create_square(x:int, y:int, length:int) :
    '''
    Takes coordinates for the top left corner of the square and the length of each side.
    Opens a window and draws a square.
    '''
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=640, height=480)  # Create a 500x500 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window
    my_canvas.create_rectangle(x, y, x+length, y+length)
    tkinter.mainloop()
    return

create_square(300, 150, 200)

create_square(100, 140, 100)

def draw_nose (x:int, y:int, canvas) :
    'Draws a nose taking a starting point x and y'
    canvas.create_arc(x, y, x+30, y+75, start = 0, extent = 180, fill='brown')
    canvas.create_oval(x+2, y+33, x+14, y+38, fill = 'black')
    canvas.create_oval(x+16, y+33, x+28, y+38, fill = 'black')
    return
def draw_mouth (x:int, y:int, canvas) :
    'Draws a mouth taking a starting point x and y'
    canvas.create_arc(x, y, x+150, y+50, start = 180, extent = 180, fill='black')
    return
def draw_eye (x:int, y:int, canvas) :
    'Draws a eye taking a starting point x and y'
    canvas.create_oval(x, y, x+70, y+90, fill = 'white')
    canvas.create_oval(x, y+20, x+70, y+90, fill='black')
    canvas.create_oval(x+10, y+30, x+60,y+80, fill='saddlebrown')
    return
def draw_face (x:int, y:int, canvas) :
    'Draws a face taking a starting point x and y'
    canvas.create_oval(x, y, x+640, y+480, fill='tan')
    draw_eye(x+150, y+150, canvas)
    draw_eye(x+420, y+150, canvas)
    draw_nose(x+300, y+220, canvas)
    draw_mouth(x+240, y+300, canvas)
    return
