def abbreviate(month: str) -> str:
    '''Takes name of month and returns three-letter abbreviation'''
    return month[0:3]
assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'

def find_area_square(side: int) -> int:
    '''Takes a number representing length of a side of a square and returns
    the area of that square'''
    return side ** 2
assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(radius: int) -> float:
    '''Takes number representing radius of a circle and returns area of circle'''
    return 3.14159 * (radius ** 2)
assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def print_even_numbers(num: list) -> int:
    '''Takes list of integers and prints each even numbers independently'''


    for i in num:
        if i % 2 == 0:
            print(i)

def calculate_shipping(weight: float) -> float:
    '''Takes value of weight of a package and returns price for shipping the package'''
    if weight < 2.0:
        return 2.00
    elif weight >= 2.00 and weight < 10.00:
        return 5.00
    elif weight >= 10.00:
        return 5.00 + (1.5 * (weight - 10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50


import tkinter

def create_square(x: int, y: int, side: int):
    '''Takes x and y coordinates of the top-left corner of the square and a side
    length and draws a square with that length'''
    my_window = tkinter.Tk()

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    my_canvas.create_rectangle(x, y, x + side, y + side)

    tkinter.mainloop()

    '''Takes x and y coordinates of top-left corner of square that inscribes a
    circle and draws a circle with the given diameter'''
    my_window = tkinter.Tk()

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    my_canvas.create_oval(x, y, x + diameter, y + diameter)

    tkinter.mainloop()

def costliest(i: list) -> str:
    '''Returns name of costliest restaurant in a given list of restaurants'''
    i.sort(key = restaurant_price)
    return i[-1].name

    without altering the list'''

import tkinter
my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=1000, height=600)
my_canvas.pack()

def draw_eye(x: int, y: int):
    my_canvas.create_oval(x, y, x + 80, y + 40)
    my_canvas.create_oval(x + 20, y, x + 60, y + 40, fill='blue')
    my_canvas.create_oval(x + 30, y + 4, x + 50, y + 40, fill='black')
    
def draw_nose(x: int, y: int):
    my_canvas.create_line(x, y, x + 20, y + 20)
    my_canvas.create_line(x + 20, y + 20, x, y + 20)
    
def draw_mouth(x: int, y: int, x2: int, y2: int):
    my_canvas.create_line(x, y, x2, y2)
    
def draw_face(x: int, y: int, x2: int, y2: int):
    my_canvas.create_oval(x, y, x2, y2)
    draw_eye(((x2 - x) / 3) + x, ((y2 - y) / 3) + y)
    draw_eye(((x2 - x) * 2) / 3 + x, ((y2 - y) / 3) + y)
    draw_nose(((x2 - x) / 2) + x, ((y2 - y) / 2) + y)
    draw_mouth(((x2 - x) / 3) + x, (((y2 - y) * 2) / 3) + y,
               (((x2 - x) * 2) / 3) + x, (((y2 - y) * 2) / 3) + y)
    tkinter.mainloop()
draw_face(100, 100, 500, 500)
