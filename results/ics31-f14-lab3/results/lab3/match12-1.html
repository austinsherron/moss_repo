def calculate_shipping(weight:float):
    if weight <2:
        return  2.00
    elif weight >= 2 and weight <10:
        return  5.00
    elif weight > 10:
        return  ((weight - 10) * 1.50) + 5.00

assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

    return (book.price*book.instock)

    index_of_most_valueable_book = 0
    for i in range(len(bookList)):
        if(inventory_value(bookList[i]) > inventory_value( bookList[index_of_most_valueable_book] )):

def draw_eye(my_canvas:tkinter.Canvas, x0:int, y0:int, x1:int, y1:int):
    my_canvas.create_oval(x0, y0, x1, y1, fill='white')
    my_canvas.create_oval(x0+10, y0+10, x1-10, y1-10, fill='blue')
    my_canvas.create_oval(x0+20, y0+20, x1-20, y1-20, fill='black')

def draw_nose(my_canvas:tkinter.Canvas, x0:int, y0:int, x1:int, y1:int):
    my_canvas.create_rectangle(x0, y0, x1, y1, fill='yellow')
    
def draw_mouth(my_canvas:tkinter.Canvas, x0:int, y0:int, x1:int, y1:int):
    my_canvas.create_oval(x0, y0, x1, y1, fill='pink')

def draw_face(my_canvas:tkinter.Canvas, x0:int, y0:int, x1:int, y1:int):
    my_canvas.create_oval(x0, y0, x1, y1, fill='brown')
    draw_eye(my_canvas, x0+75, y0+75, x1-175, y1-175)
    draw_eye(my_canvas, x0+175, y0+75, x1-75, y1-175)
    draw_nose(my_canvas, x0+145, x0+125, x1-145, y1-125)
    draw_mouth(my_canvas, x0+75, y0+225, x1-75, y1-25)
    
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1000, height=10000)
my_canvas.pack()
draw_face(my_canvas, 200, 200, 500, 500)
draw_face(my_canvas, 600, 600, 900, 900)
tkinter.mainloop()
