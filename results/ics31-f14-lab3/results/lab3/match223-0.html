import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()

def draw_eye(canvas: "tkinter.Canvas", x:float, y:float):
    canvas.create_oval(x,y,x+80, y+55, fill='white')
    canvas.create_oval(x+10, y+3, x+60, y+47, fill='brown')
    return
def draw_nose(canvas: "tkinter.Canvas", x: float, y: float):
    canvas.create_polygon(x+40, y, x, y+30, x+70, y+30, fill='yellow')
    return
def draw_mouth(canvas: "tkinter.Canvas", x: float, y: float):
    canvas.create_oval(x, y, x+80, y+25, fill='red')
    return
def draw_face(canvas: "tkinter.Canvas", x: float, y: float):
    canvas.create_oval(x, y, 250+x, 250+y, fill='white')
    draw_eye(canvas, x+25, y+50)
    draw_eye(canvas, x+140, y+50)
    draw_nose(canvas, x+85, y+105)
    draw_mouth(canvas, x+85, y+195)
    return

draw_face(my_canvas, 100, 100)
tkinter.mainloop()
