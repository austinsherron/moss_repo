def calculate_shipping(n: int) -> int:
    if n < 2:
        return (2.00)
    if n > 2 and n < 10:
        return (5.00)
    if n > 10:
        return (5.00 + ((n - 10) * 1.50))

assert calculate_shipping(1.9) == 2.00
assert calculate_shipping(2.1) == 5.00
assert calculate_shipping(9.9) == 5.00
assert calculate_shipping(10.1) == 5.1499999999999995

   my_canvas.create_oval(x + 100, y + 100, x + 150, y + 150, fill = 'blue' )
   my_canvas.create_oval(x + 115, y + 115, x + 135, y + 135, fill = 'black')
   my_canvas.create_oval(x + 25, y + 25, x + 200, y + 200)

def draw_nose(x: int, y:int) -> None:
    my_canvas.create_oval(x + 525, y + 325, x + 600 , y + 400, fill = 'red')
    
def draw_mouth(x: int, y:int) -> None:
    my_canvas.create_oval(x + 400, y + 500, x + 700, y + 600, fill = 'pink')
   
def draw_outline(x:int, y:int) -> None:
    my_canvas.create_oval(x + 200, y, x + 1000, y + 800, fill = 'white')
