def calculate_shipping(weight: int) -> float:
    if weight < 2:
        result = 2.00
    elif  weight < 10:
        result = 5.00
    elif weight > 10:
        result = 5.00 + (weight-10)*1.50
    return result

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
