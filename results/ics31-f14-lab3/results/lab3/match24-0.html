def calculate_shipping(weight: float) -> float:
    if weight<2:
        return(2.00)
    elif 10 > weight >2:
        return(5.00)
    elif weight >= 10:
        return(5.00 +(1.50 * (weight-10)))

my_window = tkinter.Tk()    # Create the graphics window

my_canvas.pack()            # Put the canvas into the window

def draw_eye(xtop:int,ytop:int,xbottom:int,ybottom:int):
    my_canvas.create_oval(xtop, ytop, xbottom, ybottom, fill= 'black')

def draw_nose(xtop:int,ytop:int,xbottom:int,ybottom:int):
    my_canvas.create_line(xtop,ytop,xbottom,ybottom,fill='red')

def draw_mouth(xtop:int,ytop:int,xbottom:int,ybottom:int):
    my_canvas.create_oval(xtop,ytop,xbottom,ybottom,fill='blue')

def draw_face():
    draw_eye(100,100,100,50)
    draw_eye(200,100,200,50)
    draw_nose(150,200,150,300)
    draw_mouth(75,300,225,330)
