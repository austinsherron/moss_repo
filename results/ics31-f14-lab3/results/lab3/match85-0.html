def calculate_shipping(pounds: float) -> float:
    "Given anything under 2 pounds returns 2. inbetween 2 and 10 pounds returns 5. more than 10 pounds returns 5 + (1.5 * each pound over 10)"
    if pounds < 2:
        return 2.00
    elif 2 <= pounds < 10:
        return 5.00
    else:
        return 5.00 + (1.50 * (pounds - 10))


assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book: Book) -> float:
    "Return inventory value of a Book given a Book"
    return book.instock * book.price

assert inventory_value(BSI[2]) == BSI[2].instock * BSI[2].price
assert inventory_value(BSI[4]) == BSI[4].instock * BSI[4].price

def top_value(inventory: list) -> Book:
    "Return highest inventory value Book in given list of Book"
    return (sorted(inventory, key=inventory_value)[-1])
assert top_value(BSI) == BSI[1]

      %(top_value(BSI).title, top_value(BSI).author, inventory_value(top_value(BSI))))

def draw_face(color:str):
    "Funtion will draw a face with input for color of eye"
    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on    
    my_canvas.pack()            # Put the canvas into the window

    my_canvas.create_oval(110, 80, 390, 420, fill='wheat')

    draw_eye(my_canvas, 140,170, color)
    draw_eye(my_canvas, 260,170, color)
    draw_nose(my_canvas)
    draw_mouth(my_canvas)
    draw_hair(my_canvas)

    tkinter.mainloop()          # Combine all the elements and display the window

def draw_eye(canvas, x, y, color):
    "Funtion will draw eye"
    canvas.create_oval(x, y, x + 100, y + 70, fill='white')
    canvas.create_oval(x + 35, y + 20, x + 65, y + 50, fill='black', outline=color, width=6)

def draw_nose(canvas):
    "Funtion will draw nose"
    canvas.create_line(200, 330, 300, 330, fill='black', width=3)
    
def draw_mouth(canvas):
    "Funtion will draw mouth"
    canvas.create_line(250, 270, 250, 300, fill='black', width=3)

def draw_hair(canvas):
    "Funtion will draw hair"
    canvas.create_polygon(250, 60, 100, 120, 90, 190, 250, 110, fill="black")
    canvas.create_polygon(250, 60, 400, 120, 410, 190, 250, 110, fill="black")
