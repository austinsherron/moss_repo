def calculate_shipping(weight: float) -> float:
    '''Returns shipping cost of pacakge depending on weight'''
    if weight < 2:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    elif weight >= 10:
        return 5.00 + (weight-10)*1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book_list: list) -> float:
    return (book_list.price * book_list.instock)

def top_value(list_of_books: list) -> list:
    inventory = 0
    answer = []
    for b in list_of_books:
        if inventory_value(b) > inventory:
            answer = b
            inventory = inventory_value(b)
    return answer
print('The highest-value book is ', top_value(BSI).title, ' by ', top_value(BSI).author, ' at a value of $ ', inventory_value(top_value(BSI)), '.', sep='')
