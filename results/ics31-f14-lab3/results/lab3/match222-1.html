def calculate_shipping(weight: float) -> float:
    '''calculates the shipping price for a package based on size'''
    if weight < 2.0:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    else:
        return 5.00 + (1.50 * (weight - 10))

def book_comparison(L: list) -> None:
    '''splits inventory into two lists and outputs which list has more books'''
    before2k = 0
    after2k = 0
    answ1 = "More titles 2000 or later "
    answ2 = "More titles before 2000 "
    before2kL = []
    after2kL = []
    #creates two lists for books before and after 2000
    for x in L:
        if x.year < 2000:
            before2kL.append(x)
        else:
            after2kL.append(x)

    #counts total number of books in inventory of each list        
    for i in range(len(before2kL)):
        before2k = before2k + int(before2kL[i].instock)
    for j in range(len(after2kL)):
        after2k = after2k + int(after2kL[j].instock)

    #outputs response
    if (before2k > after2k):
        print(answ1, "(", before2k, " vs. ", after2k, ")", sep = '')
    else:
        print(answ2, "(", after2k, " vs. ", before2k, ")", sep = '')

    return None

book_comparison(BSI)

face_window = tkinter.Tk()
face_canvas = tkinter.Canvas(face_window, width=1000, height=500)  # Create a 500x500 canvas to draw on
face_canvas.pack()            # Put the canvas into the window

def draw_eye (x: int, y: int, eye: str):
    '''draws an eye at the x and y coordinate of the specified color'''
    face_canvas.create_oval(x, y, x + 75, y + 25, fill = 'white') #whites
    face_canvas.create_oval(x + 25, y, x + 55, y + 25, fill = eye) #iris
    face_canvas.create_oval(x + 35, y + 7, x + 45, y + 17, fill = 'black') #pupil
    return None

def draw_nose (x: int, y: int, skinColor: str):
    '''draws a nose given x and y coordinates'''
    face_canvas.create_arc(x, y, x + 60, y + 80, start =145, extent = 90, fill = 'black')
    face_canvas.create_arc(x + 1, y - 1, x + 60, y + 80, start = 145, extent = 90, fill = skinColor, outline = skinColor)
    face_canvas.create_arc(x - 40, y + 90, x + 12, y + 40, start = 180, extent = 160, fill = 'black')
    face_canvas.create_arc(x - 41, y + 89, x + 12, y + 40, start = 180, extent = 160, fill = skinColor, outline = skinColor)
    return None

def draw_mouth ( x: int, y: int, z: int, w: int, mouth: str):
    '''Takes mouth coordinates and whether or not the person is smiling or frowning, outputs the mouth'''
    if mouth == 'smile':
        face_canvas.create_arc(x, y, z, w, start = 180, extent = 180, fill = 'indianred', outline = "black")
    elif mouth == 'frown':
        face_canvas.create_arc(x, y, z, w, start = 0, extent = 180, fill = 'indianred', outline = "black")
    else:
        face_canvas.create_arc(x, y, z, w, start = 180, extent = 360, outline = "black")
    
    return None

def draw_face( x: int, y: int, color: str, eye: str, mouth: str):
    '''given location, skin tone, eye color, and smile, outputs a face'''
    face_canvas.create_oval( x, y, x + 300, y + 300, fill = color)
    draw_eye(x+50, y + 75, eye)
    draw_eye(x+175, y + 75, eye)
    draw_nose(x+160, y+80, color)
    draw_mouth(x+50, y+200, x+250, y+250, mouth)
    return None

draw_face(50, 100, 'burlywood', 'blue', 'smile')
draw_face(350, 100, 'saddlebrown', 'green', 'meh')
draw_face(650, 100, 'peru', 'brown', 'frown')

tkinter.mainloop()          # Combine all the elements and display the window
