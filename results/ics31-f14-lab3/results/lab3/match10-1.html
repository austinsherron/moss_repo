def calculate_shipping(e:int)->float:
    ''' Return the price of shipping crates '''
    if e < 2:
        return 2.00
    elif 2 <= e < 10:
        return 5.00
    elif e >= 10:
        return 5.00 + (e - 10) * 1.50

def inventory_value(b:Book) -> int:
    '''takes a Book and returns the value of the inventory of the book'''
    return b.instock * b.price

def top_value(b:list) -> Book:
    '''takes a list of Books and returns the Book that has the highest-value inventory'''
    temp = b[0]
    for x in b:
        if inventory_value(x) > inventory_value(temp):
            temp = x
    print('The highest-value book is '+temp.title+' by '+temp.author+' at the value of $',int(inventory_value(temp)))
    return temp

f = tkinter.Canvas(window, width=1500, height=1500)
f.pack()

def draw_eye(x1, y1, x2, y2, w1, w2, s1, z1, z2, s2):
    f.create_oval(x1, y1, x2, y2, fill='white')


    f.create_oval(z1, z2, z1 + s2, z2 + s2, fill='black')

def draw_nose(a1, a2, b, c):
    f.create_polygon(a1, a2, a1 + b, a2 + c, a1 - b, a2 + c, fill = 'black')

def draw_mouth(x,y,z):
    f.create_line(x,y,x+2*z,y)
    f.create_line(x, y, x + z, y + z, x + 2 * z, y, fill='black')

def draw_face(x1, y1, x2, y2, z1, z2, a1, b1, b2, a2, w1, r1, w2, r2, j1, j2, q1, d1, d2, q2, n1, n2, f, h, x, y, a):
    draw_eye(x1, y1, x2, y2, z1, z2, a1, b1, b2, a2)
    draw_eye(w1, r1, w2, r2, j1, j2, q1, d1, d2, q2)
    draw_nose(n1, n2, f, h)
    draw_mouth(x, y, a)

draw_face(10, 10, 50, 30, 20, 10, 20, 30, 20, 4, 110, 10, 150, 30, 120, 10, 20, 130, 20, 4, 75, 15, 10, 30, 20, 50, 50)
