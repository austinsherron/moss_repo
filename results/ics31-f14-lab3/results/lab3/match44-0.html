def calculate_shipping(weight: float) -> float:
    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5.00 + (weight - 10) * 1.50


assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

# Eye function
def draw_eye(x: float, y: float, color: str, canvas: tkinter.Canvas):
    ''' Draws an eye with it's top left corner at (x,y) with the color specified'''

    canvas.create_oval(x, y, x + 100, y + 50, fill = "white")
    canvas.create_oval(x + 25, y, x + 75, y + 50, fill = color)
    canvas.create_oval(x + 40, y + 15, x + 60, y + 35, fill = "black") # Colors black pupil


# Nose function
def draw_nose(size: float, y_position: float, canvas: tkinter.Canvas):
    ''' Draws a nose of specified size, located y_position from the center of the canvas (300, 300).
        + y_position would cause nose to shift up, - y_position would shift down'''
    
    length_amplifier = size * 1.2   # Scales with the size of the nose in the y-direction
    
    canvas.create_oval(300 - size/2, 300 - y_position, 300 + size/2, 300 + length_amplifier)
    canvas.create_rectangle(300, 300 - y_position, 300 + size/2, 300 + length_amplifier, fill = "Bisque", outline = "Bisque")
    canvas.create_line(300, 300 - y_position, 300, 300 - y_position - length_amplifier)


# Mouth function
def draw_mouth(size: float, y_position: float, canvas: tkinter.Canvas):
    ''' Draws a mouth of specified size, located y_position from the center of the canvas (300, 300).
        + y_position would cause nose to shift up, - y_position would shift down'''

    length_amplifier = size * 0.5   # Scales with the size of the mouth in the y-direction (the curve)

    canvas.create_oval(300 - size/2, 300 - y_position, 300 + size/2, 300 - y_position + length_amplifier)
    canvas.create_rectangle(300 - size/2, 300 - y_position, 300 + size/2, 300 - y_position + length_amplifier/1.5, fill = "Bisque", outline = "Bisque")


# The entire face function
def draw_face(nose_size: float, nose_position: float, mouth_size: float,
              mouth_position: float, eye_spacing: float, eye_color: str):
    ''' Draws a 400 x 550 face with specified facial features'''

    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=600, height=600)  # Create a 600x600 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window


    # Draws oval face
    my_canvas.create_oval(100, 25, 500, 575, fill = "Bisque")

    # Draws mouth
    draw_mouth(mouth_size, mouth_position, my_canvas)

    # Draws nose
    draw_nose(nose_size, nose_position, my_canvas)


    ''' Draw eyes with specified spacing in between them'''
    # Draws left eye
    draw_eye(200 - eye_spacing/2, 150, eye_color, my_canvas)

    # Draws right eye
    draw_eye(300 + eye_spacing/2, 150, eye_color, my_canvas)
    


# Draw 3 different faces on 3 separate windows/canvases
draw_face(40, 0, 200, -55, 30, "blue")
draw_face(70, 10, 100, -60, 40, "grey")
draw_face(30, -20, 50, -50, 20, "green")
