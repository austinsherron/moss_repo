def calculate_shipping(w:float)->float:
    #calculates shipping cost
    if w<2:
        return 2.00
    elif w<10:
        return 5.00
    else:
        return 5+((w-10)*1.5)
assert calculate_shipping(1.5)==2
assert calculate_shipping(7)==5.00
assert calculate_shipping(15)==12.50

def inventory_value(b:Book)->int:
    return b.instock*b.price
def top_value(l:[Book])->Book:
    keep=l[0]
    temp=inventory_value(l[0])
    for b in l:
        if inventory_value(b)>temp:
            keep=b
            temp=inventory_value(b)
    print("The highest-value book is",keep.title,"by",keep.author,"at a value of $",temp,".")
    return keep

      my_canvas3.create_oval(a+0.457*l, b+0.4*l, a+0.527*l,b+0.47*l,fill='Gainsboro') #right pupil
      my_canvas3.create_oval(a+(0.173+0.035-0.01)*l, b+(0.4+0.035-0.01)*l, a+(0.173+0.035+0.01)*l,b+       (0.4+0.035+0.01)*l,fill='black') # iris

      my_canvas3.create_oval(a+(0.35-0.15)*l, b+0.72*l, a+(0.35+0.15)*l,b+0.82*l,fill='red') #mouth
