import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window



my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

def draw_eye(x, y):
    my_canvas.create_oval(x, y, x+200, y+100, fill='grey') #outer eye
    my_canvas.create_oval(x+70, y, x+135, y+100, fill='red') #pupil
    my_canvas.create_oval(x+90, y+35, x+110, y+65, fill='blue') #iris

def draw_nose(x, y):
    my_canvas.create_oval(x, y, x+10, y+10, fill='grey')

def draw_mouth(x, y):
    my_canvas.create_rectangle(x, y, x+100, y+50, fill='pink')



def draw_face(x,y):
    draw_eye(x,y)
    draw_eye(x+290,y)
    draw_nose(x+230,y+140)
    draw_mouth(x+180,y+190)

draw_face(10,10)

tkinter.mainloop()          # Combine all the elements and display the window
