my_window = tkinter.Tk()

def draw_eye(x:int,y:int):
    my_canvas.create_oval(x,y, x+75,y+25)
    my_canvas.create_oval(x+25,y, x+50,y+25)

def draw_nose(x:int,y:int):
    my_canvas.create_line(x,y,x-10,y+25)
    my_canvas.create_line(x-10,y+25,x+5,y+25)

def draw_mouth(x:int,y:int):
    my_canvas.create_oval(x,y, x+150,y+25)

def draw_face(x:int,y:int):


    my_canvas.create_oval(x,y, x+300,y+300)
    draw_eye(x+50,y+100)
    draw_eye(x+175,y+100)
