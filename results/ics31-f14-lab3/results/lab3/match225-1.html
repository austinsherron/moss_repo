def calculate_shipping(weight_of_package: float) -> float:
    ''' Returns the price of shipping '''
    if weight_of_package < 2:
        return 2.00
    elif weight_of_package >= 2 and weight_of_package < 10:
        return 5.00
    elif weight_of_package >= 10:
        return 5 + ( 1.50 * (weight_of_package - 10) )

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(40.75) == 51.125

def inventory_value(book: Book) -> float:
    ''' Returns inventory value by multiplying price and the amount in stock'''
    return book.price * book.instock

def top_value(book_list: list) ->  Book:
    ''' Finds highest value book in a list of books '''
    highest_value_book = book_list[0]
    for b in book_list:
        if inventory_value(b) > inventory_value(highest_value_book):
            highest_value_book = b
    return highest_value_book

top_value_book = top_value(BSI)
print('The highest-value book is {} by {} at a value of ${}'.format(top_value_book.title,
                                                                    top_value_book.author,
                                                                    inventory_value(top_value_book)))
