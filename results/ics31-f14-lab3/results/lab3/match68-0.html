def top_value(BSI: list) -> Book: 
    "Returns the book with the highest-value inventory"
    
    
    current_highest = Book('0', '0' , '0', 0, 0, 0)
    for i in BSI:
        if (inventory_value(i) > inventory_value(current_highest)):
            current_highest = i
    return current_highest

def drawEye(my_canvas_i3):
    x = 70
    my_canvas_i3.create_oval(x, x, x + 100, x + 50, outline='black')
    my_canvas_i3.create_oval(x + 30, x + 10, x + 70, x + 50, fill='black')
    my_canvas_i3.create_oval(x + 200, x, x + 300, x + 50, outline='black')
    my_canvas_i3.create_oval(x + 230, x + 10, x + 270, x + 50, fill='black')
    return my_canvas_i3

    my_canvas_i3.create_oval(25, 150, 390, 200, fill='red')
