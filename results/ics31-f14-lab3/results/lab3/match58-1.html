def calulate_shipping(weight: float) ->float:
    'rate 2 [0,2) rate 5 [2,10) rate 2 plus 1.5 per/pound [10,infinity)'
    if(weight > 0 and weight < 2):
        weight = 2.00
    elif(weight > 2 and weight < 10 or weight == 2):
        weight = 5.00
    else:
        weight = 5 + 1.5*(weight - 10)
    return weight
assert calulate_shipping(1.5) == 2.00
assert calulate_shipping(7) == 5.00
assert calulate_shipping(11) == 6.50

weight = float(input('enter a weight: '))
print(calulate_shipping(weight))
