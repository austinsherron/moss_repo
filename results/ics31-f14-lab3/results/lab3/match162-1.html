def calculate_shipping (n:float)-> float:
    " It gives the price of shipping"
    if n<2:
        return 2.00


    elif n>=2 and n<=10:
        return 5.00
    else:
        return 5+(1.50*(n-10))

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()

def create_eye(x, y, L1,L2:int,color:str):
    'creates the eyes of the face'
    my_canvas.create_oval(x, y, x + L1, y + L2, fill=color) #pupil
    my_canvas.create_oval(x+25, y, x + 50, y + L2, fill='black') #iris
    
def draw_nose(x,y,L1,L2:int,color:str):
    'draws the face'
    my_canvas.create_oval(x, y, x + L1, y + L2, fill=color)
    
def draw_mouth(x,y,L1,L2:int,color:str):
    'draws the mouth'
    my_canvas.create_oval(x, y, x + L1, y + L2, fill=color)
    
def draw_hair(x,y,L1,L2:int,color:str):
    'draws the hair'
    my_canvas.create_oval(x, y, x + L1, y - L2, fill=color)
    

def draw_face(x,y,L1,L2:int,color:str):
    'draws the face by calling the above functions'
    draw_hair(x,y+125,L1,L2-300,'black')
    my_canvas.create_oval(x, y, x + L1, y + L2, fill=color)
    create_eye(x+25,y+150,L1-225,L2-400,'white')
    create_eye(x+200,y+150,L1-225,L2-400,'white')
    draw_nose(x+140,y+225,L1-275,L2-395,'orange')
    draw_mouth(x+50,y+350,L1-100,L2-425,'red')
    
       
draw_face(300,150,300,450,'yellow') #coded so that when X/Y=2 and L1 and L2 should be kept constant

tkinter.mainloop()
