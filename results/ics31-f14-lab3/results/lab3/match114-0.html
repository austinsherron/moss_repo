def calculate_shipping(weight: float) -> float:
    '''Takes the weight of a package to be shipped
    and returns the shipping price'''
    if weight < 2.00:
        return 2.00
    elif weight < 10.00:
        return 5.00
    elif weight >= 10.00:
        return 5.00 + ((weight - 10) * (1.50))
print (calculate_shipping(1.5))
print (calculate_shipping(7))
print (calculate_shipping(15))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(n: Book) -> float:
    '''finds the value of the inventory of a particular book by multipling the
    number of books in store by the price of that book'''
    return n.instock * n.price

def top_value(b: list) -> Book:
    '''finds the most expensive book in the list'''
    highest_book = b[0]
    for book in b:
        if (book.price * book.instock) > (highest_book.price * highest_book.instock):
            highest_book = book
    return highest_book
highest_book = top_value(BSI)

import tkinter

my_window = tkinter.Tk()   

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
my_canvas.pack()

def draw_eye(x,y):
    '''creates eye with starting coordinates x,y'''
    my_canvas.create_oval(x, y, x+10, y+10, fill='black', outline='black')
def draw_nose(x,y):
    '''creates nose with starting coordinates x,y'''
    my_canvas.create_line(x, y, x+15, y+30, fill='black')
    my_canvas.create_line(x, y+30,x,y,fill='black')
    my_canvas.create_line(x, y+30,x+15,y+30,fill='black')
def draw_mouth(x,y):
    '''creates mouth with starting coordinates x,y'''
    my_canvas.create_arc(x, y, x+50, y+50, extent = -180, fill='black', outline='black')
def draw_skin (x,y):
    '''creates face with starting coordinates x,y'''
    my_canvas.create_oval(x,y,x+100,y+150)
def draw_face():
    '''creates face using the above draw_functions'''
    draw_eye(50,50)
    draw_eye(100,50)
    draw_nose(80,70)
    draw_mouth(58,90)
    draw_skin(30,30)

draw_face()

tk.mainloop()
