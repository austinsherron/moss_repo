def print_even_numbers (value: list):
    '''Print integers that divide by 2 (even numbers only)'''
    for char in value:

            print(char)

def calculate_shipping (weight: int) -> float:
    '''Return the price of the calculated shipping amount'''
    if (weight < 2):
        return 2.00
    elif (weight < 10):
        return 5.00
    else:
        return 5.00 + ((weight - 10)*(1.50))

assert calculate_shipping(15) == 12.50
assert calculate_shipping(20) == 20.00

def inventory_value(TheBook: Book) -> float:
    '''Returns the value of our inventory of that book.'''
    return TheBook.price * TheBook.instock

def top_value (aList: list) -> Book:
    '''Returns the highest value book.'''
    highest_value_book = aList[0] 
    for book in aList:
        if(inventory_value (book) > inventory_value (highest_value_book)):
            highest_value_book = book
    return highest_value_book

highest_price_book = top_value(BSI)

      "at a value of $", inventory_value(highest_price_book),'.')

import tkinter         
my_window = tkinter.Tk()    
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)   
my_canvas.pack()


    
# Eye
def draw_eye(startingX: int, startingY: int):
    my_canvas.create_oval(startingX, startingY, startingX + 200, startingY + 100, fill='white')         #Eye Shape
    my_canvas.create_oval(startingX + 50, startingY, startingX + 150, startingY + 100, fill='blue')          #Iris
    my_canvas.create_oval(startingX + 75, startingY + 25, startingX +125, startingY+75, fill='black')         #Pupil
# Nose
def draw_nose(startingX: int, startingY: int):
    my_canvas.create_line(startingX + 180, startingY + 200, startingX + 280, startingY + 200)
    my_canvas.create_line(startingX + 180, startingY + 200, startingX + 280, startingY + 125)
# Mouth
def draw_mouth(startingX: int, startingY: int):
    my_canvas.create_oval(startingX + 130, startingY + 255, startingX + 330, startingY + 300, fill='red')
    my_canvas.create_line(startingX+130, startingY + 275, startingX + 330, startingY + 275)

# Head
def draw_head(startingX: int, startingY: int):
    my_canvas.create_oval(startingX - 50, startingY - 120, startingX + 550, startingY + 400, fill='tan')
    
# Overall Face
def draw_face(startingX: int, startingY: int):
    draw_head(startingX, startingY)
    draw_eye(startingX, startingY)
    draw_eye(startingX + 300, startingY)
    draw_nose(startingX, startingY)
    draw_mouth(startingX, startingY)
    tkinter.mainloop()
draw_face(150,300)
