def calculate_shipping(weight: float)->float:
#    Returns a value dependent on weight. Shipping is flat rate for
#    under 10 dollars and variable rate at 5.00 plus 1.50 per pound over 10 lbs
    if( weight < 2):
        return 2
    elif( weight < 10):
        return 5
    else:
        return 5 + (weight-10)*1.50

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book: Book)->float:
    "Return the value of the store's stock of the given book"
    "by multiplying the price by the number of copies in stock"
    return book.price * book.instock

def top_value(booklist: 'list of books')->Book:
    "Loops through the given list of books and returns the highest"
    "value book as defined by inventory_value"
    topBook = booklist[0]
    for i in range(len(booklist)):
        if inventory_value(booklist[i]) > inventory_value(topBook):
            topBook = booklist[i]
    return topBook
print(top_value(BSI))

face_window = tkinter.Tk()
face_canvas = tkinter.Canvas(face_window, width=500, height=500)
face_canvas.pack()

def draw_eye(window, canvas, x, y):
    "Draws an eye at given x and y coordinates"
    canvas.create_oval(x, y, x + 200, y + 100)
    canvas.create_oval(x + 75, y + 100, x + 175, y, fill = 'blue')
    canvas.create_oval(x + 100, y + 30, x + 150, y + 70, fill = 'black')

def draw_nose(window, canvas, x, y):
    "Draws a dot (nose) at the given x and y coordinates"
    canvas.create_oval(x + 210, y + 80, x + 190, y + 110, fill = 'black')


def draw_mouth(window, canvas, x, y):
    "Draws a curved arc (mouth) at the given x and y coordinates"
    canvas.create_arc(x + 100, y + 150, x + 300, y + 260)

def draw_face(window, canvas):
    "Makes multiple calls to draw_nose, draw_mouth, and draw_eye to"
    "compose an image of a face on the given canvas"
    draw_eye(face_window, face_canvas, 50, 50)
    draw_eye(face_window, face_canvas, 300, 50)

    draw_nose(window, canvas, 75, 75)
    draw_mouth(window, canvas, 60, 100)

draw_face(face_window, face_canvas)
