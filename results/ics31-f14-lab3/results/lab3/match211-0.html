def calculate_shipping(weight : float) -> float :
    if ( weight < 2.0 ):
        return 2.00
    elif ( weight >= 2.0 and weight < 10.0 ):
        return 5.00
    elif (weight >= 10):
        return (5 + (weight - 10)*1.50)

assert calculate_shipping(1.5) == 2.00


assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

my_window = tkinter.Tk()    # Create the graphics window

shape = tkinter.Canvas(my_window, width=1000, height=700)  # Create a 500x500 canvas to draw on
shape.pack()            # Put the canvas into the window


def create_eye(x:int, y:int):
    shape.create_oval(x,y, x+120, y-80, fill='white' , outline='black')
    shape.create_oval(x+100,y, x+20, y-80, fill='blue')
    shape.create_oval(x+80,y-20,x+40,y-60, fill="black")

def create_mouth(x:int, y:int):
    shape.create_rectangle(x,y,x+120,y+40, fill = "red")
    shape.create_line(x,y+20,x+120,y+20)

def create_nose(x:int, y:int):
    shape.create_polygon(x,y,x-25,y+70,x+25,y+70, fill = "brown")

def create_face(x:int, y:int):
    shape.create_oval(x,y,x+400,y+500, outline="brown", width = 3 )
    create_nose(x+200,y+230)
    create_mouth(x+140,y+350)
    create_eye(x+60,y+200)
    create_eye(x+220,y+200)

create_face(5,5)
create_face(500,200)


tkinter.mainloop()          # Combine all the elements and display the window
