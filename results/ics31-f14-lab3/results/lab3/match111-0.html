def calculate_shipping(weight: float) -> float:
    if weight < 2:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    else:
        return 5.00 + (weight - 10) * 1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2) == 5.00

from collections import namedtuple     
Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')
# Restaurant attributes: name, kind of food served, phone number, best dish, price of that dish
RC = [
    Restaurant("Thai Dishes", "Thai", "334-4433", "Mee Krob", 12.50),
    Restaurant("Nobu", "Japanese", "335-4433", "Natto Temaki", 5.50),
    Restaurant("Nonna", "Italian", "355-4433", "Stracotto", 25.50),
    Restaurant("Jitlada", "Thai", "324-4433", "Paht Woon Sen", 15.50),
    Restaurant("Nola", "New Orleans", "336-4433", "Jambalaya", 5.50),
    Restaurant("Noma", "Modern Danish", "337-4433", "Birch Sap", 35.50),
    Restaurant("Addis Ababa", "Ethiopian", "337-4453", "Yesiga Tibs", 10.50) ]

def restaurant_price(restaurant: Restaurant) -> float:
    return restaurant.price
assert restaurant_price(Restaurant("Asdf", "asdf", "asdf", "asdf", 5.00)) == 5.00

for r in RC:
    print (restaurant_price(r))

def costliest(restaurant: Restaurant) -> str:
    restaurant.sort(key = restaurant_price)
    restaurant.reverse()
    return restaurant[0].name
print(costliest(RC))
assert costliest(RC) == 'Noma'

#d.4
def costliest2(restaurant: Restaurant) -> str: 
    return sorted(restaurant, key = restaurant_price, reverse = True)[0].name
print(costliest2(RC))
assert costliest2(RC) == 'Noma'

        pre2000.append(book)

    sumPre = sumPre + book.instock

def inventory_value(book: Book)-> float:
    return book.price * book.instock
def top_value(BL: Book)-> Book:
    return sorted(BL, key = inventory_value, reverse = True)[0]
print("The highest-value book is ", top_value(BSI).title, " by ", top_value(BSI).author," at a value of $ ", "%.2f" % inventory_value(top_value(BSI)),".", sep='')

def draw_eye(x1: int, y1: int, x2: int, y2: int):
    my_canvas3.create_oval(x1, y1, x2, y2, fill = 'white')
    my_canvas3.create_oval(x1 * 1.5, y1 * 1.5, x2 * .75, y2 * .75, fill = 'brown')
    #my_canvas3.create_oval(x1 * 1.75 + (.25 * x1), y1 * 1.75 + ((1/12) * y1), x2 * .5 + (.25 * x2), y2 * .5 + ((1/12) * y2), fill = 'black')

    my_canvas3.create_oval(x1 + 300, y1, x2 + 300, y2, fill = 'white')
    my_canvas3.create_oval(x1 * 1.5 + 300, y1 * 1.5, x2 * .75 + 300, y2 * .75, fill = 'brown')
    #my_canvas3.create_oval(x1 * 1.75 + 300 + (.25 * x1), y1 * 1.75 + (.1 * y1), x2 * .5 + 300 (.25 * x2), y2 * .5 + (.1 * y2), fill = 'black')

def draw_mouth(x1: int, y1: int, x2: int, y2: int):
    my_canvas3.create_oval((x2 + x1 + 300)/2 - (x1 * .5), (x2 + x1 + 300)/2 - (x1 * .5), (x2 + x1 + 300)/2 + (x1 * .5), (x2 + x1 + 300)/2 - (x1 * .5))

def draw_nose(x1: int, y1: int, x2: int, y2: int):
    my_canvas3.create_oval((x2 + x1 + 300)/2 - (x1 * .25), (x2 + x1 + 300)/2 - (x1 * .25) - (y1 * .5), (x2 + x1 + 300)/2 + (x1 * .25), (x2 + x1 + 300)/2 + (x1 * .25) - (y2 * .5), fill = "blue")
    
def draw_face(x1: int, y1: int, x2: int, y2: int):
    draw_eye(x1, y1, x2, y2)
    draw_mouth(x1, y1, x2, y2)
    draw_nose(x1, y1, x2, y2)
    
draw_face(100, 100, 300, 300)
