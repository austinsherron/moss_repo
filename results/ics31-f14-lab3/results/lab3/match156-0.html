def calculate_shipping(p: float) -> float:
    rate=0
    if(p<2):
        rate=2.00
    elif(p>=2 and p<10):
        rate=5.00
    elif(p>=10):
        rate=5.00+((int(p)-10)*1.5)
    return rate

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

p = float(input('Enter number of pounds'))
print(calculate_shipping(p))

def drawFace(x: int, y: int):
    drawEyes(x, y)
    drawNose(x + 115, y + 50)
    drawMouth(x + 60, y + 150)

def drawEyes(x: int, y: int):
    p = x + 70
    q = y + 70
    my_canvas.create_oval(x, y, p, q)
    my_canvas.create_oval(x + 15, y + 15, p - 15, q - 15, fill = 'blue')
    my_canvas.create_oval(x + 25, y + 25, p - 25, q - 25, fill = 'black')

    x += 150
    p = x + 70
    my_canvas.create_oval(x, y, p, q)
    my_canvas.create_oval(x + 15, y + 15, p - 15, q - 15, fill = 'blue')
    my_canvas.create_oval(x + 25, y + 25, p - 25, q - 25, fill = 'black')

def drawNose(x: int, y: int):
    p = x - 25
    q = y + 50
    my_canvas.create_line(x, y, p, q)
    my_canvas.create_line(p, q, x, q)

def drawMouth(x: int, y: int):
    p = x + 100
    my_canvas.create_line(
        x - 50, y - 25, x - 25, y,
        x, y, p, y, 
        p + 25, y, p + 50, y - 25, smooth = "true")

drawFace(x, y)
drawFace(x2, y2)
