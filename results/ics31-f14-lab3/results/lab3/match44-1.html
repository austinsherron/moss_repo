def inventory_value (b: Book) -> int:
    return (b.price*b.instock)

def top_value (listBooks: list) -> Book:
    highest = inventory_value(listBooks[0])
    valueBook = listBooks[0]
    for i in range(len(listBooks)-1):
        if(highest < inventory_value(listBooks[i]) ):
            highest = inventory_value(listBooks[i])
            valueBook = listBooks[i]
    return valueBook

highestValueBook = top_value(BSI)
print("The highest value book is ", highestValueBook.title, " by ",
      highestValueBook.author, " at a value of $",
      inventory_value(highestValueBook),sep="")

def draw_eye(x: int, y: int, c: str):
    # Outline eyes   
    my_canvas.create_oval(x-25,y-10,x+25,y+10,fill="white")
    # Draw irises
    my_canvas.create_oval(x-10,y-10,x+10,y+10,fill=c)
    # Draw pupils
    my_canvas.create_oval(x-6,y-6,x+6,y+6,fill="black")

def draw_nose(x:int, y:int,c:str):
    my_canvas.create_line(x,y,x,y+100,fill=c)

def draw_mouth(x:int, y:int, c:str):
    my_canvas.create_oval(x-50,y+125,x+50,y+150,fill=c)
    my_canvas.create_line(x-50,y+137,x+50,y+137)
    
def draw_face(x:int, y:int, c: str):
    draw_nose(x,y,c)
    draw_mouth(x,y,c)
    draw_eye(x-50,y,c)
    draw_eye(x+50,y,c)


draw_face(200,200,"red")

draw_face(500,500,"black")
