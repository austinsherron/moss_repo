def calculate_shipping (weight: float) -> float:
    ''' Calculates shipping rate given the weight'''
    if weight < 2:
        return 2.00
    elif 2 <= weight < 10:
        return 5.00
    else: 
        return 5.00 + 1.50 * (weight - 10)
assert calculate_shipping(1.5) == 2.00

def create_square (x_coor: int, y_coor: int, length: int) -> tkinter:
    '''Given the coordinates of the top-left point and length of a side'''
    '''a picture with those characteristics will be provided as output'''
    my_window = tkinter.Tk()    # Create the graphics window

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window

    my_canvas.create_rectangle(x_coor, y_coor, x_coor + length, y_coor + length, fill='') # Draw orange line

    tkinter.mainloop()          # Combine all the elements and display the window

    '''Returns the inventory value given a namedtuple by multiplying the number'''
    '''in stock with the price'''
    return book.instock * book.price

def top_value(list_of_books: list) -> list:
    '''Given a list of Books, this function will loop through the list searching'''
    '''for the Book with the greatest inventory price'''
    a = inventory_value(list_of_books [0])
    b = list_of_books[0]
    for i in range(len(list_of_books)):
        if inventory_value (list_of_books[i]) > a:
             b = list_of_books[i]
             a = inventory_value (list_of_books[i])
    return b

def draw_eye (x_cor: int, y_cord: int, canvas: tkinter, color: str) -> tkinter:
    '''given the x/y shift as inputs, window and desired eye color, this function produces a
    single eye with those characteristics'''
    canvas.create_oval(13+x_cor, 26+y_cord, 88+x_cor, 76+y_cord, fill='white')
    canvas.create_oval(33+x_cor, 40+y_cord, 70+x_cor, 69+y_cord, fill=color)
    canvas.create_oval(42+x_cor, 49+y_cord, 62+x_cor, 60+y_cord, fill='black')
    
def draw_nose (x_cor: int, y_cord: int, canvas: tkinter) -> tkinter:
    '''given the x/shift as inputs, the output is a nose that moves accordingly in place of the
    face'''
    canvas.create_line(102+x_cor, 80+y_cord, 85+x_cor, 100+y_cord, fill='black')
    canvas.create_line(85+x_cor, 100+y_cord, 102+x_cor, 100+y_cord, fill='black')
    
def draw_mouth (x_cor: int, y_cord: int, canvas: tkinter) -> tkinter:
    '''given the x/shift as inputs, the output is a mouth that moves accordingly in place of the
    face'''
    canvas.create_line(62+x_cor, 125+y_cord, 62+x_cor, 135+y_cord, fill='black')
    canvas.create_line(137+x_cor, 125+y_cord, 137+x_cor, 135+y_cord, fill='black')
    canvas.create_line(62+x_cor, 135+y_cord, 137+x_cor, 135+y_cord, fill='black')

def draw_face (x_cor: int, y_cord: int, color: str) -> tkinter:
    '''given the x/shift as inputs as well as eye color, the output is an entire face that
    moves accordingly within the constructed graphics window'''
    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=700, height=700)  
    my_canvas.pack() 
    my_canvas.create_oval(0+x_cor, 0+y_cord, 200+x_cor, 170+y_cord, fill='#FFE2C6')
    draw_eye (x_cor, y_cord, my_canvas, color)
    draw_eye (x_cor + 100, y_cord, my_canvas, color)
    draw_nose (x_cor, y_cord, my_canvas)
    draw_mouth (x_cor, y_cord, my_canvas)
    tkinter.mainloop()          # Combine all the elements and display the window
    
draw_face (50, 50, 'brown')

draw_face (150, 250, 'blue')

draw_face (350, 50, 'green')
