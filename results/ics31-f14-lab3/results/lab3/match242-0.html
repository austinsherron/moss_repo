def calculate_shipping(p:float)->float:
    '''Calculates shipping price based on weight p'''
    if p < 2.0:
        return 2.00
    if p >= 2 and p < 10:
        return 5.00
    if p > 10:
        return 5.00 + (p - 10)*1.5
assert calculate_shipping(1.5) == 2.00
print('1.5: ', calculate_shipping(1.5))
assert calculate_shipping(7) == 5.00
print('7: ', calculate_shipping(7))
assert calculate_shipping(15) == 12.50
