def calculate_shipping(weight: float) -> float:
    '''returns the cost of shipping based on the weight input'''
    if weight < 2:
        return 2.0
    elif weight >= 2 and weight < 10:
        return 5.0
    else:
        return 5.0 + 1.5*(weight - 10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
#test values
print('The shipping cost of a 1 pound package is', calculate_shipping(1))
print('The shipping cost of a 2 pound package is', calculate_shipping(2))
print('The shipping cost of a 11 pound package is', calculate_shipping(11))

       before 2000 is greater than or less than the number
       of books published after 2000'''
    before_2000_BSI = []
    after_2000_BSI = []
    for book in l:


        if book.year < 2000:
            before_2000_BSI.append(book.title)
        else:
            after_2000_BSI.append(book.title)
    if len(before_2000_BSI) > len(after_2000_BSI):
        print('More titles before 2000 (', len(before_2000_BSI),
              'vs.', len(after_2000_BSI), ')')
    else:
        print('More titles 2000 and after (', len(after_2000_BSI),
              'vs.', len(before_2000_BSI), ')')

def inventory_value(book: Book) -> float:
    ''' returns the inventory value
        (price * # in stock) of the given book'''
    return book.price * book.instock

def top_value(lst: list) -> Book:
    ''' sorts the given list by inventory value
        and returns the highest valued book'''
    lst.sort(key = inventory_value)
    return lst[-1]

print('The highest-value book is {} by {} at a value of {}'.format(
    top_value(BSI).title, top_value(BSI).author,
    inventory_value(top_value(BSI))))

my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width = 1000, height = 1000)
my_canvas.pack()

def draw_eye(x: int, y: int):
    '''draws an eye of dimensions 200 x 100 with topleft corner at (x, y)'''
    my_canvas.create_oval(x, y, x + 200, y + 100, fill = 'white')
    my_canvas.create_oval(x + 50, y + 25, x + 150, y + 75, fill = 'light blue')
    my_canvas.create_oval(x + 75, y + 37.5, x + 100, y + 52.5, fill = 'black')

def draw_nose(slantX: int, slantY: int, windowX: int, windowY: int):
    '''draws a nose of width 50 and height 100, starting from the
       middle of the face'''
    my_canvas.create_line(windowX + 250, windowY + 200,
                          windowX + 250 - slantX,
                          windowY + 200 + slantY)
    my_canvas.create_line(windowX + 250 - slantX,
                          windowY + 200 + slantY,
                          windowX + 250, windowY + 200 + slantY)

def draw_mouth(xLength: int, yLength: int, windowX: int, windowY: int):
    '''draws a mouth of width xLength * 2 and height yLength * 2,
       starting from (windowX + 250, windowY + 400)'''
    my_canvas.create_oval(windowX + 250 - xLength, windowY + 400 - yLength,
                          windowX + 250 + xLength,
                          windowY + 400 + yLength, fill = 'black')
def draw_face(windowX: int, windowY: int):
    '''draws a face with top-left corner starting from (windowX, windowY)'''
    my_canvas.create_oval(windowX, windowY, windowX + 500, windowY + 500, fill = 'yellow')
    draw_eye(windowX + 50, windowY + 100)
    draw_eye(windowX + 250, windowY + 100)
    draw_nose(50, 100, windowX, windowY)
    draw_mouth(100, 50, windowX, windowY)
    
draw_face(0, 0)
