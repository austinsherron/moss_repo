#c.5 - Function 'calculate_shipping' takes weight of package, returns price of shipping



def calculate_shipping(weight: float) -> float:
    "Calculate shipping cost based on weight of package"
    if weight < 2:
        rate = 2.00
    elif 2 <= weight < 10:
        rate = 5.00
    elif weight >= 10:
        rate = 5.00 + (weight - 10) * 1.50 
    return rate

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

#e.6 - Function 'inventory_value' takes a Book, and returns the value of its inventory.
#      Function top_value takes a list of Books, returns the Book object with highest invntory value

def inventory_value (b: Book) -> float:
    "Takes a book, and returns the value of its inventory (price x number in stock)"
    return b.price * b.instock

print ("Inventory value of book = ", inventory_value(Book("Gregory Maguire", "Wicked", "Fantasy", 1996, 14.89, 1000))) #Checking - function call   


def top_value (book_list: list) -> Book:
    "Takes a list of books, and returns the Book with highest inventory value"

    Sorted_list = sorted(book_list, key = inventory_value, reverse = False)

    for b in Sorted_list:
        highest_value_inventory  = b
    
    return highest_value_inventory


top = top_value (BSI)

print ("The highest-value book is", top.title, "by", top.author, "at a value  $", inventory_value(top))
