def calculate_shipping (weight: float):
    'Shipping rates based on weight'
    if weight < 2:
        return (2.00)
    if 2 <= weight < 10:
        return (5.00)
    if weight >= 10:
        return (5.00 + 1.5 * (weight - 10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print('PRICE OF SHIPPING WEIGHT OF 1.5 POUNDS')
print('$',calculate_shipping(1.5))
