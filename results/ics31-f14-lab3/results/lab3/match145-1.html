def calculate_shipping(n) ->float:
    "calculate shipping price"
    if n<2:
        return 2
    if 2<=n<10:
        return 5
    if n>10:
        return 5+(n-10)*1.5
assert calculate_shipping(1.5)==2.00

import tkinter              

my_window = tkinter.Tk()    

my_canvas = tkinter.Canvas(my_window, width=5000, height=5000)  
my_canvas.pack()            





def draw_nose(x,y):
    'draw a nose'
    my_canvas.create_line(x,y,x+50,y+50)
    my_canvas.create_line(x,y+50,x+50,y+50)


def draw_mouth(x,y):
    'draw a mouth'
    my_canvas.create_line(x-40,y+90,x+90,y+90)
    my_canvas.create_line(x-40,y+100,x+90,y+100)
    my_canvas.create_line(x-40,y+90,x-40,y+100)
    my_canvas.create_line(x+90,y+90,x+90,y+100)

def draw_eye(x,y):
    'draw an eye'
    
    my_canvas.create_oval(x-30,y-30,x-90,y-90, fill='white')
    my_canvas.create_oval(x-38,y-38,x-82,y-82, fill='blue')
    my_canvas.create_oval(x-50,y-50,x-70,y-70, fill='black')

    my_canvas.create_oval(x+30,y-30,x+90,y-90, fill='white')
    my_canvas.create_oval(x+38,y-38,x+82,y-82, fill='blue')
    my_canvas.create_oval(x+50,y-50,x+70,y-70, fill='black')   

def draw_head(x,y):
    'draw an outline around features'
    my_canvas.create_oval(x-150,y-150,x+150,y+150)



def draw_face(x,y):
    'draw the entire face at once'
    draw_nose(x,y)
    draw_mouth(x,y)
    draw_eye(x,y)
    draw_head(x,y)

draw_face(500,500)
draw_face(200,200)
