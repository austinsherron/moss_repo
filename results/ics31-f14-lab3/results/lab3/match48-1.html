#calculate_shipping: W: weight of package ->float
"Take the weight of a package and calculate the cost of shipping."
def calculate_shipping(W:float)->float:
    if W< 2:
        cost = 2.00
    elif 10 > W >= 2:
        cost = 5.00
    elif W >= 10:
        cost = 5 + (W - 10) * 1.50 
    return cost
W = float(input('What is the weight of your package?'))
print(calculate_shipping(W))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5.00
