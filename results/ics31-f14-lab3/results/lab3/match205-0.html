    ''' Return the area of the circle with radius of the parameter '''
    radius = 3.14159 * (radius ** 2)
    return radius

def print_even_numbers(num_list: list) -> str:
    ''' Print all the even numbers in the list '''
    if len(num_list) == 1:
        if num_list[0] % 2 == 0:
            print(num_list[0])
    else:
        for num in num_list:
            if num % 2 == 0:
                print(num)

print_even_numbers([1])
print_even_numbers([4])
print_even_numbers([1, 2, 8, 100, 101])

def calculate_shipping(weight: int) -> float:
    ''' Return shipping price based on weight '''
    float(weight)
    if weight < 2.00:
        return 2.00
    elif weight >= 2.00 and weight < 10.00:
        return 5.00
    else:
        return 5.00 + (weight - 10) * 1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def create_square(x: int, y: int, length: int):
    ''' Create a square with starting coordinates and length of each side '''
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width = x * 2 + length, height = y * 2 + length)
    my_canvas.pack()

my_window = tkinter.Tk()

def draw_eye(x: int, y: int):
    my_canvas.create_oval(x, y, x + 100, y + 50, fill='white')


    my_canvas.create_oval(x + 25, y, x + 75, y + 50, fill='red')
    my_canvas.create_oval(x + 40, y + 15, x + 60, y + 35, fill='black')
    return
def draw_nose(x: int, y: int):
    my_canvas.create_polygon(x, y, x - 20, y + 50, x + 20, y + 50, fill='brown')
    return
def draw_mouth(x: int, y: int):
    my_canvas.create_polygon(x, y, x + 50, y + 20, x + 100, y, fill='pink')
    return
def draw_face(x: int, y: int):
    my_canvas.create_oval(x, y, x + 500, y + 400, fill='tan')
    draw_eye(x + 100, y + 100)
    draw_eye(x + 300, y + 100)
    draw_nose(x + 250, x + 200)
    draw_mouth(x + 200, x + 300)
draw_face(10, 10)
