import tkinter 
my_window = tkinter.Tk()   

my_canvas = tkinter.Canvas(my_window, width=1500, height=1500)
my_canvas.pack()

def draw_face(x:int,y:int,xl:int,yl:int):
    my_canvas.create_oval(x-xl,y-xl,x+xl,x+yl)
    return

def draw_eye(x:int,y:int,size:int):
    my_canvas.create_oval(x-70*size,y-25*size,x+70*size, y+25*size,width=2, fill='white')
    my_canvas.create_oval(x-25*size,y-25*size,x+25*size, y+25*size,width=2, fill='black')
    return

def draw_nose(x:int,y:int,size:int):
    my_canvas.create_oval(x-15*size,y-30*size,x+15*size,y+30*size)
    my_canvas.create_oval(x-13*size,y+12*size,x-2*size,y+22*size, fill= 'black')
    my_canvas.create_oval(x+2*size,y+12*size,x+13*size,y+22*size, fill= 'black')
    return
def draw_mouth(x:int,y:int,size:int):
    my_canvas.create_line(x-20*size,y-10*size,x,y)
    my_canvas.create_line(x,y,x+20*size,y-10*size)
    my_canvas.create_line(x-20*size,y-10*size,x,y+10*size)
    my_canvas.create_line(x+20*size,y-10*size,x,y+10*size)
    return

draw_face(200,130,100,100)
draw_eye(150,100,0.3)
draw_eye(250,100,0.3)
draw_nose(200,170,1)
draw_mouth(200,250,2)



 
tkinter.mainloop()
