def calculate_shipping(weight):
    "Takes in integer variable and returns set price"
    if weight < 2:
        price = 2.00
    elif weight < 10 and weight > 2:
        price = 5.00
    else:
        price = 5 + ((weight - 10) * 1.5)
    return price

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter

my_window = tkinter.Tk()
my_face = tkinter.Canvas(my_window, width=500, height=500)
my_face.pack()

def draw_eye(x,y):
    my_face.create_oval(x, y, x+150, y+70, fill='white')
    my_face.create_oval(x+40, y, x+110, y+70, fill='black')
" A function that takes in 2 integer values to create an eye " 
    
def draw_nose(x,y):
    my_face.create_line(x, y, x+80, y + 70)
    my_face.create_line(x, y+70, x+80, y +70)
" A function that takes in 2 integer values to create a nose "

def draw_mouth(x,y):
    my_face.create_oval(x, y, x+230, y+20, fill='red')
" A function that takes in 2 integer values to create a mouth "

def draw_face():
    draw_eye(50,50)
    draw_eye(250,50)
    draw_nose(200, 150)
    draw_mouth(120, 250)
" A function that, with predefined integers, calls on all functions to draw a face "

draw_face()
tkinter.mainloop()
