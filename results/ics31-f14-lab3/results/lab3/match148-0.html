def calculate_shipping(weight:float) -> float:
    #Calculates the shipping cost from weight
    if weight < 2:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    else:
        rate = 5 +((weight-10)*1.50)
        return rate

my_window_face = tkinter.Tk()

my_canvas_face= tkinter.Canvas(my_window_face, width=900, height=900)
my_canvas_face.pack()
def draw_eye(x, y, z, n):
    '''Draws an eye onto a canvas'''
    return (my_canvas_face.create_oval (x, y, z+140, n+77, fill = 'white'),                 #Draws the eye
    my_canvas_face.create_oval (x+75, y, z+90, n+77, fill = 'blue'),
    my_canvas_face.create_oval (x+140, y+77, z, n, fill = 'black'))
def draw_nose(x, y, z, n):
    '''Draws a nose onto a tkinter canvas'''
    (my_canvas_face.create_line(x, y, x, y+z))
    (my_canvas_face.create_line(x, y+z, x-n, y))
    (my_canvas_face.create_line(x-n, y, x, y))
def draw_mouth(x, y, z, n):
    '''Draws a mouth onto a tkinter canvas'''
    (my_canvas_face.create_arc(x, y, z, n))

def draw_face(x1, y1, z1, n1, x2, y2, z2, n2, x3, y3, z3, n3):
    '''Draws an entire face onto the tkinter canvas'''
    draw_eye(x1, y1, z1, n1)
    draw_eye(100+x1, y1, 100+z1, n1)
    draw_nose(x2, y2, z2, n2)
    draw_mouth(x3, y3, z3, n3)

draw_eye(100,200,300,400)
