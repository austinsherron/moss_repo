def calculate_shipping(weight: float) -> float:
    ''' Returns the price of shipping given the weight of the package '''
    if weight < 2:
        return 2.0
    elif weight >= 10:
        return 5.0 + (weight-10)*1.5
    else: 
        return 5.0 
assert calculate_shipping(2) == 5.0
assert calculate_shipping(1.5) == 2.0


assert calculate_shipping(7) == 5.0
assert calculate_shipping(10) == 5.0
assert calculate_shipping(15) == 12.5

my_window = tkinter.Tk()    
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()  

def draw_face(x1, y1, x2, y2, eye_fill):
    ''' Draws a face where (x1,y1) is the top left and (x2,y2) is the bottom right with eye fill color '''           
    x_length = x2 - x1 # Length on x-coordinate
    y_length = y2 - y1 # Length on y-coordinate
    my_canvas.create_oval(x1, y1, x2, y2, fill='yellow')

    draw_eye(x1 + x_length/10, y1 + y_length/5,
             x1 + 2*x_length/5, y1 + 2*y_length/5, eye_fill) # Draws the first eye
    draw_eye(x1 + 3*x_length/5, y1 + y_length/5,
             x2 - x_length/10, y1 + 2*y_length/5, eye_fill) # Draws the second eye
    draw_nose(x1 + 2*x_length/5, y1 + 2*y_length/5,
              x1 + 3*x_length/5, y1 + 3*y_length/5) # Draws the nose
    draw_mouth(x1 + 3*x_length/10, y2 - 3*y_length/10,
               x2 - 3*x_length/10, y2 - y_length/10) # Draws the mouth
   
def draw_eye(x1, y1, x2, y2, eye_fill):
    ''' Draws an eye where (x1,y1) is the top left and (x2,y2) is the bottom right ''' 
    x_length = x2 - x1
    y_length = y2 - y1
    my_canvas.create_oval(x1, y1, x2, y2, fill ='white') # Outer oval
    my_canvas.create_oval(x1 + x_length/8, y1, x2 - x_length/8, y2, fill=eye_fill) # Iris
    my_canvas.create_oval(x1 + x_length/3, y1 + y_length/3,
                          x2 - x_length/3, y2 - y_length/3, fill ='black') # Pupil
   
def draw_nose(x1, y1, x2, y2):
    ''' Draws an triangle facing up in a square with corners (x1,y1) and (x2,y2) '''
    x_length = x2 - x1
    my_canvas.create_line(x1, y2, x1 + x_length/2, y1)
    my_canvas.create_line(x2, y2, x1 + x_length/2, y1)
    my_canvas.create_line(x1, y2, x2 , y2)

def draw_mouth(x1, y1, x2, y2):
    ''' Draws an oval mouth inscribed in the rectangle with corners (x1,y1) and (x2,y2) '''
    my_canvas.create_oval(x1, y1, x2, y2, fill='white')

draw_face(50,50,300,400, 'red')
draw_face(400,350,700, 800, 'brown')

tkinter.mainloop()
