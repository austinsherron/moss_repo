def calculate_shipping(pound: int) -> int:
    """A handy tool for calculating shipping cost given weight"""
    if pound < 2:
        dollar = 2.00
    elif 2 <= pound < 10:
        dollar = 5.00
    elif pound >= 10.00:
        dollar = 5 + (pound-10)*1.50
    return dollar
    

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book_entry: Book) -> float:
    '''Takes a Book and returns the inventory value of the list'''
    return book_entry.price*book_entry.instock

assert(inventory_value(new_books[0])) == 108.07499999999999

def top_value(booklist: list) -> Book:
    '''Takes a list of Books and
    returns the Book with the 
    most valuable inventory'''
    top_valued_book = 0

    for book_entry in booklist:
        if top_valued_book < inventory_value(book_entry):
            top_valued_book = inventory_value(book_entry)
            
    for book_entry in booklist:
        if top_valued_book == inventory_value(book_entry):
            return book_entry

print((top_value(BSI)))

def draw_eye(x1: int, y1: int) -> None:
    '''Takes two coordinates and creates an eye with top left at (x1,y1)'''
    my_canvas.create_oval(x1, y1, x1+150, y1+100, fill = 'white') #EYE BORDER
    my_canvas.create_oval(x1+30, y1+5, x1+120, y1+95, fill = 'teal') #IRIS
    my_canvas.create_oval(x1+60, y1+35, x1+90, y1+65, fill = 'black') #PUPIL

def draw_mouth(x1: int, y1: int) -> None:
    '''Takes two coordinates and creates a mouth with top left at (x1,y1)'''
    my_canvas.create_line(x1, y1, x1+300, y1, fill = 'red') #MOUTH

def draw_nose(x1: int, y1: int) -> None:
    '''Takes two coordinates and creates a nose with top left at (x1,y1)'''
    my_canvas.create_arc(x1, y1, x1+300, y1+100, fill = 'orange')

def draw_face(x1: int, y1: int) -> None:
    '''Takes two coordinates and draws a face (with eyes, mouth, nose, head) with top left at (x1,y1)'''
