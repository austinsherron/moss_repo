w = float(input('How heavy is the package? '))
def calculate_shipping(w: float) -> float:
    if w < 2:
        result = 2.00
    elif 2 <= w < 10:
        result = 5.00
    elif w >= 10:
        result = 5 + (1.5 *(w-10))
    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print('Shipping would cost: $', calculate_shipping(w), '\n')
