def calculate_shipping(weight:int) -> int: #C5
    '''takes the weight of the package and calculates shipping cost'''
    if weight < 2:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    else:
        x=(5 + (weight - 10)* 1.5)
        return x

def book_stock(BSI):
    bt=[]
    at=[]
    binstock = 0
    ainstock = 0
    for a in BSI:
        if float(a.year) >= 2000:
            at.append(a)
        elif float(a.year)< 2000:
            bt.append(a)
    for b in bt:
        binstock += b.instock
    for b in at:
        ainstock += a.instock
    print("More titles before 2000",binstock, 'vs.', ainstock)
