def calculate_shipping(n: float) -> float:
    '''Takes weight of a package and returns shipping price'''
    if n < 2:
        return(2.00)
    elif 2 <= n < 10:
        return(5.00)
    elif 10 <= n:
        return(((n-10) * 1.50) + 5.00)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter              # Load the library; do this just once per program
my_window = tkinter.Tk()    # Create the graphics window


my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

def draw_eye(StartingX, StartingY):
    my_canvas.create_oval(StartingX,StartingY,StartingX+200,StartingY+100, fill = 'white')
    my_canvas.create_oval(StartingX+50,StartingY,StartingX+150,StartingY+100, fill = 'blue')
    my_canvas.create_oval(StartingX+75,StartingY+25,StartingX+125,StartingY+75, fill = 'black')
def draw_nose(StartingX,StartingY):
    my_canvas.create_line(StartingX,StartingY,StartingX-50,StartingY+100)
    my_canvas.create_line(StartingX-50,StartingY+100,StartingX,StartingY+100)
def draw_mouth(StartingX,StartingY):
    my_canvas.create_line(StartingX,StartingY,StartingX+300,StartingY)


def draw_face(X,Y):
    draw_eye(X,Y)
    draw_eye(X+250,Y)
    draw_nose(X+225,Y+75)
    draw_mouth(X+75,Y+225)

draw_face(25,25)
tkinter.mainloop()
