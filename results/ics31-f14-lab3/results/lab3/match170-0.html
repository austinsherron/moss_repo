def abbreviate(name: str) -> str:
    "Return an abbreviation of the parameter"
    if(len(name) > 3):
        return name[0:3]
    else:
        return name
    assert abbreviate('January') == 'Jan'
    assert abbreviate('abril') == 'abr'
    assert abbreviate('abbreviate') == 'abb'

    "Returns the area of a square given length"
    return length ** 2
    assert find_area_square(0) == 0
    assert find_area_square(1) == 1
    assert find_area_square(5) == 25

    Restaurant("Nola", "New Orleans", "336-4433", "Jambalaya", 5.50),
    Restaurant("Noma", "Modern Danish", "337-4433", "Birch Sap", 35.50),
    Restaurant("Addis Ababa", "Ethiopian", "337-4453", "Yesiga Tibs", 10.50) ]

def restaurant_price(my_restaurant: Restaurant) -> float:
    "returns the price given a restaurant"
    return my_restaurant.price

    assert restaurant_price(RL[1]) == 5.50
    assert restaurant_price(RL[2]) == 25.50

def costliest(restaurants: [Restaurant]) -> str:
    'returns the costliest restaurant in a list'
    restaurants.sort(key=restaurant_price)
    return(restaurants[-1].name)
    assert costliest(RL) == 'Noma'

    'returns the costliest restaurants in a list without changing the list'

def inventory_value(book: Book) -> float:
    'returns the price times the number of books in stock'
    return book.price * book.instock
    assert inventory_value(Book('Return of Sherlock Holmes', 'Arthur Conan Doyle', 'action', 1905, 25.00, 20)) == 500
def top_value(books: [Book]) -> Book:
    'returns the book with the highsest inventory value'
    return sorted(books, key=inventory_value)[-1]
