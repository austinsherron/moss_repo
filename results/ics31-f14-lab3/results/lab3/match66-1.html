def calculate_shipping(weight):
    if weight < 2:
        weight = 2.00
    elif 10 > weight >= 2:
        weight = 5.00
    elif weight >= 10:
        weight = 5.00 + ((weight - 10) * 1.50)
    return(weight)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eyes(x, y):
    my_canvas.create_oval(x, y, x + 200, y + 100, fill = 'white')
    my_canvas.create_oval(x + 70, y + 30, x + 130, y + 70, fill = 'blue' )
    my_canvas.create_oval(x + 90, y + 40, x + 110, y + 60, fill = 'black')
    

def draw_mouth(x, y):
    my_canvas.create_oval(x, y, x + 150, y + 90, fill = 'black')

def draw_nose(x, y):
    my_canvas.create_line(x - 25, y - 25, x, y)
    my_canvas.create_line(x, y, x + 25, y - 25)

def draw_face(x, y):
    my_canvas.create_oval(x - 300, y - 300, x + 300, y + 300, fill = 'yellow')
    draw_eyes(x - 230, y - 200)
    draw_eyes(x + 30, y - 200)
    draw_mouth(x - 70, y + 100)
    draw_nose(x, y)

draw_face(600, 400)
