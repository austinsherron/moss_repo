def print_even_numbers (s: list) -> list:
    "Input is a list of numbers, returns the even numbers in a list."
    a = []
    i = 0
    for i in range(len(s)):
        if (s[i] % 2 == 0):
            a.append(s[i])
    return a

def calculate_shipping (x: float) -> float:
    "Input is the weight of package, returns price for shipping the package"
    if (x < 2):
        return 2.00
    elif (x < 10):
        return 5.00
    else:
        return 5 + (x - 10) * 1.5
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(100))

def inventory_value (b: namedtuple) -> float:
    "Takes a book and returns the inventory times the price."
    return b.instock * b.price

    "Returns the most expensive book from the list of books."
