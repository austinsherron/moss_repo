def calculate_shipping(value: float) -> float:
    if value < 2:
        return 2.00
    else:
        if value < 10:
            return 5.00
        else:
            return 5.00+(value-10)*1.50

import tkinter              

my_window = tkinter.Tk()   

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
my_canvas.pack()            

def draw_eye(eyex: int, eyey: int):
    my_canvas.create_oval(eyex-20, eyey, eyex+40, eyey+20, fill='beige')
    my_canvas.create_oval(eyex+3, eyey+3, eyex+17, eyey+17, fill='saddlebrown', activefill='black')
    
def draw_nose(nosex: int, nosey: int):
    my_canvas.create_polygon(nosex, nosey, nosex-20, nosey+40, nosex+20, nosey+40, fill='khaki', activefill='black')

def draw_mouth(x: int, y: int):
    my_canvas.create_oval(x, y, x+80, y+20, fill='red', activefill='black')

def draw_face(facex: int, facey: int):
    my_canvas.create_oval(facex, facey, facex+280, facey+290, fill='LightCoral')
    draw_eye(180, 210)
    draw_eye(300, 210)
    draw_nose(250, 260)
    draw_mouth(210, 330)
 
draw_face(110, 110)

tkinter.mainloop()
