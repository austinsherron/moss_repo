def calculate_shipping(w: float)->float:
    if w<2:
        return 2.00
    if 2<=w and w<10:
        return 5.00
    if w>=10:
        return 5.00+int((w-10))*1.50

def before_2000(books: list)->list:
    l=[]
    for x in range(len(books)):
        if books[x].year<2000:
            l.append(books[x])
    return l
def after_2000(books: list)->list:
    l=[]
    for x in range(len(books)):
        if books[x].year>=2000:
            l.append(books[x])
    return l

def numBooks(books: list)->int:
    count=0
    for x in range(len(books)):
