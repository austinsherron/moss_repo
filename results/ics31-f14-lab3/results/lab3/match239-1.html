def calculate_shipping(z:float) -> float:
    if z < 2:
        return 2.00
    elif z >= 2 and z < 10:
        return 5.00
    elif z >= 10:
        return 5.00 + ((z-10)*1.50)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print (calculate_shipping(1.5))
print (calculate_shipping(7))
print (calculate_shipping(15))

def draw_eye(a:int, b:int):
    my_canvas.create_oval(a-75, b-50, a+75, b+50, fill='white')
    my_canvas.create_oval(a-30, b-30, a+30, b+30, fill='blue')
    my_canvas.create_oval(a-10, b-10, a+10, b+10, fill='black')
    return

def draw_nose(a:int, b:int):
    my_canvas.create_line(a,b,a,b+50)
    my_canvas.create_line(a,b+50,a-40,b+90)
    my_canvas.create_line(a-40,b+90,a,b+90)
    return

def draw_mouth(a:int, b:int):
    my_canvas.create_oval(a,b,a+140,b+40)
    my_canvas.create_line(a,b+20,a+140,b+20)
    return

def draw_face(a:int, b:int):
    draw_eye(a-100, b-150)
    draw_eye(a+100, b-150)
    draw_nose(a, b-80)
    draw_mouth(a-100,b+110)
    my_canvas.create_oval(a-240,b-280,a+240,b+250)
    return


my_window = tkinter.Tk()    

my_canvas = tkinter.Canvas(my_window, width=600, height=600)
my_canvas.pack()        

draw_face(300,300)
tkinter.mainloop()
