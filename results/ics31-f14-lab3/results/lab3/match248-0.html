def eyz (x: int, y: int, z:int):
    '''eye at x,y with size z'''
    my_canvas.create_oval(x,y,x+z,y+z,fill ='blue')
    my_canvas.create_oval(x + int(z/20),y + int(z/20),x + z - int(z/20),y + z - int(z/20),fill = 'white')
    my_canvas.create_oval(x+int(z/10),y+int(z/10),x+z-int(z/10),y+z-int(z/10),fill = 'black')
def draw_mouth (x:int, y:int, z:int):
    '''draws mouth'''
    my_canvas.create_line(x+int(z/3), y+int(3*z/4), x+int(2*z/3), y+int(3*z/4))
def draw_nose (x:int, y:int, z:int):
    '''draws nose'''
    my_canvas.create_line(x+int(z/2), y+int(z/2), x+int(z/2+z/10), y+int(z/2+z/5))
    my_canvas.create_line(x+int(z/2),y+int(z/2+z/5),x+int(z/2+z/10), y+int(z/2+z/5))

def draw_face (x:int, y:int, z:int):
    '''draw face'''
    my_canvas.create_oval(x,y,x+z,y+z,fill ='yellow')
    eyz(x + int(z/3)- int(z/10), y + int(z/3), int(z/5))
    eyz(x + z - int(z/3), y + int(z/3), int(z/5))
    draw_mouth(x,y,z)
    draw_nose(x,y,z)
    
draw_face(100,100,400)
draw_face(500,100,200)
draw_face(500,400,200)
draw_face(750,500,100)
