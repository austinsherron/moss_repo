def calculate_shipping(x):
    if x<2:
        return 2.00
    elif 2<x<10:
        return 5.00
    elif x>10:
        return 5.00 + 1.50*(x-10)

assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
