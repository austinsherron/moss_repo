    "Print the even numbers froma a given list"
    for n in range(len(nums)):

def calculate_shipping(weight: float) -> float:
    'Return price for shipping package'
    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5.00 + (weight - 10) * 1.5
assert calculate_shipping(1) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(12) == 8.00

print(calculate_shipping(15))

def inventory_value(book: Book) -> float:
    'Returns the value of the inventory'
    return book.price * book.instock
