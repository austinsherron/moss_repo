#c.5  function that takes the weight of a package to be shipped then returns the price for shipping the package

def calculate_shipping(weight)->float:
    if weight<2.00:
        return 2.00
    elif 2.00<=weight<10.00:
         return 5.00
    elif 10.00<=weight:
       return 5.00 + (weight-10) * 1.5

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
