def calculate_shipping(weight: float) -> float:
    '''Returns the price for shipping the package given the package weight'''
    LIGHT_PRICE = float(2)
    MEDIUM_PRICE = float(5)
    HEAVY_PRICE = float(5)
    OVERWEIGHT_CHARGE = 1.5
    price = 0
    if (weight < 2):
        price = LIGHT_PRICE
    elif (weight < 10):
        price = MEDIUM_PRICE
    else:
        price = HEAVY_PRICE + OVERWEIGHT_CHARGE*(weight-10)
    return price
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book: Book) -> float:
    '''calculate the inventory value of given book'''
    price = book.price
    num = book.instock
    return price * num
def top_value (booklist: list) -> Book:
    '''list the book with the highest inventory value'''
    newlist = sorted(booklist, key=inventory_value)
    return newlist[-1]
print('The highest-value book is', top_value(BSI).title,
      'by', top_value(BSI).author, 'at a value of',
      inventory_value(top_value(BSI)))
