def calculate_shipping(w:float) -> float:
    ''' Calculates price for shipping package based on package weight '''
    if w < 2:
        return 2.00
    elif w < 10:
        return 5.00
    else:
        return 5.00 + 1.5*(w-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter

def draw_eye (x:int, y:int, d:int, canvas: tkinter.Canvas):
    ''' Draw eyes based off of x and y coordinates and diameter given '''
    canvas.create_oval (x, y, x+0.15*d, y + 0.15*d
                        , fill ='white')
    canvas.create_oval (x+.03*d, y+.03*d, x+0.12*d, y + 0.12*d, fill = 'medium spring green')
    canvas.create_oval (x+.05*d, y+.05*d, x+.1* d, y +.1*d, fill = 'black')
    return

def draw_nose(x:int, y:int, d:int, canvas: tkinter.Canvas):
    ''' Draws nose using x and y coordinates and diameter given '''
    canvas.create_oval (x, y, x+.03*d, y+.03*d, fill = 'black')

def draw_mouth(x:int, y:int, d:int, canvas: tkinter.Canvas):
    ''' Draws mouth using x and y coordinates and diameter given '''
    canvas.create_oval (x, y, x+.2*d, y+.08*d, fill = 'firebrick')

def draw_face(x:int, y:int, d:int):
    ''' Draws face using x and y coordinates and diameter given '''
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width = x+d+200, height = y+d+200)
    my_canvas.pack()
    my_canvas.create_oval((x, y, x+d, y+d), fill = 'lemon chiffon')
    draw_eye(x+.25*d, y+.25*d, d, my_canvas)#L eye
    draw_eye(x+.6*d, y+.25*d, d, my_canvas)
    draw_nose(x+ .43*d,y+.5*d,d, my_canvas)#L nostril
    draw_nose(x+.54*d, y+.5*d, d, my_canvas)#R nostril
    draw_mouth(x+.4*d, y+.7*d, d, my_canvas)#Mouth
    tkinter.mainloop()
    
draw_face(50,50,500)
draw_face(17, 28, 20)
draw_face(80, 130, 800)
