import tkinter

my_window = tkinter.Tk()  

my_canvas = tkinter.Canvas(my_window, width=1400, height=1000)  
my_canvas.pack()  

        

def draw_eye(x,y):
    return my_canvas.create_oval(x,y,x+200,y+120, fill = 'white'), my_canvas.create_oval(x+50,y+10,x+130,y+110, fill = 'magenta'),my_canvas.create_oval(x+80,y+45,x+100,y+75, fill = "gold")


def draw_nose(x,y):
    return my_canvas.create_line(x+220,y+130,x+160,y+270),my_canvas.create_line(x+220,y+130,x+270,y+270),my_canvas.create_line(x+160,y+270,x+270,y+270)


def draw_mouth(x,y):
    return my_canvas.create_oval(x+120,y+300,x+320,y+360,fill = 'red'),my_canvas.create_line(x+120,y+330,x+320,y+330)

def draw_contour(x,y):
    return my_canvas.create_oval(x-20,y-150,x+500,y+400, fill = 'grey')

def draw_face(x,y):
    return draw_contour(x,y),draw_eye(x,y),draw_eye(x+250,y),draw_nose(x,y),draw_mouth(x,y)

draw_face(30,200)
draw_face(600,200)
tkinter.mainloop()
