def calculate_shipping(weight:float)->float:
    '''Take the weight of a package and return the price for shipping the package'''
    if weight<2:        
        return 2.00
    if 2<= weight< 10:    
        return 5.00     
    else:               
        return 5.00+((weight-10)*1.50)

assert calculate_shipping(1.5) == 2.00
