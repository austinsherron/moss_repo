def calculate_shipping(i: int) -> str:
    if i < 2:
        return 2
    elif 10 > i >= 2:
        return 5
    else:
        return 5+(i-10)*1.5

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

pre_2000_list = [ ]

def add_books_in_list(book_list: list) -> int:
    total = 0
    for x in range(len(book_list)):
        total = total + book_list[x]
    return total
assert add_books_in_list([2, 6, 4]) == 12
assert add_books_in_list([2, 1, 2]) == 5
