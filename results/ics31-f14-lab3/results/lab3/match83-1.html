# c.5 - calculate_shipping()
def calculate_shipping(weight: float) -> float:
    ''' input a float weight and use it to calculate shipping costs and return
    price to the user '''
    if weight < 2.0:
        return 2.00
    elif weight < 10.0:
        return 5.00
    else:
        return 5.00 + ((weight - 10.0) // 1 * 1.5) # integer division to throw away partial pounds

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00


assert calculate_shipping(14.5) == 11.00
assert calculate_shipping(15.0) == 12.50

def inventory_value( book: Book) -> float:
    ''' return the value of the inventory '''
    return book.price * book.stock

def top_value(BookList: list) -> Book:
    best_book = BookList[0]
    for book in BookList:
        if (book.price * book.instock) > (best_book.price * best_book.instock):
            best_book = book
    return best_book

best_book = top_value(BSI)
print('The highest value book is', best_book.title, 'by', best_book.author, 'at a value of $', best_book.price * best_book.instock)

def draw_eye(x: int, canvas: str):
    ''' draw an eye with the given X value as guidance for where to start '''
    canvas.create_oval(100 + x,230, 160 + x,270, fill = 'white')
    canvas.create_oval(340 - x,230, 400 - x,270, fill = 'white')

    canvas.create_oval(115 + x,235, 145 + x,265, fill = 'brown')
    canvas.create_oval(355 - x,235, 385 - x,265, fill = 'brown')

    canvas.create_oval(125 + x,245, 135  + x,255, fill = 'black')
    canvas.create_oval(365 - x,245, 375 - x,255, fill = 'black')

def draw_mouth(m: int, canvas: str):
    ''' draw a mouth with the given x value as guidance for where to start '''
    canvas.create_oval(175,300 + m, 325,350 + m, fill = 'white')
    canvas.create_oval(175,325 + m, 325, 375 + m, fill = 'orange', outline = 'orange')

def draw_nose(n: int, canvas: str):
    ''' draw a nose with the given x value as guidance for where to start '''
    canvas.create_polygon(250,210 + n, 225 - n/4,250 + n, 275 + n/4,250 + n, fill = 'yellow' )

def draw_hair(x: int, canvas: str):
    ''' draw hair on the head of the face sometimes '''
    if x >= 50:
        # draw hair if x >= 50
        canvas.create_line(150,125, 100,75, fill = 'black')
        canvas.create_line(200,100, 175,50, fill = 'black')
        canvas.create_line(250,75, 250,25, fill = 'black')
        canvas.create_line(300,100, 325,50, fill = 'black')
        canvas.create_line(350,125, 400,75, fill = 'black')

def draw_face(eye_start: int, mouth_start: int, nose_start: int, canvas: str, face: str):
    ''' draw a face with the given variables affecting the start locations for facial features
    use eye_start variable to affect hair'''
    face = tkinter.Tk()
    canvas = tkinter.Canvas(face, width=500, height=500)
    canvas.pack()

    canvas.create_oval(100,50, 400,450, fill = 'orange')
    draw_eye( eye_start, canvas)
    draw_mouth(mouth_start, canvas)
    draw_nose(nose_start, canvas)
    draw_hair(eye_start, canvas)

# call three different versions of draw_face()
