def calculate_shipping(t: float) -> float:
    'returns price for shipping'
    if t < 2.00:
        return 2.00
    if 2 < t < 10:
        return 5.00
    if t > 10:
        return (5 + ((t-10)*1.50))
    

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
