def print_even_numbers(my_list: str) -> str:
    " Print each even number of the list input by the user "
    for num in range(len(my_list)):
        number=int(my_list[num])

            print(my_list[num])

def calculate_shipping(weight: float) -> float:
    " Returns the cost of shipping for the parameter value "
    if weight < 2:
        rate = 2.00
    elif (2 <= weight < 10):
        rate = 5.00
    else:
        rate = 5.00 + (weight-10)*1.5
    return rate
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print("The shipping cost for a 3 pound package is", calculate_shipping(3))
#to take input: print(calculate_shipping(float(input('Enter the weight of the package'))))

my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()

def draw_eye(x: int, y: int, length: int, color: str):
    ''' Draw an eye with respect to the first two parameters as the starting
    coordinates and the third and fourth parameters as the diameter and fill respectively


    '''
    my_canvas.create_oval(x-length/2, y-length/2, x+length/2, y+length/2, fill=color)
    return

def draw_nose(x: int, y: int, color: str):
    ''' Draw a nose with respect to the first two parameters as the starting
    coordinates and the third as the fill


    '''
    my_canvas.create_polygon(x-5, y, x+5, y, x, y+5, x-5, y, fill=color)

def draw_mouth(x: int, y: int, length: int, color: str):
    ''' Draw a mouth with respect to the first two parameters as the starting
    coordinates and the third as the fill
    '''


    my_canvas.create_arc(x-length/2, y, x+length/2, y+length, start = 0, extent=180, fill=color)

def draw_face(x: int, y: int, width: int, height: int, color: str):
    ''' Draw a face with respect to the first two parameters as the starting
    coordinates and the third as the fill, and then call draw_eye twice and
    the other functions once.


    '''
    my_canvas.create_oval(x-width/2, y-height/2, x+width/2, y+height/2, fill=color)
    draw_eye(x-width/6, y-25, 10, 'black')


    draw_eye(x+width/6, y-25, 10, 'black')
    draw_nose(x, y, 'black')
    draw_mouth(x, y+25, 30, 'pink')
    return

draw_face(250, 250, 150, 200, '#CEAB69')
