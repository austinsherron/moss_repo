def calculate_shipping(rate: float) -> float:
        if rate < 2:
            return (2.00)
        elif 2 < rate <10:
            return (5.00)
        elif rate >= 10:
            return (5.00+(rate-10)*1.50)



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
