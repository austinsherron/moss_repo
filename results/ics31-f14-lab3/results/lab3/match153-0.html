def calculate_shipping(x: float) -> float: #define function to calculate fee
    if x < 2.00: #if the weight is under 2 pounds
        fee = 2.00
    elif 2.00 < x < 10.00: #if the weight is under 10 and more than 2
        fee = 5.00
    else:    #if the weight is more than 10
        fee = 5.00+(x-10.00)*1.50 #rate is 10 plus 1.5 per pound
    return fee

assert calculate_shipping(1.5) == 2.00 #check he function
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
print('A 16 pound package will cost $',calculate_shipping(16))

def sortBy2000(bookList: list) -> str:
    'sort book by the year befor or later than 2000,and compare the number of book instock'
    book_before2000 = []  #define two empty list 
    book_after2000 = []
    numberOfBooksBefore2000 = 0   #define two variables
    numberOfBooksAfter2000 = 0
    for i in bookList: #scan the book in the BSI
        if i.year < 2000: # find out the book which is published before 2000
             book_before2000.append(i) #add the book into new list
             numberOfBooksBefore2000 = numberOfBooksBefore2000 + i.instock #accumulate the number of books instock
        else:  #find out the book which is published in 2000 or later
            book_after2000.append(i)  #add books to new list
            numberOfBooksAfter2000 = numberOfBooksAfter2000 + i.instock 

    #compare the number of book before 2000 and in or later 2000
    if numberOfBooksBefore2000 > numberOfBooksAfter2000: 
        print ('More titles before 2000', '(', numberOfBooksBefore2000, 'vs', numberOfBooksAfter2000, ')')
    else:
        print ('More titles 2000 or later', '(', numberOfBooksAfter2000, 'vs', numberOfBooksBefore2000, ')')
sortBy2000(BSI)
