def calculate_shipping(s: float) -> float:
    " Return the price for shipping package given pounds"
    if s < 2:
        return 2.00
    elif s >= 2 and s < 10:
        return 5.00
    elif s >= 10:
        return 5.00+(s-10)*1.5

assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
