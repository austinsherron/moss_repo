def abbreviate(word):

def find_area_square(length):
    area = length ** 2
    return area

assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(radius):
    area = radius ** 2 * 3.14159
    return area

assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def calculate_shipping(weight):
    if( weight < 2):
        return 2.00
    elif( weight < 10):
        return 5.00
    else:
        return 5.00 + 1.50 * (weight - 10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value( book ):
    return book.price * book.instock

def top_value( books ):
    mostExpensiveIndex = 0
    counter = 0
    for book in books:
        if( inventory_value(book) > inventory_value(books[mostExpensiveIndex]) ):
            mostExpensiveIndex = counter
            counter += 1
    return books[mostExpensiveIndex]

def draw_eye( my_canvas, x ):
    my_canvas.create_oval(x,100,x+200,200, fill='white')
    my_canvas.create_oval(x+50,100,x+150,200, fill='pink')
    my_canvas.create_oval(x+125,100,x+75,200, fill='blue')

def draw_nose( my_canvas, x ):
    my_canvas.create_line( x,200,x+100,400 )
    my_canvas.create_line( x,400,x+100,400 )
    
def draw_mouth( my_canvas, x ):
    my_canvas.create_oval( x,450,x+500,650 )


def draw_face():
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
    my_canvas.pack()
    draw_eye( my_canvas, 200 )
    draw_eye( my_canvas, 500 )
    draw_nose( my_canvas, 450 )
    draw_mouth( my_canvas, 200 )
    tkinter.mainloop()
    return
