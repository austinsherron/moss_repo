def print_even_numbers(x:list) -> list:
      New= list()
      for num in x:
            if num%2 == 0:
                  New.append(num)
      return New

def calculate_shipping(weight:float) ->float:
      if weight>0:
            if 0<weight <2:
                  return 2.00
            if 2<weight<10:
                  return 5.00
            if weight>10:
                  return 1.5*(weight-10)+5.00
      else:
            return 'error'

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def print_value_raise(BSI:list) ->list:
      New_value_BSI=BSI
