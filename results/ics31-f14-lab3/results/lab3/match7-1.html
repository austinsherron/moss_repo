def abbreviate(month: str) -> str: 
    '''Take any string as input and return its first three characters'''
    return(month[0:3])
assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'

def find_area_square(side_length: int) -> int:
    '''Takes input as side length of a square and returns the area of that square'''
    return(side_length ** 2)
assert find_area_square(1) == 1
assert find_area_square(5) == 25

def find_area_circle(radius: int) -> float:
    '''Takes the radius of a circle and returns the area of that circle'''


    return((radius ** 2)*3.14159)
assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def print_even_numbers(numbers: list) -> list:
    '''Takes a list of integers and prints each even number on the list'''


    for n in range(0, len(numbers)):
        if(numbers[n]%2==0):
            print(numbers[n])
print_even_numbers([2, 47, 31, 99, 20, 19, 23, 105, 710, 1004])

def calculate_shipping(weight: float) -> float:
    '''Takes the weight of a package returns the price for shipping the package'''


    if(weight < 2):
        return(2.00)
    elif(weight >= 2 and weight < 10):
        return(5.00)
    else:
        return(5 + 1.50 * (weight - 10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
