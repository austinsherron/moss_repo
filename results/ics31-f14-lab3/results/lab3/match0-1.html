def calculate_shipping(w:float) -> float:
    '''returns the price for shipping the package'''
    if w < 2:
        result = 2.00
    elif 2 <= w < 10:
        result = 5.00
    elif w >= 10:
        result = (5.00 + ((int(w-10)) * 1.50))
    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(10.5) == 5.00
assert calculate_shipping(11.5) == 6.50
assert calculate_shipping(15) == 12.50

def total_stock(titles:list) -> int:
    '''returns the sum of all instock figures of all "titles" in the list'''
    total_books = 0
    for books in titles:
        total_books += books.instock
    return total_books
assert total_stock(before_2000) == 180
assert total_stock(after_2000) == 175

if total_stock(before_2000) > total_stock(after_2000):

          total_stock(after_2000),").", sep='')

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()

def draw_eye(x, y, x1, y1):
    '''draws in the tkinter window an eye described by the arguments'''
    my_canvas.create_oval(x, y, x1, y1, fill='white') 
    my_canvas.create_oval(x+20, y+20, x1-5, y1-5, fill='light blue')
    my_canvas.create_oval(x+25, y+25, x1-10, y1-10, fill='black' )

def draw_nose(x, y, x2, y2):
    '''draws in the tkinter window a nose described by the arguments'''
    my_canvas.create_oval(x, y, x2, y2, fill='yellow')

def draw_mouth(x, y, x1, y2):
    '''draws in the tkinter window a mouth described by the arguments'''
    my_canvas.create_oval(x, y, x1, y2, fill='red')

def draw_face(x, y, x1, y1):
    '''draws in the tkinter window a face described by the arguments'''
    my_canvas.create_oval(x, y, x1, y1, fill='white')
    draw_eye(x+75, y+100, x1-225, y1-200)
    draw_eye(x+225, y+100, x1-75, y1-200)
    draw_nose(x+175, y+225, x1-225, y1-225)
    draw_mouth(x+150, y+300, x1-175, y1-25)

#tests draw_face function
#for "perfect face": x = y, x1 = y1, x1-x = 350, and y1-y = 350
