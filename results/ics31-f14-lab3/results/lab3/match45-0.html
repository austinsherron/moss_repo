    if(weight< 2):
        return(UNDER_TWO_RATE)
    elif (weight<10):
        return(UNDER_TEN_RATE)
    else:
        return UNDER_TEN_RATE + (weight - 10) * EXTRA_RATE

my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window


  
def draw_eye(x:int, y:int)-> None:
    '''return all eyes in parameter'''
    my_canvas.create_oval(x, y, x+50, y-50, fill='white') # Draw orange line
    my_canvas.create_oval(x+10,y,x+40,y-50, fill='blue')
    my_canvas.create_oval(x+15,y-10,x+35,y-40, fill='black')

def draw_mouth (x:int, y:int )-> None:
    '''return all mouth in parameter'''
    my_canvas.create_oval(x, y, x+150, y+100, fill='pink')
    my_canvas.create_oval(x+15,y+30,x+130,y+100, fill='red')

def draw_nose (x:int, y:int )-> None:
    '''return all nose in parameter'''
    my_canvas.create_arc(x, y, x+150, y+50, fill='orange')


def draw_face(x:int, y:int ) -> None:
    '''return face in parameter'''
    my_canvas.create_oval(x, y, x+375, y+300, fill='brown')
    draw_eye(x + 50, y +100)
    draw_eye(x + 250,y + 100)
    draw_mouth(x + 110, y + 160)
    draw_nose(x + 100, y + 120)

draw_face(50,100)
