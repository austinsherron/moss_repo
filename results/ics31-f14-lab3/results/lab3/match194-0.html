def calculate_shipping(weight: float):
    '''Takes weight of package and returns the price for shipping'''
    if weight < 2:
        result = 2.00
    if weight >= 2 < 10:
        result = 5.00
    if weight >= 10:
        result = (5.00 + 1.5*(weight-10))
    return result

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(5) == 5.00
assert calculate_shipping(10) == 5.00

def draw_eye(window, x:int, y:int, D:int):
    window.create_oval(x, y, x + D, y + D, fill = 'mistyrose2')
    window.create_oval(x, y, x + D/2, y + D/2, fill = 'white')
    window.create_oval(x, y, x + D/4, y + D/4, fill = 'black')
    
def draw_nose(window, x:int, y:int, D) :
    window.create_oval(x, y, x + D, y + D, fill = 'yellow')

def draw_mouth(window, x:int, y:int, D:int) :
    window.create_oval(x, y, x + D, y + D, fill = 'Red')
 
def draw_face(window,x:int, y:int, D:int):
    window.create_oval(x, y, x + D, y + D, fill = 'midnight blue')

    draw_nose(window, x+D/2.5, y + D/2, D/4.5)
