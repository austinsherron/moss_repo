def calculate_shipping(weight) -> float:
    " Calculates the price of shipping "
    if weight < 2:
     return 2.00
    if 2 < weight < 10:
     return 5.00
    else:
     return 5.00 + (1.50 * (weight - 10))



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book:list) -> int:
    "Takes in a book and returns the value of our inventory"
    return book.instock * book.price


def top_value(book:list) -> str:
    "Takes in a list of books and returns the book object that has the highest-value instock"
    BSI_inventory = sorted(book, key = inventory_value)
    print("The highest-value book is", BSI_inventory[-1].title, "by", BSI_inventory[-1].author, "at a value of", BSI_inventory[-1].price * BSI_inventory[-1].instock, ".")
    return BSI_inventory[-1]

top_value(BSI)

def draw_eye(x, y, d1, a, b, d2):
    " Draw an eye "
    f_canvas.create_oval(a, b, a+d2, b+d2, fill='white')
    f_canvas.create_oval(x, y, x+d1, y+d1, fill='black')
    
def draw_nose(x, y, d):
    " Draw a nose "
    f_canvas.create_oval(x, y, x+d, y+d, fill='black')

def draw_mouth(x, y, d):
    " Draw a mouth "
    f_canvas.create_oval(x, y, x+d, y+d, fill='black')

def draw_outline(x, y, d):
    f_canvas.create_oval(x, y, x+d, y+d, fill='yellow')
def draw_face(x, y):
    draw_outline(60+x, 45+y, 200)
    draw_eye(40+x,55+y,50, 40+x,55+y,100)
    draw_eye(150+x,55+y,50, 150+x,55+y,100)
    draw_nose(130+x, 155+y, 10)
    draw_nose(140+x, 155+y, 10)
