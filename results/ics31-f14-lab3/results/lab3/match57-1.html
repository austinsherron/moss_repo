print('---------Part (f)------------')

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window,width=1500,height=1500)
my_canvas.pack()
def draw_eye (x:int,y:int)->int:
    my_canvas.create_oval(x+100,y+100,
                          x+150,y+150,fill = 'black')
    my_canvas.create_oval(x+120,y+120,
                          x+130,y+130,fill = 'brown')
def draw_nose (x:int,y:int)->int:
    my_canvas.create_oval(x+180,y+150,
                          x+220,y+180,fill = 'blue')
def draw_mouth(x:int,y:int)->int:
    my_canvas.create_oval(x+150,y+200,
                          x+250,y+250,fill = 'red')
def draw_face(x,y)->int:
    my_canvas.create_oval(x,y,
                          x+400,y+400)
    draw_eye(x,y)
    draw_eye(x+150,y)
    draw_nose(x,y)
    draw_mouth(x,y)


draw_face(50,50)
draw_face(400,350)
tkinter.mainloop()
