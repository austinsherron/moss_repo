def calculate_shipping(w: float) -> float:
    if (w < 2.00):
        return 2.00
    elif (2 <= w < 10):
        return 5.00
    elif (w >= 10):
        n = w - 10
        return n * 1.50 + 5.00
assert calculate_shipping(1) == 2.0
assert calculate_shipping(2.00) == 5.0
assert calculate_shipping(3.00) == 5.0
assert calculate_shipping(12.00) == 8.0
assert calculate_shipping(10.00) == 5.00
assert calculate_shipping(12.50) == 8.75

my_eye = tkinter.Tk()
my_drawing2 = tkinter.Canvas(my_eye, width = 2000, height = 1000)
my_drawing2.pack()

def draw_eye(x: int, y: int, x2: int, y2: int) -> 'Eye':
    my_drawing2.create_oval(x, y, x2, y2, outline = 'black', fill = 'white')
    my_drawing2.create_oval(x-100, y, x2+100, y2, outline = 'black', fill = 'blue')
    my_drawing2.create_oval(x-175, y-125, x2+175, y2+125, outline = 'black', fill = 'black')

def draw_mouth(x:int, y: int, x2: int, y2: int) -> 'Mouth':
    my_drawing2.create_oval(x, y, x2, y2, outline = 'black', fill = 'white')
    my_drawing2.create_line(x, y+25, x2, y2-25, fill = 'black')

def draw_nose(x:int, y:int, x2: int, y2:int) -> 'Nose':
    my_drawing2.create_line(x, y, x2, y2, fill = 'black')
    my_drawing2.create_line(x, y, x2, y, fill = 'black')

def draw_face(x:int, y:int, x2: int, y2: int) -> 'Face':
    ''' x and y must be at least 500 apart
        x2 and y2 must be at least 300 apart '''
    draw_eye(x, y, x2, y2)
    draw_eye(x+500, y, x2+500, y2)
    draw_mouth(x-225, y+300, x2+725, y2+550)
    draw_nose(x, y+150, x2+450, y2+250)
draw_face(850, 350, 450, 150)
