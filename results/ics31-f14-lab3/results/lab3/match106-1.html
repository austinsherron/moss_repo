def calculate_shipping(weight: int) -> float:
    if weight < 2:
        return 2.00
    elif 2 <= weight < 10:
        return 5.00
    elif weight > 10:
        return 5.00 + 1.5*(weight-10)
    else:
        print("ERROR")
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

    max = -1
    index = 0
    for i in range(len(book_list)):

my_window_2 = tkinter.Tk()

face_canvas = tkinter.Canvas(my_window_2, width=2000, height=1000)  # Create a 500x500 canvas to draw on
face_canvas.pack()            # Put the canvas into the window

def draw_eye(x: int, y: int, length: int):
    face_canvas.create_oval(x, y, x+length, y+(length/2), fill = 'orange')
    face_canvas.create_oval(x+(length/3), y, x+length-(length/3),y+(length/2), fill = 'black') 
    
def draw_mouth(x: int, y: int, length: int):
    face_canvas.create_oval(x, y, x+length, y+(length/3), fill = 'red')

def draw_nose(x: int, y: int, baselength: int):
    face_canvas.create_polygon(x, y, x+(baselength/2), y + baselength, (x + baselength/2)-baselength, y+baselength, x,y)

def draw_face(x: int, y: int, length: int):
    '''Uses draw_eye, draw_mouth, and draw_nose to create a face that starts at
    the position given by parameters x and y, and that scales to the size given
    by the parameter length.'''
    lengthx = length*.75 #width (horizontal) of the face
        
    face_canvas.create_oval(x, y, x + lengthx, y + length, fill = 'yellow')

    lefteyex = x + lengthx/7 #The x position of the left eye
    lefteyey = y + length/6  #The y position of the left eye
    eyelength = length/6 #length of the eye
    draw_eye(lefteyex, lefteyey, eyelength) #left eye
    righteyex = x + lengthx - eyelength - (lengthx/7)
    draw_eye(righteyex, lefteyey, eyelength) #right eye
    draw_mouth( ((x + (lengthx/2)) - (lengthx / 3)), y + length/1.5, length/2) #draw   
    draw_nose(x + lengthx/2, y+length/3, length/5)

# draw 2 faces
draw_face(20, 20, 600) #bigger, top left corner
draw_face(700, 100, 300) #smaller, towards the right of the bigger one
print("On tkinter window")

tkinter.mainloop()
