def calculate_shipping(n: float) -> float:
    ''' Return price for shipping of the input n '''
    if n < 2:
        return 2
    elif 2 <= n <= 10:
        return 5
    elif n > 10:
        return ((n - 10) * 1.5) + 5
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(10))
print(calculate_shipping(75))

def create_eye(x: float, y:float, d: float) -> None:
    ''' Creates an eye '''
    return my_canvas.create_oval(x , y , x + d, y + d, fill = "white")
def create_nose(x: float, y: float, d: float) -> None:
    ''' Create a nose '''

def create_mouth(x: float, y: float, d: float) -> None:
    ''' Draws a mouth '''
    return my_canvas.create_rectangle(x, y , x + d, y + d, fill = "white")
def create_head(x: float, y: float, d: float) -> None:
    ''' Draws a head '''
    return my_canvas.create_oval(x, y, x + d, y + d, fill = "white")
