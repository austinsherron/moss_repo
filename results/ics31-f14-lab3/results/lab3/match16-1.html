def calculate_shipping(x:int) -> int:
    if x <2:
        return 2
    if 2 <= x <= 10:
        return 5
    if x> 10 :
        return (x-10)*1.5+5

    
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

my_window2 = tkinter.Tk()


my_canvas2 = tkinter.Canvas(my_window2, width = 500, height = 500)
my_canvas2.pack()

def draw_eye(x,y,z):
    my_canvas2.create_oval(x,y,x+2*z,y+z)


    my_canvas2.create_oval(x+0.5*z,y,x+1.5*z,y+z,fill = 'blue')
    my_canvas2.create_oval(x+0.75*z,y+0.25*z,x+1.25*z,y+0.75*z,fill = 'black')
def draw_nose(x,y,z):
    my_canvas2.create_polygon(x,y,x+z,y+2*z,x-z,y+2*z)


def draw_face(x,y,z):
    
    my_canvas2.create_oval(x,y,2*(x+z),2*(y+z))
    draw_eye(x+z,y+z,z*0.5)
    draw_eye(x+2*z,y+z,z*0.5)
    draw_nose(2*x,1.8*y,z*0.5)
    my_canvas2.create_line(1.5*x,2.5*y,2.5*x,2.5*y,smooth=True,splinesteps=100)
draw_face(100,100,50)
