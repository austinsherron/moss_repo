def abbreviate(s: str) -> str:
    '''take any string as input and returns its first three characters'''
    if len(s) <= 3:
        return s
    else:
        return s[0:3]
assert abbreviate('March') == 'Mar'
assert abbreviate('hello') == 'hel'

    '''takes a number representing the length of one side of a square and returns the area of that square'''

def calculate_shipping(w: int) -> int:
    '''returns the price for shipping based on weight'''
    if w<2:
        return 2.00
    elif w<10:
        return 5.00
    else:
        return 5.00+1.5*(w-10)
assert calculate_shipping(12)==8
assert calculate_shipping(3)==5

def inventory_value(b:Book) -> float:
    '''takes a Book and returns the value of the inventory of the book'''
    return b.instock * b.price
assert inventory_value(BSI[0])==1208.9

def top_value(b:list) -> Book:
    '''takes a list of Books and returns the Book that has the highest-value inventory'''
    temp = b[0]
    for x in b:
        if inventory_value(x) > inventory_value(temp):
            temp = x
    print('The highest-value book is '+temp.title+' by '+temp.author+' at the value of $',int(inventory_value(temp)))
    return temp
assert top_value(BSI)==BSI[1]

me = tkinter.Canvas(window, width=1000, height=1000)
me.pack()

def draw_eye(x1,y1,x2,y2,i1,i2,s1,p1,p2,s2):
    me.create_oval(x1,y1,x2,y2,fill='white')
    me.create_oval(i1,i2,i1+s1,i2+s1,fill='blue')


    me.create_oval(p1,p2,p1+s2,p2+s2,fill='pink')

def draw_nose(n1,n2,w,h):
    me.create_polygon(n1,n2,n1+w,n2+h,n1-w,n2+h, fill = 'green')

def draw_mouth(x,y,s):
    me.create_line(x,y,x+2*s,y)
    me.create_line(x,y,x+s,y+s,x+2*s,y,fill='black',smooth='true')

def draw_face(x1,y1,x2,y2,i1,i2,s1,p1,p2,s2,xx1,yy1,xx2,yy2,ii1,ii2,ss1,pp1,pp2,ss2,n1,n2,w,h,x,y,s):
    draw_eye(x1,y1,x2,y2,i1,i2,s1,p1,p2,s2)
    draw_eye(xx1,yy1,xx2,yy2,ii1,ii2,ss1,pp1,pp2,ss2)
    draw_nose(n1,n2,w,h)
    draw_mouth(x,y,s)

draw_face(10,10,50,30,20,10,20,30,20,4,110,10,150,30,120,10,20,130,20,4,75,15,10,30,20,50,50)
