def calculate_shipping(weight:int)->int:
    'Returns price of shipping'
    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5.00 + 1.50 * (weight - 10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(inventory = namedtuple('Book', 'author title genre year price instock')):
    'Returns the value of our inventory of that book'
    return(inventory.instock * inventory.price)

def top_value (LB = list):
    'Returns the book with the highest value inventory'
    LBS = (sorted(LB, key = inventory_value, reverse = True))
    print("The highest-value book is", LBS[0].title, "by", LBS[0].author, "at a value of", inventory_value(LBS[0]))

import tkinter
my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()
    
def draw_eye(x = int, y = int):
    my_canvas.create_oval(x,y,x+100,y+50, fill = 'white')
    my_canvas.create_oval(x+35,y+10,x+70,y+40, fill = 'green')
    my_canvas.create_oval(x+40,y+15,x+65,y+35, fill = 'black')
    
def draw_nose(x = int, y = int, w = int):
    'Take in a start point and width and draw a nose'
    my_canvas.create_polygon(x, y, x - (.5*w), y + 100, x + w, y + 100, x, y)

def draw_mouth(x = int, y = int, w = int):
    'Take in a start point and width and draw a mouth'
    my_canvas.create_oval(x, y + 50, x + w, y + 100)

def draw_head(x = int, y = int, w = int):
    'Takes start point and creates a head'


    my_canvas.create_rectangle(x, y , x + 2.5*w, y + 4*w)
def draw_face(x = int, y = int, w = int):
    draw_eye(x+50,y+100)
    draw_eye(x+150,y+100)
    draw_nose(x+150, y+200, .35*w)
    draw_mouth(x+105, y+300, 1.2*w)
    draw_head(x+25, y+25, w)

draw_face(0,0,100)
draw_face(400,400,100)

tkinter.mainloop()
