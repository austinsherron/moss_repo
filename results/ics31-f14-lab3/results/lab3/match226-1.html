def calculate_shipping(weight: float)-> float:
    '''Takes in the weight of a package and returns the shipping price of that package'''
    LESS_TWO_RATE = 2.00
    LESS_TEN_RATE = 5.00
    MORE_TEN_RATE = 1.50
    if(weight < 2):
        return(LESS_TWO_RATE)
    elif(weight < 10):
        return(LESS_TEN_RATE)
    else:
        return(LESS_TEN_RATE + (MORE_TEN_RATE * (weight - 10)))



assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book: Book) -> float:
    '''Takes in a book and returns the total value of that book in inventory'''
    return(book.price * book.instock)

def top_value(inv_value: "list of Books") -> Book:
    '''Takes in a list of books and returns the book with the highest total value in that list'''
    max_value = 0
    max_index = 0
    for i in range(len(inv_value)):
        if(inventory_value(inv_value[i]) > max_value):
            max_index = i
            max_value = inventory_value(inv_value[i])
    return (inv_value[max_index])

total = top_value(BSI)
print("The highest value book is", total.title, "by", total.author, ", at a value of $", inventory_value(total))

my_window = tkinter.Tk()
my_canvas2 = tkinter.Canvas(my_window, width = 500, height = 500)
my_canvas2.pack()

def draw_eye(x: int, y: int) -> None:
    '''Takes in coordinates and draws an eye based on them'''
    my_canvas2.create_oval(x, y, x+100, y+100, fill='white')
    my_canvas2.create_oval(x+25, y+25, x+75, y+75, fill='purple')
    my_canvas2.create_oval(x+40, y+40, x+60, y+60, fill='black')    
    return

def draw_nose(x: int, y: int) -> None:
    '''Takes in coordinates and draws a nose based on them'''
    my_canvas2.create_line(x, y, x-40, y+100)
    my_canvas2.create_line(x-40, y+100, x+40, y+100)    
    return

def draw_mouth(x: int, y: int) -> None:
    '''Takes in coordinates and draws a mouth based on them'''
    if x < 250:
        my_canvas2.create_arc(x+50, y, x+250, y+75, start=0, extent=-180, fill = 'black')
    else:
        my_canvas2.create_arc(x+50, y, x-150, y+75, start=0, extent=-180, fill = 'black')
    return

def draw_face(x: int, y: int) -> None:
    '''Takes in coordinates and draws a face based on them'''
    my_canvas2.create_oval(10, 10, 490, 490)
    if x < 250:
        draw_eye(x, y)
        draw_eye(x+200, y)
        draw_nose(x+150, y+50)
        draw_mouth(x, y+200)
    else:
        draw_eye(x, y)
        draw_eye(x-200, y)
        draw_nose(x-50, y+50)
        draw_mouth(x, y+200)   
                
    tkinter.mainloop()
    return

draw_face(100, 100)
