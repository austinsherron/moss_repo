def find_area_square(n: int) -> int:
    "Return the square of the input."
    return n**2

def find_area_circle(n: int) -> float:
    "Return the area of a circle with a radius equal to the input."
    return 3.14159 * (n ** 2)

def print_even_numbers(n: list):
    "Print all even numbers in the supplied list."
    for x in n:
        if x % 2 == 0:
            print(x)
    return
print("All even numbers between 0 and 18:")
print_even_numbers(range(20))
print("\n")

def calculate_shipping(n: float) -> float:
    "Return the shipping cost of a package with a weight equal to the input."
    if n < 2:
        return 2
    elif n < 10:
        return 5
    else:
        return 5 + ((n-10) * 1.5)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(15) == 12.50
print("Shipping price of a package weighing 1.5 pounds:", calculate_shipping(1.5),
      "\nShipping price of a package weighing 7 poinds:", calculate_shipping(7),
      "\nShipping price of a package weighing 15 pounds:", calculate_shipping(15))
print("\n")

def raise_prices_ten_percent(n: "list of Books"):
    '''Increases the price of all books in the supplied list by 10 percent.'''
    x = 0
    while x < len(n):
        n[x] = Book(n[x].author, n[x].title, n[x].genre, n[x].year, n[x].price*1.1, n[x].instock)
        x = x + 1
    return
print("This next function will raise the prices of all books by 10 percent.")

print("The edited list is as follows:\n", BSI)

import tkinter


def draw_face(canvas: "tkinter.Canvas", x: int, y: int):
    canvas.create_oval(x, y, 250+x, 275+y, fill='orange')
    draw_eye(canvas, x+25, y+50)
    draw_eye(canvas, x+140, y+50)
    draw_nose(canvas, x+85, y+105)
    draw_mouth(canvas, x+85, y+195)
    return



def draw_eye(canvas: "tkinter.Canvas", x: int, y: int):
    canvas.create_oval(x, y, x+75, y+50, fill='white')
    canvas.create_oval(x+12, y+3, x+57, y+47, fill='black')
    return

def draw_nose(canvas: "tkinter.Canvas", x: int, y: int):
    canvas.create_polygon(x+40, y, x, y+70, x+80, y+70, fill='green')
    return

def draw_mouth(canvas: "tkinter.Canvas", x: int, y: int):
    canvas.create_oval(x, y, x+80, y+25, fill='red')
    return

my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()

print("A tkinter window with a face on it should display now. Close it to move on (the program is finished).")
