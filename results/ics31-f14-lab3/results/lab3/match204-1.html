def calculate_shipping (package: float) -> float:
    if package<2:
        result= 2.00
    elif 2<=package<10:
        result= 5.00
    else:
        result= 5.00 + (package - 10)*1.50
    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5.00

def draw_eye(x: int, y: int, x2:int, y2:int):
    my_canvas.create_oval(x+50,y+200,x2+350,y2+300, fill='white')
    my_canvas.create_oval(x+150,y+200,x2+250,y2+300, fill='sky blue')
    my_canvas.create_oval(x+175,y+225,x2+225,y2+275, fill='black')

    my_canvas.create_oval(x, y, x+z, y+z, fill='pink')
    
def draw_mouth(x: int, y: int, z: int):
    my_canvas.create_line(x, y, x+z, y)

    draw_eye(x,y,x,y)
    draw_eye(x+400, y, x+400, y)
