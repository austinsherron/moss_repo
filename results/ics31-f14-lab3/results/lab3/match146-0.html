def calculate_shipping(value: int) -> float:
    "Take in a weight and return the shipping price for that weight)"
    if value < 2:
        return 2.00
    elif value >= 2 and value <10:
        return 5.00
    else:
        return 5.00 + (1.50*(value-10))
