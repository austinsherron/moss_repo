def calculate_shipping(weight:float) -> float:
    " Returns shipping value based on weight "
    if weight < 2:
        return 2
    if 2 <= weight < 10:
        return 5
    if weight >= 10:
        return 5 + 1.50*(weight-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print( calculate_shipping(7))

def draw_eye(x1: float, y1:float, d:float):
    " Draws an eye"
    
    
    my_canvas.create_oval(x1 , y1, x1 + d * d/6, y1 + d, fill = 'white')
    my_canvas.create_oval(x1+d*d/24, y1, x1 +d*d/12 +d*d/24 , y1 + d, fill = 'aquamarine')
    my_canvas.create_oval(x1+d*d/24+d*d/36, y1+d*d/64, x1 +d*d/12 +d*d/24 - d*d/36 , y1 + d - d*d/64, fill = 'black')
    
    
    
    
def draw_nose(x1:float, y1:float, x2:float, y2:float, x3:float, y3:float):
