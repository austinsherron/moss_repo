def calculate_shipping(x):
    if x < 2:
        return(2.00)
    if 2 <= x < 10:
        return(5.00)
    if x >= 10:
        return(5.00 + (1.50 * (x - 10.00)))

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

my_canvas.pack()            # Put the canvas into the window



def draw_eye(x, y: int) -> None:
   "Draws eye without duplicating code"
   my_canvas.create_oval(x + 100, y - 300, x + 300, y - 200)
   my_canvas.create_oval(x + 150, y - 300, x + 250, y - 200, fill = "blue")
   my_canvas.create_oval(x + 175, y - 275, x + 225, y - 225, fill = "white")
   
   my_canvas.create_oval(x - 100, y - 300, x - 300, y - 200)
   my_canvas.create_oval(x - 150, y - 300, x - 250, y - 200, fill="blue")
   my_canvas.create_oval(x - 175, y - 275, x - 225, y - 225, fill = "white")


def draw_nose(x, y: int) -> None:
   "Draws nose without duplicating code"
   my_canvas.create_line(x, y - 100, x - 50, y + 100)
   my_canvas.create_line(x, y - 100, x + 50, y + 100)
   my_canvas.create_line(x - 50, y + 100, x + 50, y + 100)


def draw_mouth(x, y: int) -> None:
   "Draws mouth without duplicating code"
   my_canvas.create_oval(x - 150, y + 200, x + 150, y + 300, fill="red")
   my_canvas.create_line(x - 150, y + 250, x + 150, y + 250)

   "Draws face"

   draw_nose(x, y)
