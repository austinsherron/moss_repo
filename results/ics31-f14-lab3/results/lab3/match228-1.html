def calculate_shipping(x):
    if x < 2:
        return(2.00)
    elif x < 10:
        return(5.00)
    else:
        return(5 + (x - 10) * 1.5)

def draw_eyes(x:int, y:int, r:int):
    return my_canvas.create_oval(x+r/8,y+2*r/8,x+3*r/8,y+7*r/16, fill='white'), my_canvas.create_oval(x+5*r/8,y+2*r/8,x+7*r/8,y+7*r/16, fill='white'), my_canvas.create_oval(x+11*r/16,y+9*r/32,x+13*r/16,y+13*r/32, fill='black'),my_canvas.create_oval(x+3*r/16,y+9*r/32,x+5*r/16,y+13*r/32, fill='black')
def draw_nose(x:int, y:int, r:int):
    return my_canvas.create_oval(x+7*r/16,y+.5*r,x+9*r/16,y+5*r/8, fill='tan'),my_canvas.create_oval(x+7*r/16,y+24*r/40,x+23*r/48,y+5*r/8, fill='black'), my_canvas.create_oval(x+25*r/48,y+5*r/8,x+27*r/48,y+24*r/40, fill='black')
def draw_mouth(x:int, y:int, r:int):
    return my_canvas.create_oval(x+3*r/8,y+3*r/4,x+5*r/8,y+7*r/8, fill='pink'), my_canvas.create_oval(x,y+3*r/16,x+r,y-5*r/16, fill='black')
def draw_face(x:int, y:int, r:int):
    return my_canvas.create_oval(x,y,x+r,y+r, fill='tan'), draw_eyes(x,y,r), draw_nose(x,y,r), draw_mouth(x,y,r)

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on

my_canvas.pack()# Put the canvas into the window
