def calculate_shipping (weight: float) -> float:
    '''Returns the price for shipping the package'''
    #print (weight)
    if  weight < 2.00:
        return (2.00)
    elif 2.00 <= weight < 10.00:
        return (5.00)
    elif weight >= 10.00:
        return (5.00 + ((weight - 10.00) * 1.50))
assert calculate_shipping(1.5) == 2.00

assert calculate_shipping(15) == 12.50
