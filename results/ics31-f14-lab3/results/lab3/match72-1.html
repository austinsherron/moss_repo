#function calculate_shipping calculates the price of the shipping based upon the item's weight
def calculate_shipping(weight: float) -> float:
   if weight < 2:
      result = 2
   elif 2 <= weight <10:
      result = 5
   else:
      result = (1.5*(weight-10) + 5)
   return result

print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))
