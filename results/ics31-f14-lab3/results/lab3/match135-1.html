def calculate_shipping(weight: float)-> int:
    '''Returns rate of shipping based on weight'''
    rate = 0.00
    if(weight < 2.00):
        rate = 2.00
    elif(weight > 2.00 and weight < 10.00):
        rate = 5.00
    else:
        rate = 5.00 + ((weight - 10) * 1.50)
    return(rate)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(5))
print(calculate_shipping(12))
print(calculate_shipping(1))

    '''Draws a nose based on parameters passed to the function'''
    my_canvas2.create_polygon(xcoord, ycoord, xcoord + (sides/2), ycoord + (sides *2), xcoord + sides, ycoord, fill = fillcolor)

    '''Draws a mouth based on parameters passed to the function'''
    my_canvas2.create_arc(xcoord,ycoord, xcoord + arcsize, ycoord + arcsize, extent = angle, fill = fillcolor)
