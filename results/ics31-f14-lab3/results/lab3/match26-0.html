def calculate_shipping(k:int) -> int:
    if k < 2:
        return 2.00
    elif k >= 2 and k < 10:
        return 5.00
    elif k >= 10:
        return 5.00 + 1.5 * (k -10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter

my_window_f = tkinter.Tk()



my_canvas_f = tkinter.Canvas(my_window_f, width=400, height=400)
my_canvas_f.pack() 

def draw_face(x:int, y:int):
    my_canvas_f.create_oval(x, y, x+250, y+300)
    return



def draw_eye(x:int, y:int):
    my_canvas_f.create_oval(x, y, x+80, y+35)
    my_canvas_f.create_oval(x+40, y+10, x+40+20, y+10+20, fill='brown')
    return



def draw_nose(x:int, y:int):
    my_canvas_f.create_line(x-10, y+10, x+10, y+10)
    my_canvas_f.create_line(x-10, y+10, x, y-5)
    my_canvas_f.create_line(x+10, y+10, x, y-5)
    return

def draw_mouth(x:int, y:int):
    my_canvas_f.create_arc(x-60, y, x+20, y)
    return

draw_face(60, 40)
draw_eye(90, 130)
draw_eye(200, 130)
