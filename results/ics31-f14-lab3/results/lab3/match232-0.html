def print_even_numbers (value: list) -> None:
    listLength = len(value)
    counter = 0
    
    while counter < listLength:
        if (value[counter] % 2 == 0):
            print(value[counter])
        counter = counter + 1
    return

def calculate_shipping (value: float) -> float:
    if (value < 2):
        return 2.00
    elif (value >= 2 and value < 10):
        return 5.00
    elif (value >= 10):
        over = value - 10
        return 5.00 + (1.50 * over)
assert calculate_shipping(1.5) == 2.00


assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(1.5))
print(calculate_shipping(7))
print(calculate_shipping(15))

def top_value(n: list) -> Book:
    count = 0
    value = 0
    book = 0

    while count < len(n):
        if (n[count].instock * n[count].price) > value:
            value = (n[count].instock * n[count].price)
            book = count
        count = count + 1
    return n[book]

def draw_face(nose_size: float, nose_position: float, mouth_size: float,
              mouth_position: float, eye_spacing: float, eye_color: str):
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=650, height=650)
    my_canvas.pack()
    my_canvas.create_oval(100, 15, 500, 550, fill = "Yellow")
    draw_mouth(mouth_size, mouth_position, my_canvas)
    draw_nose(nose_size, nose_position, my_canvas)
    draw_eye(200 - eye_spacing*1.75, 210, eye_color, my_canvas)
    draw_eye(300 + eye_spacing*1.75, 210, eye_color, my_canvas)

def draw_eye(x: float, y: float, color: str, canvas: tkinter.Canvas):
    canvas.create_oval(x, y, x + 100, y + 20, fill = "red")
    canvas.create_oval(x + 25, y, x + 75, y + 20, fill = color)
    canvas.create_oval(x + 40, y + 15, x + 60, y + 5, fill = "black")

def draw_nose(size: float, y_position: float, canvas: tkinter.Canvas):
    canvas.create_oval(300 - size*1.75, 300 - y_position, 300 + size/2, 300 + size)
    canvas.create_rectangle(300, 300 - y_position, 300 + size, 300 + size/2, fill = "Yellow", outline = "Yellow")
    canvas.create_line(300, 300 - y_position/2, 300, 300 - y_position - size)

def draw_mouth(size: float, y_position: float, canvas: tkinter.Canvas):
    canvas.create_oval(300 - size/2, 300 - y_position, 300 + size/2, 300 - y_position + size)
    canvas.create_rectangle(300 - size, 300 - y_position, 300 + size/2, 300 - y_position + size/1.5, fill = "Yellow", outline = "Yellow")
    
draw_face(70, 10, 100, -60, 40, "brown")
