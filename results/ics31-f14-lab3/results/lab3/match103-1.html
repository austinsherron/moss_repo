my_window = tkinter.Tk()
my_face = tkinter.Canvas(my_window, width = 500, height = 500)
my_face.pack()
def draw_eye(x, y, d, z:float) -> None:
#Create pair of eyes and irises
    my_face.create_oval(.2*d + x,.35*z+ y,.35*d +x,.425*z+ y, fill = 'white')
    my_face.create_oval(.65*d + x,.35*z+ y,.80*d +x,.425*z+ y, fill = 'white')
    my_face.create_oval(.225*d + x,.35*z+ y,.325*d +x,.425*z+ y, fill = 'brown')
    my_face.create_oval(.675*d + x,.35*z+ y,.775*d +x,.425*z+ y, fill = 'brown')
    

def draw_nose(x, y, d, z: float) -> None:
#Create nose
    my_face.create_arc(.33*d +x, .5*z + y, .63*d + x, y+.8*z, fill= 'tan')
    my_face.create_oval(.5*d +x, .615*z + y, .525*d + x, y+.645*z, fill= 'pink')
    my_face.create_oval(.575*d +x, .615*z + y, .6*d + x, y+.645*z, fill= 'pink')
def draw_mouth(x, y, d, z: float) -> None:
#Create aghasted mouth
    my_face.create_oval (x+.375*d, y+.8*z, x+.675*d, y+.9*z , fill = 'pink')

def draw_face(x, y, d, z: float) -> None:
#Create face + hair and all of the other functions
    my_face.create_rectangle(x+.2*d, y+.5*z, x+.8*d, y-.3*z, fill = 'pink')
    my_face.create_oval(x, y, x+d, y+z, fill = 'tan')
    draw_eye(x,y,d,z)
    draw_nose(x,y,d,z)
    draw_mouth(x,y,d,z)


draw_face(100, 100, 200, 250)
draw_face(10, 300, 100, 100)
draw_face(300,300,100,150)
#x and y is the coordinates of the face,  d is the horizontal distance o/ face
#z is the vertical distance o/ face
tkinter.mainloop()
