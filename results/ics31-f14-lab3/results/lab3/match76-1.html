def calculate_shipping (n: float) -> str:
    '''Returns shipping rates based on weight'''
    if n < 2:
        return (2.00)
    elif n >= 2 and n <10:
        return (5.00)
    else:
        return (5+(n-10)*1.5)

assert calculate_shipping (7) == 5.00
assert calculate_shipping (15) == 12.5

def create_tkinter () -> 'canvas variable':
    '''creates the tkinter window'''
    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=900, height=1000)  # Create a 900x1000 canvas to draw on
    my_canvas.pack() # Put the canvas into the window
    return (my_canvas)

def counting_books (n:list, s:int) -> int:
    '''Returns the amount of book before and after the split date (s)'''
    titles_after = []
    titles_before = []
    for i in range(len(n)):
        if n[i].year >= s:
            titles_after.append(n[i])
        else:
            titles_before.append(n[i])
            
    titles_after_instock = 0
    titles_before_instock = 0
    for i in range(len(titles_after)):   #Adding the number of books after 2000
        titles_after_instock = titles_after_instock + titles_after[i].instock
    for i in range(len(titles_before)):    #Add the number of books before 2000
        titles_before_instock = titles_before_instock + titles_before[i].instock

    
    if titles_before_instock > titles_after_instock: #prints number of books before 2000 is > than after 2000, prints statement
        print("More titles before ", s, " (", titles_before_instock, " vs. ", titles_after_instock, ")", sep='')
    else: #opposite of the if statement
        print("More titles ", s, " or later", " (", titles_after_instock, " vs. ", titles_before_instock, ")", sep='')
