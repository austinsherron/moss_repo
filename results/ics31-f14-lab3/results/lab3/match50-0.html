def calculate_shipping (weight: float) -> float:
    ''' Calculates the shipping price based off of the weight '''
    rate = 0


    if weight < 2:
        rate = 2.00
    elif 2 <= weight < 10:
        rate = 5.00
    else:
        rate = 5.00 + (1.50 * (weight - 10))
    return rate
assert calculate_shipping(1) == 2.0
assert calculate_shipping(2) == 5.0
assert calculate_shipping(3) == 5.0
assert calculate_shipping(10) == 5.0
assert calculate_shipping(11) == 6.5

def inventory_value (book: Book) -> float:
    ''' Returns the inventory value of a specific Book '''
    return book.instock * book.price

def top_value (books: list) -> Book:
    ''' Loops through the list of books and returns the book with the highest-
        valued inventory. '''
    highest_value = books[0]
    for book in books:
        if inventory_value(book) > inventory_value(highest_value):
            highest_value = book
    return highest_value

highest_value = top_value(BSI)

    ''' Parameters are upper left and lower right coordinates of the eye and color.
        Width of iris is 80. Width of pupil is 40. '''

def draw_nose (x1: int, y1: int, w: int, h: int, color: str) -> None:
    ''' (x1, y1) is the lower left corner of the triangle. Other points based
        off of the width and height '''
    my_canvas.create_polygon(x1, y1, x1 + w, y1, x1 + w/2, y1 - h, fill = color)

def draw_mouth (x1: int, y1: int, w: int, h: int) -> None:
    ''' (x1, y1) is upper left corner of the actual mouth. Width and height
        are the width and height of the mouth. '''
    my_canvas.create_arc(x1, y1 - h, x1 + w, y1 + h, start = 180, extent = 180)
