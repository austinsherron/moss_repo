def calculate_shipping (n:int)->int:
    '''Return the rate of shipping based on the pounds input'''
    if n < 2:
        return 2.00
    if 2 < n < 10:
        return 5.00
    if n > 10:
        return 5.00 + (1.50 * (n-10))

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
