def calculate_shipping(weight: int) -> int:
    'Calculates the price of shipping given its weight'
    if weight < 2:
        return 2.00
    elif weight >= 2 and weight < 10:
        return 5.00
    elif weight >= 10:
        return (5.00 + 1.50 * (weight-10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(5) == 5.00
assert calculate_shipping(11) == 6.50
print(calculate_shipping(1.5))
print(calculate_shipping(5))
print(calculate_shipping(15))
