def calculate_shipping (n:float) -> float:
    '''calculate the shipping cost'''
    if n < 2:
        return (2.00)
    elif 2 <= n < 10:
        return (5.00)
    elif n >=10:
        return ((5.00) + (1.50* (n-10)))
