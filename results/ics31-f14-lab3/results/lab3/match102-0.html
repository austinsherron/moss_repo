def abbreviate(month: str) -> str:
    '''Return the first three characters of given string.'''
    return month[0 : 3]

assert abbreviate('January') == 'Jan'
assert abbreviate('abril') == 'abr'
assert abbreviate('string') == 'str'

def find_area_square(length: float) -> float:
    '''Return area of square with given side'''
    return length ** 2

assert find_area_square(1) == 1
assert find_area_square(5) == 25
assert find_area_square(2.5) == 6.25

def find_area_circle(radius: float) -> float:
    '''Returns the area of circle with given radius'''
    return 3.14159 * (radius ** 2)
assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975
assert find_area_circle(10) == 314.159


def print_even_numbers(numbers: list):
    '''Prints the even numbers in a given list of numbers'''
    for i in numbers:
        if i % 2 == 0:
            print(i)
    return

print_even_numbers([2, 47, 31, 99, 20, 19, 23, 105, 710, 1004])


def calculate_shipping(weight: float) -> float:
    '''Returns the price for shipping with given weight of package'''


    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5.00 + ((weight - 10) * 1.50)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
assert calculate_shipping(20) == 20.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(10) == 5.00


import tkinter

def create_square(x: float, y: float, length: float):
    '''Draws a square where x and y are the coordinates of the upper left
    corner and length is the side of square'''
    my_window = tkinter.Tk()

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    my_canvas.create_rectangle(x, y, x + length, y + length, fill = 'yellow')

    tkinter.mainloop()
    return

create_square(100, 100, 100)
create_square(80, 100, 50)


def create_circle(x: float, y: float, diameter: float):
    '''Draws a circle where x and y are the coordinates of the upper left
    corner and diameter is the diameter of circle'''
    my_window = tkinter.Tk()

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    my_canvas.create_oval(x, y, x + diameter, y + diameter, fill = 'red')
    tkinter.mainloop()
    return

create_circle(100, 100, 100)
create_circle(150, 90, 75)

def inventory_value(book: Book) -> float:
    '''Returns the value of the inventory of the book'''
    return book.price * book.instock

assert inventory_value(BSI[0]) == 660

def top_value(BL: 'list of Books') -> Book:
    ''' Returns the book with the highest inventory value'''
    books = sorted(BL, key = inventory_value)
    return books[-1]

                              genre='Technology', year=2011, price=33.0,
                              instock=50)

      "at a value of $", inventory_value(new_book))

def draw_eye(x: float, y: float):
    '''Draws an eye where x and y are the coordinates of the upper left
    corner and diameter is the diameter of the eye'''

    my_canvas.create_oval(x, y, x + 100, y + 100, fill = 'white')
    my_canvas.create_oval((x + 25), (y + 25), (x + 75),
                          (y + 75), fill = 'blue')
    my_canvas.create_oval((x + 30), (y + 30),
                          (x + 60), (y + 60), fill = 'black')
    return

def draw_nose(x: float, y: float):
    '''Draws a nose where x and y are the coordinates of the upper left corner'''

    my_canvas.create_line(x, y, (x + 100), (y + 100), x, (y + 100))
    return

def draw_mouth(x: float, y: float):
    '''Draw a mouth where x and y are the coordinates of the upper left corner'''

    my_canvas.create_line(x, y, (x + 200), y)
    return

def draw_face(x: float):
    '''Draws a face where (x, x) is the coordinate of the upper left corner'''

    draw_eye(x, x)
    draw_eye((x + 400), x)
    draw_nose((x + 200), (x + 100))
    draw_mouth((x + 100), (x + 300))
    return

my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()
