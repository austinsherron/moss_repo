def calculate_shipping(value: float) -> float:
    "Calculates shipping price based on weight in lbs"
    if value < 2:
        return 2.00
    elif value >= 2 and value <= 10:
        return 5.00
    else:
        pounds_over = value - 10
        return 5.00 + (pounds_over * 1.5)
        


assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(value: Book) -> float:
    ''' Returns the value of a book that is in the inventory'''
    return value.price * value.instock

assert inventory_value(BSI[0]) == 2752.20000000000045
assert inventory_value(BSI[2]) == 1120.8999999999999
assert inventory_value(BSI[4]) == 760.1000000000001


def top_value(value: [Book]) -> Book:
    '''Takes a list of Books and gives back highest value Book'''
    high_book = Book('Dummy', 'Dummy', 'Dummy', 0, 0.01, 0)
    for x in range(0, len(value)):
        if inventory_value(value[x]) > inventory_value(high_book):
            high_book = value[x]
    return high_book

book1 = top_value(BSI)
print("The highest-value book is", book1.title, "by", book1.author, "at a value of $", inventory_value(book1))

my_window = tkinter.Tk()    # Create the graphics window
my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window
    
def create_mouth(lips_color: str):
    '''Create a mouth'''
    my_canvas.create_polygon(100, 300, 250, 315, 400, 300, 300, 325, 200, 325, 100, 300, fill=lips_color)
    my_canvas.create_polygon(100, 300, 250, 285, 400, 300, 300, 275, 250, 280, 200, 275, 100, 300, fill=lips_color)
    my_canvas.create_polygon(100, 300, 250, 285, 400, 300, 250, 315, 100, 300, fill='#660000')
    return

def create_nose(size: str):
    '''Create a Nose (big, normal, small)'''
    if size == 'normal':
        my_canvas.create_line(225, 150, 210, 250)
        my_canvas.create_line(275, 150, 290, 250)
        my_canvas.create_line(210, 250, 250, 265)
        my_canvas.create_line(290, 250, 250, 265)
    elif size == 'small':
        my_canvas.create_line(235, 150, 220, 250)
        my_canvas.create_line(265, 150, 280, 250)
        my_canvas.create_line(220, 250, 250, 265)
        my_canvas.create_line(280, 250, 250, 265)
    elif size == 'big':
        my_canvas.create_line(210, 150, 195, 250)
        my_canvas.create_line(290, 150, 305, 250)
        my_canvas.create_line(195, 250, 250, 265)
        my_canvas.create_line(305, 250, 250, 265)
    return

def create_eyes(eye_color: str):
    '''Creates two eyes'''
    my_canvas.create_oval(100, 100, 200, 150, fill='white')
    my_canvas.create_oval(125, 100, 175, 150, fill=eye_color)
    my_canvas.create_oval(140, 115, 160, 135, fill='black')

    my_canvas.create_oval(300, 100, 400, 150, fill='white')
    my_canvas.create_oval(325, 100, 375, 150, fill=eye_color)
    my_canvas.create_oval(340, 115, 360, 135, fill='black')
    return

def create_face(face_color: str, eye_color: str, nose_size: str, lips_color: str):
    '''Creates a face'''
    
    #Create the face
    my_canvas.create_oval(25, 25, 475, 425, fill=face_color)

    #Call face drawing functions
    create_mouth(lips_color)
    create_nose(nose_size)
    create_eyes(eye_color)
    
    tkinter.mainloop()          # Combine all the elements and display the window
    return
