def calculate_shipping(weight:int)->float:
    '''returns the shipping given weight of the package'''

        result=2.00
    elif 2<=weight<10:
        result=5.00
    else:
        result=(((weight-10)*1.50)+5.00)
    return result

import tkinter             

my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()


def draw_eye (x:int,y:int)->None:
    '''returns a drawing of an eye based on oval coordinates'''
                
    my_canvas.create_oval(x, y, x+200, y-100, fill='white')
    my_canvas.create_oval(x+50, y, x+150, y-100, fill='blue')
    my_canvas.create_oval(x+75, y-75, x+125, y-25, fill='black')

    
def draw_nose(x:int,y:int,z:int)->None:
    '''returns a drawing of a nose based on oval coordinates'''
    my_canvas.create_oval(x, y, x + z, y + z, fill='pink')


def draw_mouth(x:int,y:int,z:int)->None:
    '''returns drawing of a mouth based on line coordinates'''
    my_canvas.create_line(x,y,x+z,y)

def draw_face(x:int, y:int)->None:
    '''returns the drawing of a face based on predefined eye,nose, and mouth funtions'''
    draw_eye(x+ 75, y+300 )
    draw_eye(x+325, y+300)
    draw_nose(x + 250, y+400,100)
    draw_mouth(x + 200, y+600, 200)


draw_face(0,0) #First call to draw_face

draw_face(340,340) #Second call to draw_face with different values
tkinter.mainloop()
