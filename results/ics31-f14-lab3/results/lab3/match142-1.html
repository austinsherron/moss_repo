def calculate_shipping(weight: float) -> float:
    if (weight < 2):
        rate = 2.00
    elif (weight < 10 and weight >= 2):
        rate = 5.00
    elif (weight >=10):
        rate = 5.00 + 1.50*(weight-10)
    return rate

assert calculate_shipping(1.5) == 2.00


assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window
my_canvas = tkinter.Canvas(my_window, width=1280, height=1024)  # Create a 500x50
my_canvas.pack()            # Put the canvas into the window

    
def draw_eye(x: int, y:int, color: str) -> None:
    # Put the canvas into the window
    my_canvas.create_oval(x, y, x+200, y-150, fill='white') # Draw orange line
    my_canvas.create_oval(x+40, y-30, x+160, y-120, fill=color) # Draw orange line
    my_canvas.create_oval(x+70, y-55, x+130, y-95, fill='black') # Draw orange line
  
def draw_nose(x: int, y:int, size: int):
    my_canvas.create_oval(x, y, x+50+size, y-60-size, fill='black') # Draw orange line

def draw_mouth(x: int, y:int):
    my_canvas.create_oval(x, y, x+400, y-100, fill = 'black')

def draw_face(x: int, y:int, color: str, eye_color: str, nose_size: int):
    my_canvas.create_oval(x+750, y, x+20, y+600, fill =color)
    draw_eye(100+x,250+y, eye_color)
    draw_eye(400+x, 250+y, eye_color)
    draw_nose(320+x,350+y, nose_size)
    draw_mouth(200+x,500+y)


draw_face(10,0, "red", "green", 50) #(Starting x, Starting y, face color, eye color, nose size)
draw_face(500, 300, "blue", "yellow", 20)
tkinter.mainloop()          # Combine all the elements and display the window
