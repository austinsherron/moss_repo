def abbreviate(string : str)-> str:
    "return the first three letters of the parameter"
    return (string[0] + string [1] + string [2])

assert abbreviate('april') == 'apr'

def calculate_shipping(weight : float)-> float:
    if weight < 2:
        return 2.00
    elif weight < 10:
        return 5.00
    else:
        return 5.00 + 1.5 * (weight-10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import random
import tkinter

tk = tkinter.Tk()
canvas = tkinter.Canvas(tk, width = 500, height = 500)

colors = ['red', 'blue', 'light blue', 'brown', 'black']

def draw_eye(x: int , y: int, color: str):
    canvas.create_oval((x-25),(y-25),(x+25), (y+25), fill = color)
    canvas.create_oval((x-12),(y-12),(x+12), (y+12), fill = 'black')
    canvas.pack()

def draw_nose( x:int, y:int):
    canvas.create_oval(x-8,y-8,x+8,y+8, fill = 'black')
    canvas.pack()

def draw_mouth(x:int , y:int ):
    canvas.create_arc(x-75,y-75,x+75,y+75,start = 180, extent = 180)
    canvas.pack()

def draw_face(color: str):
    canvas.create_oval(100,50,400,375)
    draw_eye(200, 150,color )
    draw_eye(300,150,color)
    draw_nose(250,200)
    draw_mouth(250,250)

draw_face(colors[random.randrange(0, len(colors))])
