def calculate_shipping(x:int):
    if x<2:
       result = 2.00
    elif 2<x<10:
       result = 5.00
    elif x>10:
       result = 5.00+1.5*(x-10)
    return result
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
