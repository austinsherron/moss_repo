def print_even_numbers(integers:list):
    '''Takes the list of integers and prints out even numbers'''
    for x in range(0,len(integers)):
        if (integers[x] % 2 == 0):
            print(integers[x])
    return

def calculate_shipping(weight:float) -> float:
    '''Takes weight of package and returns price for shipping'''
    if (weight <= 2):
        return 2.00
    elif ((weight >= 2) and (weight < 10)):
        return 5.00
    else:
        return 5.00 + ((weight - 10) * 1.50)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book_parameter: Book) -> float:
    '''Returns the inventory value'''
    return book_parameter.price * book_parameter.instock

def top_value(book_list_parameter: []) -> Book:
    '''Returns the book with the highest inventory'''
    max_index = 0

    for x in range(0, len(book_list_parameter) - 1):
        if (inventory_value(book_list_parameter[max_index]) < inventory_value(book_list_parameter[x + 1])):
            max_index = x + 1

    return book_list_parameter[max_index]

print('The highest-value book is ', (top_value(BSI)).title,
    ' at a value of $', inventory_value(top_value(BSI)), sep='')
