def calculate_shipping(weight:float) -> float:
    if weight < 2:
        return 2.00
    elif 2 <= weight < 10:
        return 5.00
    else:
        return 5.00 + (weight * 1.50 - weight)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

        my_canvas.create_oval(x+20,y+5,x+80,y+45)
        my_canvas.create_oval(x+40,y+10,x+60,y+35)

def draw_mouth(x,y:int):
        my_canvas.create_oval(x,y,x+100,y+50)

def draw_nose(x,y:int):
        my_canvas.create_rectangle(x,y,x+30,y+30)

    draw_mouth(x+75,y+100)
    draw_eye(x,y)
    draw_eye(x+150,y)
    draw_nose(x+110,y+55)
