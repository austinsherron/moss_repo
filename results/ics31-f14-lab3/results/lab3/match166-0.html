def calculate_shipping(weight:float) -> float:
    "return the price for shipping of the package given the weight"
    if weight<2:
        return 2
    elif weight<10:
        return 5
    else:
        return 5 + 1.50*(weight-10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value (book: Book) -> float:
    " Return value of inventory of given book"
    return book.price * book.instock

def top_value(books: list) -> Book:
    " Return the book with the highest inventory value given a list of books"
    top=books[0]
    for i in range(len(books)):
        if inventory_value(books[i]) > inventory_value(top):
            top=books[i]
    return top

print ("The highest-value book is", top_value(BSI).title, "by", top_value(BSI).author, "at a value of $", inventory_value(top_value(BSI)))

def draw_face():
    "Draws a face"
    my_window=tkinter.Tk()
    my_canvas=tkinter.Canvas(my_window, width= 500, height=500)
    my_canvas.create_oval(100, 50, 400, 450)
    draw_eye(150, 200, 40, my_canvas)
    draw_eye(275, 200, 40, my_canvas)
    draw_nose(250, 250, 60, my_canvas)
    draw_mouth(200, 360, 110, my_canvas)
    my_canvas.pack()
    tkinter.mainloop()

def draw_eye(x: int, y: int, diameter: int, canvas):
    "Draws an eye given upper left x and y coordinate, iris diameter, and canvas"
    canvas.create_oval(x, y, x + 2 * diameter, y + diameter)
    canvas.create_oval(x + diameter / 2, y, x + 3 * diameter / 2, y + diameter, fill="blue")
    canvas.create_oval(x + diameter / 2 + diameter * 0.25, y + diameter * 0.25, x + 3 * diameter / 2 - diameter * 0.25, y + diameter - diameter * 0.25, fill="black")

def draw_nose(x: int, y: int, height: int, canvas):
    "Draws a nose given upper left x and y coordinate, height of the nose, and canvas"
    canvas.create_polygon(x, y, x, y + height, x + height * 0.5, y + height)

def draw_mouth(x:int, y:int, length: int, canvas):
    "Draws a mouth given upper left x and y coordinate, length of mouth, and canvas"
    canvas.create_line(x, y, x + length, y)
