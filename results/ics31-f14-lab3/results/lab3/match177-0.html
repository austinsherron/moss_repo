    '''returns area of a circle using the radius'''

def calculate_shipping(w: int) -> int:
    '''returns shipping cost'''
    if w < 2:
        return 2.00
    elif w < 10:
        return 5.00
    elif w >= 10:
        return 5.00 + (w - 10) * 1.50
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(n: list) -> float:
    '''returns the value of the book instock'''

    '''returns the highest value Book from a list'''
