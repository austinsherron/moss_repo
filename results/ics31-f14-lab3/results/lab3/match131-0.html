import tkinter              # Load the library; do this just once per program

def create_square(x,y,length_of_side):
    my_window = tkinter.Tk()    # Create the graphics window

    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window

    
    my_canvas.create_line(x, y, x+length_of_side, y)
    my_canvas.create_line(x, y, x, y + length_of_side)
    my_canvas.create_line(x, y + length_of_side, x +length_of_side ,y+length_of_side)
    my_canvas.create_line(x + length_of_side, y, x + length_of_side, y + length_of_side)

    tkinter.mainloop()     
create_square(100,200,30)

import tkinter

my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=5000, height=1000)  #


def draw_eye(x,y,z,a,b):

    my_canvas.pack()
    my_canvas.create_oval((x), (y),(x+z), (y+z),fill = 'green') #small eye
    my_canvas.create_oval((x+a), (y),(x+z+a), (y+z),fill = 'green') #small eye
    my_canvas.create_oval((x+b), (y+b),(x+z), (y+z),fill = 'black')
    my_canvas.create_oval((x+a+b), (y+b),(x+z+a), (y+z),fill = 'black')
    
    
def draw_nose(x,y,z,a,b): 

    my_canvas.pack()

    my_canvas.create_oval((x+z),(y+z),(x+2*z),(y+2*z))

def draw_mouth(x,y,z,a,b):
    my_canvas.pack()

    my_canvas.create_line((x),(y+3*z),(x+3.3*z),(y+3*z))


    


def draw_face(x,y,z,a,b):

    my_canvas.pack()
    my_canvas.create_oval(x-1*z,y-1*z,x+4.5*z,y+4.5*z)
    draw_eye(x,y,z,a,b)
    
    draw_nose(x,y,z,a,b)
    
    draw_mouth(x,y,z,a,b)
    
draw_face(100,100,150,300,25)
draw_face(1000,100,150,300,25)
