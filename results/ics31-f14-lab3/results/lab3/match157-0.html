    #take the given string and prints the first three characters
    return n[0] + n[1] + n[2]

def calculate_shipping (d: float) -> float:
    #calculate the shipping cost of certain packages weight
    if d < 2:
        return 2.00
    elif 2 <= d < 10:
        return 5.00
    else:
        return 5.00 + 1.50 * (d - 10)

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter 

def draw_face():
    #draws a yellow face with blue eyes and white mouth, and nose
    my_window = tkinter.Tk()   

    my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)  
    my_canvas.pack()

    my_canvas.create_oval(50, 50, 950, 950, fill='yellow')
    def draw_eye1():
        #draws left eye
        my_canvas.create_oval(100, 160, 400, 340, fill='white')
        my_canvas.create_oval(185, 190, 325, 310, fill='lightblue')
        my_canvas.create_oval(215, 210, 290, 290, fill='black')

    draw_eye1()

    def draw_eye2():
        #draws right eye
        my_canvas.create_oval(600, 160, 900, 340, fill='white')
        my_canvas.create_oval(685, 190, 825, 310, fill='lightblue')
        my_canvas.create_oval(715, 210, 790, 290, fill='black')

    draw_eye2()

    def draw_nose():
        #draws two lines that make up the nose
        my_canvas.create_line(450, 500, 520, 400, fill='black')
        my_canvas.create_line(450, 500, 550, 500, fill='black')
        
    draw_nose()

    def draw_mouth():
        #draws a white circle that is the mouth
        my_canvas.create_oval(350, 700, 650, 850, fill='white')
    draw_mouth()
    
    tkinter.mainloop()
