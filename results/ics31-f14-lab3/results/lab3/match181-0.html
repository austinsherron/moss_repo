def calculate_shipping(value:float) -> float:
    " Returns price of shipping for package "
    if value < 2:
        return(2.0)
    elif value == 2 or value < 10:
        return(5.0)
    elif value == 10:
        return(5.0)
    elif value > 10:
        value = 5 + 1.50 * (value - 10)
        return(value)
               
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width = 1500, height = 1500)
my_canvas.pack()

# face outline
def draw_faceoutline(x: int, y: int):
    " Creates face outline at coordinates (x, y) "
    my_canvas.create_oval(x - 260, y - 300, x + 260, y + 300)

# eyes


def draw_eye(x: int, y:int):
    " Creates eyes at coordinates (x, y) "
    my_canvas.create_oval(x - 60, y - 20 , x + 60, y + 20) 
    my_canvas.create_oval(x - 20, y - 20, x + 20, y + 20, fill= 'blue'), 
    my_canvas.create_oval(x - 10, y - 10, x + 10, y + 10, fill='black')

# nose
def draw_nose(x: int, y: int):
    " Creates nose at coordinates (x, y) "
    my_canvas.create_line(x, y, x + 50, y + 25)
    my_canvas.create_line(x, y + 50, x + 50, y + 25)

# mouth
def draw_mouth(x: int, y: int):
    " Creates mouth at coordinates (x, y) "
    my_canvas.create_line(x, y, x + 100, y)

# whole face
def draw_face(x: int, y:int):
    " Creates whole face at coordinates (x, y) "
    draw_faceoutline(x + 350, y + 350)
    draw_eye(x + 240, y + 300)
    draw_eye(x + 490, y + 300)
    draw_nose(x + 350, y + 400)
    draw_mouth(x + 310, y + 520)

# wow twinsies
