def calculate_shipping(weight: float) -> float:
    if (weight < 2):
        return 2.00
    elif (weight < 10):
        return 5.00
    else:
        return 5.00 + 1.50 * (weight - 10)
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

print(calculate_shipping(15))

def inventory_value(book: Book) -> float:
    return (book.price * book.instock)

def top_value(lst: list) -> Book:
    return sorted(lst, key = inventory_value)[-1]

my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width = 600, height = 600)
my_canvas.pack()
def draw_eye(x: int, y: int):
    my_canvas.create_oval(x+25, y+100, x+125, y+150)
    my_canvas.create_oval(x+50, y+100, x+100, y+150)
    my_canvas.create_oval(x+60, y+110, x+90, y+140)
def draw_nose(x: int, y: int):
    my_canvas.create_line(x+165, y+100, x+125, y+200)
    my_canvas.create_line(x+125, y+200, x+180, y+200)

def draw_mouth(x: int, y: int):
    my_canvas.create_oval(x+100, y+250, x+210, y+280) 


def draw_face(x: int, y: int, diameter: int) :
    my_canvas.pack()
    my_canvas.create_oval(x, y, x+diameter, y+diameter)
    draw_eye(x, y)
    draw_eye(x+160, y)
    draw_nose(x, y)
    draw_mouth(x, y)    

draw_face(150,150,300)
