#calculate_shipping returns the price for shipping the package
def calculate_shipping(weight):
  
    if(weight<2):
        price=float(2.00)
        return(price)
    elif(weight<10):
        price=float(5.00)
        return(price)
    elif(weight>10):
        price=float(5.00+1.50*(weight-10))
        return(price)
print(calculate_shipping(1.5))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50
