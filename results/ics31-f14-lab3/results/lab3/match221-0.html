def find_area_square(s:int) -> int:
    "Find the area of the square given the length of one side."
    return(s ** 2)
assert find_area_square(1) == 1
assert find_area_square(5) == 25


def find_area_circle(r: int) -> float:
    "Find the area of the circle given the radius of the circle."
    return( 3.14159 * (r ** 2))
assert find_area_circle(1) == 3.14159
assert find_area_circle(5) == 78.53975

def print_even_numbers (lst:list):
    "Print the even numbers in a list."
    for element in lst:
        if element % 2 == 0:
            print(element)

def calculate_shipping(weight:float)-> float:
    "Calculates the shipping price given the weight of the package."
    if weight < 2:
        return(2.00)
    elif weight >= 2 and weight < 10:
        return(5.00)
    elif weight >= 10:
        return(5.00 + (1.50 * (weight - 10)))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye(canvas: tkinter.Canvas, starting_x: int, starting_y: int, color:str) -> tkinter.Canvas:
    canvas.create_oval(starting_x,starting_y, starting_x + 50, starting_y + 50, outline = color)
    return canvas

def draw_mouth(canvas: tkinter.Canvas, color:str) -> tkinter.Canvas:
    canvas.create_line(200, 350, 300, 350, fill = color)
    return canvas

def draw_nose(canvas: tkinter.Canvas, color:str) -> tkinter.Canvas:
    canvas.create_polygon(250, 230, 200, 300, 300, 300, fill = color)
    return canvas

# One-Time Pad

def xor_sequence(msg: str, key: str) -> int:
    "XOR's the sequences msg and key, assumes len(msg) == len(key)."
    i = 0
    result = ""
    for i in range(0,len(msg)):
        char_a = int(msg[i])
        char_b = int(key[i])
        result = result + str(char_a ^ char_b)
    return result

message = '01010101'
key = '10101010'
ciphertext = xor_sequence(message,key)
print("The message is", message)
print("The cipher is:", ciphertext)
print("The recovered message is", xor_sequence(key,ciphertext))
