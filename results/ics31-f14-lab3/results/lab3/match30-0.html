def calculate_shipping(weight: int) -> str:
    if weight < 2:
        cost = "$2.00"
    elif weight >= 2 and weight <10:
        cost = "$5.00"
    else:
        cost = "$" + str(5+ 1.5*(weight-10))
    return cost
assert calculate_shipping(1) == '$2.00'
assert calculate_shipping(2) == '$5.00'
assert calculate_shipping(5) == '$5.00'
assert calculate_shipping(10) == '$5.0'
assert calculate_shipping(12) == '$8.0'
print('Calculate shipping for 3 pounds', calculate_shipping(3))
print('Calculate shipping for 6 pounds', calculate_shipping(6))
print('Calculate shipping for 11 pounds', calculate_shipping(11))
