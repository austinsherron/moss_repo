    '''Takes the radius of a circle and returns the area of the circle'''

    return pi*radius**2

def calculate_shipping(weight: float) -> float:
    '''Return shipping price with package weight'''
    if weight < 2:
        return 2.00
    if weight >= 2 and weight<10:
        return 5.00
    if weight >=10:
        return 5.00 + (weight-10)*1.5
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value(book: Book) -> float:
    '''calculates inventory value of book'''
    price = book.price
    stock = book.instock
    return price * stock


def top_value (booklist: list) -> Book:
    '''list the highest inventory value book'''
    newlist = sorted(booklist, key=inventory_value)
    return newlist[-1]
print('The highest book value is', top_value(BSI).title,
      'by', top_value(BSI).author, 'at a value of',
      inventory_value(top_value(BSI)))

import tkinter
def draw_face():
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width = 500, height = 500)    
    my_canvas.pack()
    my_canvas.create_oval(0,0,500,500,fill='orange')
    draw_eye(my_canvas,0,100)
    draw_eye(my_canvas,200,100)
    draw_nose(my_canvas)
    draw_mouth(my_canvas)
    tkinter.mainloop()

def draw_eye(canvas, x: int, y: int):
    canvas.create_oval(75+x,y,225+x,y+150,fill='white')
    canvas.create_oval(125+x,y+50,175+x,y+100,fill='red')
    canvas.create_oval(140+x,y+65,155+x,y+80,fill='black')
    return

def draw_mouth(canvas):
    canvas.create_arc(300,500,0,350,fill='skyblue')
    return

def draw_nose(canvas):
    canvas.create_polygon(250,200,200,300,300,300,fill='green')
    return

draw_face()
