def calculate_shipping(weight: float) -> float:
    "Calculates the coast to ship a package."
    if weight < 2:
        rate = 2.00
    elif 2 <= weight < 10:
        rate = 5.00
    else:
        rate = 5.00 + 1.50 * (weight - 10.0)
    return rate

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye(x1: int, y1: int, x2: int, y2: int, color: str) -> None:
    "Draws an eye based on inputed points."
    dx = x2 - x1
    dy = y2 - y1
    my_canvas.create_oval(x1 + dx/4, y1, x2 - dx/4, y2, fill = color)  # Draw iris
    my_canvas.create_oval(x1 + dx/3, y1 + dy/5, x2 - dx/3, y2 - dy/5, fill = 'black')  # Draw pupil
    my_canvas.create_oval(x1, y1, x2, y2)                  # Draw eye whites



def draw_nose(x1: int, y1: int, x2: int, y2: int, x3: int, y3: int) -> None:
    "Draws a nose given 3 points, the second point being the intersection."
    my_canvas.create_line(x1, y1, x2, y2)
    my_canvas.create_line(x2, y2, x3, y3)



def draw_mouth(x1: int, y1: int, x2: int) -> None:
    "Draws a straight-line mouth given a point and another x-coordinate."
    my_canvas.create_line(x1,y1,x2,y1)



def draw_face(x1: int, y1: int, x2: int, y2: int, eye_color: str) -> None:
    "Draws the entire face given two points, scaling included."    
    dx = x2 - x1
    dy = y2 - y1
    my_canvas.create_oval(x1, y1, x2, y2)
    draw_eye(x1 + dx*0.25, y1 + dy*0.25, x1 + dx*0.45, y1 + dy*0.35, eye_color)
    draw_eye(x1 + dx*0.55, y1 + dy*0.25, x1 + dx*0.75, y1 + dy*0.35, eye_color)
    draw_nose(x1 + dx*0.5, y1 + dy*0.4, x1 + dx*0.45, y1 + dy*0.55, x1 + dx*0.55, y1 + dy*0.55)
    draw_mouth(x1 + dx*0.35, y1 + dy*0.67, x1 + dx*0.65)
