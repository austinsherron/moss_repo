def calculate_shipping(n:float)->float:
  '''Takes weight and gives shipping price'''
  if n < 2:
    return 2.00
  elif n >= 2 and n < 10:
    return 5.00
  else:
    return (5.00 + (1.50 * (n-10)))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye(yo_canvas):
  x = 100
  yo_canvas.create_oval(x, x + 100, x + 300,x + 225, fill = 'snow2') #left outer eye
  yo_canvas.create_oval(x + 90, x + 100,x + 215,x + 225, fill = 'medium spring green') #pupil
  yo_canvas.create_oval(x + 130,x + 140,x + 180,x + 190, fill = 'black') #iris
  
  yo_canvas.create_oval(x + 400,x + 100,x + 700,x + 225, fill = 'snow2') #right outer eye
  yo_canvas.create_oval(x + 490,x + 100,x + 615,x + 225, fill = 'medium spring green') #pupil
  yo_canvas.create_oval(x + 530,x + 140,x + 580,x + 190, fill = 'black') #iris

  yo_canvas.create_line(x + 50,x + 75,x + 280,x + 75) #left eyebrow
  yo_canvas.create_line(x + 420,x + 40,x + 650,x + 40) #right eyebrow
