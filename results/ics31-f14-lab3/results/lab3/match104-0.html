    return n ** 2

def calculate_shipping (n:int):
    if n < 2:
        return 2
    elif n < 10:
        return 5
    else:
        return 5 + (1.50 * (n -10))

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=500, height=500)
my_canvas.pack()

def draw_eye (x:int, y:int, d:int):
    x2 = x + d
    y2 = y + d
    inner_x = x + (3/4 * d)
    inner_x2 = x2 - (3/4 * d)
    inner_y = y + (3/4 * d)
    inner_y2 = y2 - (3/4 * d)
    outer_x = x + (2/3 * d)
    outer_x2 = x2 - (2/3 * d)
    outer_y = y + (2/3 * d)
    outer_y2 = y2 - (2/3 * d)
    my_canvas.create_oval(x,y,x2,y2, fill='white')
    my_canvas.create_oval(inner_x,inner_y,inner_x2,inner_y2, fill='blue')
    my_canvas.create_oval(outer_x,outer_y,outer_x2,outer_y2, fill='black')
def draw_face(x:int, y:int,d:int):
    x2 = x + d
    y2 = y + d
    my_canvas.create_oval(x,y,x2,y2, fill='yellow')
def draw_mouth (x:int, y:int, d:int):
    x2= x + d
    my_canvas.create_line(x,y,x2,y, fill='black')
def draw_nose(x:int,y:int, d:int):
    x2= x + d
    x3= x + d
    y3= y - d
    my_canvas.create_line(x,y,x2,y, fill='black')
    my_canvas.create_line(x,y,x3,y3,fill='black')
draw_face(0,0,500)
draw_eye(100,75,100)
draw_eye(300,75,100)
draw_mouth(100,350,300)
draw_nose(150,250,100)
