def calculate_shipping (calc : float) -> float:
    "Calculates the shipping rate"
    if calc < 2:
        result = (2.00)
    elif 2 <= calc < 10:
        result = (5.00)
    elif calc >= 10:
        result = (5 + (calc - 10) * 1.5)
    return result
assert calculate_shipping(1) == 2.00
assert calculate_shipping(5) == 5.00
assert calculate_shipping(11) == 6.50
