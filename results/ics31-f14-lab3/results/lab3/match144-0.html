def calculate_shipping(n:float):
    '''take weight of a package and return the price to ship it'''
    if n < 2:
        return 2.00
    elif (n >= 2) and (n < 10):
        return 5.00
    elif n >= 10:
        return 5 + (1.5 * (n - 10))
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

def draw_eye (my_canvas, x ):
    my_canvas.create_oval(x, 50,x+100, 100, fill='white')
    my_canvas.create_oval(x+25, 50, x+75,100, fill='blue')
    my_canvas.create_oval(x+63, 50, x+37,100, fill='black')

def draw_nose(my_canvas, x ):
    my_canvas.create_line( x, 100, x+50, 200)
    my_canvas.create_line( x, 200, x+50, 200)

def draw_mouth(my_canvas, x ):
    my_canvas.create_oval (x,300, x+350, 500)

def draw_face():
    my_window = tkinter.Tk()
    my_canvas = tkinter.Canvas(my_window, width=600, height=600)
    my_canvas.pack()
    draw_eye(my_canvas, 100)
    draw_eye(my_canvas, 350)
    draw_nose(my_canvas, 300)
    draw_mouth(my_canvas,100)

    tkinter.mainloop()
    return

draw_face()
