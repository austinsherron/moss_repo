def calculate_shipping(weight: float) -> float:
    "Prints the rate of shipping"
    if weight < 2:
        rate = (2.00)
    elif weight < 10 and weight >=2:
        rate = (5.00)
    elif weight >= 10:
        rate = 5.00 + (weight-10)*1.5
    return rate
