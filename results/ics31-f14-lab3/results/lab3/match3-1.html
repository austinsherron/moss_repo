    '''returns the first three characters of the inputted string'''

    return s[0:3]

def calculate_shipping(weight_pounds:float) ->float:
    '''returns the price for shipping the package'''
    if weight_pounds < 2:
        return 2.00
    elif weight_pounds < 10:
        return 5.00
    total = 5.00
    weight_pounds = math.floor(weight_pounds - 10)
    total = total + (1.5*weight_pounds)
    return total

def draw_face(color:str, size:int, length:int) ->None:
    '''draws a face with tkinter'''
    my_window = tkinter.Tk()   
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)
    my_canvas.pack()

    def draw_eye(x:int, y:int, color:str) ->None:
        '''draws an eye'''
        my_canvas.create_oval(x, y, x+200, y+100, fill='white')


        my_canvas.create_oval(x + 50, y, x+150, y+100, fill=color)
        my_canvas.create_oval(x + 75, y+25, x+125, y+75, fill='black')
    draw_eye(0,0,color)
    draw_eye(300,0,color)
    def draw_nose(size:int) ->None:
        '''draws a nose'''
        my_canvas.create_line(250, 100, 250+size, 200)
        my_canvas.create_line(250+size, 200, 250, 200)
    draw_nose(size)
    def draw_mouth(length:int) ->None:
        '''draws a mouth'''
        my_canvas.create_line(250-length/2, 300, 250+length/2, 300)
    draw_mouth(length)
    tkinter.mainloop()        
draw_face('brown', 100.5, 200.5)
draw_face('yellow', 20.5, 100.5)
