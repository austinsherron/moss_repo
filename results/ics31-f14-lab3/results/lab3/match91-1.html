def calculate_shipping (weight: float) -> float:
    "returns the price for shipping packages"
    if weight < 2:
        price = 2.00
    elif weight < 10:
        price = 5.00
    else:
        price = 5 + 1.5* (weight-10)
    return price
assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(2) == 5.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(10) == 5.00
assert calculate_shipping(15) == 12.50

def inventory_value (books : Book) -> float:
    "return the total value of a title"
    return books.price * books.instock
assert inventory_value(BSI[0]) == 143
assert inventory_value(BSI[2]) == 281.6
assert inventory_value(BSI[-2]) == 242

def top_value (books: list) -> Book:
    "return the namedtuple for books with the highest value"
    topBook = books[0]
    for index in range(len(books)):
        if inventory_value(topBook) < inventory_value(books[index]):
            topBook = books[index]
    return topBook
print ("highest-value book is", top_value(BSI).title ,"by", top_value(BSI).author ,"at a value of $", inventory_value(top_value(BSI)))

window2 = tkinter.Tk()
canvas2 = tkinter.Canvas(window2, width = 900, height = 900)
canvas2.pack()
def draw_eye (x: int, y: int, eyeColor: str):
    "draws two eyes with various colors"
    canvas2.create_oval (x, y, x+100, y+50, fill = 'white')
    canvas2.create_oval (x+25, y, x+75, y+50, fill = eyeColor)
    canvas2.create_oval (x+40, y+15, x+60, y+35, fill = 'black')
def draw_nose (x: int, y: int, size: int):
    "draws a nose with various sizes"
    xSize1 = 12 + 12*size
    ySize1 = 30 + 30*size
    xSize2 = 2 + 2*size
    ySize2 = 20 + 20*size
    canvas2.create_line (x, y, x+xSize1, y+ySize1, x-xSize2, y+ySize2, smooth='true')
def draw_mouth (x: int, y: int, size: int):
    "draws a mouth with various shapes"
    xSize1 = 27*size
    xSize2 = 270 - 27*size
    ySize1 = 13*size
    ySize2 = 135 - 13*size
    canvas2.create_line (x+xSize1, y+ySize1, x+135, y+ySize2, x+xSize2, y+ySize1, smooth = 'true', width = 5)
def draw_face (x: int, y: int, eyeColor, noseSize, mouthShape):
    "draws a face with two eyes, a nose, and a mouth"
    canvas2.create_oval(x, y, x+370, y+370)
    draw_eye(x+50, y+80, eyeColor)
    draw_eye(x+220, y+80, eyeColor)
    draw_nose(x+190, y+120, noseSize)
    draw_mouth(x+50, y+240, mouthShape)

draw_face(50, 50, 'cyan', 7, 7)
draw_face(400, 400, 'gold', 3, 0)
