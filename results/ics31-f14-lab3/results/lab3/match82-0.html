def calculate_shipping(pounds:float)-> float:
    '''Calculates shipping by determining the weight of pounds then
    calculating according to the shipping rate of each category of pounds'''
    if pounds < 2.0:
        return 2.00
    elif pounds >= 2 and pounds < 10:
        return 5.00
    elif pounds >= 10:
        return (5 + (1.5*(pounds - 10)))

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 12.50

#Functions for drawing on tkinter
def draw_eye(x: int, y: int, distanceBetweenEyes: int, diameter: int):
    '''Asks for x-y coordinates, the distance between its eyes and the diameter/how big the eyes are supposed to be. Also draws pupils according to the eye'''
    my_canvas.create_oval(x, y, x+diameter, y+diameter)
    my_canvas.create_oval(x+distanceBetweenEyes, y, x + distanceBetweenEyes + diameter, y+diameter)
    my_canvas.create_oval(x+35, y+35, x + (diameter/1.25), y + (diameter/1.25), fill = 'black')
    my_canvas.create_oval(x + distanceBetweenEyes + 35, y + 35, x + distanceBetweenEyes + (diameter/1.25), y + (diameter/1.25), fill = 'black')
    return
def draw_mouth(x: int, y: int, width: int, length: int, concavity: int):
    '''Draws a mouth using an arc'''
    my_canvas.create_arc(x, y, x+width, y+ length, style = "arc", extent = concavity)
    return
def draw_nose(x: int, y: int, length: int):
    '''Draws a nose using lines'''
    my_canvas.create_line(x, y, x - length, y + length)
    my_canvas.create_line(x - length, y + length, x, y + length)
    return
def draw_moustache(x: int, y: int, length: int,width: int ):
    '''Creates a moustache according to x, y, length, width'''
    my_canvas.create_rectangle(x, y, x + width, y -length , fill = 'black')
    return
def draw_eyebrows(x: int, y: int, length:int, elevation: int):
    my_canvas.create_line(x, y, x + length, y - elevation)
