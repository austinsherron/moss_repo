def calculate_shipping(weight: float) -> float:
    if weight < 2:
        return 2.00
    elif 2 <= weight < 10:
        return 5.00
    elif weight >= 10:
        return weight * 1.50

assert calculate_shipping(1.5) == 2.00
assert calculate_shipping(7) == 5.00
assert calculate_shipping(15) == 22.50
