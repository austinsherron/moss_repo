def rotate(x:str,k:int) -> str:
    '''rotates the str as per the key k'''
    t=x[k:]+x[0:k]
    return t
def caeser_encrypt(x:str,k:int) -> str:
    '''encrypts the str x as per the key k'''
    y=x.lower()
    alph='abcdefghijklmnopqrstuvwxyz'
    cipher=rotate(alph,k)

    t=y.maketrans(alph,cipher)
    return y.translate(t)
def caeser_decrypt(x:str,n:int):
    '''decrypts the str x as per the key n'''
    y=x.lower()
    d=caeser_encrypt(y,26-n)
    return d
