from random import randrange

def last_name():
    '''randomly generates a last name'''
    name = ''
    a = open('surnames.txt')
    data = a.readlines()
    for i in data[randrange(1,1000)]:
        if i.isupper() or i.islower():
            name += i
    return name

def random_name(n:int):
    '''Randomly generate the number of names, formated "surname, first name",
passed into the parameter'''
    list_names = []
    for i in range(n):
        a = last_name()
        b = first_name()
        name = (a[0]+a[1:].lower() +', '+b[0] + b[1:].lower())
        list_names.append(name)
    return list_names

def copy_file(n:str):
    if n == 'line numbers':
        '''If the parameters are "line number" then it will copy the file to a
newly created file wuth line numbers formated to the right on the left margin'''
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        line_number = 1
        for line in infile:
            outfile.write('{:5}:{}'.format(line_number,str(line)))
            line_number += 1
        infile.close()
        outfile.close()
    elif n == 'Gutenberg trim':
        '''If the parameters are "Gutenberg trim" then it will copy only the body
of the text from the start line to the end line'''
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        data = infile.readlines()
        start_line = False
        end_line = False
        for i in data:
            if 'START OF THIS PROJECT' in i:
                start_line = True
            if 'END OF THIS PROJECT' in i:
                end_line = True
            if start_line == True and end_line == False:
                outfile.write(i)
        infile.close()
        outfile.close()
    elif n == 'Statistics':
        '''If the parameters are "Statistics" it will copy the text and
and append the statistics of the text at the bottom of the text file'''
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'a+')
        data = infile.readlines()
        result = 0
        empty_lines = 0
        for line in data:
            outfile.write(line)

        for i in data:
            if i.isspace():
                result += 1
                empty_lines +=1
        sum_char = 0
        for i in data:
            for k in i:
                if k.isupper() or k.islower():
                    sum_char += 1        
        outfile.write('\n{:5}   {}'.format((len(data)),'lines in the file'))
        outfile.write('\n{:5}   {}'.format((result),'empty lines'))
        outfile.write('\n{:7.1f} {}'.format((sum_char/len(data)),'average characters per line'))
        outfile.write('\n{:7.1f} {}'.format((sum_char/(len(data)-(empty_lines))),'average characters per non-empty line'))
        infile.close()
        outfile.close()
    else :
        '''If the parameters are something other than the three options above,


it will just write the text to a new file'''
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        for line in infile:
            outfile.write(line)
        infile.close()
        outfile.close()
