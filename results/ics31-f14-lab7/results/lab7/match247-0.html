def Caesar_break(s:str):
    '''Returns the decrypted string without the use of a key.'''
    s = s.lower()
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    for k in range(26):
        new = ''
        for i in s:
            if i in alphabet:
                new = new+alphabet[(alphabet.find(i)+k)%26]
            else:
                new = new+i
        if new[0:new.find(' ')] in lowercaseDictionary:
            break
    return new

def copy_file(s:str):
    '''Copies the specified file into a new file. Different parameters have different effects on the copying process
       1. Inputting "line numbers" will copy the file and place line numbers at the start of each line
       2. Inputting "Gutenberg trim" will copy the file while cutting out the housekeeping text at start and end


       3. Inputting "statistics" will copy the file and print out the statistics of the file at the end.'''
    
    if s == 'line numbers':
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        lineNumber = 1
        for line in infile:
            outfile.write("{:5d}: ".format(lineNumber)+line)
            lineNumber += 1
        infile.close()
        outfile.close()
    elif s == 'Gutenberg trim':
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        result = []
        for line in infile:
            if line.startswith('*** START'):
                result = []
            if line.startswith('*** END'):
                result.append(line)
                break


            result.append(line)
        for i in result:
            outfile.write(i)
        infile.close()
        outfile.close()
    elif s == 'statistics':
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        lineNumber = 0
        emptylines = 0
        number_of_characters = 0
        for line in infile:
            if line == '\n':
                emptylines += 1
            lineNumber += 1
            for i in line:
                number_of_characters += 1
            outfile.write(line)
        AverageCharPerLine = number_of_characters/lineNumber
        AverageCharPerNone = number_of_characters/(lineNumber-emptylines)
        outfile.write("\n{:5d}   ".format(lineNumber)+'lines in the file\n')
        outfile.write("{:5d}   ".format((emptylines))+'empty line\n')
        outfile.write("{:7.1f} ".format((AverageCharPerLine))+'average characters per line\n')
        outfile.write("{:7.1f} ".format((AverageCharPerNone))+'average character per non-empty line\n')
        infile.close()
        outfile.close()
    else:
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
