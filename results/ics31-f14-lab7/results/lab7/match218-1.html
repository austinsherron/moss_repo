from random import *
infile = open("wordlist.txt", "r")

    list.append(x.strip())

DICT = {}
for w in list:
    DICT[w] = 0

infile.close()

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'


def rotate(n: int) -> str:
    return ALPHABET[n:] + ALPHABET[:n]

translation = []
for n in range(26):
    translation.append(str.maketrans(rotate(n), ALPHABET))

def Caesar_encrypt(plaintext: str, n: int) -> str:
    encrypt = str.maketrans(ALPHABET, rotate(n))
    return plaintext.lower().translate(encrypt)



def Caesar_break(cyphertext: str) -> str:
        words = 0
        var1 = 0
        for x in range(26):
            hits = 0       
            for word in cyphertext.strip().split():
                words += 1 
                if word.translate(translation[x]) in DICT:
                    hits += 1
            if hits > var1:
                var1 = hits
                plaintext1 = cyphertext.translate(translation[x])
        return plaintext1

def copy_file(word: str ):
    """ Copys a file and returns the same copy"""
    count = 0
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors = 'ignore')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    if word == "line numbers":
        for line in infile:
            count += 1
            outfile.write('{:5}: {}'.format(count, line))
    if word == "Gutenberg trim":
        for line in infile:
            if "*** START OF THIS PROJECT GUTENBERG EBOOK THE ADVENTURES OF SHERLOCK HOLMES ***" in line:
                for line in infile:
                    outfile.write(line)
                if "*** END OF THIS PROJECT GUTENBERG EBOOK THE ADVENTURES OF SHERLOCK HOLMES ***" in line:
                    break
                break
    if word == "statistics":
        empty_line_count = 0
        char_count = 0
        
        for line in infile:
            count += 1
            if line == "\n":
                empty_line_count += 1
            split_line = line.split()
            for i in split_line:
                char_count += len(i)
            char_count_per_nonempty_line = char_count/(count - empty_line_count)
            char_count_per_line = char_count/count
            
        print('{0:5d}'.format(count), '\tlines in the list')
        print('{0:5d}'.format(empty_line_count), '\tempty lines')
        print('{:7.1f}'.format(char_count_per_line), 'average characters per line')
        print('{:7.1f}'.format(char_count_per_nonempty_line), 'average characters per non-empty line')
            
                    
                
    infile.close()
    outfile.close()
