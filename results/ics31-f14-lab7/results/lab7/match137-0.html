def random_names(x: int) -> list:
    '''Generates a list with the specificed number of random names'''
    names = []
    while len(names) < x:
        names.append(random_name_generator('surname') + ", " + random_name_generator('firstname') + '\n')
    return ''.join(names)

            namefile = open("femalenames.txt", "r")
    for i in namefile.readlines():
        lower_case = i.split('\t')[0].lower().capitalize()
        name.append(lower_case)
    namefile.close()
    return name[n]

alphabet = 'abcdefghijklmnopqrstuvwxyz'

def rotated(key : int)->str:
    '''rotates alphabet by given key'''
    newalpha = (alphabet[0: key])
    neweralf = (alphabet[key: ])
    stuff = neweralf + (newalpha)
    return stuff
def rotated_alphabet(code:str, key:int) -> str:
    '''Rotates alphabet by given key'''
    if key >25:
        return (code[key%26:len(code)] +code[0:key%26])
    else:
        return (code[key:len(code)] +code[0:key])

def Caesar_break (ciphertext: str)-> str:
    '''We will have the 26 possible keys
    and will choose the one with the most# of correct words'''
    each_string = []
    new_message = ciphertext.lower()
    for each_number in range(0,27):
        table = str.maketrans(rotated_alphabet(alphabet,each_number),alphabet)
        each_string.append(new_message.translate(table))
    hit_list = actual_word_counter(each_string)
    a = 0
    for each_word_counter in hit_list:
        if each_word_counter > a:
            a = each_word_counter
    position = hit_list.index(a)
    return(each_string[position])      
        
def actual_word_counter(message_list: list) -> str:
    '''Counts the number of legit words in each string and
    prints the one that is the most correct one'''
    hit_list = []
    Fuck = open("wordlist.txt", 'r')
    crud = Fuck.read().split("\n")
    Fuck.close()
    for each_message in message_list:
        result = 0
        for each_word in each_message.split():
            if each_word in crud:
                result += 1
        hit_list.append(result)
    return hit_list

def copy_file() -> None:
    '''Copies the file of the first input into the file of the 2nd input,
    which will make a new file if that file did not exist before'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    for line in infile:
        outfile.write(line)
    infile.close()
    outfile.close()

def copy_file_with_linenumbers_Gutenberg(big_str:str) -> None:
    '''Takes in a str. If it is 'Gutenberg trim', the outfile will
    copy only the body of a Project Gutenberg file,
    omitting the "housekeeping" material at the front and end.


    Else it will do what copy_file_with_linenumbers does.'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    result = ''
    if big_str == 'line numbers':
        for line in infile:
            result += 1
            outfile.write('{:6}: {:}'.format(result,line))
    elif big_str == 'Gutenberg trim':
        for line in infile:
            result += line
        START = result.find('*** START')
        END = result.find('*** END')
        for line in result[START:END+result.find('\n')]:
            outfile.write(line)
    else:
        for line in infile:
            outfile.write(line)  
    infile.close()
    outfile.close()

def stats(strings: list) -> None:
    '''takes a list of strings and prints statistics of each line'''
    empty = []
    numsum = []
    non_empty_numsum = []
    
    for i in range(0, len(strings)):
##        if strings[i] == '\n':
        if strings[i].strip()=='':
            empty.append(i)
        else:
            non_empty_numsum.append(len(strings[i]))
        numsum.append(len(strings[i]))
    average = sum(numsum)/(i+1)
    empaverage = (sum(non_empty_numsum))/ (len(strings) - len(empty))
    return ("{:6}   lines in the list \n{:6}   empty Lines \n   {:5.1f} average characters per line \n   {:5.1f} average characters per non-empty line \n".format(
        len(strings), len(empty), average, empaverage))
