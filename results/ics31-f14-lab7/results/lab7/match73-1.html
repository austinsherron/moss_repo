from random import randrange 
def random_names(x:int)->list:
    '''t takes an integer and returns a list of
    that many strings, with each string a randomly generated name'''
    infile=open ('surnames.txt','r')
    surnames=infile.read()
    infile.close()

    infile=open ('femalenames.txt','r')
    femalenames=infile.read()
    infile.close()
    
    infile=open ('malenames.txt','r')
    malenames=infile.read()
    infile.close()

    surnames=surnames.split()[::4]
    femalenames=femalenames.split()[::4]
    malenames=malenames.split()[::4]

    list_of_names=[]
    for i in range(x):
        random_surname=surnames[randrange (len(surnames))]
        random_femalename=femalenames[randrange (len(femalenames))]
        random_malename=malenames[randrange(len(malenames))]

        randrange(len(first_name))
        first = first_name[randrange(len(first_name))]
        a_name = random_surname +', '+first
        list_of_names.append (a_name)
    return list_of_names

def rotate(letter:str,key:int)->str:
    '''rotates a letter based on the key'''
    alphabet='abcdefghijklmnopqrstuvwxyz' * 30
    intab=alphabet[alphabet.find(letter)]
    outtab= alphabet[alphabet.find(letter)+ key]
    trantab=str.maketrans(intab,outtab)
    new_string= letter.translate(trantab)
    return new_string

def Caesar_encrypt(message:str,key:int)->str:
    '''encrypts a message according to the key'''
    new_string=''
    for i in message.lower():
        l=rotate(i,key)
        new_string+=l
    return new_string

def Caesar_decrypt(message:str,key:int)->str:
    '''decrypts a message according to the key'''
    return Caesar_encrypt(message,key * -1)

def Caesar_break(cyphertext:str)->str:
    '''cracks a caesar encryption'''
    infile=open('wordlist.txt','r')
    wordlist=infile.readlines()
    for i in range(len(wordlist)):
        wordlist[i]=wordlist[i].lower().strip()
    infile.close()
    count_list=[]
    decrypt_list=[]
    for i in range(26):
        decrypt_list.append(Caesar_decrypt(cyphertext, i))
    for sentence_word in decrypt_list:
        count = 0
        s=sentence_word.split()
        for i in s:
            if i in wordlist:
                count+=1
        count_list.append(count)
    return decrypt_list[count_list.index(max(count_list))]
