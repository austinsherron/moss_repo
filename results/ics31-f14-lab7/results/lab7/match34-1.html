def capitalization(s:str)->str:
    """fixes capitalization
    """
    result=''
    r = s.split()
    result = r[0][0].upper()+ r[0][1:].lower()+ " " + r[1][0].upper() + r[1][1:].lower()
    return result

alphabet = "abcdefghijklmnopqrstuvwxyz"

def rotate(x: int) -> str:
    """ Rearrange the alphabet by shifting it "x" places
    """
    return alphabet[x:] + alphabet[:x]



def ceasar_break(s:str)->str:
    """Code breaking
    """
    word = ''
    count = 1
    d = 0
    z = []
    w = open('wordlist.txt').readlines()
    for i in w:
        z.append(i[:-1])
    for i in range(26):
        table = str.maketrans(rotate(i), alphabet)
        r = s.translate(table)
        for q in r.split():
            if q in z:
                d += 1
                if d > count == True:
                    word = r
                    count = d
    return word

def line_num(l:"list of str")->str:
    """ Display the list of strings with the associated number
    """
    t= ''
    for x in range(len(l)):
        t += "{}: {}".format(x+1,l[x-1])
    return t

def empty_string(s: str) -> bool:
    """ Decides whether or not the string is empty
        1) Empty     = return True
        2) Not empty = return False
    """
    empty = True
    if s == "":
        return True
    else:
        for letter in s:
            if letter != " ":
                empty = False
        if empty == False:
            return False
        else:
            return True



def stats(s:"list of strings")-> str:
    """ Display the stats of the list of strings:
        lines in list, empty lines, avg characters per line, avg characters per non-empty line
    """ 
    lines = len(s)
    empty = 0
    avg_line = 0
    avg_ne_line = 0


    for x in s:
        if empty_string(x) == True:
            empty += 1
    for x in s:
        for letter in x:
            avg_line += 1
    avg_line = avg_line/lines
    avg_ne_line = avg_line/(lines-empty)


    print("{:5} Lines in the list\n\
{:5} empty lines\n\
{:7.1f} average characters per line\n\
{:7.1f} average characters per non-empty line".format(lines, empty, avg_line, avg_ne_line))
