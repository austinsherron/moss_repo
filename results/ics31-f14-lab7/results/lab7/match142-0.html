from random import randrange

def names(file:str)->list:
    " takes a file and returns a list of the names"
    infile = open(file,'r')
    content = infile.read()
    names = content.split()
    newnames = []
    infile.close()
    for part in names:
        if part[0] not in '1234567890':
            newnames.append(part)
    return newnames

def nameselect(list1:list,list2:list)->str:
    firstname = list1[randrange(len(list1))]
    lastname = list2[randrange(len(list2))]
    full = (lastname+', '+firstname)
    return full

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def Caesar_encrypt(word:str,code:int)->str:
    "Takes a string and a number and returns a string that has the letters\
    shifted to the amount number inputted"
    if code > 26:


        code = code%26
    newstr = ALPHABET[code:] + ALPHABET[:code]
    table = str.maketrans(ALPHABET,newstr)
    return word.translate(table)

def Caesar_decrypt(word:str,code:int) ->str:
    "Takes a string and a number and returns a string that has the letters\
    shifted back the amount number inputted"
    if code > 26:


        code = code%26
    newstr = ALPHABET[code:] + ALPHABET[:code]
    table = str.maketrans(newstr,ALPHABET)
    return word.translate(table)

def Caesar_break(m:str)->str:
    "takes a encrypted string and searches through each ceaser encrypt possibility, \
returns string with most english words"
    infile = open('wordlist.txt')
    dictionary = infile.read()
    words = dictionary.split()
    infile.close()
    combo = []
    numb = []
    for i in range(26):
        newstr = ALPHABET[i:] + ALPHABET[:i]
        table = str.maketrans(newstr,ALPHABET)
        code = m.translate(table)
        point = 0
        for r in words:
            for f in code.split():
                if f == r:
                    point +=1
        numb.append(point)
        combo.append(code)
    return combo[numb.index(max(numb))]

def stats(liststr:list):
    "takes a list of strings and prints the amount of strings in the list the empty strings in the list the average \
characters per line and the average characters per non-empty line"
    lines = (len(liststr) - 1)
    empty = liststr.count('')
    sumcharacters = []
    for characters in liststr:
        sumcharacters.append(len(characters))
    avg_character = sum(sumcharacters)/lines
    avg_nonempty = sum(sumcharacters)/(lines - empty)
    print('{:5} {:5}'.format(lines,"lines in the list"))
    print("{:5} {:5}".format(empty,"empty lines"))
    print("{:5.1f} {:5}".format(avg_character,"average characters per line"))
    print("{:5.1f} {:5}".format(avg_nonempty,"average characters per non-empty line"))
