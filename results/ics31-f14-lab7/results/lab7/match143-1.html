ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate(num:int) -> str:
    if num > 26:
        num = num % 26
    new_alphabet = ' '
    return ALPHABET[num:] + ALPHABET[:num]

def Caesar_encrypt(message:str, key:int) -> str:
    lower = message.lower()
    table = str.maketrans(ALPHABET, rotate(key))
    return lower. translate(table)

def Caesar_decrypt(message: str, key:int) -> str:
    table = str.maketrans(ALPHABET, rotate(-key))
    return message.translate(table)

def copy_file(a: str):
    '''If the paramter is "line numbers" the copied file includes line numbers
    at the start of each line. If the paramter is "Gutenberg trim" then the
    copied file includes line numbers and omits the unnecessary information
    before the text begins and after it ends. If the paramter is "statistics"
    it will copy the file and print statistics.'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    counter = 1 
    if a == 'line numbers':
        for line in infile:
            outfile.write('{0:5}: {1}'.format(counter, line))
            counter += 1
        infile.close()
        outfile.close()
    elif a == 'Gutenberg trim':
        write = False
        for line in infile:
            if line.startswith('*** START'):
                write = True
            if write == True:
                outfile.write(line)
            elif line.startswith('*** END'):
                break
        infile.close()
        outfile.close()
    elif a == 'statistics':
        lineList = []
        emptyLines = []
        chars = 0
        for line in infile:
            outfile.write(line)
            lineList.append(line)
            chars += len(line)
            if line.isspace():
                emptyLines.append(line)
        print('{0:6} {1}'.format(len(lineList) , 'lines in the file'))
        print('{0:6} {1}'.format(len(emptyLines), 'empty lines'))
        print('{0:6.1f} {1}'.format(chars/len(lineList), 'average characters per line'))
        print('{0:6.1f} {1}'.format(chars/(len(lineList)-len(emptyLines)), 'avergage characters per non-empty line'))
        infile.close()
        outfile.close()
    else:
        for line in infile:
            outfile.write(line)
        infile.close()
        outfile.close()
