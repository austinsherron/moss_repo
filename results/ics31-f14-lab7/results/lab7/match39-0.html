from random import randrange

def random_first_names(num: int) -> list:
    '''Takes an integer and returns a list of that many strings
    where each string is a randomly generated female or name
    first name.'''
    a = open('femalenames.txt','r')
    b = open('malenames.txt','r')
    girl_name = a.readlines()
    male_name = b.readlines()
    allNames = girl_name + male_name


    random_index = []
    newList = []
    newListTwo = []
    for n in range(num):
        random_index.append(randrange(len(allNames)))
    for r in random_index:
        newList.append(allNames[r])

        newListTwo.append(s.split()[0])

ALPHABET = "abcdefghijklmnopqrstuvwxyz"

def rotated(ALPHABET: str, num: int) -> str:
    '''Returns a rotated alphabet with the specified number of characters
    taken off the front and added on the end of the string.'''
    for c in ALPHABET:
        ALPHABET2 = ALPHABET[num:] + ALPHABET[:num]        
    return ALPHABET2
assert rotated(ALPHABET, 2) == 'cdefghijklmnopqrstuvwxyzab'
assert rotated(ALPHABET, 5) == 'fghijklmnopqrstuvwxyzabcde'
assert rotated(ALPHABET, 1) == 'bcdefghijklmnopqrstuvwxyza'

def Caesar_encrypt(s: str, num: int) -> str:
    '''Returns the ciphertext for decryption.'''
    for c in s:
        s2 = s.lower()
    return s2.translate(s2.maketrans(ALPHABET, rotated(ALPHABET, num)))
assert Caesar_encrypt("Hi", 2) == 'jk'
assert Caesar_encrypt("abc", 5) == 'fgh'
assert Caesar_encrypt('School', 1) == 'tdippm'

def convertToLowerWithoutNewline() -> list:
    '''Returns a list that converts the wordlist file into
    lowercase and replaces each newline with nothing.'''
    f = open('wordlist.txt', 'r')
    wordlist = f.readlines()
    f.close()
    newlist = []
    
    for n in wordlist:
        x = n.replace("\n", "")
        x = x.lower()
        newlist.append(x)

    return newlist

def stats(ls: list) -> None:
    '''Prints statistics from a list of strings.'''
    numLine = len(ls)
    emptyLine = 0
    numChar = 0
    numCharEmpt = 0
    for i in ls:
        if( i == '\n'):
            emptyLine += 1
    for c in ls:
        numChar += len(c)
    for c in ls:
        if c.strip() != '':
            numCharEmpt += len(c)
    avgPerLine = numChar / numLine
    avgPerNon = numChar / (numLine - emptyLine)
    a = ("{:5} {:>19}".format(numLine, "lines in the list"))
    b = ("{:5} {:>13}".format(emptyLine, "empty lines"))
    c =("{:7.1f} average characters per line".format(avgPerLine))
    d =("{:7.1f} average characters per non-empty line".format(avgPerNon))
    e = a + '\n' + b + '\n' + c + '\n' + d
    return e

def copy_file(s:str) -> list:
    '''Returns lines if the parameter is 'line numbers',
    Returns body of Project Gutenberg file if parameter is
    'Gutenberg trim' while omitting the housekeeping material
    at the front nad the end, and prints out the statistics
    if the parameter is 'statistics'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
  
    
    linecontent = infile.readlines()
    if s == "line numbers":
        for x in range(len(linecontent)):
            outfile.write("{:5}: {:}".format(x+1,linecontent[x]))
    elif s == 'Gutenberg trim':
        w = False
        for x in linecontent:
            if '*** START' in x:
                w = True
                
            elif '*** END' in x:
                break
            elif w:
                outfile.write(x) 
    elif s == 'statistics':
        outfile.write(stats(linecontent))      
    else:
        for line in linecontent:
            outfile.write(line)
    infile.close()
    outfile.close()
