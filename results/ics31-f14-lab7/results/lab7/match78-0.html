ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotate_alphabet(a: int) -> str:
    '''Takes a number and produces a rotated string'''
    return ALPHABET[a:] + ALPHABET[0:a]

def Caesar_encrypt(a: str, b: int) -> str:
    '''Encrypts a message.'''
    encryption = str.maketrans(ALPHABET,
                             rotate_alphabet(b))
    return a.lower().translate(encryption)


          
def Caesar_decrypt(a: str, b: int) -> str:
    '''Decrypts an ecrypted mesage with a key.'''
    decryption = str.maketrans(rotate_alphabet(b),
                             ALPHABET)
    return a.lower().translate(decryption)

def Caesar_break(ciphertext: str ) -> str:
    '''Decrypts an ecrypted message without a key.'''
    hits = 0
    key = 0
    for x in range(26):
        temp = 0
        decryption = Caesar_decrypt(ciphertext, x)
        for word in decryption.split():
            if word in dictionary:
                temp += 1
            if hits < temp:
                hits = temp
                key = x
    result = Caesar_decrypt(ciphertext, key)
    return result
