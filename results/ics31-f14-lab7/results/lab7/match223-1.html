def random_surname(l: list) -> str:

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate(num: int) -> str:
    """ Rotates order of ALPHABET by input number.
    """
    return ALPHABET[num:] + ALPHABET[:num]

def Caesar_decrypt(message:str, key:int) -> str:
    """ Decrypts a message by a Caesar cipher.
    """
    table = str.maketrans(ALPHABET, rotate(-(key % 26)))
    return message.translate(table)

d = open('wordlist.txt')
dictionary = d.read()
dictionary = dictionary.lower()
wordlist = dictionary.split('\n')
d.close()

def Caesar_break(ciphertext: str) -> str:
    """ Decrypts a Caesar ciphertext through brute force.
    """
    count_list = []
    decryptions = []
    for i in range(26):
        plaintext = Caesar_decrypt(ciphertext, i)
        plaintext = plaintext.split(' ')
        for word in plaintext:
            count = 0
            if word in wordlist:
                count += 1
            count_list.append(count)
            decryptions.append(' '.join(plaintext))
    index = count_list.index(max(count_list))
    print(decryptions[index])

def stats(l:list)-> int:
    """ Prints the statistics of a list of strings.
    """
    average_per_line = []
    average_per_nline = []
    count_of_nline = 0
    for i in l:
        average_per_line.append(len(i))
    for i in l:
        x = i.strip(' ')
        if len(x) != 0:
            average_per_nline.append(len(i))
        else:
            count_of_nline +=1
    lines = len(l)
    empty_lines = count_of_nline
    average_character = sum(average_per_line)/len(l)
    average_character2 = sum(average_per_nline)/(len(l)-count_of_nline)
    print("{:5}{:2} lines in the list".format(lines,' '))
    print("{:5d}{:2} empty lines".format(empty_lines,' '))
    print("{:7.1f} average characters per line".format(average_character))
    print("{:7.1f} average characters per non-empty line".format(average_character2))
