def Caesar_decrypt(x:str, num: int) -> str:
    "Returns the plaintext"
    x = x.lower()
    alphabet1 = 'abcdefghijklmnopqrstuvwxyz'
    alphabet2 = ''
    i = (num%26)
    alphabet2 = alphabet2.join(alphabet1[i:].split() + alphabet1[0:i].split())
    x = x.translate(str.maketrans(alphabet2, alphabet1))
    return x

def Caesar_break(ciphertext:str) -> str:
    decrypted = []
    for word in range(0,27):
        n = Caesar_decrypt(ciphertext, word)
        for i in decrypted:
            if n == i:
                decrypted.append(n)
    return decrypted
