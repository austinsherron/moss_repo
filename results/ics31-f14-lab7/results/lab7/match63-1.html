def Caesar_break (cipher: str) -> str:
    ALPHABET = ("abcdefghijklmnopqrstuvwxyz")
    cipher = cipher.lower()
    for k in range (1,26):
        k = 26 - k
        table = str.maketrans(ALPHABET[:],ALPHABET[k:]+ALPHABET[:k])
        plaintext = cipher.translate(table)
        if r.find(plaintext) != -1:
            return plaintext
