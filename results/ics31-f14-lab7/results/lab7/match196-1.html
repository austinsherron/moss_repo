from random import randrange

#create the namelist function which returns the namelist of the three files
def namelist(filename):
    """return the name list of the file"""
    infile=open(filename,'r')
    linelist=infile.readlines()
    infile.close()
    result=[]
    for line in linelist:
        wordlist=line.split('\t')
        result.append(wordlist[0].lower().capitalize())
    return result

#store the three namelists in surname,femalename, and malename respectively.

alphabet='abcdefghijklmnopqrstuvwxyz'
def rotated_alphabet(n):
    return alphabet[n:]+alphabet[:n]
print(rotated_alphabet(1))

def Caesar_encrypt(message: str, key: int)-> str:
    """returns the ciphertext-encrypted message"""
    encrypt = message.maketrans(alphabet, rotated_alphabet(key))
    return message.translate(encrypt)

print(Caesar_encrypt('Brianna',20))
print(Caesar_encrypt('chasm',1))

def Caesar_decrypt(message: str, key:int)-> str:
    """Returns the plaintext-each letter in the original message"""
    encrypt = message.maketrans(rotated_alphabet(key), alphabet)
    return message.translate(encrypt)
print(Caesar_decrypt('dibtn',1))

    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    linelist = infile.readlines()
    if s=="line numbers":
        for linenum in range(len(linelist)):
