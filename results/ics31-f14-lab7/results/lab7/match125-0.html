ALPHABET = "abcdefghijklmnopqrstuvwxyz"

def rotate_alphabet(n: int) -> str:
    return ALPHABET[n%26:] + ALPHABET[0:n%26]
assert rotate_alphabet(1) == "bcdefghijklmnopqrstuvwxyza"

def Caesar_encrypt(s: str, n: int) -> str:
    table = str.maketrans(ALPHABET, rotate_alphabet(n))
    return s.lower().translate(table)
assert Caesar_encrypt("This is encryption", 24) == "rfgq gq clapwnrgml"

def Caesar_decrypt(s: str, n: int) -> str:
    table = str.maketrans(ALPHABET, rotate_alphabet(-n))
    return s.lower().translate(table)
assert Caesar_decrypt("rfgq gq clapwnrgml", 24) == "this is encryption"

def get_crypt_list(s: str) -> list:
    all_encrypt = []
    for n in range (26):
        all_encrypt.append(Caesar_encrypt(s, n))
    return all_encrypt

def Caesar_break(s: str) -> str:
    all_crypt = get_crypt_list(s)
    NL = []
    for x in range(len(all_crypt)):
        L = all_crypt[x].split()
        NL.append(0)
        for y in range(len(L)):
            for z in range(len(wordlist)):
                if L[y] == wordlist[z]:
                    NL[x] += 1
    number = NL.index(max(NL))
    return all_crypt[number]

def find_number_of_lines(L: list) -> int:
    line_counter = 0
    for x in range(len(L)):
        if "\n" in L[x]:
            line_counter += 2
        else:
            line_counter += 1
    return line_counter
assert find_number_of_lines(["", "\n", "asdfghjkl"]) == 4

def find_number_of_empty_lines(L: list) -> int:
    empty_line_counter = 0
    for x in range(len(L)):
        if L[x] == "":
            empty_line_counter += 1
    return empty_line_counter
assert find_number_of_empty_lines(["","","\n\n"," ","hjgdskfl"]) == 2

def find_characters_per_line(L: list) -> int:
    characters = 0
    for x in range(len(L)):
        for i in range(len(L[x])):
            characters += 1
    return characters
assert find_characters_per_line(["123456789"]) == 9

def stats(L: list) -> str: 
    return "{:5d} {:53s}".format(find_number_of_lines(L), "lines in the file") + "\n" + "{:5d} {:53s}".format(find_number_of_empty_lines(L), "empty lines") + "\n" + "{:7.1f} {:50s} ".format(find_characters_per_line(L)/find_number_of_lines(L), "average characters per line") + "\n" + "{:7.1f} {:50s}".format(find_characters_per_line(L)/(find_number_of_lines(L)-find_number_of_empty_lines(L)), "average characters per non-empty line")

def copy_file(s: str):
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors='ignore')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    if s == 'line numbers':
        line_count = 1
        for line in infile:
            outfile.write(str(line_count) + ": " + line)
            line_count += 1
            
    elif s == "Gutenberg trim":
        datalines = infile.read().split("\n")
        start = 0
        end = 0


        for x in range(len(datalines)):
            if "*** START" in datalines[x]:
                start = x+1
            elif "*** END" in datalines[x]:
                end = x
                break
        for i in range(start, end):
            outfile.write(datalines[i] + "\n")

    elif s == 'statistics':
        datalines = infile.read().split("\n")
        for x in range(len(datalines)):
            outfile.write(datalines[x] + "\n")
        outfile.write(stats(datalines))
        
    else:
        for line in infile:
            outfile.write(line)
    infile.close()
    outfile.close()

copy_file("statistics")
