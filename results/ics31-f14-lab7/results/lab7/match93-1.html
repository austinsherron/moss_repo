def stats(sl:list):
    '''print statistics'''
    

    return '\n{:6.1f} {}\n{:6.1f} {}\n{:6.1f} {}\n{:6.1f} {}\n'.format(len(sl),'lines in the list',sl.count('\n'),'empty lines',
