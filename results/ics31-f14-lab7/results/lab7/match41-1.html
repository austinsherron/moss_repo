from random import randrange

def generateSurname(p: int)-> list:
    ''' Return a list of randomly generated surnames. '''
    infile = open('surnames.txt', 'r')
    data_surnames = infile.readlines()


    random1 = []
    list1 = []
    new_list1 = []
    for x in range(p):
        random1.append(randrange(len(data_surnames)))
    for c in random1:
        list1.append(data_surnames[c])
    for v in list1:
        new_list1.append(v.title().split()[0])
    return new_list1

def copy_file(p:str):
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors='ignore')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')

    if p == 'line numbers':
        infileText = infile.readlines()
        for i in range(len(infileText)):
            outfile.write('{:>5} : {}'.format(i+1,infileText[i]))

    elif p== "Gutenberg trim":
        G= 0
        E=0
        infileText = infile.readlines()
        for i in range(len(infileText)):
            if ("*** START") in infileText[i]:
                G += i

            if ("*** END") in infileText[i]:
                E += E+ i
                break
        for i in infileText[G + 1:E]:
            outfile.write(i)

    elif p== "statistics":
        infileText = infile.readlines()
        for i in range(len(infileText)):
            print("{:>8} {:25}".format(len(infileText), "lines in the list"))
            empty_line = 0
            for c in infileText:
                if c.strip() == "":
                    empty_line += 1
            print("{:>8} {:25}".format(empty_line, "empty lines"))
            total_characters = 0
            for c in infileText:
                total_characters += len(c)
            print("{:>8.1f} {:25}".format(total_characters/len(infileText), "average characters per line"))
            tot_char_non_emp = 0
            for c in infileText:
                if c.strip() != "":
                    tot_char_non_emp += len(c)
                    
            print("{:>8.1f} {:25}".format(tot_char_non_emp/(len(infileText)-empty_line), "average characters per non-empty line"))
            break
    else:
        for line in infile:
            outfile.write(line)
    infile.close()
    outfile.close()
