word_list = open('wordlist.txt')
words = word_list.readlines()
new_words = []
for i in range(len(words)):
    m = words[i].replace('\n', '')
    new_words.append(m)

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def Rotate(n: int) -> str:
    '''Returns a rotated alphabet witht the specified number of
       characters taken off the front and added to the end of the str'''
    m = ALPHABET[n:], ALPHABET[:n]
    return ''.join(m)

def Caesar_break(s: str) -> str:
    '''Returns the plaintext'''
    m = s.lower()
    t = []
    for i in range(26):
        table = str.maketrans(Rotate(i),
                              ALPHABET)
        q = m.translate(table)
        t.append(q)
    for i in range(len(t)):
        if t[i] in new_words:
            return t[i]
        elif t[i].capitalize() in new_words:
            return t[i].capitalize()
        elif t[i].upper() in new_words:
            return t[i].upper()

assert Caesar_break('hswohila') == 'alphabet'
assert Caesar_break('Cniqn') == 'Algol'
assert Caesar_break('CHFE') == 'AFDC'
print(Caesar_break('Klkqkov'))
