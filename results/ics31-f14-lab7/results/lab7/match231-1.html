ALPHABET='abcdefghijklmnopqrstuvwxyz'

def Caesar_encrypt(msg1:str, key:int) -> str:
    '''encrypt given message'''
    s =key % 26
    table=str.maketrans(ALPHABET,ALPHABET[s:]+ ALPHABET[0:s])


    return msg1.translate(table)

def Caesar_decrypt (msg1:str, key:int) -> str:
    '''decrypt given message'''
    table=str.maketrans((ALPHABET[key:]+ ALPHABET[0:key]),ALPHABET)
    return msg1.translate(table)

def copy_file(str1:str):
    '''makes a copy of a file depending on parameter'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors='ignore')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    char=0
    blank_count=0
    count=0
    if str1 == 'line numbers':
        for line in infile:
            count+=1
            outfile.write(str(count)+':'+ ' '+ line)
    elif str1 == 'Gutenberg trim':
        for line in infile:
            if '*** START' in line:
                for line in infile:
                    outfile.write(line)
                if '*** END' in line:
                    break
    elif str1 == 'statistics':
        for line in infile:
            if line == '\n':
                blank_count += 1
            count+=1
            outfile.write(line)
            char+=len(line)
        outfile.write('\n' + '\n')
        outfile.write(str(count)+' '+'lines in the file')
        outfile.write('\n'+ str(blank_count) + ' ' + 'empty lines')
        outfile.write('\n'+ str(char/count) + ' ' + 'average characters per line')
        outfile.write('\n'+ str(char/(count-blank_count)) + ' '+ 'average characters per non-empty line') 
        infile.close()
        outfile.close()
