def random_name(b: list) -> str:
    '''Returns a random name, male or female'''
    num = randrange(len(b))
    name_info = b[num].split()
    return name_info[0].capitalize()

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotated_alphabet(a: int):
    '''Takes an int and returns the alphabet translated by the given int'''
    return ALPHABET[a%26:] + ALPHABET[:a%26]

def word_count(d:str)->int:
    '''Counts the number of times a word appears in the dictionary'''
    total=0
    f1= d.split()
    for y in f1:
        if y in dictionary:
            total += 1
    return total

assert word_count("ASAP AWACS Aachen") == 3

def Caesar_break(a:str)->str:
    '''Returns the plaintext of an encrypted message'''
    r = 0
    s = ''
    for x in range(26):
        f= a.translate(str.maketrans(rotated_alphabet(x), ALPHABET))
        if word_count(f) >= r:
            r = word_count(f)
            s = f
    return s

def empty_lines(a: list) -> int:
    '''Returns a number of empty lines in a list of str'''
    sum = 0
    for x in a:
            if x.isspace():
                sum += 1
    return sum

def count_characters(a: list) -> int:
    '''Counts the number of characters in a list of str'''
    total = 0
    for x in a:
        total += len(x)
    return total

def count_characters2(a: list) -> int:
    '''Counts the number of characters per non-empty line'''
    total = 0
    for x in a:
        if not x.isspace():
            total += len(x)
    return total

def stats(a: list):
    '''Prints statistics of a list of str'''
    print("{:5}".format(len(a)), '  lines in the list')
    print("{:5}".format(empty_lines(a)), '  empty lines')
    print("{:7.1f}".format(count_characters(a)/len(a)), 'average characters per line')
    print("{:7.1f}".format(count_characters2(a)/(len(a)-empty_lines(a))),'average characters per non-empty line')
