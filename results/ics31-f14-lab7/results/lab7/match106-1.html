ALPHABET ='abcdefghijklmnopqrstuvwxyz'
def encrypt(y: int) -> str:
    if y > 25:
        y = y%26
    result = ALPHABET[y:]+ALPHABET[:y]
    return result

def Caesar_encrypt(s:str,n:int)->str:
    table = str.maketrans(ALPHABET,encrypt(n))
    s = s.translate(table)
    return s
def Caesar_decrypt(s:str,n:int)->str:
    table = str.maketrans(encrypt(n),ALPHABET)
    s = s.translate(table)
    return s
