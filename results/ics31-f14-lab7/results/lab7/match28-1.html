    for i in range(num):

        s = infile.readlines()
        infile.close()
        last_name = s[randrange(1, 1001)].split()[0].lower().title()

        infile = open('femalenames.txt', 'r')
        fn = infile.readlines()
        infile.close()

        infile = open('malenames.txt', 'r')
        mn = infile.readlines()

        male_name = mn[randrange(1,1001)].split()[0].lower().title()

        a = randrange(1,3)
        if a == 1:
            print (last_name + ', ' + female_name)
        else:

def Caesar_break(s: str) -> str:
    infile = open('dictionary.txt' , 'r')
    d = infile.read()
    infile.close()
    a = []
    ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
    for i in range(len(ALPHABET)):
        table = str.maketrans(ALPHABET, (ALPHABET[i:] + ALPHABET[0:i]))
        a.append((s.translate(table)))
    b = []
    for i in range(len(a)):
        b.append(str(a[i]).split())
    for i in range(len(b)):
        for j in range(len(b[i])):
            if b[i][j] in d and b[i][j-1] in d:
                c = i
    e = []
    table = str.maketrans(ALPHABET, (ALPHABET[c:] + ALPHABET[0:c]))
    e.append((s.translate(table)))
    return ' '.join(e)

def copy_file(x):


    file = input("Name of file to copy:")
    infile = open(file, 'r', errors= 'ignore')
    NL = infile.readlines()
    copiedfile = input("Name of copied file: ")
    outfile = open(copiedfile, 'w')
    start = False
    for i in range(len(NL)):
        if x == 'line numbers':
            outfile.write('{0:5}: {1}'.format(i+1, NL[i]))
        elif x == 'Gutenberg trim':
            if '*** END' in NL[i]:
                break
            if start:
                outfile.write(NL[i])
            if '*** START' in NL[i]:
                start = True
        else:
            outfile.write(NL[i])
    if x == 'statistics':
        line_stats(NL)
    infile.close()
    outfile.close()
    
def stats_average(g):
    var = 0


    if len(g) == 0:
        return 0
    for i in range(len(g)):
        var += len(g[i])
    return var/len(g)

def line_stats(h):
    EL = []


    FL = []
    for i in h:
        if i == '\n':
            EL.append(i)
        if i != '\n':
            FL.append(i)
    print('''
{0:5} lines in the file
{1:5} empty lines
{2:5.1f} average characters per line
{3:5.1f} average characters per non-empty line'''\
          .format(len(h),len(EL),stats_average(h),stats_average(FL)))
