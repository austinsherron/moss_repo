from random import randrange
def random_names(n:int)->list:

    result = ''
    x = []
    d = []
    y = []
    f = []
    z = []
    m = []
    infile = open('surnames.txt','r')
    data = infile.readlines()
    for i in range(n):
        x.append((randrange(len(data))))
    for p in x:
        d.append(data[p+1].split('\t')[0].lower().title())
    infile.close()
    for i in range(n):
        if randrange(2) == 1:
            infile = open('malenames.txt','r')
            data = infile.readlines()
            z.append((randrange(len(data))))
        else:
            infile = open('femalenames.txt','r')
            data = infile.readlines()
            y.append((randrange(len(data))))
    for p in z:
         m.append(data[p+1].split('\t')[0].lower().title())
    for p in y:
         f.append(data[p+1].split('\t')[0].lower().title())
    for i in f:
        m.append(i)
    for i,v in zip(m,d):
        result+=('{}, {}\n'.format(i,v))
    return result

    a=''
    for i in r.lower():
        c=al.find(i)
        if c == -1:
            a+=i
        elif (c+p)>=25:
            x=(c+p)%26
            a+=al[x]
        else:
