def Caesar_decrypt(m:str, n: int) -> str:
    "Returns the plaintext"
    m = m.lower()
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    alphabet2 = ''
    x = (n%26)
    alphabet2 = alphabet2.join(alphabet[x:].split() + alphabet[0:x].split())
    m = m.translate(str.maketrans(alphabet2, alphabet))
    return m
print(Caesar_decrypt('BJET', 12))
print(Caesar_decrypt('BJET', 23))
def Caesar_break(f: str) -> str:
    key = []
    last = ''
    g = f.split()
    for h in g:
        for i in range(26):
            key.append(Caesar_decrypt(h, i))
    for i in key:
        for j in words:
            if i == j:
                if len(g)==1:
                    last+= i
                if len(g)>1:
                    k = last.split()
                    if len(k)==0:
                        last += i
                    else:
                        last += ' ' + i
    return last

def stats(a: list) -> str:
    "Lists the stats of a list of strings"
    ave = 0
    print('{0:6}'.format(len(a)), '  lines in the list')
    empty = 0
    for emp in a:
        if emp == '':
            empty += 1
    print('{0:6}'.format(empty), '  empty lines')
    for emp in a:
        ave += len(emp)
    st = (ave/len(a))
    ts = (ave/(len(a)-empty))
    print('{:8.1f}'.format(st), 'average characters per line')
    print('{:8.1f}'.format(ts), 'average characters per non-empty line')
