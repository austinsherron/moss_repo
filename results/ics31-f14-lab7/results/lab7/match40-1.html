def encrypt (plaintext: str, key: int) -> str:
    cyphertext =''
    table = str.maketrans(lower,lower[key%26:] + lower[:key%26])
    cyphertext = plaintext.translate(table)

def decrypt (plaintext: str, key: int) :
    cyphertext =''
    table = str.maketrans(lower,lower[-key%26:] + lower[:-key%26])
    cyphertext = plaintext.translate(table)
    return cyphertext
