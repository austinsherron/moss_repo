def remove_punctuation(string:str)-> str:
    '''removes all punctuation in a string'''
    s = ''
    letter_list = ['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
    for letter in string:
        if letter.lower() not in letter_list:
            s+=' '
        else:
            s+=letter
    return s

def split_clean_str(s: str)->list:
    '''splits and cleans a string'''
    s=remove_punctuation(s)
    return s.split()

def Caesar_decrypt(cipherText: str, y: int)-> str:
    '''decrypts an encrypted string'''
    x = y%26
    plainText = ''
    for i in range(len(cipherText)):
        letter = cipherText[i].lower()
        plainText += chr(ord(letter)-x)
    return plainText

def copy_file(operation: str):
    '''takes one parameter (string) if it is 'line numbers' then it will add line numbers
            to the string, if it is 'Gutenburg Trim it will trim the excess of the book,
            if it is 'statistics' it will print the statistics''''

        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        count = 1
        for line in infile:
            outfile.write('{}. {}'.format(count,line))
            count+=1
        infile.close()
        outfile.close()
    if operation == 'Guterberg Trim':
        line_list = []

        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        for line in infile:
            line_list.append(line)
        for x in range(len(line_list)):
            if '*** START' in line_list[x]:
                start = x
                break
        for x in range(len(line_list)):
            if '*** END' in line_list[x]:
                end = x
                break
        for y in range(start, end+1):
            outfile.write(line_list[y]) 
        infile.close()
        outfile.close()
    if operation == 'statistics':
        infile_name = input("Please enter the name of the file to copy: ")
        infile = open(infile_name, 'r')
        outfile_name = input("Please enter the name of the new copy:  ")
        outfile = open(outfile_name, 'w')
        lines = 0
        empty = 0

            outfile.write(line)
            lines+=1

        outfile.write('{:10.5f}   average characters per non-empty line \n'.format(char/(lines-empty)))
        infile.close()
        outfile.close()
