import random

def random_names(number: int) -> 'list of strings':


    surname_list = read_text('surnames.txt')
    female_list = read_text('femalenames.txt')
    male_list = read_text('malenames.txt')
    genders = ['male', 'female']
    name_list = []
    for i in range(number):
        surname = random.choice(surname_list)
        first_name = ''
        if random.choice(genders) == 'male':
            first_name = random.choice(male_list)
        elif random.choice(genders) == 'female':
            first_name = random.choice(female_list)
        name_list.append('{}, {}'.format(surname.capitalize(), first_name.capitalize()))
    return name_list

    

def read_text(filename: str) -> 'list of strings':


    returnList = []
    file = open(filename, 'r')
    fileRaw = file.readlines()
    for line in fileRaw:
        tabIndex = line.index('\t')
        returnList.append(line[0:tabIndex])
    return returnList

Alphabet = 'abcdefghijklmnopqrstuvwxyz'
def rotate_alphabet(number: int) -> str:
    """Rotates the alphabet according to number"""
    new_alphabet = ''
    for i in range(len(Alphabet)):
        if i + number >= 26:
            n = (i + number) - 26
            new_alphabet = new_alphabet + (Alphabet[n])
        else:
            n = i
            new_alphabet = new_alphabet + (Alphabet[n + number])
    return new_alphabet

def Caesar_decrypt (message: str, number: int) -> str:
    """Decrypts message using Caesar cipher"""
    table = str.maketrans(rotate_alphabet(number), Alphabet)
    s = message.translate(table)
    return s
