def stats(l:list):
    '''takes a list of strings and prints statistics as follows:
 16824  lines in the list
   483  empty lines
  53.7  average characters per line
  65.9  average characters per non-empty line'''
    emp=0
    strlength=0
    print('{:5}  lines in the list'.format(len(l)))
    for i in l:
        strlength+=len(i)
        if i == '\n':
            emp += 1
    print('{:5}  empty lines'.format(emp))
    print('{:7.2f}  average characters per line'.format(strlength/len(l)))
    print('{:7.2f}  average characters per non-empty line'.format(strlength/(len(l)-emp)))
