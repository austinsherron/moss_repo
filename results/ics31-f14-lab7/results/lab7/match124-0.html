def random_names(n:int)-> list:
    '''Creats a list of random names. Lastname,Firstname'''

    sur=[]
    infile = open("surnames.txt", 'r')
    surnames = infile.readlines()
    infile.close()
    for i in surnames:
        #process string 'i' so that you get the contents before \t
        y = i.split(sep= '\t')
        sur.append(y[0])
    
    female=[]
    infile = open("femalenames.txt", 'r')
    femalenames = infile.readlines()
    infile.close()
    for i in femalenames:
        #process string 'i' so that you get the contents before \t
        x = i.split(sep= '\t')
        female.append(x[0])
    
    male= []
    infile = open("malenames.txt", 'r')
    malenames = infile.readlines()
    infile.close()
    for i in malenames:
        #process string 'i' so that you get the contents before \t
        z = i.split(sep= '\t')
        male.append(z[0])
    
    result= []
    for i in range(n):
        full_name= ""
        num= randrange(1,1001)
        full_name+= sur[num].lower().capitalize()+ ", "
        MF= randrange(1,3)
        if MF==1:
            num= randrange(1,1001)
            full_name+= female[num].lower().capitalize()
        else:
            num= randrange(1,1001)
            full_name+= male[num].lower().capitalize()
        result.append(full_name)
    return result

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate(n:int)->str:
    '''rotates alphabet to create encryption'''
    return ALPHABET[n:] + ALPHABET[:n]

def Caesar_break(s:str)-> str:
    '''takes encrytped string and returns that string decrypted '''
    message = ''
    result = []
    most = 0
    for i in range(26):
        table = str.maketrans(rotate(i), ALPHABET)
        result.append(s.translate(table))
    for i in result:
        word_list = i.split()
        hits = 0
        for word in word_list:
            if word in dic:
                hits += 1
        if hits > most:
            most = hits
            message = i
    return message
