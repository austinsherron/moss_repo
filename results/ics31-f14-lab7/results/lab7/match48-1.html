al='abcdefghijklmnopqrstuvwxyz'
def Caesar_encrypt(r:str,p:int)->str:
    a=''
    for i in r.lower():
        c=al.find(i)
        if c == -1:
            a+=' '
        elif (c+p)>=26:
            x=(c+p)%26
            a+=al[x]
        else:
            x=(c+p)
            a+=al[x]
    return a



def Caesar_decrypt(r:str,p:int)->str:
    a=''
    for i in r.lower():
        c=al.find(i)
        if c == -1:
            a+=' '
        elif (c-p)<0:


            x=(c-p)%26
            a+=al[x]
        else:
            x=(c-p)
            a+=al[x]
    return a

def Caesar_break(ciphertext:str)->str:
    c=0
    n=0
    s=ciphertext.split()
    dlist=data4.split()
    for i in range(0,27):
        for o in s:
            word=Caesar_decrypt(o, i)
            if word in dlist:
                n+=1
            elif n > c:
                c = n
                r = i

    return Caesar_decrypt(data4, r)
