def random_names (n:int) -> str:
    newnames = []
    f = surnames.read()
    g = f.split()
    h = 0
    lastname = []
    while h < len(g):
        lastname.append(g[h])
        h += 4
    q = femalenames.read()
    r = q.split()
    s = 0
    girlnames = []
    while s < len(r):
        girlnames.append(r[s])
        s += 4
    a = malenames.read()
    b = a.split()
    c = 0
    boynames = []
    while c < len(b):
        boynames.append(b[c])
        c += 4
    firstname = boynames + girlnames
    for p in range (n):
        o = ('{}, {}'.format(lastname[randrange(len(lastname))], firstname[randrange(len(firstname))]))
        newnames.append(o)
    return newnames
