from random import randrange
def file_list(a:'file of names')->list:
    """takes a file as a parameter and converts it into a list( just for the purpose
    to help in creating a list of names with the text files we were provided)"""
    empty=[]
    infile=open(a,'r')
    data=infile.read()
    stuff=data.split()
    for i in range(0,len(stuff),4):
        empty.append(stuff[i].capitalize())
    return empty
def single_random_name(name:list)->str:
    '''takes a list and returns a generated name'''
    for i in range(len(name)):
        num=randrange(len(name))
    return name[num]
def female_male()->list:
    '''takes no parameters and returns a list, male or female'''
    if randrange(3)==2:
        return file_list('femalenames.txt')
    return file_list('malenames.txt')
def random_names(n:int)->list:
    '''takes an integer and returns a list of that many strings,
    with each string a randomly generated name as "Lastname, Firstname"'''
    name_list=[]
    firstname=female_male()
    surnames=file_list('surnames.txt')
    for i in range(n):
        name_list.append(single_random_name(surnames)+','+' '+single_random_name(firstname))

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def translate_punctuation(sentc: str) -> str:
    '''replaces punctuations with blanks'''
    table = str.maketrans("""!?.:;*@#$%^&"()-_=+][{}\/>,<`~|""", "                               ")
    return sentc.translate(table)
def rotate_alphabet(num: int)-> str:
    rotated=''
    rotated+=ALPHABET[num:]+ALPHABET[:num]
    return rotated

def Caesar_encrypt(message: str, key: int) ->str:
    '''takes two arguments: a string containing the message the plaintext for encryption
    and an int for the key, indicating how far down the alphabet to find each substitute letter.
    The encryption function returns the ciphertext'''
    encrypted = ''
    message = message.lower()
    table = str.maketrans(ALPHABET, rotate_alphabet(key))
    encrypted = message.translate(table)
    return encrypted
assert Caesar_encrypt('cat',3)== 'fdw'

def Caesar_break(message: str) -> str:
    '''takes takes a ciphertext string (encrypted using a Caesar cipher as we did last week)
    and returns the plaintext for that string, without having the key.'''
    decrypted = message
    highesthit = 0
    infile=open('wordlist.txt','r')
    data=infile.read().lower()
    data=data.split()
    for i in range(26):
        table= str.maketrans(rotate_alphabet(i), ALPHABET)
        word= message.translate(table)
        trans_word=translate_punctuation(word)
        stuff=trans_word.split()
        for k in range(len(stuff)):
            hit=0
            if stuff[k] in data:
                    hit+=1
                    if hit>highesthit:
                        highesthit=hit
                        decrypted=word
    return decrypted

def stats(L: list):
    '''takes a list of strings and prints statistics as follows:
    16824   lines in the list
    483   empty lines
    53.7 average characters per line
    65.9 average characters per non-empty line'''
    total_characters=0
    empty_count = 0

    print('{0:4}  '.format(len(L)), 'lines in the list')
    

    for i in L:
        if i.strip() == '':
            empty_count += 1
            
    print('{0:4}  '.format(empty_count), 'empty lines')
    for i in L:
        total_characters+=len(i)

    
    print('{0:6.1f}'.format((total_characters)/(len(L))), 'average characters per line')

    non_empty_lines = len(L) - empty_count
    print('{0:6.1f}'.format(total_characters/non_empty_lines), 'average characters per non-empty line')
