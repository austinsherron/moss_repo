def random_names(num: int) -> list:
    s = ' '
    ln = len(random_lastnames())
    fn = len(random_first_names())
    l = [ ]
    for i in range(0, num):
        s = '{0:}, {1:}'.format(random_lastnames()[randrange(0, ln)], random_first_names()[randrange(0,fn)])
        l.append(s)
    return l

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def alphabet_rotate(num: int):
    s = ALPHABET.replace(ALPHABET[0:num], '') + ALPHABET[0:num]
    return s

def Caesar_decrypt(s:str, num:int):
    table = str.maketrans(alphabet_rotate(num), ALPHABET)
    return s.translate(table)

def Caesar_break(s: str) -> str:
    L = []
    for i in range(0, 26):
       L.append(Caesar_decrypt(s, i))
    return L

def random_words() -> list:
    infile=open('wordlist.txt', 'r')
    l = []
    for line in infile:
        x = infile.readline()
        lst = x.split()
        l.append(lst[0])
    return l

def word_count(s: str) -> int:
    count = 0
    l = s.split()
    L = random_words()
    for i in l:
        if i in L:
            count += 1
    return count

def brute_break(s: str):
    number = 0
    word = ''
    for i in Caesar_break(s):
        if word_count(i) > number:
            number = word_count(i)
            word = i
    return word
