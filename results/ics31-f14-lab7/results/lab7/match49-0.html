def delete_punctuation (a:str)-> str:
    table = str.maketrans('!@#$%^&*()_+=-[]{}|\:"?><;/.,',
                          '                             ')
    b = a.translate(table)
    return b

def rotated_alphabet(a:int)->str:
    ALPHABET = 'abcdefghijklmnopqrstuvwxyz'       
    New_One = ALPHABET[a:]+ALPHABET[:a]
    return New_One

def copy_file (a:str):
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors='ignore')
    data = infile.readlines()
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    if a == 'line numbers':
        format_str = '{0:5}: {1}'
        for i in range(len(data)):
            outfile.write(format_str.format(i+1, data[i]))
    elif a == 'Gutenberg trim':
        for i in range(len(data)):
            if '*** START OF THIS PROJECT' in data[i]:
                b = i
            if '*** END OF THIS PROJECT' in data[i]:
                c = i
        outfile.write(''.join(data[b : c+1]))
    elif a == 'statistics':
           b = len(data)
           c = 0
           d = 0
           for i in data:
               if len(i.strip())==0:
                   c+=1
           for i in data:
               d += len(i)
               e = d/b
               f = d/(b-c)
           g=('{:5}   lines in the list'.format(b)+'\n'+
              '{:5}   empty lines'.format(c)+'\n'+
              '{:7.1f} average characters per line'.format(e)+'\n'+
              '{:7.1f} average characters per non-empty line'.format(f))
           result = ''.join(data) + '\n\n' + str(g) + '\n'
           outfile.write(result)
    else:
        outfile.write(''.join(data))
        
    infile.close()
    outfile.close()
