alphabet = 'abcdefghijklmnopqrstuvwxyz'

def create_rotated_alphabet(a: str, n: int) -> str:
    ''' create a version of the alphabet that is n spaces away for use in
    the Caesar cypher '''
    rotated_alphabet = ''
    for i in range(26):
        if i+n < 26:
            rotated_alphabet += a[i+n]
        elif i+n < 0:
            rotated_alphabet += a[i+n + 26]
        else:
            rotated_alphabet += a[i+n - 26]
    return rotated_alphabet

def Caesar_encrypt(message:str, key:int, a: str) -> str:
    ''' encrypt a message using the Caesar cypher: replace a letter with
    a letter that is key spaces along the alphabet '''
    table = str.maketrans(a, create_rotated_alphabet(a, key))
    return message.translate(table)

def Caesar_decrypt(message:str, key:int, a: str) -> str:
    ''' Decrypt an encoded message by reversing the key '''
    table = str.maketrans(a, create_rotated_alphabet(a, -key))
    return message.translate(table)
