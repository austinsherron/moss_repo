ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotate(num):
    newstring = ''
    front = ALPHABET[0:num]
    end = ALPHABET[num:]
    newstring = end + front  
    return newstring

    
def Caesar_decrypt(str2, key:int)->str:
    '''Takes in 2 parameters and returns the plaintext'''

    new = rotate(key)
    table = str2.maketrans(new, ALPHABET)
    str2.lower()
    return str2.translate(table)

def stats (L: list):
    """Takes a list of strings and prints statistics."""
    line = 0
    e_lines = 0
    chars = 0
    for lines in L:
        line += 1
        if lines == '' or "":
            e_lines += 1
        for char in lines:
            chars += 1
    return( ('{:5d}   lines in the list'.format(line)+ '\n {:5d}   empty lines'.format(e_lines)) + '\n {:7.1f} average characters per line'.format(chars/line) + '\n {:7.1f} average characters per non-empty line'.format(chars/(line - e_lines)))
