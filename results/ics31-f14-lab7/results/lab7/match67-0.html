def encrypt(s:str, n:int):


    if n > len(s):
        ns = s[n%len(s):] + s[:n%len(s)]
    else:
        ns = s[n:]+ s[:n]
    return ns
def Caesar_encrypt(s:str,n:int)->str:
    table = str.maketrans(ALPHABET,encrypt(ALPHABET,n))
    table2 = str.maketrans(alphabet,encrypt(alphabet,n))
    s = s.translate(table)


    s = s.translate(table2)
    return s
def Caesar_decrypt(s:str,n:int)->str:
    '''decrypt a strings, return a string without changing the capital'''
    table = str.maketrans(encrypt(ALPHABET,n),ALPHABET)
    table2 = str.maketrans(encrypt(alphabet,n),alphabet)
    s = s.translate(table)
    s = s.translate(table2)
    return s
def SofR(s:str)->[str]:
    result = []
    for i in range(0,26):
        result.append(Caesar_decrypt(s,i))
    return result

def copy_file(a:str):
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r', errors='ignore')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    i=1
    if a == 'line numbers':
        for line in infile:
            new_line=("{:5}: {:}".format(i,line))
            i+=1
            outfile.write(new_line)
    elif a == 'Gutenberg trim':
        line = infile.read().split('\n')
        for ls in range(len(line)):
            if "*** START" in line[ls]:
                for lr in range(ls,len(line)):
                    if "*** END" in line[lr]:
                        outfile.write(line[lr]+'\n')
                        break
                    else:
                        outfile.write(line[lr]+'\n')
    elif a == 'statistics':
        l = infile.read().split('\n')
        for enil in l:
            outfile.write(enil+'\n')
        ll=len(l)
        empty=0
        av=0
        for i in range(len(l)):
            av=av+(len(l[i])-l[i].count(' '))
            if l[i]=='':
                empty=empty+1
        av2=av/len(l)
        avnonem=(av)/(len(l)-empty)
        outfile.write("{:7} {:}".format(ll,'lines in the list')+'\n')
        outfile.write("{:7} {:}".format(empty,'empty lines')+'\n')
        outfile.write("{:7.1f} {:}".format(av2,'average characters per line')+'\n')
        outfile.write("{:7.1f} {:}".format(avnonem,'average characters per non-empty line')+'\n')

    else:
        for line in infile:
            outfile.write(line)
        infile.close()
        outfile.close()
