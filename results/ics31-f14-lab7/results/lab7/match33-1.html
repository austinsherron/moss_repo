Alpha = 'abcdefghijklmnopqrstuvwxyz'

def Caeser_decrypt(m:str, n:int):
    '''decrypts the str utilizing the int as key'''
    n = n%26
    Ciph = Alpha[n:] + Alpha[0:n]
    b = m.lower()
    a = b.maketrans(Ciph,Alpha)
    return b.translate(a)


def Caeser_break(code:str) ->str:
    '''takes in a coded cipher and returns the uncoded string'''
    Dictionary_ref = []
    counter_list = []
    table = str.maketrans('`~!@#$%^&*()_+{}|\][:;<?><,./"',
                          '                              ')


    code = code.translate(table)
    for i in range(26):
        Dictionary_ref.append(Caeser_decrypt(code, i))
    for v in range(26):
        counter = 0
        word_list = Dictionary_ref[v].split()
        for word in word_list:
            if word in dictionary:
                counter += 1
        counter_list.append(counter)
    highest_count = counter_list.index(max(counter_list))
    return Caeser_decrypt(code,(highest_count))

    blank = 0
    char = 0
    count = len(string)
    print ('{:8} Lines in the list'.format(count))
    for i in range(count):
        if i == '':
            blank += 1
    print('{:8} Empty lines'.format(blank))
    for i in string:
        for v in i:
            char += 1
    print('{:8.3} Average characters per line'.format(char/count))

def copy_file(string: str)->None:
    infile_name = input("Please enter the name of the file to copy: ")


    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    line_number = 1
    for line in infile:
        if string == 'line numbers':
            outfile.write('{:6}: '.format(line_number) + line)
            line_number += 1
        elif string == 'Gutenberg trim':
            data = infile.readlines()


            new_data = []
            for i in range(len(data)):
                index = data[i].find('*** START OF THIS PROJECT GUTENBERG EBOOK')
                if index != -1:
                    new_data = data[(i + 1):]
            for v in range(len(data)):
                index = data[v].find('*** END OF THIS PROJECT GUTENBERG EBOOK')
                if index != -1:
                    new_data = new_data[:v]
            book_str = '\n'.join(new_data)
            outfile.write(book_str)
        elif string == 'statistics':
            outfile.write(line)
            data = infile.readlines()
            stats(data)
        else:
            outfile.write(line)
    infile.close()
    outfile.close()
