from random import randrange

def stringCount(file: str, target_string: str):
    infile=open(file)
    content=infile.read()
    return content.count(target_string)

def make_list(file:str):
    '''makes the txt file into a list'''
    infile=open(file,"r+")
    y = []
    z = []
    for i in range(stringCount(file,"\n")):
        x = infile.readline()
        y.append(x)
    for i in y:
        j = i.split()
        z.append(j[0])
    return z


def random_names(n:int):
    '''returs a list of many strings with each string randomly generated'''
    a = make_list("surnames.txt")
    b = make_list("malenames.txt")
    c = make_list("femalenames.txt")
    d = b + c
    y = []
    for i in range(n):
        j = a[randrange(0,len(a))]
        w = d[randrange(0,len(d))]
        y.append(j.capitalize() + ", " + w.capitalize())
    return y

def rotated(string: str, num: float):
    if num >= 26:
        num -= 26
    return string[num:] + string[:num]
def Caesar_encrypt(m:str, k:int) -> str:
    ''' entering plaintext returns cipher text'''
    real_alpha = 'abcdefghijklmnopqrstuvwxyz'
    real_ALPHA = real_alpha.upper()
    plus_upper_alpha = real_alpha + real_ALPHA
    new_alpha = rotated(plus_upper_alpha,k)
    trans = str.maketrans(plus_upper_alpha, new_alpha)
    new_string = ''
    for i in m:
        new_str = i.translate(trans)
        new_string += (new_str)  
    return new_string

def Caesar_decrypt(m:str, k:int) -> str:
    '''Entering ciphertext and returns plaintext'''
    real_alpha = 'abcdefghijklmnopqrstuvwxyz'
    real_ALPHA = real_alpha.upper()
    plus_upper_alpha = real_alpha + real_ALPHA
    new_alpha = rotated(plus_upper_alpha,k)
    trans = str.maketrans(new_alpha, plus_upper_alpha)
    new_string = ''
    for i in m:
        new_str = i.translate(trans)
        new_string += (new_str)  
    return new_string

def empty_line(s: list) -> int:
    '''Return number of empty lines in the function'''
    result = []
    for i in s:
        if i == '\n':
            result.append(i)
    return len(result)

def total_characters(s: list) -> int:
    '''Returns total number of characters in a line'''
    result = []
    for i in s:
        result.append(len(i))
    return sum(result)
