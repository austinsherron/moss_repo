def read_file(file_to_read: 'File of name attributes') -> ['Names in the file']:
    '''Takes a file of name attributes and returns a list of the
       names in that file'''
    infile = open(file_to_read, 'r')
    content = infile.readlines()
    infile.close()
    name_list = []

    for line in range(len(content)):
        split = content[line].split()
        name_list.append(split[0])
    return name_list

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def replace_punctuation(phrase: str) -> str:
    '''Takes a phrase and replaces all of the punctuation in the phrase
       with blanks'''
    table = str.maketrans('''!@#$%^&*()_+=-{}[]:";'<>?,./''', '''                            ''')
    new_phrase = phrase.translate(table)
    return new_phrase


def rotate_ALPHABET(shift: int) -> str:
    '''Takes a number and rotates the string ALPHABET by taking the
       characters up to the index of that number and placing them at
       the end of the string'''
    temp = ALPHABET[:((shift + 26)%26)]
    result = ALPHABET[((shift + 26)%26):] + temp
    return result
assert rotate_ALPHABET(23) == 'xyzabcdefghijklmnopqrstuvw'

def Caesar_decrypt(message: str) -> list:
    '''Translates using the rotate_ALPHABET function'''
    decrypted_message_possibilities = []
    for a in ALPHABET_list:
        table = str.maketrans(a, ALPHABET)
        decrypted_message_possibilities.append(message.translate(table))
    return decrypted_message_possibilities

def split_Caesar_decrypt(message: str) -> list:
    Caesar_list = Caesar_decrypt(message)
    split_list = []
    for s in Caesar_list:
        split_list.append(s.split())
    return split_list
def check_split_Caesar_decrypt(message: str) -> str:
    list_to_check = split_Caesar_decrypt(message)
    infile = open('wordlist.txt', 'r')
    check_list = []
    content = infile.read()
    infile.close()
    lowered_content = content.lower()
    for lst in list_to_check:
        check = 0
        for s in lst:
            s = s.lower()
            s = replace_punctuation(s)
            if s in lowered_content:
                check += 1
        check_list.append(check)
    index = -1
    for i in check_list:
        index += 1
        if i == len(list_to_check[0]) - 1:
            return list_to_check[index]
def Caesar_break(message: str) -> str:
    break_list = check_split_Caesar_decrypt(message)
    Caesar_break_string = ''
    for s in break_list:
        Caesar_break_string = Caesar_break_string + s + ' ' 
    return Caesar_break_string

def find_empty_line(strlist:list)->int:
    '''takes a list of strings and finds the number of empty lines'''
    emptystr = []
    for line in strlist:
        if line == '\n':
            emptystr.append(line)
    return len(emptystr)

def count_line(strlist:list)->int:
    '''finds the amount of characters in each line of list'''
    characterlen = []
    for line in strlist:
       characterlen.append(len(line))
    return sum(characterlen)
