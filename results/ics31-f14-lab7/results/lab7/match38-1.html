def random_names (x:int)->[str]:

    surnames1 = []
    while c < len(b):
        surnames1.append(b[c])
        c += 4
    d = malenames.read()
    e = d.split()
    f = 0
    malenames1 = []
    while f < len(e):
        malenames1.append(e[f])
        f+=4
    g = femalenames.read()
    h = g.split()
    i = 0
    femalenames1 = []
    while i < len(h):
        femalenames1.append(h[i])
        i+=4
    firstnames = malenames1 + femalenames1
    for j in range(x):
        k = ('{}, {}'.format(surnames1[randrange(len(surnames1))],
                              firstnames[randrange(len(firstnames))]))
        listofnames.append(k)
    return listofnames
