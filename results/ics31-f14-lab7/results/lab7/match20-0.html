import string
ALPHABET = 'abcdefghijklmnopqrstuvwxyz'


def rotation(n: int) -> str:
    '''Rotates the alphabet, starting it at given number index'''
    B = ALPHABET[n:] + ALPHABET[0:]
    return B
    

def Caesar_encrypt(message: str, num: int) -> str:
    ''' Takes a string and encrypts it by a number shift in the alphabet '''
    x = str.maketrans(ALPHABET, rotation(num))
    return message.lower().translate(x)

def Caesar_decrypt(message: str, num: int) -> str:
    ''' Takes a string and decrypts it by a number shift in the alphabet '''
    x = str.maketrans(rotation(num), ALPHABET)


    return message.translate(x)

def rotation26() -> list:
    '''Returns a list of 26 rotated alphabets'''
    L = []
    for i in range(26):
        L.append(rotation(i))
