Alphabet = 'abcdefghijklmnopqrstuvwxyz'

def c_alphabet(number: int)-> str:
    '''Returns the changed alphabet according to the key'''    
    x = Alphabet.strip(Alphabet[:number]) + Alphabet[:number]
    return(x)


assert c_alphabet(1) =='bcdefghijklmnopqrstuvwxyza'
assert c_alphabet(3) =='defghijklmnopqrstuvwxyzabc'


def Caesar_encrypt(message: str, key: int) -> str:
    '''Returns the encrypted message given key'''
    table1 = message.maketrans(Alphabet,c_alphabet(key%26))
    return message.translate(table1)
assert Caesar_encrypt('i love programming',15) =='x adkt egdvgpbbxcv'


def Caesar_decrypt(crypt: str, key: int) -> str:
    '''Returns the message decrypted given key'''
    table2 = crypt.maketrans(c_alphabet(key), Alphabet)
    return crypt.translate(table2)
assert Caesar_decrypt('x adkt egdvgpbbxcv',15) == 'i love programming'

print(Caesar_encrypt('Each partner should do this part independently: Make up a message without telling your partner what it is. Encrypt the message with a key of your choosing.',6))
      


def Caesar_list(message: str)-> list:
    '''Takes encrypted message and returns the list of translations'''
    translationlist = []
    splitlist = []
    for i in range(1,27):
        table2 = message.maketrans(c_alphabet(i), Alphabet)
        translationlist.append(message.translate(table2))
    for x in translationlist:
        splitlist.append(x.split())
    return splitlist
                        



def dictionary(sentence:list)-> int:
    '''takes a list of words and counts the number of words in each list'''
    wordcount = 0
    infile = open('dictionary.txt','r')
    word = infile.read()
    wordlist = word.split()
    infile.close
    for word in sentence:
        wordcount += (wordlist.count(word))
    return wordcount




def Caesar_break(encrypted_message: str):
    '''Returns the decrypted message'''
    list1 = Caesar_list(encrypted_message)
    wordcount = []
    for item in list1:
        wordcount.append(dictionary(item))
    key = wordcount.index(max(wordcount))+1
    return Caesar_decrypt(encrypted_message, key)

def line_word_sum(sentence: str)-> int:
    '''Returns sum of the characters on a line'''
    temp_sum=0
    for j in sentence.split():
        temp_sum += len(j)
    return temp_sum

def stats(l:list):
    '''Takes a list of strings and prints statistics'''
    x=0
    total = 0
  
    #lines in the list
    linelist =('{0:5}   lines in the list'.format(len(l)))
    #empty lines
    for i in l:
        if i == ' ':
            x = x+1
        elif i == '':
            x = x+1
    emptylines = ('{0:5}   empty lines'.format(x))
   
    #average characters per line
    for i in l:
        total += line_word_sum(i)
    avgchar =('{0:7.1f} average characters per line'.format(total/len(l)))
    #average characters per non empty line
    avgperline =('{0:7.1f} average characers per non empty line'.format(total/(len(l)-x)))
    return '\n' +'\n'+ linelist + '\n' + emptylines + '\n' + avgchar + '\n' + avgperline

    


def copy_file(parameter:str):
    '''makes a copy of the inputed file'''
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")
    outfile = open(outfile_name, 'w')
    
    if parameter == 'Gutenberg trim':
        data = infile.read()
        content = data.split('\n')
        start = content.index('*** START OF THIS PROJECT GUTENBERG EBOOK THE ADVENTURES OF SHERLOCK HOLMES ***')
        stop = content.index('*** END OF THIS PROJECT GUTENBERG EBOOK THE ADVENTURES OF SHERLOCK HOLMES ***')
        for l in range(start,stop +1):
            outfile.write(str(content[l] + '\n'))
        outfile.close()   
    elif parameter == 'line numbers':
        data = infile.read()
        content = data.split('\n')
          
        for l in range(len(content)):
            outfile.write('{:5}: {} \n'.format(l+1, str(content[l])))
        outfile.close()
    elif parameter == 'statistics':
        
        data = infile.read()
        content = data.split('\n')
        for line in data:
            outfile.write(line)
        outfile.write(str(stats(content)))
        
        
        outfile.close()
        
    else: 
        for line in infile:
            outfile.write(line)
        infile.close()
        outfile.close()
