def create_random_name():
   '''Creates one random name'''
   surnames = []
   last = open("surnames.txt", 'r')
   for i in last:
      datalines = last.readline().split()
      surnames.append(datalines[0])
      x = random.randrange(0, len(surnames))
   last.close()

   first_names = []

   first.close()

   print(surnames[x].capitalize()+ ", " + first_names[y].capitalize())

def random_names(n: int) -> str:

    ''' returns the alphabet with beginning characters added to end '''

def Caesar_encrypt(plaintext:str,key:int) -> str:
   ''' returns the encrypted plaintext with the key '''
   table = str.maketrans(ALPHABET,rotated_alphabet(ALPHABET,key % 26))
   new_text = plaintext.lower()
   return new_text.translate(table)

def Caesar_decrypt(message:str,key:int) -> str:
    ''' returns the decrypted message '''
    table = str.maketrans(rotated_alphabet(ALPHABET,key % 26),ALPHABET)
    return message.translate(table)
