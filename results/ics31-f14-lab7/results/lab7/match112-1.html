from random import randrange
def create_namelist_from_file(filename:str) -> list:
    infile = open(filename)
    content = infile.readlines()
    infile.close()
    result = []
    for i in content:
        result.append(i.split()[0])
    return result

Surnames = create_namelist_from_file('Surnamelist.txt')
FemaleNames = create_namelist_from_file('FemaleNamelist.txt')
MaleNames = create_namelist_from_file('MaleNameList.txt')


def Single_random_name(s:list, f:list, m:list) -> str:
    result = ''
    RandomSurname = s[randrange(0, len(s))]
    RandomMale = m[randrange(0, len(m))]
    RandomFemale = f[randrange(0, len(f))]
    RandomGender = randrange(0, 1)
    if RandomGender == 0:
        return RandomSurname  + ',' + RandomFemale
    else:
        return RandomSurname + ',' + RandomMale


def random_names(x:int) -> list:
    result = []
    for i in range(x):
        result.append(Single_random_name(Surnames, FemaleNames, MaleNames)) 
    return result

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'
def rotate(x:str, y:int)-> str:
    s = x[y:]
    s += x[:y]
    return s
def Caesar_decrypt(x:str, y:int)-> str:
    orig = x.maketrans(rotate(ALPHABET, y), ALPHABET)
    t = x.translate(orig)
    return t
Caesar_decrypt('dbu',1)

def average(s: list):
    ''' Totals up values in list and divides by length of list
    to find average '''
    total = 0
    if len(s) == 0:
        return 0
    for i in range(len(s)):
        total += len(s[i])
    return total/len(s)

def stats(x: list):
    ''' Prints stats of list '''
    emptyLines = []
    nonEmpty = []
    for i in x:
        if i.isspace():
            emptyLines.append(i)
        else:
            nonEmpty.append(i)
    print('{0:5}   lines in the list\n{1:5}   empty lines\n{2:7.1f} average characters per line\n{3:7.1f} average characters per non empty line'.format(len(x),len(emptyLines),average(x),average(nonEmpty)))
