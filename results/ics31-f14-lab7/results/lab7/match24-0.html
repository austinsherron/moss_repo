def surname(n:str)->list:
    ''' Takes a file and returns only the surnames in the list'''
    infile= open(n, 'r')
    data= infile.readlines()
    infile.close()
    result=[]
    for i in data:
        result.append(i.split())
    last_name=[]
    for i in result:
        last_name.append(i[0].title())
    return last_name
    
def Names(n:str, m:str)->list:
    ''' Takes a file and returns only the female and
    male names in the list'''
    infile= open(n,'r')
    infile1=open(m,'r')
    data= infile.readlines()
    data1= infile1.readlines()
    infile.close()
    infile1.close()
    result=[]
    for i in data:
        result.append(i.split())
    for i in data1:
        result.append(i.split())
    name=[]
    for i in result:
        name.append(i[0].title())
                   
    return name

def random_names(n:int)-> [str]:
    ''' Returns a list of n (an int) randomly generated names'''
    surname1= surname('surnames.txt')
    names1= Names('malenames.txt', 'femalenames.txt')

    result=[]
    for i in range(n):
        result.append(surname1[randrange(0,len(surname1))] + ' , ' + names1[randrange(0,len(names1))])
    return result

ALPHABET = 'abcdefghijklmnopqrstuvwxyz'

def rotated_alphabet(k:int) -> str:
    ''' Rotates the alphabet by given key (an int)'''
    return ALPHABET[k:] + ALPHABET[:k]

def dictionary_list(n:str)->list:
    ''' Takes a dictionary file and returns list of words'''
    infile= open(n, 'r')
    data= infile.readlines()
    infile.close()
    word_list =[]
    for i in data:
        word_list.append(i.split()[0])
    word_list_lower= []
    for i in word_list:
        word_list_lower.append(i.lower())
    return word_list_lower

def copy_file(n:str):
    '''Given input of n, copies file accordingly'''
    
    infile_name = input("Please enter the name of the file to copy: ")
    infile = open(infile_name, 'r')
    outfile_name = input("Please enter the name of the new copy:  ")


    outfile = open(outfile_name, 'w')
        

    if n == 'line numbers':
        data= infile.readlines()
        for i in range(len(data)):
            outfile.write("{:5}:  {}".format(i+1,data[i]))
            
        
    
    elif n== 'Gutenberg trim':
        beginning=0
        end=0
        data=infile.readlines()
        
        for i in range(len(data)):
            if "*** START" in data[i]:
                beginning+=i
                continue
            if "*** END" in data[i]:


                end= end+ i
                break
        
        for i in data[beginning+1: end]:
            outfile.write(i)
            
                
    elif n=='statistics':
        
        empty_lines=0
        total_characters=0
        x=1
        for i in infile:
            x+=1
            
            if i.strip()=='':
                empty_lines+= 1
           
       
            total_characters+= len(i)
            outfile.write(i)
            
        print('\n')
        print("{:6d}{:4} lines in the file".format(x,' '))
        print("{:6d}{:4} empty lines".format(empty_lines,' '))
          
        print("{:8.1f}{:2} average characters per line".format(total_characters/x, ' '))
        print("{:8.1f}{:2} average characters per non-empty line".format(total_characters/(x - empty_lines),' '))
        
        for line in infile:
            outfile.write(line)
           

    else:
        
        for line in infile:
            outfile.write(line)
    
    
    infile.close()
    outfile.close()
