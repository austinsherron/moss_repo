from collections import namedtuple
Count = namedtuple('Count', 'letter number')

def letter_count(words: str, letters: str) -> list:
    ''' Returns the count of certain letters in given statement/word
as namedtuple'''
    c_list = []
    for var in letters:
        number = 0
        for x in words:
            if x == var:
                number += 1
        c_list.append(Count(var, number))
    return c_list

print(letter_count('The cabbagge has baggage', 'abcd'))
