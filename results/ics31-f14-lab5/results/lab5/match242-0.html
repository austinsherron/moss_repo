Count = namedtuple('Count', 'letter number')
def letter_count(a:str, b:str) -> list:
    '''returns the counts of certain letters as a list of namedtuples'''
    result = [ ]   
    for x in b:
        count = 0
        for n in a:
            if n == x:
                count += 1
        result.append(Count(x,count))
    return result
print(letter_count('hello', 'he'))
print(letter_count('The thunder struck me', 'rum'))

assert letter_count('hello', 'he') == [Count(letter='h', number=1), Count(letter='e', number=1)]
assert letter_count('The thunder struck me', 'rum') == [Count(letter='r', number=2), Count(letter='u', number=2), Count(letter='m', number=1)]
