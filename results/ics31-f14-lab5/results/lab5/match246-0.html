Count = namedtuple("Count", "letter number")
def letter_count(x: str, y: str) -> [Count]:
    '''Returns the number of letters in a given phrase'''
    result = []
    for i in y:
        result.append(Count(i, x.count(i)))
    return result
assert letter_count("The cow jumped over the moon", "ojc") == [Count("o", 4),
                                                               Count("j", 1),
                                                               Count("c", 1)]
assert letter_count('Anteater', 'zot') == [Count('z', 0), Count('o', 0),
                                           Count('t', 2)]
assert letter_count("Apples and oranges", "") == []
print(letter_count("ICS is the best", "iet"))
