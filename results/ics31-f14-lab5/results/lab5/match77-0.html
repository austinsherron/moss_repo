def Dishlist_average (L: 'list of Dish') -> float:
    '''Returns average of prices in a list of Dish'''
    price_list = Dishlist_prices(L)

Count = namedtuple('Count', 'letter number')

def count_single_letter(word: str, let: str) -> Count:
    count = word.count(let)
    return Count(letter = let, number = count)
assert count_single_letter('Decepticons mobilize', 'e') == Count(letter ='e', number = 3)
assert count_single_letter('The dog jumped over the fence', 'o') ==\
       Count(letter ='o', number = 2)
assert count_single_letter('EVERYTHING IS AWESOME!!!!', '!') == Count(letter ='!', number = 4)

def letter_count(word: str, letters: str) -> [Count]:
    '''Returns a list of Count in which the first parameter
    is the string being counted and the second parameter includes
     the characters of the string you wish to be counted'''
    count_list =[]
    for char in letters:
        count_list.append(count_single_letter(word,char))
    return count_list
    
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',
       number=5), Count(letter='b',number=3), Count(letter='c',number=1),
       Count(letter='d',number=0)]
assert letter_count('The dog hopped', 'o') == [Count(letter = 'o', number = 2)]
assert letter_count('The fluffy pink blanket', 'aeiou') == [Count(letter = 'a',
        number = 1), Count(letter = 'e', number = 2),
        Count(letter = 'i', number = 1) , Count(letter = 'o', number = 0),
        Count(letter = 'u', number = 1)]
