def Dishlist_display(DLT:[Dish]):
    '''Takes a list of Dishes and returns one large string consisting of the string representation of each
    dish followed by a newline ('\n') character. '''
    result='';

        result=result+'[';

                       result=result+str(DLT[i][j]);
                       if j!=(len(DLT[i])-1):

def Restaurant_is_cheap (R:Restaurant, n:int) -> bool:
    '''Takes a restaurant and a number and returns True if the avg. price of the
    Restaurant's menu is less than or equal to a number.'''
    sums = 0.0;            
    for p in R.menu:
        sums = sums + p.price
    avg = sums/len(R.menu)
    if avg > n:
        return False
    else:
        return True

Count = namedtuple('Count', 'letter number');
def letter_count(string, filt: str) -> Count:
    '''Counts the number of times specified character(s) appear in a string.'''
    result = [];
    for i in filt:
        click=0;
        for j in string:
            if j==i:
                click=click+1
        result.append(Count("letter='"+i+"'", "number="+str(click)))
    return result

print ('Count of a, b, c, d, and e in aaabbcccdde: \n')
print(letter_count('aaabbcccdde','abcde'))
print ('\nCount of the alphabet in The quick brown fox jumps over the lazy dog: \n')
print(letter_count('The quick brown fox jumps over the lazy dog','abcdefghijklmnopqrstuvwxyz'))
