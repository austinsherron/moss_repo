Count = namedtuple('Count', 'letter number')
def letter_count(x: str, y: str) -> [namedtuple]:
    '''Return a namedtuple for each character displaying each character and the count of the character.'''
    result = []
    number = 0
    letter = ''
    for i in y:
        for j in x:
            if i == j:
                letter = i
                number += 1
            else:
                letter = i
        result.append(Count(letter, number))
        number = 0
        letter = ''
    return result

assert letter_count('Hello World', 'aeiou') == [Count(letter='a', number=0), Count(letter='e', number=1), Count(letter='i', number=0), Count(letter='o', number=2), Count(letter='u', number=0)]
assert letter_count('Hello World', 'abc') == [Count(letter='a', number=0), Count(letter='b', number=0), Count(letter='c', number=0)]
