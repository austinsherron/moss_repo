Count = namedtuple('Count', 'letter number')

def letter_count(s1: str, s2: str) -> list:
    """ Takes two strings as parameters and returns the counts of certain
        letters in the first parameter """
    result = []
    for letter in s2:
        count = 0
        for l in s1:
            if l == letter:
                count += 1
        result.append(Count(letter, count))
    return result
                
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
