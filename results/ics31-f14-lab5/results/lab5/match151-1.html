Count = namedtuple('Count', 'letter number')

def letter_count(x: str, y: str) -> list:
    '''Return list of namedtuple that identifies counts of each letter of string y in x'''
    c =[]
    for single_str in y:
        c.append(Count(letter=single_str, number=x.count(single_str)))
    return c

assert letter_count('Abba', 'ab') == [Count(letter='a', number=1), Count(letter='b', number=2)]
print(letter_count('Abba', 'ab'))
assert letter_count('Hello there!', 'aeiou') == [Count(letter='a', number=0), Count(letter='e', number=3), Count(letter='i', number=0), Count(letter='o', number=1), Count(letter='u', number=0)]
print(letter_count('Hello there!', 'aeiou'))
assert letter_count('    Yo aslfkfsdf ', ' \t\n') == [Count(letter=' ', number=6), Count(letter='\t', number=0), Count(letter='\n', number=0)]
