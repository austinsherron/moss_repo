def Dishlist_display(x: list)->str:
    result = ''
    for i in x:
        result += ('Name:      ' + i.name + '\n' +
                   'Price:     %.2f' % i.price + '\n' +
                   'Calories:  %i' % i.calories +'\n')
    return result

Count = namedtuple('Count', 'letter number')
def counter(x: str, y: str)->Count:
    return Count(y,x.count(y))

print(counter('nicki', 'i'))

def letter_count(x: str, y:str)->list:
    result = []
    for i in range(len(y)):
        result.append(counter(x, y[i]))
    return result

print(letter_count('bananas', 'bans'))
