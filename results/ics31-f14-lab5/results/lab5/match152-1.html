def Dish_str(x: Dish) -> str:
    ''' Takes a Dish and returns it as 'name ($price): calories' '''
    formatString = "{} (${}): {} cal"
    return formatString.format(x.name, x.price, x.calories)
assert Dish_str(N1) == "Paht Woon Sen ($9.5): 330 cal"
assert Dish_str(N2) == "Fried Rice ($3.0): 210 cal"

def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    ''' Returns true if two dishes' names and calories are the same '''
    return (dish1.name == dish2.name) and (dish1.calories == dish2.calories)

def Dish_change_price(x: Dish, number: float):
    ''' Adds a percentage of the price of the dish to the price of the dish '''
    b = x._replace(price = x.price + (x.price * (number * 0.01)))
    return b
assert Dish_change_price(N1, 100) == Dish(name='Paht Woon Sen', price=19.0, calories=330)
assert Dish_change_price(N2, -50) == Dish(name='Fried Rice', price=1.5, calories=210)

Count = namedtuple("Count", "letter number")
def letter_count(x: str, letters: str) -> list:
    ''' Takes two strings and tests how many times each letter in the second
        string occurs in the first string '''
    result = []
    for i in letters:
        y = x.count(i)
        result.append(Count(i, y))
    return result
