def Dishlist_average(dishL:list)->float:
    '''Takes a list of Dishes and returns the average price of those dishes'''
    output = 0
    priceL = Dishlist_prices(dishL)
    for x in priceL:
        output += x
    if output != 0: #prevents division by 0 error for empty lists
        return output / len(dishL)
    else:
        return 0

Count = namedtuple('Count', 'letter number')

def letter_count(s1:str,s2:str)->list:
    '''Returns a list of the count of the letters in second parameter
    found in the first parameter'''
    output = []
    for y in s2.lower():
        output.append(Count(y,s1.lower().count(y)))
    return output

print(letter_count('The cabbage has baggage', 'abcd'))
print(letter_count('The cabbage has baggage', ''))
print(letter_count('The cabbage has baggage', ' t'))
