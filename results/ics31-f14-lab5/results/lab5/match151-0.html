from collections import namedtuple
Count = namedtuple ('Count', 'letter number')


def letter_count_one(s1:str, s2:str)-> Count:
    "Return a namedtuple. The namedtuple's fields are defined by how many times a \
    single string appears in the given word"
    numb =  s1.count(s2)
    c = Count( s2, numb)
    return c

assert letter_count_one ( 'chicken', 'c' ) == Count(letter='c', number=2)

def letter_count ( s1:str, s2:str) -> list:
    "Return a List of namedtuples. Each namedtuple's value depends on the \
    ammount of times an index string appears in the given word"
    count = []
    for i in s2:
        count.append(letter_count_one (s1, i))
    return count

assert letter_count('cheese','ce')==[Count(letter='c', number=1), Count(letter='e', number=3)] 
print (letter_count('some_message', ' \t\n')) == [Count(letter=' ', number=0), Count(letter='\t', number=0), Count(letter='\n', number=0)]
