# C.2 - Dish_str function
def Dish_str(d: Dish) -> str:
    ''' Takes in a Dish and returns a human-readable str with
    the dish attributes '''
    readable = d.name + ' ($' + str(d.price) + '): ' + str(d.cal) + ' cal'
    return readable

# C.3 - Dish_same function
def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    ''' Takes in two Dish objects and returns True if their name and
    calorie count attributes are the same '''
    return (dish1.name == dish2.name and
            dish1.cal == dish2.cal)

assert not Dish_same(d1, d2)
assert not Dish_same(d1, d3)
assert Dish_same(d1, d1)
assert Dish_same(d3, d4)

# C.4 - Dish_change_price function
def Dish_change_price(d: Dish, percent_change: float) -> Dish:
    ''' Takes in a Dish object and a number representing a percentage price change
    and returns a Dish object equal to the input Dish except with the price attribute
    changed by the input percentage '''
    return (Dish(d.name, d.price + d.price * (percent_change / 100) , d.cal))

# C.10 - Dishlist_average
def Dishlist_average(DL: [Dish]) -> float:
    ''' Takes in a list of Dish and returns their average price '''
    average = 0.0
    if len(DL) != 0:
        average = sum(Dishlist_prices(DL)) / len(DL)
    return average

Count = namedtuple('Count', 'letter number')

def letter_count(input_str: str, search_letters: str) -> 'list of Count':
    ''' Takes in two str as parameters. Returns the counts of a given character
    from the first input string. The character to count is specified by the second
    input string. '''
    result = []
    for char in search_letters:
        number_of_char = input_str.count(char)
        char_as_Count = Count(char, number_of_char)
        result.append(char_as_Count)
    return result

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5),
                                                           Count(letter='b',number=3),
                                                           Count(letter='c',number=1),
                                                           Count(letter='d',number=0)]
assert letter_count('white space  is  bad \n', ' \n') == [Count(letter=' ',number=6),
                                                          Count(letter='\n',number=1)]
assert letter_count(''' spaces
                    with letters

                    and new lines....


                    a plenty !''', ' \n!~') == [Count(letter=' ',number=66),
                                                Count(letter='\n',number=6),
                                                Count(letter='!',number=1),
                                                Count(letter='~',number=0)]
