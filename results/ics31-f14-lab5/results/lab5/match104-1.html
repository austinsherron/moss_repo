Count = namedtuple('Count', 'letter number')

def char_count(message: str, char: str) -> Count:
    '''Takes a string and a single character and returns a single Count
    namedtuple with the character and the number of times it occurs in the string'''
    return Count(char, message.count(char))
assert char_count("Mississippi", "s") == Count('s',4)
assert char_count("Mississippi", "p") == Count('p',2)

def letter_count(message: str, letters: str) -> 'list of Counts':
    '''Returns the counts of certain letters in the first parameter; the
    second parameter is a string that specifies which letters to count'''
    C = []
    for letter in letters:
        C.append(char_count(message, letter))
    return C
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b', number=3),
                                                           Count(letter='c', number=1), Count(letter='d', number=0)]
assert letter_count('''Four    score
and
seven       years
ago''', ' \t\n') == [Count(letter=' ', number=11), Count(letter='\t', number=0), Count(letter='\n', number=3)]
