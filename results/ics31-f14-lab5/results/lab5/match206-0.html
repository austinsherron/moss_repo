def Restaurant_is_cheap(r: Restaurant, n:int) -> bool:
    "Returns True or False if the restaurant menu price average is less than the number stated"
    price = 0
    result = ''
    for x in r.menu:
        price += x.Price

    if price <= n:
        result = True
    else:
        result = False
    return result

Count = namedtuple('Count', 'letter number')
def letter_count (word: str, letters:str) -> "Collection of Count":
    "Return namedtuple with whatever letters put in and how many of each letter"
    namedtuple_list = []
    for i in letters:
        namedtuple_list.append(Count(i, word_count(word, i)))
    return namedtuple_list

def word_count (string:str, individual_letters: str):
    'Return the number of letters that appear on the "word" string'
    number = 0
    for i in string:
        if i == individual_letters:
            number += 1
    return number
