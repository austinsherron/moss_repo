def letter_count(s:str, s2:str)->int:
    '''  returns the counts of certain letters in the first parameter;
the second parameter is a string that specifies which letters to count.'''
    Count = namedtuple('Count', 'letter number')
    result = []
    for a in s2:
       C = Count(a ,s.count(a))
       result.append(C)
    return result
