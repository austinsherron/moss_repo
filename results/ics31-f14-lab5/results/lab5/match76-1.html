from collections import namedtuple
Count = namedtuple('Count', 'letter number')
    
def letter_count(message: str, letters: str) -> str:
    ''' Count the number of specified letters in a specified string '''
    list_result = []
    count_result = 0
    for i in letters:                               # Loop through each letter in given string of letters
        for j in message:                           # Count how many times one letter is repeated
            if i == j:
                count_result += 1             
        list_result.append(Count(i, count_result))  # Append a new Count with the letter
        count_result = 0                            # Reset count of how many times one letter is repeated
    return list_result
