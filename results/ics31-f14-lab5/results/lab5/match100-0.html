from collections import namedtuple
Count = namedtuple('Count', 'letter number')



def count_one_character(s: str, c: str) -> 'Count':
    ''' takes a string and a single character and
       returns a single Count namedtuple with the
       character and the number of times it occurs in the string'''
    return Count(c, s.count(c))
assert count_one_character('bob', 'b') == Count(letter='b', number=2)
print(count_one_character('bob', 'b'))
print()

def letter_count(s: str, c: str) -> 'list of Counts':


    R = []
    for l in c:
        R.append(count_one_character(s, l))
    return R
assert letter_count('bob', 'bo') == [Count(letter='b', number=2),
                                     Count(letter='o', number=1)]
print(letter_count('The cabbage has baggage', 'abcd'))
