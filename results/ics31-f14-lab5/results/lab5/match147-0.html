def Dish_str(dish: Dish) -> str:
    ''' return formatted output for dish '''
    return '{} (${}): {} cal'.format(dish.name, dish.price, dish.num_calories)

def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    ''' return true if name and calories are the same '''
    return dish1.name == dish2.name and dish1.num_calories == dish2.num_calories

def Dish_change_price(dish: Dish, price_change: float) -> Dish:

def Dishlist_display(dish_list: list) -> str:
    ''' return formatted string of dish list '''
    result = ''
    for i in range(len(dish_list)):
        result = result + Dish_str(dish_list[i])
        if i != len(dish_list)-1: # prevents newline after the last index
            result += '\n'
    return result

def Dishlist_average(dish_list: list) -> float:
    ''' returns the average price of all dishes in the dish list '''

Count = namedtuple("Count", "letter number")
def letter_count(word1: str, word2: str) -> [Count]:
    ''' return a list of Count that includes the letter searched for and the frequency
        that it occurs'''
    result = []
    for i in range(len(word2)):
        result.append(Count(word2[i], word1.count(word2[i])))
    return result
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5),
                                                           Count(letter='b',number=3),
                                                           Count(letter='c',number=1),
                                                           Count(letter='d',number=0)]
