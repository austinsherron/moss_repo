Count = namedtuple('Count', 'letter num')
def small_count(c:str, n: str)-> Count:
    '''Checks the string in this index'''
    counter = 0
    for i in range(len(c)):
        if c[i] == n:
            counter += 1
    C = Count(n,counter)
    return C
        
print(small_count('hello','l'))
    
def letter_count(c: str, n: str) -> list:
    '''Returns the amount of times string n is found in string c'''
    Clist = []
    for i in range(len(n)):
        Clist.append(small_count(c, n[i]))
    return Clist

print(letter_count('hello','he'))
