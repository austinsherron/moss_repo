from collections import namedtuple

Dish = namedtuple('Dish', 'name price calories')

d1 = Dish('Pepperoni Pizza', 2.00, 550)
d2 = Dish('Vanilla Ice Cream', 1.50, 300)

def Dish_str(d: Dish) -> str:
    '''Take a dish type and return it\'s datafields as a string'''
    return d.name + ' ($' + str(d.price) + '): ' + str(d.calories) + ' cal'

assert Dish_str(d1) == 'Pepperoni Pizza ($2.0): 550 cal'
assert Dish_str(d3) == 'Side Salad ($3.0): 120 cal'


def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    '''Compare two dishes' names and calorie counts'''
    return (dish1.name == dish2.name and dish1.calories == dish2.calories)

assert Dish_same(d1, d1)
assert Dish_same(d3, d4)
assert not Dish_same(d2, d4)

def Dish_change_price(d: Dish, change: int) -> Dish:
    '''Change a dish's price with the given percentage'''
    new_dish = d._replace(price = d.price + d.price*(change/100))
    return new_dish

assert Dish_change_price(d1, 100) == Dish('Pepperoni Pizza', 4.00, 550)

def Dish_is_cheap(d: Dish, threshold: float) -> bool:
    '''Checks if a dish is cheaper than a given price point'''
    return d.price < threshold

assert Dish_is_cheap(d1, 3.00)
assert not Dish_is_cheap(d3, 2.00)

Count = namedtuple('Count', 'letter number')

def letter_count(M: str, L: str) -> list:
    '''Return Count objects that show the number of each letter
       provided in the given string'''
    result = []
    for i in L:
        currcnt = 0
        for j in M:
            if j == i:
                currcnt += 1
        result.append(Count(i, currcnt))
    return result

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
assert letter_count('The quick brown fox jumps over the lazy dog', 'aeiouy') == [
    Count(letter='a',number=1),
    Count(letter='e',number=3),
    Count(letter='i',number=1),
    Count(letter='o',number=4),
    Count(letter='u', number=2),
    Count(letter='y', number=1)]
assert letter_count('Pizza is awesome!', 'zy') == [Count(letter='z', number = 2), Count(letter='y', number = 0)]
