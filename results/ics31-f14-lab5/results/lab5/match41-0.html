Count = namedtuple('Count', 'letter number')

def letter_count(message: str, letters: str) -> object:
    ''' Return a Count namedtuple with the first field as the letter and the
    second field as the letter count '''
    new_list = []
    for i in range(len(letters)):
        new_list.append(Count(letters[i],message.count(letters[i])))
    return new_list
assert letter_count('This is a test', 'aif ') == [Count(letter='a', number=1),
                                                 Count(letter='i', number=2),
                                                 Count(letter='f', number=0),
                                                 Count(letter=' ', number=3)]
assert letter_count('asdf', '') == []
