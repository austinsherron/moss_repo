def Dish_str(D: Dish) -> str:
    '''Returns Dish items'''
    name = D.name
    price = str(D.price)
    calories = str(D.calories)

    return str_1

def Dish_change_price(d1: Dish, num: float) -> Dish:
    '''Returns a Dish that's the same as the parameter except that its price is changed as follows: The number
    (positive or negative) represents a percentage change in price (so that 100 would double the price
    and â€“50 would cut it in half)'''

    d2 = (d1.price * (num / 100) + d1.price)

    d3 = d1._replace(price = d2)



    return d3

def Dishlist_prices(dish_list: [Dish]) -> list:
    '''Returns a list of numbers containing just the prices of the dishes on that list'''

    prices = []
    for i in dish_list:
        prices += [i.price]

    return prices

def Dishlist_average(dish_list: list) -> float:
    '''Returns the average price of those dishes'''

    prices = Dishlist_prices(dish_list)
    total = 0
    j = 0
    for i in prices:
        total += i
        j += 1
        
    return total / j

Count = namedtuple('Count', 'letter number')


def letter_count(string1, string2) -> [Count]:
    '''Returns the counts of certain letters in the first parameter;
    the second parameter is a string that specifies which letters to count'''
    count_list = []
    for i in string2:
        num = 0
        for j in string1:
            if i == j:
                num += 1
        count_list.append(Count(i, num))

    return count_list

assert letter_count("hello how are you", "ao") == [Count(letter='a', number=1), Count(letter='o', number=3)]
