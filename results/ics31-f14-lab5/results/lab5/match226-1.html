from collections import namedtuple

Dish = namedtuple('Dish', 'name price calories')
d1 = Dish('spaghetti', 9.99, 500)
d2 = Dish('lasagna', 14.99, 700)

def Dish_str(n: Dish) -> str:
    '''Return the values contained in the parameter in the following format:
       NAME ($PRICE): (CALORIES) cal'''
    return '{0} (${1}): {2} cal'.format(n.name, n. price, n. calories)
assert Dish_str(d1) == 'spaghetti ($9.99): 500 cal'
assert Dish_str(d2) == 'lasagna ($14.99): 700 cal'
assert Dish_str(d3) == 'pizza ($6.99): 600 cal'

def Dish_same(n1: Dish, n2: Dish) -> bool:
    '''Returns the evaluation of a comparison between dishes using their name
       and their number of calories'''
    if n1.name == n2.name and n1.calories == n2.calories:
        return True
    else:
        return False


assert Dish_same(d1, d1) == True
assert Dish_same(d1, d2) == False
assert Dish_same(d2, Dish('lasagna', 12.99, 700)) == True

def Dish_change_price(n: Dish, x: float) -> Dish:
    '''Returning a new namedtuple with a different price, specified by a change
       due to an inputted value'''
    return n._replace(price=n.price + n.price*(x/100))

                                          =500)
assert Dish_change_price(d2, -50) == Dish(name='lasagna', price=7.495, calories
                                          =700)

def Dish_is_cheap(n: Dish, x: float) -> bool:

def single_letter_count(string: str, letter: str) -> Count:
    ''' Returns the number of times a specific letter is present in another string'''
    return Count(letter, string.count(letter))
    
assert single_letter_count('apple', 'p') == Count(letter='p', number=2)




def letter_count(string1: str, string2: str) -> 'list of Counts':
    ''' Returns the number of times each character is present in a string '''
    result = []
    for i in string2:
        result.append(single_letter_count(string1, i))
    return result

assert letter_count('banana', 'an') == [Count(letter='a', number=3), Count(letter='n', number=2)]
assert letter_count('dinner', 'ne') == [Count(letter='n', number=2), Count(letter='e', number=1)]
