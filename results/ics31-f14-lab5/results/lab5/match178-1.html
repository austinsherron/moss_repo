def Dish_raise_price(d: list) -> list:
    '''takes a dish and raises the price by 2.5'''
    new_price = d.price + 2.5
    new_dish = d._replace(price = new_price)
    return new_dish
assert Dish_raise_price(r2.menu[0]) == Dish(name='Homard Bleu', price=47.5, calories=750)

def Menu_raise_price(m: list) -> list:
    '''takes a menu and raises the price of dishes prices by 2.5'''
    for x in range(len(m)):
        m[x] = Dish_raise_price(m[x])
    return m

assert Menu_raise_price(r4.menu) == [Dish(name='Homard Bleu', price=47.5, calories=750), Dish(name='Tournedos Rossini', price=67.5, calories=950), Dish(name="Selle d'Agneau", price=62.5, calories=850)]

def Restaurant_raise_prices(r: list) -> list:
    """takes a restaurant and raises price of dishes by 2.5"""
    new_menu = Menu_raise_price(r.menu)
    new_rest = r._replace(menu = new_menu)
    return new_rest


assert Restaurant_raise_prices(r5) == Restaurant(name='Taillevent', cuisine='French', phone='01-44-95-15-01', menu=[Dish(name='Homard Bleu', price=47.5, calories=750), Dish(name='Tournedos Rossini', price=67.5, calories=950), Dish(name="Selle d'Agneau", price=62.5, calories=850)])
