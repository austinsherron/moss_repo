Count=namedtuple('Count','letter number')
def letter_count(s1, s2):


    '''Takes two parameters and returns the counts of certain letters
in the first parameter; the second parameter is a string that specifies
which letters to count.'''
    found_letters = []
    search_letters = [] 
    for letter in s2:
        search_letters.append(letter)
    for s_letter in search_letters:
        found_letters.append(Count(s_letter, s1.count(s_letter)))
    return found_letters
