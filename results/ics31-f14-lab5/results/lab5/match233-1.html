def Dishlist_average(dishlist:list)->float:
    ''' returns the average price of the dishes in the dish list '''
    total = 0
    count = 0
    for x in dishlist:
        total = total + x.price
        count = count + 1
    return (total/count)

Count = namedtuple('count','character number')
def letter_count(str1:str,str2:str) -> list:
    ''' returns a list of count namedtuples counting the number of characters in str1 based on the characters in str2 '''
    countlist = []
    counter = 0
    for x in str2:
        counter = 0
        for y in str1:
            if(x==y):
                counter+=1
        countlist.append(Count(x,counter))
    return countlist

print(letter_count('the quick brown fox','tqicn'))
print(letter_count('aaa bbb ccc ddd','abcd'))
