Count = namedtuple('Count','letter number')


def letter_count(phrase:str, letters:str)->'list of counts':
        '''return a list of namedtuples that counts the number of times a letter appears in a phrase'''
        CL=[]
        for one in letters:
                counter=0
                for two in phrase:
                        if two == one:
                                counter += 1

                CL.append(Count(one,counter))
        return CL


assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
