def Dish_raise_price(D: Dish) -> Dish:
    '''Returns the Dish with the price raised by $2.50'''
    d = D._replace(price = D.price + 2.50)
    return d

def Menu_raise_prices(M: list) -> list:
    '''Returns the menu with the price of every Dish raised by $2.50'''
    m = []
    for d in M:
        m.append(Dish_raise_price(d))
    return m

def Restaurant_raise_prices(R: Restaurant) -> Restaurant:
    '''Returns the Restaurant with the price of every Dish in its menu
    raised by $2.50'''
    m = R.menu
    r = R._replace(menu = Menu_raise_prices(m))
    return r

def Collection_raise_prices(C: list) -> list:
    '''Returns the Collection with the price of every Dish at every restaurant
    raised by $2.50'''
    result = []
    for r in C:
        result.append(Restaurant_raise_prices(r))
    return result

Count = namedtuple('Count', 'letter number')
def letter_count(s1:str,s2:str) ->'list of Count':
    counter=[] 
    for letter in s2:
        counting=Count(letter,0)
        letscount=0
        for letter2 in s1:
            if letter == letter2:
                letscount+=1
        c = counting._replace(number=counting.number+letscount)
        counter.append(c)               
    return counter
