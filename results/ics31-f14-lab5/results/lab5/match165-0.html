Count = namedtuple("Count", "letter number")

def letter_count(s1: str, letters: str) -> list:
    ''' returns a list of Count representing letters in s1'''
    count_list=[]
    for i in letters:
        counter = 0
        for j in s1:
            if i== j:
                counter += 1
        count_list.append(Count(i, counter))
    return count_list
assert (letter_count('The cabbage has baggage', 'abcd') ==
        [Count(letter='a',number=5),
         Count(letter='b',number=3),
         Count(letter='c',number=1),
         Count(letter='d',number=0)])
