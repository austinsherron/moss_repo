def Dish_str(n:namedtuple)->str:
    return n.name + '($' + str(n.price) + '):' + str(n.calories) + 'cal'



def Dish_same(n:namedtuple, b:namedtuple)->bool:
    return n.name == b.name and n.calories == b.calories

def Dishlist_change_price(n:list, b:float)->list:
    for i in range(len(n)):
        n[i] = Dish_change_price(n[i], b)
    return n

def Dishlist_prices(n:list)->list:
    result = []
    for i in n:
        result.append(i.price)
    return result

def Dishlist_average(n:list)-> float:
    return sum(Dishlist_prices(n))/len(n)

def Dishlist_keep_cheap(n:list, b:float)->list:
    result = []
    for i in n:
        if i.price<b:

def letter_count(n:str, b:str)->list:
    result= []
    Count = namedtuple('Count', 'letter number')
    for i in range(len(b)):
        n.count(b[i])
        result.append(Count(b[i], n.count(b[i])))
    return result
