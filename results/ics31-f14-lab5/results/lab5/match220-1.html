def Dish_str(dish: Dish) -> str:
    '''Takes a Dish and returns a string in the given format'''
    return (dish.name+' ($ '+str(dish.price)+' ) : '+str(dish.calories)+' cal')

def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    '''Takes two dishes as arguments and returns True if their names and
    calories are same and False otherwise'''
    return dish1.name == dish2.name and dish1.calories == dish2.calories

assert Dish_same(d1, d2) == False
assert Dish_same(d3, d1) == False
assert Dish_same(d1, d4) == True

def Dish_change_price(dish: Dish, num: float) -> Dish:
    '''Takes a dish and a number and reflects the percent change in the price
    of the dish'''
    return dish._replace(price = dish.price + (dish.price * (num/100)))

                                          price=8.0, calories=550)

def Dishlist_all_cheap(dishes: [Dish], num: float) -> bool:
    '''Takes a list of dishes and a number and returns True if the price of
    every dish on the list is less than the number and False otherwise'''
    check = 1

def Dishlist_average(dishes: [Dish]) -> float:
    '''Takes a list of Dishes and returns the average price of those Dishes'''
    total = 0
    for i in dishes:
        total += i.price
    return total / len(dishes)

Count = namedtuple('Count', 'letter number')

def letter_count(string: str, letters: str) -> Count:
    ''' Function returns the counts of certain letters in the first parameter,
    the second parameter is a list that specifies what letters to count'''
    result = []
    for i in letters:
        result.append(Count(i, string.count(i)))
    return result
