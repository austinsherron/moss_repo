Count = namedtuple('Count', 'letter number')

def letter_count(string: str, contains: str) -> list:
    "Takes in a string and compared string.  Calculates how many of each character are in the string"
    "Returns the number of each compared character"
    counter = 0
    result = []
    for x in range(0, len(contains)):
        for y in range(0, len(string)):
            if string[y] == contains[x]:
                counter += 1
                named_tuple = Count(string[y], counter)
            named_tuple = named_tuple._replace(number = counter)
        counter = 0
        named_tuple = named_tuple._replace(letter = contains[x])
        result.append(named_tuple)
    return result
