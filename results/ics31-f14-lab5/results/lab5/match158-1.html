Count = namedtuple('Count', 'letter number')

def count_numbers(s:str,letter:str)->int:


    n=0
    for a in range(0,len(s)):
        if s[a:a+1]==letter:
            n=n+1
    return n

def letter_needs(s:str,position:int)->str:
    return s[position:position+1]

def letter_count(s1:str,s2:str)->list:


    CL=[]
    for a in range(0,len(s2)):
        letter=letter_needs(s2,a)
