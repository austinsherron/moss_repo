def Dish_str(dish: Dish) -> str:
    ''' Return a string representing the dish. '''
    return dish.name + ' (${:2.2f}'.format(dish.price) + '): ' + str(dish.calories) + ' cal'

Count = namedtuple('Count', 'letter number')

def count(s1: str, s2: str) -> Count:
    ''' Return a Count namedtuple with the character and the number of times it occurs in the first string. '''
    num = 0
    for char in s1:
        if char == s2:
            num += 1
    return Count(s2, num)
assert count('Apple', 'p') == Count(letter='p',number=2)
assert count('BANANA', 'A') == Count(letter='A',number=3)
assert count('orange', 'o') == Count(letter='o',number=1)
assert count('drums', 'v') == Count(letter='v',number=0)

def letter_count(s1: str, s2: str) -> list:
    ''' Return the counts of certain letters in the first parameter. '''
    result = []
    for char in s2:
        result.append(count(s1, char))
    return result
assert letter_count('The cabbage has baggage', 'abcd') == (
[Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)])
assert letter_count('Hello \n \t \n Goodbye', ' \t\n') == [Count(letter=' ',number=4), Count(letter='\t',number=1), Count(letter='\n',number=2)]
