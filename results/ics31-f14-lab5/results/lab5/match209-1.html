def Dish_str(Dish: 'Dish') -> str:
    '''Returns a str in this form "Paht Woon Sen ($9.50): 330 cal"'''

def Dish_change_price(dish: Dish, n: float) -> Dish:
    '''Returns Dish with price increased/decreased by n percentage'''
    D = dish._replace(price = ((n/100) * dish.price) + dish.price)

Count = namedtuple('Count', 'letter number')

def count_str(word: str, l: str) -> list:
    '''Takes a string and a single character and returns a single
    Count namedtuple'''
    L = (Count(letter=l, number=word.count(l)))
    return L
assert count_str('apple', 'p') ==  Count(letter='p', number=2)

def letter_count(phrase: str, counted: str) -> list:
    '''Return counts of certain letters 1st parameter, 2nd parameter
    specifies which letters to count'''
    L = []
    for x in range(len(counted)):
        L.append(count_str(phrase, counted[x]))
    return L
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
