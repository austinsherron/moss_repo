def Collection_raise_prices(r:collection)->collection:
    result=[]
    for d in r:
        for g in d.menu:
            f=[]
            p = g.price+2.50
            new_price=g._replace(price=p)
            f.extend(new_price)
            new_menu=d._replace(menu=f)
            result.extend(new_menu)
    return result
