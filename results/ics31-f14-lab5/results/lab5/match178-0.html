def Dish_raise_prices(b: Dish) -> Dish:
    '''Raises the price of each dish by $2.50, and returns this new
       value.'''
    updatedDish = b._replace(price = float(b.price + 2.50))
    return updatedDish
            
def Menu_raise_prices(c: list) -> list:
    '''Raises the price of all the dishes in the list by $2.50 and
       returns the list'''
    for i in range(0, len(c)):
        c[i] = Dish_raise_prices(c[i])
    return c

def Restaurant_raise_prices(R: Restaurant) -> Restaurant:
    '''Raises the price of all the dishes in the given restaurant by $2.50
       and returns the restaurant with altered menu prices'''
    R = R._replace(menu = Menu_raise_prices(R.menu))       
    return R

Count = namedtuple('Count', 'letter number')
def single_letter_count(message:str, letter: str) -> Count:
    '''The function returns the count of a single letter
       in the first parameter; the second parameter is a
       string that specifies which letter to count.'''
    c1 = Count(letter, message.count(letter))
    return c1
def letter_count(message:str, letters: str)-> list:
    '''The function returns the counts of certain letters
       in the first parameter; the second parameter is a
       string that specifies which letters to count.'''
    CL = []
    for i in letters:
        CL.append(single_letter_count(message, i))
    return CL

print(letter_count('I like dogs', 'sd'))
