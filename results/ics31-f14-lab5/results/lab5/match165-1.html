Count = namedtuple('Count', 'letter number')
def letter_count(message: str, thingytocount:str)->list:
    '''Counts number of letters in a string'''
    total = []
    for a in thingytocount:
        c = 0
        for b in message:
            if a == b:
                c = c+1
        total.append(Count(a,c))
    return total
print('Testing function letter_count:')
