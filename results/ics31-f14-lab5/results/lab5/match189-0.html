def letter_count(w: str, y: str) -> list:


    '''returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to count
    '''
    Count_n = namedtuple('Count_n', 'letter number')
    Count_list = []
    for i in y:
        if (i in w):
            Count_list.append(Count_n(i, w.count(str(i))))
    return Count_list
