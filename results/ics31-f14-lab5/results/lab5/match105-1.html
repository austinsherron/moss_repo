Count = namedtuple('Count', 'letter number')

def count_one_character(string:str, character:str) -> Count:
    '''reutrns a Count namedtuple with the number of times a character occurs in the string'''
    from collections import namedtuple
    Count = namedtuple('Count', 'letter number')
    y = string.count(character)
    p = Count(letter=str(character), number=int(y))
    return p

print(count_one_character('The cabbage has baggage', 'g'))

    
def character_count(string:str, letters:str) -> [Count]:
    '''returns a list of Count namedtuples'''
    result = []
    for x in letters:
        y = 0
        for n in string:
            if x in n:
                y+=1
        result.append(count_one_character(string, x))
    return result 
            

print(character_count('The cabbage has baggage', 'abcd'))
