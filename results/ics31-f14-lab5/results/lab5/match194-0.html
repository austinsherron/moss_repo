def Dishlist_display(list_dishes:list)->str:
    "take a list of dishes, returns a string representation of each dish follwed by a newline"

def Dish_raise_price(dish):
    "takes and raises price by 2.50"
    new_dish = dish._replace(price = dish.price + 2.50)
    return new_dish

def Menu_raise_prices(menu):
    "takes a menu and returns the menu with the prices raised by 2.50"
    new_menu = []
    for dish in menu:
        new_dish = Dish_raise_price(dish)
        new_menu.append(new_dish)
    return new_menu

def Restaurant_raise_prices(Restaurant)->Restaurant:
    "takes a restaurant and returns the restaurant with the prices in menu raised"
    Restaurant = Restaurant._replace(menu = Menu_raise_prices(Restaurant.menu))

Count = namedtuple('Count','letter number')

def letter_count(a:str,b:str)->list:
    "function takes two strings and returns a list of count named tuples of second string in first string"
    countlist = []
    for letter in b:
        countlist.append(Count(str(letter),(a.count(letter))))
    return countlist
print(letter_count('bannana','an'))
