def Dishlist_all_cheap(L:list,p:float)->bool:
    '''Takes a list of Dishes and a number and returns True if the price of every dish on the list is less than that number

    '''Takes a list of Dishes and a number representing a percentage change and returns a list of Dishes with
    each price changed by the specified amount
    '''

def Dishlist_average(L:list)->list:
    '''Takes a list of Dishes and returns the average price of those dishes
    '''

    p=Dishlist_prices(L)

Count=namedtuple('Count','letter number')


def letter_count(some_message:str,letters:str)->list:
    ''' This function returns the counts of certain letters in the first parameter; the second parameter is a string
    that specifies which letters to count
    '''
    new_list=[]
    for i in range(len(letters)):
        new_list.append(Count(letter=letters[i],number=some_message.count(letters[i])))
    return new_list
