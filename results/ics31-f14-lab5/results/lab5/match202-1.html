def Dish_change_price(d: Dish, num: int) -> Dish:
    """ Takes a Dish and a int, returns the Dish with a change in
    price based on the int(taken as a percentage)

def Menu_raise_prices(m: Restaurant.menu) -> Restaurant.menu:
    """ Raises price of all dishes on menu by 2.50

Count = namedtuple('Count', 'letter number')

def count_one_char(s1: str, s2: str) -> int:
    return Count(s2, s1.count(s2))
assert count_one_char('hi', 'h') == Count(letter='h', number=1)
assert count_one_char('hello world', ' ') == Count(letter=' ', number=1)

def letter_count(s1: str, s2: str) -> int:


    result = []
    for i in s2:
        result.append(count_one_char(s1, i))
    return result
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5),
                                                            Count(letter='b', number=3),
                                                            Count(letter='c', number=1),
                                                            Count(letter='d', number=0)]
