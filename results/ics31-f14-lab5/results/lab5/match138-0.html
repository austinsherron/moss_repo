def Dish_change_price(d: Dish, percent: float)-> Dish:
    '''takes a Dish and a float as a percentage and if the percentage is positive it
       increases the price of the dish by that percent and if negatice it decreases
       the price by the percentage'''
    if(percent > 0):
        newDish = Dish(d.name, d.price + d.price * percent / 100, d.calories)
    elif(percent < 0):
        newDish = Dish(d.name, d.price - d.price * (percent * (-1)) / 100, d.calories)
    return newDish

assert Dish_change_price(d1, 100) == Dish('Pizza', 11.98, 800)
assert (Dish_change_price(d2, -50)).price == 1.495

def letter_count(s1: str, s2: str) -> list:
    '''Takes 2 strings, counts the characters in s2 that is in s1 and returns a
       namedtuple that contains the fields of a letter and a number'''
    CountedLetters = namedtuple('CountedLetters', 'letter number')
    countedLettersList = []
    for i in s2:
        count = 0
        for j in s1:
            if i in j:
                count += 1
        countedLettersList.append(CountedLetters(i, count))
    return countedLettersList
