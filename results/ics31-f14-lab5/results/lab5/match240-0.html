def Price_average(r:Restaurant)-> float:
      avg = 0
      for d in r.menu:
            avg += d.price
      avg = round(avg/len(r.menu),2)
      return avg


def Collection_select_cheap(collection:list,num:float)->list:
      '''returns a list of all the Restaurants in the collection whose average price is less than or equal to that number'''
      result=[]
      for r in collection:
            check = Price_average(r)
            if (check <= num):
                  result.append(r)
      return result

Count = namedtuple('Count', 'letter number')
def letter_count(s1: str, s2: str) -> list:
      '''The function returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to coun'''
      result = []
      for c in s2:
            result.append(Count(c, 0))
      for c in s1:
            for i in range(len(result)):
                  if c == result[i].letter:
                        result[i] = result[i]._replace(number = result[i].number + 1)
                        break
      return result
