from collections import namedtuple
Dish = namedtuple('Dish', 'name price calories')



d1 = Dish('Burger', 5, 500)

def Dish_str(meal: Dish)->str:
    '''Returns dish as a str'''
    return meal.name + '(' + str(meal.price) + ')' + ':' + str(meal.calories) + 'cal'

assert Dish_str(d1) == 'Burger(5):500cal'

def Dish_same(m1: Dish, m2: Dish)->bool:
    '''if the name and calories are the the same, return true'''
    result = ''
    if (m1.name == m2.name and m1.calories == m2.calories):
        result = True
    elif (m1.name != m2.name and m1.calories != m2.calories):
        result = False
    return result
assert Dish_same(d3, d4) == True
assert Dish_same(d1, d2) == True
assert Dish_same(d1, d3) == False

def Restaurant_is_cheap(a: Restaurant, n: int)->bool:
    '''See if the restaurant is cheap based on input'''

    result = ''

Count = namedtuple('Count', 'letter number')
def count(phrase: str, letter: str)->namedtuple:
    timesOccurred = phrase.count(letter)
    return Count(letter=letter, number=timesOccurred)
assert count("I'm a little teapot, short and stout", 'l') == Count(letter='l', number=2)

def letter_count(phrase: str, letters: str)->list:
    counted = []
    for x in letters:
        counted.append(count(phrase, x))
    return counted
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b', number=3), Count(letter='c', number=1), Count(letter='d', number=0)]
