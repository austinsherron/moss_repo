Count = namedtuple('Count', 'letter number')


def count_number(string: str, character: str) -> int:
    '''Returns amount of times character appears in string'''
    appear = 0
    for l in string:
        if character == l:
            appear += 1
    return appear
    
def letter_count(x: str, y: str) -> list:
    '''Returns count of each specified letter appearing in first string'''
    count_list = []
    for l in y:
        count_list.append(Count(l, count_number(x, l)))
    return count_list

assert letter_count('The cabbage has baggage', 'ab') == [Count(letter='a', number=5), Count(letter='b', number=3)]
