Count = namedtuple('Count', 'letter number')

def letter_count(s:str, w:str)->Count:
    """Count the amount of times the letters in the second string appear in
       the first string
    """
    t = []
    l = []
    for i in w:
        t.append(i)
    for i in range(len(t)):
        l.append(Count(t[i], s.count(t[i])))
    return l
assert letter_count('anthony', 'ant') == [Count(letter='a',number=1),
                                          Count(letter='n',number=2),
                                          Count(letter='t',number=1)]
print(letter_count('anthony', 'ant'))
