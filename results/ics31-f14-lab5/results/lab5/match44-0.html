def Restaurant_is_cheap(rest: Restaurant, avg_max = float) -> bool:
    ''' Returns boolean statement whether or not the average price of
        Restaurant menu is less than avg_max '''
    sum_prices = 0
    for d in rest.menu:
        sum_prices += d.price
    return (sum_prices / len(rest.menu)) <= avg_max

Count = namedtuple('Count', 'letter number')

def count_one_character(astring: str, letter: str) -> int:
    ''' Returns the number of letters in astring '''
    return astring.count(letter)


assert count_one_character('hello', 'l') == 2

def letter_count (astring: str, bstring: str) -> [Count]:
    ''' Returns a list of Count for each letter of bstring in astring'''
    count_list = []
    for b in bstring:
        count_tuple = Count('', '')
        count_tuple = count_tuple._replace(letter = (b)) 
        count_tuple = count_tuple._replace(number = count_one_character(astring, b))
        count_list.append(count_tuple)       
    return count_list
assert letter_count("hello \t  ", "h\tl") == [Count(letter='h', number=1),
                                              Count(letter='\t', number=1),
                                              Count(letter='l', number=2)]
assert letter_count('kjshdf/.,/.,/./.,/.,/', '/,k') == [Count(letter='/', number=6),
                                                        Count(letter=',', number=4),
                                                        Count(letter='k', number=1)]
