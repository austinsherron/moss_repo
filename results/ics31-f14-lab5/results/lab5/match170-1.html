def Dishlist_average(dishList: list) -> float:
    '''Takes a list of dishes and returns the average of all the prices of the dishes'''

    for i in dishList:

Count = namedtuple('Count', 'letter number')

def letter_count(s1: str, s2: str) -> list:
    ''' Returns a list of Count namedtuples using the letters in s2 to count from
        s1 '''
    newList = []
    for i in s2:
        count = 0
        for n in s1:
            if i == n:
                count += 1
        newList.append(Count(i, count))
    return newList
assert letter_count('i love apples', 'aeiou') == [Count(letter='a', number=1),
                                                  Count(letter='e', number=2),
                                                  Count(letter='i', number=1),
                                                  Count(letter='o', number=1),
                                                  Count(letter='u', number=0)]
assert letter_count(''' hello
how are you ''', ' \t\n') == [Count(letter=' ', number=4),
                               Count(letter='\t', number=0),
                               Count(letter='\n', number=1)]
print(letter_count('i love apples', 'aeiou'))
print(letter_count(''' hello
how are you ''', ' \t\n'))
