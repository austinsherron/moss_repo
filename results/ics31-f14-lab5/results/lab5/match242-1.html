def Dish_str (D: Dish) -> str:
    '''takes a dish D as input, and returns a string in the format

def Dish_change_price(D: Dish, change: float) -> Dish:
    ''' returns a Dish that has the same properties as the input Dish,
    but with its price altered according to the input number change.
    If change is positive, the price is increased by that percentage.
    If change is negative, the price is decreased by that percentage'''
    newprice = D.price + (D.price * (change * (10 **(-2)) ))
    newdish = D._replace( price = newprice )
    return newdish

assert Dish_change_price( d1, 200 ).price == 8.97

def Dishlist_change_price( Dishlist: list, change: float) -> list:
    ''' returns a list similar to the input list, but with each dish's price
    changed according to the input number. If the input number is positive,
    the price is increased by that percentage. If the input price is negative,
    the price is decreased by that percentage'''
    result = Dishlist[:]
    for i in range (len(result)):
        result[i] = result[i]._replace( price = Dish_change_price( result[i], change).price )

Count = namedtuple('Count', 'letter number')

def letter_count(message: str, to_count: str) -> list:
    ''' Takes two strings, message and to_count, then takes each letter in
    to_count and checks how many times it appears in message. A Count
    namedtuple is created to store the data for each letter in to_count,
    and this function returns a list of the Counts created.'''
    result = []
    for i in range(len(to_count)):
        item_count = Count(to_count[i], message.count(to_count[i]))
        result.append(item_count)
    return result

assert letter_count('Mississippi', 'is') == [Count(letter='i', number=4), Count(letter='s', number=4)]
assert letter_count('America', 'a') == [Count(letter='a', number=1)]
assert letter_count('Ungulgate', 'uU') == [Count(letter='u', number=1), Count(letter='U', number=1)]
