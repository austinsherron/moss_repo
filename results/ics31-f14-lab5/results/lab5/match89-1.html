from collections import namedtuple
Count = namedtuple('Count', 'letter number')


def letter_count(message:str, letters:str) -> 'list of Count namedtuples':
    "Returns the counts of certain letters in the first parameter as a list of Count namedtuples"
    counts_list = []
    letters = letters.strip()


    for i in range(len(letters)):
        num = message.count(letters[i])
        c = Count(letters[i], num)
        counts_list.append(c)
    return counts_list

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b', number=3), Count(letter='c', number=1), Count(letter='d', number=0)]
assert letter_count('The cabbage has baggage', 'aeiou') == [Count(letter='a', number=5), Count(letter='e', number=3), Count(letter='i', number=0), Count(letter='o', number=0), Count(letter='u', number=0)]
assert letter_count('The cabbage has baggage', '\t\n') == []

print('The cabbage has baggage,', 'abcd:', letter_count('The cabbage has baggage', 'abcd'))
print('The cabbage has baggage,', 'abcd:', letter_count('The cabbage has baggage', 'aeiou'))
print('The cabbage has baggage,', 'abcd:', letter_count('The cabbage has baggage', ' \t\n'))
