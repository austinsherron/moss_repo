from collections import namedtuple
Dish = namedtuple('Dish', 'name price calories')
d1 = Dish('Mac&cheese', 7.00, 500)

def Dish_str(d:Dish)-> str:
    return str(d.name) + '($' + str(d.price) + '): ' + str(d.calories) + ' cal'

def Dish_same(d_1:Dish, d_2:Dish)-> bool:
    if d_1.calories == d_2.calories:
        if d_1.name == d_2.name:
            return True
    else:
        return False

assert Dish_same(d1, d2)
assert not Dish_same(d1, d3)


def Dish_change_price(d:Dish, n:int)-> Dish:
    return d._replace(price = d.price + (d.price*(n/100)))

from collections import namedtuple
Count = namedtuple('Count', 'letter number')

def letter_count(s1:str, s2:str)-> namedtuple:
    '''Return counts of the first string using the second string'''
    new_list = [] 
    for i in range(len(s2)):
        new_list.append(Count(s2[i], s1.count(s2[i])))
    return new_list    
             
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5),
                                                           Count(letter='b',number=3),
                                                           Count(letter='c',number=1),
                                                           Count(letter='d',number=0)]       

assert letter_count('Hello world', 'aeiou') == [Count(letter='a',number=0),
                                                Count(letter='e',number=1),
                                                Count(letter='i',number=0),
                                                Count(letter='o',number=2),
                                                Count(letter='u',number=0)]
