def count_one_character(s:str, s1:str) ->Count:
    'Return a count nametuple of times s1 occurs within s'
    return "Count(letter='" + s1 + "',number=" + str(s.lower().count(s1.lower()))  + ')'
    ####This will return a string will "" surround the entire string. 
        


def letter_count(s:str, s1:str) ->list:
    'Take in two strings where the first string acts as a string and the'
    'second string speicifies which strings to look for. Return a list'
    'of nametuples that will tell how many of each string from s2 is in s'
    lst =[]
    for x in range(len(s1)):
        a = count_one_character(s,s1[x])
        lst.append(a)
    return lst
assert letter_count('The cabbage has baggage','abcd') == ["Count(letter='a',number=5)","Count(letter='b',number=3)",
                                                          "Count(letter='c',number=1)", "Count(letter='d',number=0)"]
