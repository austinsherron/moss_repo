Count=namedtuple('Count','letter number')

def letter_count(s:str,c:str)->'the counts of certain letters':
    '''The function returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to count.
    Thus, letter_count(some_message, 'aeiou') would return counts of the vowels in the string some_message and letter_count(some_message, ' \t\n')
    would return counts of the white-space characters (space, tab, and newline). '''
    empty=[]
    for i in c:
        empty.append(Count(i,s.count(i)))
    return empty
assert letter_count('abcdefga','a')==[Count('a',2)]
assert letter_count('   ',' ')==[Count(' ',3)]
assert letter_count('?!hello','h?l')==[Count('h',1),Count('?',1),Count('l',2)]
assert letter_count('some_message','aeiou')==[Count(letter='a', number=1), Count(letter='e', number=3), Count(letter='i', number=0), Count(letter='o', number=1), Count(letter='u', number=0)]
assert letter_count('\n\n\n\n','\n') == [Count('\n',4)]
assert letter_count('\t\t\t\t\t','\t')==[Count('\t',5)]
assert letter_count('ABdggdajsbaEiOu','Aa')==[Count('A',1),Count('a',2)]
