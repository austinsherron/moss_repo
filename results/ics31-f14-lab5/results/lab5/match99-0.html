def Dish_str(d:Dish)->str:
    '''Takes a Dish and returns a string with it's name, price and calories'''
    result = ''
    result += d.name + ' ($' + str(d.price)+  '): ' +str(d.calories) + ' cal'
    return result

Count = namedtuple('Count', 'message result')
def character_count(a: str, b: str) -> namedtuple:
    '''returns a namedtuple with the origianl str and the amount of times the second string appears'''
    result = 0
    for c in a:
        if c == b:
            result+=1
    return Count(a,result)

def letter_count(a: str, b: str) -> list:
    '''returns a list of the amount of times each character in the second string appears in the first'''
    b.split()
    result = []
    for c in b:
        r = character_count(a,c)
        result.append(r.result)
    return result

print(letter_count('ajd;fj a', ' a;'))
