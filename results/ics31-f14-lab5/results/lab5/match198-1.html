Count = namedtuple("Count", 'letter number')

def letter_count(a: str, b: str) -> int:
    name_list = []

    for character1 in b:
        count_character = 0
        for character2 in a:
            if character1 == character2:
                count_character += 1
        name_list.append(Count(character1, count_character))
    return name_list

        
assert(letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b', number=3), Count(letter='c', number=1), Count(letter='d', number=0)])
assert(letter_count('I want pizza', 'az') == [Count(letter='a', number=2), Count(letter='z', number=2)])
