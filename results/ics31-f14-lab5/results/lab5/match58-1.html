def Dish_str(dish: Dish) -> str:
    '''Returns str representation'''
    Dishstr = ''
    Dishstr += dish.name + ' ($' + str(dish.price) + '): '
    Dishstr += str(dish.calories) + ' cal'
    return Dishstr
assert Dish_str(Dish1) == 'Cheeseburger ($2.5): 500 cal'
assert Dish_str(Dish2) == 'Pizza ($4.0): 450 cal'
assert Dish_str(Dish3) == 'French Fries ($1.0): 300 cal'

def Dish_same(dish1: Dish, dish2: Dish) ->  bool:

def Dish_change_price(dish: Dish, change: int) -> Dish:
    '''
    Takes a Dish and a number and returns a Dish changed in price.
    '''
    return dish._replace(price = dish.price * (1+(change/100)))

def Dishlist_average(dishlist: list) -> float: 
    '''  Takes a list of Dishes and returns the average price of those dishes.'''
    if len(dishlist) == 0:
        return 0.0
    prices = Dishlist_prices(dishlist)
    avg = sum(prices) / (len(prices))
    return avg
assert Dishlist_average(DL) == 28/9

Count = namedtuple("Count", "letter number")
def count_letter(haystack: str, needle: str) -> Count:
    '''Returns the counts of a letter in the first parameter'''
    return Count(needle, haystack.count(needle))

def letter_count(haystack: str, needle: str) -> list:
    '''Returns the counts of certain letters in the first parameter'''
    countlist =[]
    for checker in needle:
        countlist.append(count_letter(haystack,checker))
    return countlist
        
assert letter_count('some_message', 'aeiou') == [Count(letter='a', number=1), Count(letter='e', number=3), Count(letter='i', number=0), Count(letter='o', number=1), Count(letter='u', number=0)]
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
