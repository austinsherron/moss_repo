Count = namedtuple('Count', 'letter number')

def letter_single_count(to_be_counted: str, what_to_count: str) -> Count:
    '''Takes in two strings with the second string representing what to count in the first string.
    Returns a single Count namedtuple.
    '''
    return Count(what_to_count, to_be_counted.count(what_to_count[:]))

assert letter_single_count('hello', 'l') == Count('l', 2)

def letter_count(to_be_counted: str, what_to_count: str) -> list:
    '''Takes in two strings with the second string representing what to count in the first string.
    Returns a list of Count namedtuples.
    '''
    result = []
    for char in what_to_count:
        result.append(letter_single_count(to_be_counted, char))
    return result

assert letter_count('hello', 'elo') == [Count(letter='e', number=1), Count(letter='l', number=2), Count(letter='o', number=1)]
