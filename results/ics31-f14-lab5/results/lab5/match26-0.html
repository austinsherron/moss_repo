def Menu_raise_prices(L:str, n:float)->list:


    for price in range(len(L)):
       L[price] = L[price]._replace(price = L[price].price + n)
    return L

def Restaurant_raise_prices(L:str, n:float)->list:
    for menu in range(len(L)):
        return Menu_raise_prices(L.menu,n)

def Collection_raise_prices(L:list, n:float)->list:
    for r in range(len(L)):
        Restaurant_raise_prices(L[r],n)
    return L

Count = namedtuple('Count','letter number')

def letter_count(t:str, c:str)->list:
    result = []


    for q in c:
        x = 0
        for p in t:
            if q in p:
                x = x + 1
        result.append(Count(letter=str(q), number=int(x)))
    return result
