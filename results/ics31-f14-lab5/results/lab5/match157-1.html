Count = namedtuple('Count', 'letter number')

def letter_count(s1: str, s2: str) -> list:
    '''returns the counts of certain letters in the first parameter,
    the second letter specifies which letters to count'''
    l = []
    for a in s2:
        c = Count(a, s1.count(a))
        l.append(c)
    return l
