def Dish_change_price(Dish:Dish, n:int)->Dish:
    """takes a Dish and a number and returns a Dish that's the same


    as the parameter except that its price is changed as follows: The
    number (positive or negative) represents a percentage change in
    price (so that 100 would double the price and â€“50 would cut it in
    half)."""
    i = Dish.price+(Dish.price*(n/100))
    j = Dish._replace(price = i)
    return j

from collections import namedtuple
Count = namedtuple('Count', 'letter number')
def letter_count (s:str, t:str) -> list:
    """The function returns the counts of certain letters in the
    first parameter; the second parameter is a string that specifies
    which letters to count. Thus, letter_count(some_message, 'aeiou')
    would return counts of the vowels in the string some_message. The
    function should return its counts as a list of Count namedtuples,
    where a Count namedtuple has two fields, letter and number."""
    emp = []
    for i in t:
        if i in s:
            emp.append(Count(i,s.count(i)))
    return(emp)
assert letter_count('hello', 'leo') == [Count(letter='l', number=2), Count(letter='e', number=1), Count(letter='o', number=1)]
assert letter_count('fail safe', ' ilf') == [Count(letter=' ', number=1),
                                             Count(letter='i', number=1),
                                             Count(letter='l', number=1),
                                             Count(letter='f', number=2)]
