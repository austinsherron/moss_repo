Count = namedtuple( 'Count', 'letter number' )

def letter_count(first: str, second: str) -> list:
    #letter_count("whatever", "aweoialewdss")
    tempList = [];
    for x in second:
        c = 0
        for y in first:
            if (x.lower() == y.lower()):
                c = c + 1
        tempList.append(Count(x,c))
    return tempList
