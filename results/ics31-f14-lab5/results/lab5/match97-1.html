def letter_count(mes:str, sage:str) -> list:
    """Return a string that returns the counts of certain letters in the first parameter
    assert(letter_count('Nikki', 'ki')) == [Count(letter='k', number=2), Count(letter='i', number=2)]
    """


    from collections import namedtuple
    Count = namedtuple('Count', 'letter number')
    amount = []
    for f in sage:
        t = 0
        for g in mes:
            if f==g:
                t+=1
                amount.append(Count(f, t))
    return amount
print(letter_count('Water bottle', 'attl'))
