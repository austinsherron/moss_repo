def Dish_str(d:Dish)->str:
    '''Returns a string in this form: Paht Woon Sen ($9.50): 330 cal'''

def Dish_change_price(l:Dish, n:float)->Dish:
    '''Returns a Dish that's the same as the parameter except that it's price is changed by the percentage of the number.'''

Count = namedtuple('Count', 'letter number')
def letter_count(s1:str, s2:str)->Count:
    '''Returns the counts of certain letters in the first parameter.'''
    C = []
    for n in s2:
        count = 0
        for i in s1:
            if i == n:
                count = count + 1
        C.append(Count(n, count))
    return C
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
assert letter_count('There are vowels', 'er') == [Count(letter='e',number=4), Count(letter='r',number=2)]
print(letter_count('The cabbage has baggage', 'abcd'))
