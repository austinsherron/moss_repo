def Dish_str (a: Dish) -> str:
    ''' Returns Dish name, price, and calories '''
    x = (a.name + " ($" + str(a.price) + "): " + str(a.calories) + " cal")


    return(x)

def Dish_same (D1: Dish, D2: Dish) -> bool:
    ''' Takes 2 dishes and returns True if names and calorie counts are equal'''
    return D1.name == D2.name and D1.calories == D2.calories

def Dish_change_price (D: Dish, n: int) -> Dish:
    ''' Takes Dish and changes price by percentage '''
    price_change = D._replace(price = D.price + (D.price * n/100))
    return price_change

def Dishlist_average (D: 'list') -> float:
    ''' takes list of dishes and returns average price of those dishes '''


    total = 0
    for dish in Dishlist_prices(D):
        total += dish
    average = total / len(Dishlist_prices(D))

Count = namedtuple('Count', 'letter number')
def letter_count (m: str, l: str) -> list:
    ''' Takes message and letters to count and returns the counts of \
        certain letters in the first parameter '''
    result = [ ]
    n = Count('', 0)
    for i in l:
        n = n._replace(letter = i, number = m.count(i))
        result.append(n)
    return result
assert letter_count('The cabbage has baggage', 'abcd') == [
    Count(letter='a', number=5),
    Count(letter='b', number=3),
    Count(letter='c', number=1),
    Count(letter='d', number=0)]
assert letter_count('i like food! Do you like food?', 'aeiou') == [
    Count(letter='a', number=0),
    Count(letter='e', number=2),
    Count(letter='i', number=3),
    Count(letter='o', number=6),
    Count(letter='u', number=1)]
