from collections import namedtuple
Count = namedtuple('count', 'letter number')
'''Define a Count namedtuple that takes in letters and numbers
and returns a single Count namedtuple with the character and number
of times it occurs in the string'''

def letter_count(str1: str, str2: str) -> list:
    '''Returns the counts of certain letters in the first parameter
    while the second parameter specifies which letters to count.'''
    result = []
    for x in str2:
        result.append(Count(x,0))

    for x in str1:
        for indx in range(len(result)):
            if x == result[indx].letter:
                result[indx] = result[indx]._replace(number = result[indx].number +1)
    return result


print(letter_count('The cabbage has baggage', 'abcd'))
assert letter_count('Paper', 'aeiou')== [Count(letter='a', number = 1), Count(letter ='e', number = 1), Count(letter='i', number = 0), Count(letter='o', number = 0), Count(letter='u', number = 0)]
assert letter_count('\nHello there', ' \t\n') == [Count(letter = ' ', number = 1), Count(letter = '\t', number = 0), Count(letter = "\n", number = 1)]
