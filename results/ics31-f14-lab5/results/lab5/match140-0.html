from collections import namedtuple
Count = namedtuple('Count', 'letter number')
def letter_count(w: str, x: str)-> list:
    """Counts amount of letters in second string that appear on first string"""
    p = []
    for l in x:
        number = Count(l, w.lower().count(l))
        p.append(number)
    return p
assert letter_count('cabbage', 'abc') == [Count(letter='a', number=2), Count(letter='b', number=2), Count(letter='c', number=1)]
                
print(letter_count('Chico State lifts on', 'cho'))
