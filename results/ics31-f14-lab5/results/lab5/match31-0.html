Count = namedtuple('Count', 'letter number')   #Define a namedtuple called count with letter and number

def letter_count (string1:str, string2:str)->list:
    '''Takes two strings and call count_character and return a list of Count '''
    result = []   # Declares an empty list
    for i in range(len(string2)):
        result.append(count_character(string1, string2[i]))
    return result

def count_character (string:str, letter:str)->Count:
    '''Takes a string and a letter to count the number of times appear and return Count'''
    result = 0
    for i in range(len(string)):
        if string[i] == letter:
            result = result + 1
    return Count(letter, result)

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
