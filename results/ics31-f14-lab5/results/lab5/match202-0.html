Dish = namedtuple('Dish', 'name price calories')
d1 = Dish('sushi', 8.00, 250)

def Dish_str(food: Dish) -> str:
    'takes a Dish object and returns string in form of name (price): calories'
    return food.name + ' ($' + str(food.price) + '): ' + str(food.calories) + ' cal'
assert Dish_str(d1) == 'sushi ($8.0): 250 cal'
assert Dish_str(d2) == 'burger ($3.0): 400 cal'

#(c.3)
def Dish_same(food1: Dish, food2: Dish) -> bool:
    '''takes two dishes as arguments and returns True if the name and calorie
    counts are the same'''
    return (food1.name == food2.name) and (food1.calories == food2.calories)
assert Dish_same(d1, d1) == True
assert Dish_same(d1, d2) == False
assert Dish_same(d1, d4) == True
assert Dish_same(d3, d4) == False

def Dish_change_price(food: Dish, change_in_price: float) -> Dish:
    '''takes a Dish and a number and returns a Dish that has its price changed
    by change_in_price%'''
    food = food._replace(price = food.price*(1.00 + change_in_price/100))
    return food

Count = namedtuple('Count', 'letter number')

def character_count(string: str, char: str) -> Count:
    ''' takes in a string and a single character. returns a single Count namedtuple
    with char and the number of times char occurs in string'''
    result = 0
    for i in string:
        if i == char:
            result += 1
    return Count(char, result)
assert character_count('hello, how are you?', 'o') == Count(letter='o', number=3)
assert character_count('o\n m\n g\n', '\n') == Count(letter='\n', number=3)

def letter_count(s1: str, s2: str) -> 'list of Count':
    '''takes in two strings. counts the number of letters in s1 specified by s2
    using Count. returns a list of Count namedtuples.'''


    result = []
    for i in s2:
        result.append(character_count(s1, i))
    return result 
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5),
                                                           Count(letter='b',number=3),
                                                           Count(letter='c',number=1),
                                                           Count(letter='d',number=0)]
assert letter_count('The \t cabbAge has \n bAggage', '\t\nAa') == [Count(letter='\t',number=1),
                                                                    Count(letter='\n',number=1),
                                                                    Count(letter='A',number=2),
                                                                    Count(letter='a',number=3)]
