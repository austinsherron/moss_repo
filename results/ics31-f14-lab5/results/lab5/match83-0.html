def Dish_str(d: Dish) -> str:
    '''Returns a string in this form: Name of Dish ($Price): Calories'''
    return d.name + " ($" + str(d.price) + '): ' + str(d.calories) + " cals"

#c.3
def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    '''Returns True if the names of the two dishes and their calorie counts
    are equal (and False otherwise)'''
    return dish1.name == dish2.name and dish1.calories == dish2.calories
assert Dish_same(d1, d2) == False
assert Dish_same(d2, d4) == True
assert Dish_same(d1, d5) == False

Count = namedtuple("Count" , "letter number")

def letter_count(m: str, c: str)-> list:
    'returns list of Counts'
    result = []
    for i in c:
        result.append(Count(i,m.count(i)))
    return result
