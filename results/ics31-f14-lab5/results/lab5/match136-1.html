         of the menu is less than or equal to that number '''

Count = namedtuple('Count', 'letter number')
def letter_count(string:str, letters:str) -> list:
    ''' Takes a string and string of letters. Returns a list of Count with the
         letter and number of times the letter occurs in the string '''
    result = []
    for i in letters:
        result.append(count_character(string, i))
    return result

def count_character(string:str, character:str) -> Count:
    ''' takes a string and a single character and returns a single Count namedtuple
         with the character and the number of times it occurs in the string '''
    return Count(character, string.count(character))

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b', number=3), Count(letter='c', number=1), Count(letter='d', number=0)]
assert letter_count('The cabbage has baggage', ' \t\n') == [Count(letter=' ', number=3), Count(letter='\t', number=0), Count(letter='\n', number=0)]
print(letter_count('The cabbage has baggage', 'abcd'))
