Dish = namedtuple('Dish', 'name price calories')
d1 = Dish('Pasta', 9.00, 350)
d2 = Dish('Cheeseburger', 5.00, 400)
d3 = Dish('Pizza', 10.00, 600)

def Dish_str(x: Dish) -> str:
    '''Takes dish and returns string that displays its components'''
    y = x.name + ' ($' + str(x.price) + '): ' + str(x.calories) + ' cal'
    return y
assert Dish_str(d1) == 'Pasta ($9.0): 350 cal'
assert Dish_str(d2) == 'Cheeseburger ($5.0): 400 cal'
assert Dish_str(d3) == 'Pizza ($10.0): 600 cal'

def Dish_same(a: Dish, b: Dish) -> bool:
    '''Takes two dishes and returns a boolean of whether their names and
    calorie counts are equal'''
    return(a.name == b.name and a.calories == b.calories)
assert Dish_same(d1, d2) == False
assert Dish_same(d1, d1) == True
assert Dish_same(d1, Dish('Pasta', 9.00, 350)) == True

def Dish_change_price(x: Dish, i: int) -> Dish:
    '''Takes dish and a number and returns the same dish with the number
    percent change in price'''
    return Dish(x.name, x.price + (x.price * (i / 100)), x.calories)
assert Dish_change_price(d1, 100) == Dish(name='Pasta', price=18.0, calories=350)
assert Dish_change_price(d2, 50) == Dish(name='Cheeseburger', price=7.5, calories=400)
assert Dish_change_price(d3, -50) == Dish(name='Pizza', price=5.0, calories=600)

def Dish_is_cheap(a: Dish, x: int) -> bool:
    '''Takes dish and returns boolean of whether dish price is less than the
    number'''


    return a.price < x
assert Dish_is_cheap(d1, 10) == True
assert Dish_is_cheap(d2, 3) == False
assert Dish_is_cheap(d3, 10) == False

def Dishlist_display(x: [Dish]) -> str:
    '''Takes a list of dishes and returns a large string consisting of the
    string representation of each dish followed by a newline character'''
    result = ''
    for i in x:
        result += (str(i) + '\n')
    return result

def Dishlist_all_cheap(i: [Dish], a: int) -> bool:
    '''Takes a list of dishes and a number and returns True if the price of
    every dish in the list is cheaper than the number'''
    for k in i:
        if not Dish_is_cheap(k, a):
            return False
    return True
assert Dishlist_all_cheap(DL, 12) == True
assert Dishlist_all_cheap(DL, 11) == False



def Dishlist_change_price(x: [Dish], y: int) -> [Dish]:
    '''Takes a list of dishes and a number representing percent change and

def Dishlist_prices(a: [Dish]) -> [float]:
    '''Takes a list of dishes and returns a list of just the prices of the
    dishes'''
    k = []
    for i in a:
        k.append(i.price)
    return k

def Dishlist_average(x: [Dish]) -> float:
    '''Takes a list of dishes and returns the average price'''
    total = 0.0
    for i in Dishlist_prices(x):
        total += i
    return total / len(x)
assert Dishlist_average(DL) == 5.888888888888889
assert Dishlist_average(DL2) == 4.875
assert Dishlist_average([d1, d2, d3]) == 8.0



def Dishlist_keep_cheap(a: [Dish], b: int) -> [Dish]:
    '''Takes a list of dishes and returns a list of dishes that have prices
    lower than the given number'''
    result = []
    for i in a:
        if Dish_is_cheap(i, b):
            result.append(i)
    return result
