def Dishlist_all_cheap(D:list, N:float)->bool:
    '''Takes a list of Dishes and a number and returns true
if the price of every dish on the list is less than that number'''
    for i in range(len(D)):
        if not D[i].price < N:

def letter_count(m , n):
    result = []

    return result
