Count = namedtuple('Count', 'letter number')
def letter_count(message: str, letters: str)-> list:
'''Returns a list of namedtuples containing the character counted and the number found'''
     result = []
     for char in letters:
          tot = message.count(char[0:])
          result.append(Count(char, tot))
     return result
assert (letter_count('abbba', 'ab')) == [Count(letter='a', number=2), Count(letter='b', number=3)]
assert (letter_count('AaAa', 'Aa')) == [Count(letter='A', number=2), Count(letter='a', number=2)]
