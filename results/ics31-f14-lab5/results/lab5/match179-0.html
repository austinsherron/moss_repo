def Dish_str(dish:Dish) -> str:
    '''Returns dish information in a string'''
    return dish.name + ' ($' + '{:0.2f}'.format(dish.price) + '): ' + str(dish.calories) + ' cal'

assert Dish_str(d2) == 'Bibimbap ($8.50): 750 cal'

def Dish_same(dish1:Dish, dish2:Dish) -> bool:
    '''Returns true if the two dish arguments have the same number of calories'''
    if (dish1.name == dish2.name):
        if (dish1.calories == dish2.calories):
            return True
    else:
        return False

d4 = Dish('Kimbap', 4.00, 400)
d5 = Dish('Kimbap', 2.50, 400)

assert Dish_same(d3, d4)
assert not (Dish_same(d1, d2))
assert Dish_same(d4, d5)

def Dish_change_price(dish:Dish, percentage_change:int) -> Dish:
    '''Returns dish object with the changed price'''
    return dish._replace(price = (percentage_change + 100)/100 * dish.price)

Count = namedtuple('Count', 'letter number')

def character_count(string:str, letter:str) -> Count:
    ''' '''
    count = Count(letter, 0)
    
    for j in range(len(string)):
        if (string[j] == letter):
            count = count._replace(number = count.number + 1)
            
    return count
    
def letter_count(string:str, letters:str) -> list:
    '''Takes a string and the letters to count to return a list of Counts'''
    result = []
    
    for i in range(len(letters)):
        result.append(character_count(string, letters[i]))

    return result
