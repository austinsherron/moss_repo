def Dish_change_price(d:Dish,n:float)->Dish:
    '''takes a Dish and a number and returns a Dish that's the same as the

Count = namedtuple('Count', 'letter number')

def letter_count(s:str, d:str)->list:
    '''returns count of certain letters in first parameter; second parameter is string that specifies which letters to count'''
    L = []
    for w in d:
        Count_1 = Count(w, 0)
        L.append(Count_1)
    for n in s:
        for i in range(len(L)):
            if n == L[i].letter:
                L[i] = L[i]._replace(number = L[i].number + 1)
                break
  
    return L    

assert letter_count('Dogs have friends', 'aos') == [Count(letter='a', number=1), Count(letter='o', number=1), Count(letter='s', number=2)]
