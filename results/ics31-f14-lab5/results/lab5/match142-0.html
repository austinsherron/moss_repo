def Dish_str(d: Dish) -> str:
    '''Converts a dish into a string'''
    price = d.price
    cals = d.calories


    return (d.name + ' ($' + str(price) + '): ' + str(cals) +' cal')

def Dish_same(d: Dish, e: Dish) -> bool:
    '''Returns true if two dishes' names and calories are equal'''
    if d.calories == e.calories and d.name == e.name:
        return True
    else:
        return False

def Dishlist_average(lst: list) -> float:
    '''Returns the average of the dishlist's prices'''

Count = namedtuple('Count', 'letter number')

def letter_count(m: str, letters: str) -> Count:
    '''Takes two strings and returns the number of times each letter in the second string appears in the first string.'''
    result = []
    for a in range(len(letters)):
        n = Count(letters[a], m.count(letters[a]))
        result.append(n)
    return result
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
print(letter_count('Hi friends', 'ie'))
