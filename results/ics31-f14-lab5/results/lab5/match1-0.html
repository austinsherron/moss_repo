    '''Takes a Resturant and a number and returns True if the average
    price of the Resturant's menu is less than or equal to that number'''
    total = 0

Count = namedtuple('Count','letter number')

def letter_count(n:str, k:str)->'list of':
    '''Returns the counts of certain letters in the first paramater and
    the second paramater is a string that specifies which letters to count'''
    count_list = []
    a = 0
    for i in k:
        count = 0
        for j in range(len(n)):
            if i == n[j]:
                count = count + 1
        count_list.append(Count(i,count))
        a+=1
    return count_list
