Collection = [r1, r2, r3]

def Dish_change_prices(x: Dish, s: float) -> Dish:
    x = x._replace(price = x.price + s)
    return x

def Menu_change_prices(x: list, s:float) -> list:
    for i in range(len(x)):
        x[i] = Dish_change_prices(x[i], s)
    return i

def Restaurant_change_prices(x: Restaurant, s:float) -> Restaurant:
    x = x._replace(menu = Menu_change_prices(x.menu, s))
    return x

def Collection_change_prices(x: list, s: float):
    for i in x:
        Restaurant_change_prices(i, s)
    return x
print(Collection_change_prices(Collection, 10.00))
