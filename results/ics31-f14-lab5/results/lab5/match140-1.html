Count = namedtuple("Count","letter number")
c1 = Count('a',2)

def letter_count(a: str, b:str)->list:
    '''takes a string and counts the number of characters of another string and creates a list of namedtuples'''
    new_list = []
    for i in b:
        n = 0
        if i in a:
            n += 1
        c = Count(i,n)
        new_list.append(c)
    return new_list
assert letter_count('bad','q') == [Count(letter='q',number=0)]
assert letter_count('cat','a b') ==[Count(letter='a', number=1), Count(letter=' ', number=0), Count(letter='b', number=0)]
