def Dish_same(d:Dish,f:Dish)->bool:

Count = namedtuple('Count', 'letter number')

def letter_count(s:str,t:str):
    '''returns the counts of certain letters in the first parameter;
    the second parameter is a string that specifies which letters to
    count.'''

    c1 = Count('a', 0)
    c_list = [Count('a', 0), Count('b', 0)]
    c_list=[]
    for char in t:
        c_list.append(Count(char, 0))

    for char in s:
        for indx2 in range(len(c_list)):
            if char == c_list[indx2].letter:
                c_list[indx2]=c_list[indx2]._replace(number=c_list[indx2].number+1)
    return c_list
assert letter_count('string','si') == [Count(letter='s', number=1), Count(letter='i', number=1)]
assert letter_count('banana','bananas') == [Count(letter='b', number=1), Count(letter='a', number=3), Count(letter='n', number=2), Count(letter='a', number=3), Count(letter='n', number=2), Count(letter='a', number=3), Count(letter='s', number=0)]
assert letter_count('tree','cat') == [Count(letter='c', number=0), Count(letter='a', number=0), Count(letter='t', number=1)]
