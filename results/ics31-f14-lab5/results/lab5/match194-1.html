# A collection is a list of restaurants

def Collection_new() -> list:
    ''' Return a new, empty collection
    '''
    return [ ]

def Collection_str(C: list) -> str:
    ''' Return a string representing the collection
    '''
    s = ""
    for r in C:
        s = s + Restaurant_str(r)
    return s

def Collection_search_by_name(C: list, name: str) -> list:
    """ Return list of Restaurants in input list whose name matches input string.
    """
    result = [ ]
    for r in C:
        if r.name == name:
            result.append(r)
    return result
    # alternative (using a list comprehension):
    # return [r for r in C if r.name == name]

def Collection_add(C: list, R: Restaurant) -> list:
    """ Return list of Restaurants with input Restaurant added at end.
    """
    C.append(R)
    return C

def Collection_remove_by_name(C: list, name: str) -> list:
    """ Given name, remove all Restaurants with that name from collection.
    """
    result = [ ]
    for r in C:
        if r.name != name:
            result.append(r)
    return result
    #    Alternative:
    #    return [r for r in self.rests if r.name != name]

def Dish_str(Dish)->str:
    "Takes a Dish and returns a specifically formatted string"
    return Dish.name+" ($"+str(Dish.price)+"): "+str(Dish.calories)+" cal"




def Menu_display(menu:list)->str:
    "take a list of dishes, returns a string representation of each dish follwed by a newline"
    menustr = ''
    for i in range(len(menu)):
        menustr = menustr + Dish_str(menu[i])+"\n"
    return menustr

#CHANGE PRICE



def Dish_change_price(Dish, price_change:int)->Dish:
    "Returns the given dish with the price changed by the given percentage"
    Dish = Dish._replace(price = Dish.price + Dish.price * (price_change/100))
    return Dish

def Dishlist_change_price(list_dishes:list,percent:int)-> list:
    "takes a list of dishes anda number representing a percentage change and returns a list of dishes with each price change by the specified amount"
    price_list = []
    for i in list_dishes:
        price_list.append(Dish_change_price(i,percent))
    return price_list



def Restaurant_change_price(Restaurant, percent: int):
    "takes a restaurant and returns a restaurant with changed price by percent parameter"
    Restaurant = Restaurant._replace(menu = Dishlist_change_price(Restaurant.menu, percent))
    return Restaurant

def Collection_change_price(collection, percent: int):
    "takes a collection of restaurants and returns the collection with the price of each dish in each restaurant in the collection by 2.50"
    
    new_collection = []
    for restaurant in collection:
       new_restaurant = Restaurant_change_price(restaurant, percent)
       new_collection.append(new_restaurant)
    return new_collection
# AVERAGE
def Dishlist_prices(list_dishes:list)->list:
    'takes a list of dishes and returns a list of the prices'
    nums = []
    for i in list_dishes:
        nums.append(i.price)
    return nums
def Dishlist_average(list_dishes:list)->float:
    'takes a list of dishes and returns average price'
    price_list = Dishlist_prices(list_dishes)
    avg = (sum(price_list))/(len(price_list))
    return avg


def Restaurant_average(restaurant)->float:
    "takes a restaurant and returns the average price of the dishes in the restaurant"
    average_price = Dishlist_average(restaurant.menu)
    return average_price
def Collection_select_cheap(collection,num:float)->list:
    "takes a collection and returns all the restaurants whose average price is less than or equal to the num"
    new_collection = []
    for restaurant in collection:
        if Restaurant_average(restaurant) <= num:
            new_collection.append(restaurant)
    return new_collection
