Count = namedtuple('Count', 'letter number')


def letter_count (A: str, B: str)-> list:
    '''function letter_count takes in 2 strings; the first being a string of letters and the second specify    which letters to count. The function will return the counts as namedtuple which two fields (letter and number)'''
    result=[]
    for i in B:
        result.append(Count(i, A.count(i)))
    return result


print(letter_count('Peter the Anteater', 'aeo'))
assert letter_count('Peter the Anteater', 'aeo') ==[Count(letter='a', number=1), Count(letter='e', number=5), Count(letter='o', number=0)]
assert letter_count('University of California, Irvine', 'aeio') == [Count(letter='a', number=2), Count(letter='e', number=2), Count(letter='i', number=5), Count(letter='o', number=2)]
