def menuRaisePrices(m: list) -> list:
    "Returns a list of dishes with the prices of the dishes each raised by 2.50"
    result = []
    for i in range(len(m)):
        result.append(dishRaisePrice(m[i]))
    return result

def restaurantRaisePrices(r: Restaurant) -> Restaurant:
    "Returns a restaurant with each dish on its menu raised by 2.50"
    r = r._replace(menu = menuRaisePrices(r.menu))
    return r

def collectionRaisePrices(C: list) -> list:
    '''Returns a collection of restaurants with the prices of every dish at each restaurant
    raised by 2.50'''
    for i in range(len(C)):
        C[i] = restaurantRaisePrices(C[i])
    return C

Count = namedtuple("Count", 'letter number')

def letter_count(x: str, y: str) -> list:
    "Returns counts of vowels in a particluar string"
    CountList = []
    
    for i in range(len(y)):
        c1 = Count(y[i], x.count(y[i]))
        CountList.append(c1)
        
    return CountList


print(letter_count("help", "lp"))
print(letter_count("help", "as"))
print(letter_count("The cabbage has baggage", "abcd"))
