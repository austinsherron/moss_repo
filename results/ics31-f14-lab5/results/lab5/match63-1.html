# A collection is a list of restaurants

def Collection_new():
    ''' Return a new, empty collection
    '''
    return [ ]

def Collection_str(C):
    ''' Return a string representing the collection
    '''
    s = ""
    for r in C:
        s = s + Restaurant_str(r)
    return s

def Collection_search_by_name(C, name):
    """ Return list of Restaurants in input list whose name matches input string.
    """
    result = [ ]
    for r in C:
        if r.name == name:
            result.append(r)
    return result
    # alternative (using a list comprehension):
    # return [r for r in C if r.name == name]


def Collection_add(C, R):
    """ Return list of Restaurants with input Restaurant added at end.
    """
    C.append(R)
    return C

def Menu_add(M, D):
    """ Return list of Restaurants with input Restaurant added at end.
    """
    M.append(D)
    return M

def Collection_remove_by_name(C, name):
    """ Given name, remove all Restaurants with that name from collection.
    """
    result = [ ]
    for r in C:
        if r.name != name:
            result.append(r)
    return result
    #    Alternative:
    #    return [r for r in self.rests if r.name != name]

def Menu_remove_by_name(M, name):
    """ Given name, remove all Restaurants with that name from collection.
    """
    result = [ ]
    for r in M:
        if r.name != name:
            result.append(r)
    return result

def Dish_str(self):
    return (
        "Name:     " + self.name + "\n" +
        "Price:    $" + str(self.price) + "\n" +
        "Calories: " + str(self.calories) +" cal"+"\n\n")
    
def Menu_str(M):
    s = ""
    for r in M:
        s = s + Dish_str(r)
    return s

def Dish_change_price(D,num):
    D=D._replace(price=D.price*(1.00+num/100.00))

def Menu_change_prices(M,num):

        M[i]=Dish_change_price(M[i], num)
    return M

def Restaurant_change_prices(R,num):
