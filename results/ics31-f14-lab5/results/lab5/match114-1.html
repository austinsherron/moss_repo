from collections import namedtuple
Restaurant = namedtuple('Restaurant', 'name cuisine phone menu')
# Constructor:   r1 = Restaurant('Taillevent', 'French', '01-11-22-33-44', 'Escargots', 23.50)

def Restaurant_str(self: Restaurant) -> str:
    return (
        "Name:     " + self.name + "\n" +
        "Cuisine:  " + self.cuisine + "\n" +
        "Phone:    " + self.phone + "\n" +
        "Menu:     " + Menu_str(self.menu)        
            )
def Restaurant_get_info() -> Restaurant:
    '''Prompt user for fields of Restaurant; create and return.'''
    return Restaurant(
        input("Please enter the restaurant's name:  "),
        input("Please enter the kind of food served:  "),
        input("Please enter the phone number:  "),
        
        Menu_enter())

def Restaurant_change_price(r: Restaurant, num:int) -> Restaurant:
    '''Returns the restaurant with all the dish's prices changed by number input'''
    return r._replace(menu = Menu_raise_prices(r.menu,num))

#### COLLECTION
# A collection is a list of restaurants

def Collection_new() -> list:
    '''Return a new, empty collection'''
    return [ ]

def Collection_str(C: list) -> str:
    '''Return a string representing the collection'''
    s = ""


    for r in C:
        s = s + Restaurant_str(r)
    return s

def Collection_search_by_name(C: list, name: str) -> list:
    '''Return list of Restaurants in input list whose name matches input string.'''
    result = [ ]
    for r in C:
        if r.name == name:
            result.append(r)
    return result
    
def Collection_add(C: list, R: Restaurant) -> list:
    '''Return list of Restaurants with input Restaurant added at end.'''
    C.append(R)
    return C

def Collection_remove_by_name(C: list, name: str) -> list:
    '''Given name, remove all Restaurants with that name from collection.'''
    result = [ ]
    for r in C:
        if r.name != name:
            result.append(r)
    return result
 
def Collection_change_prices(c: list, num: int):
    '''Changes price of each Restaurant in the collection.'''
    for i in range(len(c)):
        c[i] = Restaurant_change_price(c[i], num)

#### Dish
Dish = namedtuple('Dish', 'name price calories')
d1 = Dish('Big Mac', 20.00, 2000)
d2 = Dish('The Whopper', 15.00, 100)
d3 = Dish('Wendy Burger', 2.00, 3000)

def Dish_str(dish: Dish) -> str:
    '''Returns string form of a Dish'''
    return str(dish.name) + ' ($' + str(dish.price) + ') : ' + str(dish.calories) + ' cal'
assert Dish_str(d1) == 'Big Mac ($20.0) : 2000 cal'

def Dish_get_info() -> Dish:
    '''Prompt user for fields of Restaurant; create and return.'''
    return Dish(
        input("Please enter the Dish's name:  "),
        float(input("Please enter the price of that dish:  ")),
        int(input("Please enter the calories of that dish:  ")))

def Dish_raise_price(d: Dish, num:int) -> Dish:
    '''Returns a Dish with its price changed by number input'''
    return d._replace(price = d.price + d.price/100*num)
assert Dish_raise_price(d1, 100) == Dish('Big Mac', 40.00, 2000)

def Menu_enter()-> 'list of Dishes':
    '''If input is yes, prompts user to create a Dish and adds it on the
    growing list of dishse. If no, the function returns the compiled list
    of Dishes'''
    dish_list = []
    while True:
        response = input('Do you want to add a Dish?: ')
        if response == 'yes':
            dish_list.append(Dish_get_info())
        if response == 'no':
            return dish_list

def Menu_str(m: 'list of Dish') -> str:
    '''Returns display string for menu of dishes'''
    result = ''
    for d in m:
        result+= Dish_str(d) + '\n'
    return result

def Menu_raise_prices(m: 'list of Dish',num:int) -> 'list of Dish':
    '''Returns a Menu(list of Dish) with all its dishes changed by number input'''
    m2 = m[:]
    for i in range(len(m2)):
        m2[i] = Dish_raise_price(m2[i],num)
    return m2

Count = namedtuple('Count', 'letter number')
def letter_count(s: str, c:str) -> 'list of Count':
    '''Returns the counts of certain leters in the first parameter; the second
       parameter is a string that specifies which letters to count'''
    count_list = []
    for i in range(len(c)):
        if c[i] == '\\':
            count_x = Count(c[i]+c[i+1],0)
            i+=1
        else:
            count_x = Count(c[i],0)
        count_list.append(count_x)
    for i in range(len(count_list)):
        letter = count_list[i].letter
        for n in range(len(s)): 
            if(s[n] == '\\'):
               character = s[n] + s[n+1]
               n+=1
            else:
               character = s[n]
            if character == letter:
                count_list[i] = count_list[i]._replace(number = count_list[i].number+1)
    return count_list
assert letter_count('The cabbage has baggage', 'Abcd') == [Count(letter='A',number=0), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
assert letter_count('abc \t \n',' \t\n') == [Count(letter=' ', number = 2),
                                             Count(letter ='\t', number = 1),
                                             Count(letter = '\n', number = 1)]
