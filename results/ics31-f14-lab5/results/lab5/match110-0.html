def dish_str(D: namedtuple) -> str:
    "This function turns a namedtuple into a formatted string."
    return D.name + "  ($" + str(D.price) + "):" + str(D.calories) + " cal"
print(dish_str(MacAndCheese))

#C3
def dish_same(D1: namedtuple, D2: namedtuple) -> bool:
    "Takes two dishes as arguments and returns True if the names of the dishes and their calorie counts are equal."
    if (D1.calories == D2.calories and D1.name == D2.name):
        return True
    else:
        return False
assert dish_same(Cheeseburger, CheeseyBread) == False
assert dish_same(Cheeseburger, CheeseyBread1) == True

def restaurant_is_cheap(r: namedtuple, f: float) -> bool:
    "Takes a restaurant and a number and returns true if the average price is less than or equal to the number"
    empty = 0.0
    for i in r.menu:
        empty += i.price
    empty = empty/len(r.menu)
    if empty <= f:
        return True
    else:
        return False
