def Dish_str(x: Dish) -> str:
    ''' Takes a Dish and return name (price): Calories'''
    return x.name + '($'+ str(x.price) + '):'+ str(x.calories) + 'cal'

def Dish_same(l: Dish, p: Dish) -> bool:
    '''Takes two dishes and returns a bool based of name and calories'''
    if l[0]== p[0] and l[-1] == p[-1]:
        return True
    else:
        return False

def Dish_change_price(b: Dish, num: int) -> str:
    ''' Takes in a dish and number and changes the price'''
    if num == 100:
        return b._replace( price = b.price*2)
    elif num == -50:
        return b._replace(price = b.price/2)

def letter_count (m:str,p:str):
    '''The function should return its counts as a list of Count namedtuples, where a Count namedtuple has two fields, letter and number.'''
    Count = namedtuple ('Count','letter number')
    result = []
