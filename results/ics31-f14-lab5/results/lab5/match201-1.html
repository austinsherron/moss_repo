Count = namedtuple('Count', 'letter number')

def one_letter_count(word:str, letter:str)-> Count:
    '''takes in a string and a single letter, goes through the string and comes up with how many times the letter came up in the string'''
    result = 0
    for i in word:
        if i == letter:
            result += 1
    C1 = Count(letter, result)
    return C1
assert one_letter_count('mississippi', 's') == Count(letter='s', number=4)
assert one_letter_count('abcdef', 's') == Count(letter='s', number=0)
print(one_letter_count('dog','o'))

def letter_count(word:str, letters:str)->[Count]:
    '''takes in a word and letters, returns the amount of times each letter appears in the word/string'''
    resultCount = []
    for i in letters:
        resultCount.append(one_letter_count(word, i))
    return resultCount

assert letter_count('redhood','od') == [Count(letter='o',number=2), Count(letter='d',number=2)]
assert letter_count('What the heck',' \t\n') == [Count(letter=' ',number=2), Count(letter='\t',number=0), Count(letter='\n',number=0)]
print(letter_count('mississippi','sp'))
