def Restaurant_is_cheap(x:Restaurant, y:float) -> bool:
    '''Takes a Restaurant and if the average price of the menu is less than the float
    given, it will return True. Else it will return False'''
    dishes = x.menu
    amount = 0
    for dish in dishes:
        amount += dish.price
    average = amount/len(dishes)
    if average <= y:
        return True
    else:
        return False

    counter_f = 0

                counter_a +=1

                listofcount.append(count_a)
            elif characters == y[-2]:
                counter_c +=1
            elif characters == y[-3]:
                counter_d += 1
            elif characters == y[-4]:
                counter_e += 1
            elif characters == y[-5]:
                counter_f +=1
            elif characters == y[-1]:
                counter_b +=1
            else:
                pass
        
            count_b = Count(y[-1], counter_b)
            count_c = Count(y[-2], counter_c)
            count_d = Count(y[-3], counter_d)
            count_e = Count(y[-4], counter_e)
            count_f = Count(y[-5], counter_f)

            listofcount.append(count_b)
