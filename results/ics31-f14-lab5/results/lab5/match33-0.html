def Dish_str(d:Dish) -> str: #returns string of dish name price and calories
    result = ''
    result = result+d.name
    result = result+' ($'
    result = result+str(d.price)
    result = result+'):'
    result = result+str(d.calories)
    result = result+' cal'
    return result

def Dish_same(x:Dish, y:Dish) -> bool: #Takes two dishes,
    #returns true if name and calorie are equal, false otherwise
    z = False
    if x.name == y.name:

            z = True
    return z

def Dish_change_price(d:Dish, p:float) -> Dish: #Takes a dish and a number, returns a
    #dish with price changed by a percentage p
    x = d.price
    r = p/100
    z = x*r
    y = x+z
    return d._replace(price=y)

count = namedtuple('count','string occur')

def letter_count(x:str,y:str) -> list:
    ''' Counts the number of occurrences of characters in input y in x. Returns a list of count namedtuples
    that each include both the character counted and the amount of times it occured in x.'''
    temp = []
    result = []
    w = list(y)
    c = list(x)
    for i in range(len(w)):
        for q in range(len(c)):
            if (w[i] == c[q]) == True:
                temp.append(w[i])
    for g in range(len(w)):
        result.append(count((w[g]),(temp.count(w[g]))))
    return result

assert len(letter_count('Hello there honey', 'Fine THEN')) == len('Fine THEN')
