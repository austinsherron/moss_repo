def Dishlist_price(d: [Dish]) -> [float]:
    '''Returns list of dishes' price'''
    result = []
    for x in range(len(d)):
        result.append(d[x].price)
    return result

Count = namedtuple('Count', 'letter number')

def one_char_count(word: str, char: str) -> Count:
    '''Returns the number of that letter in the word'''
    C = Count(char, word.count(char))
    return C

assert one_char_count('tomato', 't') == Count('t', 2)
assert one_char_count('the potato', ' ') == Count(' ', 1)
assert one_char_count('tomato', 'b') == Count('b', 0)

def letter_count(word: str, letter: str) -> [Count]:
    '''returns the number of letters in the word'''
    result = [ ]
    string_list = list(letter)
    for x in string_list:
        if x in word:
            result.append(one_char_count(word, x))
    return result

assert letter_count('tomato', 'ot') == [Count('o', 2), Count('t', 2)]
assert letter_count('lettuce', 'ue') == [Count('u', 1), Count('e', 2)]
assert letter_count('\ttomato \n', '\t \n') == [Count('\t', 1), Count(' ', 1), Count('\n', 1)]
