Count = namedtuple('Count', 'letter number')
#
def letter_count(a:str, b:str) -> list:
    counts = []
    letter = ''
    for x in b:
        letter = x
        number = 0
        for y in a:
            if y == x:
                number += 1
        counts += [Count (letter, number)]
    return counts
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]  # function testing
assert letter_count('Happy ICS 31! I love programming!', 'abcd\t\n ') == [Count(letter='a', number=2), Count(letter='b', number=0), Count(letter='c', number=0), Count(letter='d', number=0), Count(letter='\t', number=0), Count(letter='\n', number=0), Count(letter=' ', number=5)]  # function testing
print(letter_count('Snoopy loves Woodstock!', 'one!.')) # function testing
