def Dish_change_price (d: Dish, n: int) -> Dish:
    '''Takes a Dish and number input (percentage) and changes the
    price of the dish object based on that number input''' 
    price_change = (d.price * (n/100))+ d.price
    new_dish = d._replace(price = price_change)
    return new_dish

Count = namedtuple('Count', 'letter number')
def letter_count(string: str, letters: str) -> list:
    '''Counts the number of letters string and returns as Count objects'''
    count_list = []
    for i in letters:
        count_list.append(Count(i, string.count(i)))
    return count_list

assert letter_count('ayyy lmao', 'y') == [Count(letter='y', number=3)]
print(letter_count('The quick brown fox jumped over the lazy dog', 'aeiou'))
