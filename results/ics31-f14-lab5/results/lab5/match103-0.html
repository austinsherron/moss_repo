def Dish_str(dish:"Dish object") -> str:

def Collection_select_cheap(collection:[Restaurant],number:float)->[Restaurant]:
    """Returns a list of restaurants whose average price is cheaper than given number"""
    return_collection = []
    for restaurant in collection:
        sumnum = 0
        for dish in restaurant.menu:
            sumnum += dish.price
        avgnum = sumnum / len(restaurant.menu)
        if avgnum < number:
            return_collection.append(restaurant)
    return return_collection

Count = namedtuple("Count","letter number")

def letter_count(input_string,count_chars) -> [Count]:
    """ Returns a list of Count namedtuples containing the counts of each
    letter in input_string that is present in count_chars"""
    count_list = []
    for char in count_chars:
        count_list.append(Count(char,input_string.count(char)))
    return count_list

assert letter_count("test","tsk") == [Count(letter="t",number=2),
                                      Count(letter="s",number=1),
                                      Count(letter="k",number=0)]
