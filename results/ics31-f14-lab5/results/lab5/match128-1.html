def Dish_same(item1:Dish, item2:Dish) -> bool:
    '''returns true if dish names and calories are equal'''
    if item1.name == item2.name:
        if item1.calories == item2.calories:
            return True
        else :
            return False
    else :
        return False

import math
def Dish_change_price(item:Dish, num:int) -> Dish:
    '''changes price based off of given number'''
    if num>0:
        amount = num/100 * item.price
        new_price = amount + item.price
    else :
        amount = (math.fabs(num))/100 * item.price
        new_price = item.price - amount
    new_dish = Dish(item.name, new_price, item.calories)
    return(new_dish)
assert Dish_change_price(d5,100).price == 40.0
assert Dish_change_price(d5,-50).price == 10.0

def Dishlist_average(items:list) -> float:

Count = namedtuple('Count', 'letter, number')

def small_part(item1:str, one:str) -> Count:
    '''returns number of times items in second string appear in first strint'''
    num = 0
    for i in range(len(item1)):
        if item1[i] == one:
            num = num + 1
    return(Count(one, num))



def letter_count(item1:str, item2:str) -> list:
    '''returns number of times items in second string appear in first string'''
    result = []
    for i in range(len(item2)):
        result.append(small_part(item1, item2[i]))
    return result
print(letter_count('He llo\n', ' ae\nilo'))
