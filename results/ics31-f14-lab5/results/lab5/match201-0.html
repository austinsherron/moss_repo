def Dishlist_average(L:list)->float:
    ''' Takes a list of Dishes and returns the average price of those dishes '''

Count = namedtuple('Count', 'letter number')

def letter_count(m:str, s:str)->list:
    ''' Returns the counts of certain leters in the first parameter (the second parameter is a string that
specifies which letters to count). The function should return its counts as a list of Count namedtuples,
where a Count namedtuple has two fields, letter and number. '''
    countlist =[]
    x = 0
    for i in range(len(s)):
        x=m.count(s[i]) # count is a method (not capitalized)
        countlist.append(Count(s[i],x))
    return countlist
assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5), Count(letter='b',number=3),
