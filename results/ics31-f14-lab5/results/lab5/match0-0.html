def letter_count(string: str, search: str) -> list:
    '''
    *********************************************
    LETTER_COUNT
    ---------------------------------------------
    Description:
    The function returns the counts of certain
    letters in the first parameter; the second
    parameter is a string that specifies which
    letters to count

    PARAMETERS:
    string: str  - str to search
    search: str  - what to search for

    RETURNS:
    list
    *********************************************
    '''
    result = []
    for i in search:
        # create tuple and initialize values
        if i.isalpha():
            c = Count(i.upper() + ' or ' + i.lower(), count(string, i.upper()) + count(string, i.lower()) )
        else:
            c = Count( i, count(string, i) )

        # check for doubles
        if c not in result:
            result.append( c )
    return result
assert letter_count('The cAbbage hAs baggAge', 'abcd') == \
[   Count(character='A or a', occurences=5),
    Count(character='B or b', occurences=3),
    Count(character='C or c', occurences=1),
    Count(character='D or d', occurences=0) ]
assert letter_count('RubBer baBby Buggy Bumpers', 'be\n') == \
[   Count(character='B or b', occurences=7),
    Count(character='E or e', occurences=2),
