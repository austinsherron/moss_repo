Count = namedtuple('Count', 'letter number')
def single_letter_count( a: str, b: str) -> Count:
       '''takes two strings as parameters and returns count of certain
letters in the first parameter using the specification from second parameter,
only one single letter for second string'''
       countlist2 = []
       countlist2.append((Count(b, a.count(b))))
       return countlist2

def letter_count (a: str, b: str) -> list:
       ''' takes two strings as parameters and returns count of certain
letters in the first paraments using the specification from second parameter'''
       countlist = []
       for c in b:
              countlist.append(single_letter_count(a, c))
       return countlist
           
assert letter_count('wweweer','we') == [[Count(letter='w', number=3)], [Count(letter='e', number=3)]]
assert letter_count('kkuusshh','kush') == [[Count(letter='k', number=2)], [Count(letter='u', number=2)], [Count(letter='s', number=2)], [Count(letter='h', number=2)]]
assert not letter_count('check','chk') == [[Count(letter='k', number =0)]]
