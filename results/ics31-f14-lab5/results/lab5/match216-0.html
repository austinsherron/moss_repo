Count = namedtuple('Count', 'letter number')
def letter_count(x: str, y: str) -> list:
#    """Counts the number of letters in the strings"""
    result = []
    for letter in y:
        n = 0
        for letter2 in x:
            if letter == letter2:
                n = n + 1
        result.append(Count(letter,n))
    return result
print(letter_count('The cabbage has baggage', 'abcd'))
