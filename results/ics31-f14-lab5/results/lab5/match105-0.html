#a function called letter_count that has two strings as parameters. The function returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to count
def letter_count(str1: str, str2: str) -> list:
    '''Takes in two strings, returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to count'''
    counter = []
    for c in str2:
        counter.append(Count(letter= c, number=str1.count(c)))
    return counter
