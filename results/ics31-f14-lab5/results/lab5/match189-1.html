def Dishlist_change_price(Dishes:list, num:int) -> list:
    ''' Takes a list of dishes and a number representing
        a percentage change and returns a list of Dishes
        with each price changed by the specified amount. '''
    for dish in Dishes:
        dPrice = Dishes.pop(0)
        newprice = dPrice._replace(price = dPrice.price + (dPrice.price * (num/100)))
        Dishes.append(newprice)
    return Dishes

def Dishlist_average(Dishes: list) -> float:
    ''' Takes a list of Dishes and returns the average
        price of those dishes. '''
    sum_list = 0
    for i in Dishes:
        sum_list += i.price
    return round(sum_list / len(Dishes),2)

# Note: A collection in this case is a list of Restaurant objects.
def Collection_select_cheap(collection: list, num: int) -> list:
    ''' Takes a collection and a number and returns a list of all the
        restaurants in the collection whose average price is less than or
        equal to that number. '''
    result = []
    for List in collection:
        if Dishlist_average(List.menu) <= num:
            result.append(List)

def letter_count(word: str, character: str) -> str:


    ''' Returns the counts of certain letters in the first parameter; the
        second parameter is a string that specifies which letters to count '''
    List = []
    for each_letter in character:
        List.append("Letter = " + each_letter + ", number="+ str(word.count(each_letter)))
