Count = namedtuple('Count', 'letter number')

def single_letter_count(some_message: str, character: str) -> Count:
    '''Returns the number of times the character appears in the message within a Count namedtuple'''
    n = 0
    for i in some_message:
        if character == i:
            n += 1
    return Count(character, n)

def letter_count(word: str, letters: str) -> 'list of Count':
    ''' '''
    result = []
    for i in letters:
        result.append(single_letter_count(word, i))
    return result

assert letter_count('big', 'g') == [Count(letter='g', number=1)]
assert letter_count('   ', ' ') == [Count(letter=' ', number=3)]
assert letter_count('Aa', 'a') == [Count(letter='a', number=1)]
assert letter_count('CATccaaat','cat') == [Count(letter='c', number=2),
                                           Count(letter='a', number=3),
                                           Count(letter='t', number=1)]
assert letter_count('\t' + 'This message was written to' + '\n' +\
                    'test the white-space assertion statement.', ' \t\n') == [Count(letter=' ', number=8),
                                                                             Count(letter='\t', number=1),
                                                                             Count(letter='\n', number=1)]
