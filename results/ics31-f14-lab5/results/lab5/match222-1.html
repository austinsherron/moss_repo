def Collection_raise_prices(inputList):
#add $2.50 to every dish in the list
    inni = []
    for i in inputList:
        ini = []
        for ii in i.menu:
            inp = ii.price
            ii = ii._replace(price= inp + 2.50)
            ini.append(ii)
        i = i._replace(menu = ini)

def Collection_change_price(inputList, percent):
#add the input percentage to every dish in the list
    inni = []
    for i in inputList:
        ini = []
        for ii in i.menu:
            inp = ii.price
            ii = ii._replace(price= inp + inp*percent)
            ini.append(ii)
        i = i._replace(menu = ini)
        inni.append(i)
    return inni

def letter_count(in1:str, in2:str):
#returns the counts of certain letters in the first parameter; the second parameter is a string that specifies which letters to count
    result = []
    num = 0
    for i in in1:
        if i in in2:
            result.append(Count(i, num))
        num = num + 1
    return result
