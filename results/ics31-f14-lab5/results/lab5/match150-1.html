def Dish_str(n: Dish) -> str:
    ''' Returns the string representation of the
        Dish '''
    return (n.name + " ($" + str(n.price) + "): " +

Count = namedtuple('Count', 'letter number')

def letter_count(x: str, y: str) -> int:
    ''' Returns how many times y appears in x in namedtuple form '''
    L = []
    for i in y:
        L.append(Count(i, x.count(i)))
    return L

assert letter_count('The cabbage has baggage', 'abcd') == [Count('a', 5),
                                                           Count('b', 3),
                                                           Count('c', 1),
                                                           Count('d', 0)]

assert letter_count('''Lots     of
spaces''', ' \t\n') == [Count(' ', 5),
                  Count('\t', 0),
                  Count('\n', 1)]
