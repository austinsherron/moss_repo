Count = namedtuple('Count', 'letter number')

def letter_count(message: str, C:str) -> list:
    count_list = []
    for x in C:
        upper_x = x.upper()
        lower_x = x.lower()
        Num = message.count(upper_x)+ message.count(lower_x)
        count_list.append(Count(x,Num))
    return count_list

print(letter_count('The cabbage has baggage', 'abcd'))
print(letter_count('THE CABBAGE HAS BAGGAGE', 'abcd'))
