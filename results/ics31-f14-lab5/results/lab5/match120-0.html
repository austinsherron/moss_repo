def Dish_change_price(f:Dish, percent: float) -> Dish:
    '''change price of dish by percent inputed'''
    if percent > 0:
        return f._replace(price = f.price + (percent/100 * f.price))
    elif percent < 0:
        return f._replace(price = f.price - (-percent/100 * f.price))

def letter_count(s:str, compared_to:str):
    ''' counts the specific letters in a string'''
    l_of_Count = []
    for char in compared_to:
        l_of_Count.append(Count(char, s.count(char)))
