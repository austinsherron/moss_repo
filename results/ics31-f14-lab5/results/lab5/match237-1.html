def Dish_str(dish: 'Dish object') -> 'string representing dish':
    '''Takes a dish and returns a string representing the dish's name,
    price, and calories'''
    return dish.name + ' ($' + str(dish.price) + '): '+ str(dish.calories) + ' cal'

Count = namedtuple('Count','letter number')

def letter_count(some_message:str, s:str) -> list:
    '''return a list of all the characters counted in the some_message'''
    templist = []
    tempint = 0


    for c in s:
        tempint = 0
        for x in some_message:
            if x ==  c:
                tempint +=1
        
        templist.append(Count(c,tempint))
    return templist
