def Dish_change_price(d: Dish, n: float) -> Dish:
    ''' takes a Dish and a number and returns a Dish that's the same as the parameter except that its price is changed as follows: The number (positive or negative) represents a percentage change in price '''
    if n > 0:
        return d._replace(price = d.price * (1 + (n / 100)))
    elif n < 0:
        n = -(n)
        return d._replace(price = d.price * (1 - (n / 100)))

assert Dish_change_price(d1, 100) == Dish('Pizza', 10.0, 300)

Count = namedtuple('Count', 'letter number')

def letter_count(s1: str, s2: str) -> "list<Count>":
    ''' takes 2 strings and returns counts of each character of 2nd string in 1st string '''
    result = []
    list_counter = []
    for x in range(len(s2)):        # counts of each character of 2nd string in 1st string
        counter = 0
        for i in range(len(s1)):
            if s1[i] == s2[x]:
                counter += 1
        list_counter.append(counter)
    for y in range(len(s2)):        # creats a list of Count objects
        result.append(Count(s2[y], list_counter[y]))
    return result
