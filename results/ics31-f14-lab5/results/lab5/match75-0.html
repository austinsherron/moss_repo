def Dish_str (d:Dish) -> str:
    ''' Prints out dish's name, price, and calories as a string '''
    new = ''
    L = [d.name, " ($", d.price, "): ", d.calories, " cal"]
    for i in L:
        new+=str(i)
    return new

Count = namedtuple('Count','letter number')

def character_count (s:str, l:str) -> int:
    '''  Takes a string and a single character and returns a Count with
    the character and the number of times it occurs in the string. '''
    return Count(letter=l,number=s.count(l))
assert character_count('thesis','i') == Count(letter='i',number=1)
assert character_count('thesis','s') == Count(letter='s',number=2)
assert character_count('hi there',' ') == Count(letter=' ',number=1)
assert character_count('''hey there
Bob''','\n') == Count(letter='\n',number=1)

def letter_count (s: str, l: str) -> int:
    ''' Counts the number of each character in the second specified string
    in the first string and returns a list of Counts. '''
    count = []
    for i in l:
        count.append(character_count(s,i))
    return count
assert letter_count('The cabbage has baggage', 'abcdt')==[Count(letter='a',number=5),
                                                         Count(letter='b',number=3),
                                                         Count(letter='c',number=1),
                                                         Count(letter='d',number=0),
                                                         Count(letter='t',number=0)]
print('\nPart (g)\n"abcdt" in "The cabbage has baggage":')
for i in letter_count('The cabbage has baggage', 'abcdt'):
    print(i.letter,"appears",i.number,"times")
