def Dishlist_all_cheap(DL: "list of Dish", p: float) -> bool:
    """ Traverse through the list of dishes
        1) Return True if every dish's price is less than the specified price
        2) Return False if any dish's price is greater than the specified price
    """
    counter = 0     # counter -> keeps track of the amount of True's
    for d in DL:
        if Dish_is_cheap(d, p) == True:
            counter = counter + 1
        if Dish_is_cheap(d, p) == False:
            return False
        if counter == len(DL):      # counter = length -> means every Dish is cheaper than p
            return True

def Dishlist_average(DL: "list of Dish") -> float:
    """ Calculate the average price for the given list of dishes
    """
    sum = 0
    if len(DL) == 0:
        return 0
    for d in DL:
        sum += d.price
    return sum/len(DL)

Count = namedtuple("Count", "letter number")

def letter_count(s1: str, s2: str) -> "list of Count":
    """ s1 = the string being searched
        s2 = a string that specifies which letters to count.

        Returns a list of Count objects displaying how many times
        each character of s2 was found in s1
    """ 
    CL = [ ]
    for x in range(len(s2)):
        count = 0           # count -> keeps track of every instance of s2[x]; count resets back to 0 after every loop of x
        if s2[x] == "\\":   # if-statement -> deals with \t and \n if found in s2
            for y in range(len(s1)):
                if s1[y:y+2] == s2[x:x+2]:
                    count += 1
        else:               #count -> deals with every other character in s2
            for y in range(len(s1)):
                if s1[y] == s2[x]:
                    count += 1
        CL.append(Count(s2[x], count))
    return CL
assert letter_count("Hello", "\ne") == [Count("\n", 0),
                                        Count("e", 1)]
assert letter_count("\n\n\t\t", "\n\t") == [Count("\n", 2),
                                            Count("\t", 2)]
assert letter_count("alphabet", "aeiou") == [Count("a", 2),
                                             Count("e", 1),
                                             Count("i", 0),
                                             Count("o", 0),
                                             Count("u", 0)]
                
for c in letter_count("\tThe dog ran into \n\tthe purple fence.", "\n aeioutnh\t"):
    print(c)
