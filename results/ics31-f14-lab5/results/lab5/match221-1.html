def Restaurant_str(self: Restaurant) -> str:

        "Phone:    " + self.phone + "\n" +

def Restaurant_get_info() -> Restaurant:

    return Restaurant(
        input("Please enter the restaurant's name:  "),

def Restaurant_change_price(r: Restaurant, percent_change: float) -> Restaurant:
    ''' Takes in a Restaurant and a number representing a percentage change in price
    and returns a Restaurant with the same attributes as the input parameter whose price
    attribute has been modified by the percentage change input '''
    return Restaurant(r.name, r.cuisine, r.phone, Menu_change_price(r.menu, percent_change))

def Restaurant_is_cheap(r: Restaurant, compare_price: float) -> bool:
    ''' Takes in a Restaurant and a comparison price and returns True if the
    average price of the menu for the Restaurant is less than the comparison
    price. '''
    return Menu_average(r.menu) < compare_price

#### COLLECTION
# A collection is a list of restaurants

def Collection_new() -> list:
    ''' Return a new, empty collection
    '''
    return [ ]

def Collection_str(C: list) -> str:
    ''' Return a string representing the collection
    '''
    s = ""
    for r in C:
        s = s + Restaurant_str(r)
    return s

def Collection_search_by_name(C: list, name: str) -> list:
    """ Return list of Restaurants in input list whose name matches input string.
    """
    result = [ ]
    for r in C:
        if r.name == name:
            result.append(r)
    return result
    # alternative (using a list comprehension):
    # return [r for r in C if r.name == name]

def Collection_add(C: list, R: Restaurant) -> list:
    """ Return list of Restaurants with input Restaurant added at end.
    """
    C.append(R)
    return C

def Collection_remove_by_name(C: list, name: str) -> list:
    """ Given name, remove all Restaurants with that name from collection.
    """
    result = [ ]
    for r in C:


        if r.name != name:
            result.append(r)
    return result
    #    Alternative:
    #    return [r for r in self.rests if r.name != name]

def Collection_change_price(C: list, percent_change: float) -> list:
    ''' Takes in a collection (list of Restaurant) and a number representing a percentage change in
    price and returns a new collection composed of the original Restaurants with the prices changed
    by the percentage indicated in the second input parameter '''
    raised_collection = []
    for r in C:
        raised_collection.append(Restaurant_change_price(r, percent_change))
    return raised_collection

def Collection_select_cheap(C: [Restaurant], lowest_price: float) -> [Restaurant]:
    ''' Returns a list of the Restaurant from the input collection whose average menu price is
    less than the number parameter taken in that represents the lowest price allowed '''
    cheap_RL = []
    for rest in C:
        if Restaurant_is_cheap(rest, lowest_price):
            cheap_RL.append(rest)
    return cheap_RL


#### Dish
Dish = namedtuple('Dish', 'name price cal')

def Dish_str(d: Dish) -> str:
    ''' Takes in a Dish and returns a human-readable str with
    the dish attributes '''
    readable = d.name + ' (${:2.2f}'.format(d.price) + '): ' + str(d.cal) + ' cal'
    return readable

def Dish_get_info() -> Dish:
    """ Prompt user for fields of Restaurant; create and return.
    """
    return Dish(
        input("Please enter the dish's name: "),
        float(input("Please enter the price of the dish: ")),
        int(input("Please enter the number of calories in the dish: ")))

def Dish_change_price(d: Dish, percent_change: float) -> Dish:
    ''' Takes in a Dish object and a number representing a percentage price change
    and returns a Dish object equal to the input Dish except with the price attribute
    changed by the input percentage '''
    return (Dish(d.name, d.price + d.price * (percent_change / 100) , d.cal))

# A menu is a list of Dishes

def Menu_enter() -> list:
    ''' Continues to prompt the user to add new Dish objects to a list and
    then returns the list of user-defined Dish objects once the user indicates
    that the Menu is complete '''
    Menu = []
    while True:
        response = input("\nDo you want to add a dish to this menu (y/n)? ")
        response = response.lower()
        if response=='y':
            Menu.append(Dish_get_info())
        elif response=='n':
            return Menu
        else:
            invalid_command(response)

def Menu_str(m: list) -> str:
    ''' Creates a human-readable string for a menu
    '''
    menu_str = ""
    for d in m:
        menu_str = menu_str + Dish_str(d) + '\n'
    return menu_str

def Menu_change_price(menu: 'list of Dish', percent_change: float) -> 'list of Dish':
    ''' Takes in a list of Dish and returns a new list composed of the input Dishlist
    with the prices changed by the percentage change value taken in as the second
    parameter '''
    result = []
    for d in menu:
        result.append(Dish_change_price(d, percent_change))
    return result

def Menu_average(m: [Dish]) -> float:
    ''' Takes in a list of Dish and returns their average price '''
    average = 0.0
    if len(m) != 0:
        average = sum(Menu_prices(m)) / len(m)
    return average
            
def Menu_prices(m: 'list of Dish') -> [float]:
    ''' Takes in a list of Dish objects (a menu) and returns a list composed of only the price
    attribute for each Dish from the input list '''
    result = []
    for d in m:
        result.append(d.price)
    return result
