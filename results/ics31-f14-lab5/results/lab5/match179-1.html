Count = namedtuple('Count', 'letter number')

def letter_count(str1: str, str2: str) -> list:
    ''' Receives two string, and returns a list contains counts of certain letters in
        the first parameter. '''
    temp = []
    for x in str2:
        count = 0
        for y in str1:
            if y == x:
                count += 1
        temp.append(Count(x, count))    
    return temp

print(letter_count('The cabbage has baggage', 'abcd'))
