def Dish_str(x:Dish) -> str:
    '''Takes a dish and returns the information in a string'''
    y = str(x.name) + " ($" + str(x.price) + "): " + str(x.calories) + " cal"
    return y

def Dish_same(x:Dish, y:Dish) -> bool:
    '''Returns True if names and calories are equal and False otherwise'''
    if x.name == y.name and x.calories == y.calories:
        return True
    else:
        return False


assert Dish_same(d1, d2) == False
assert Dish_same(d3, d3) == True

def Dish_change_price(x:Dish, y:int) -> Dish:
    '''Takes a dish and a number and returns dish with price changed according to number percentage'''
    return Dish(x.name, x.price + (x.price * (y/100)), x.calories)
assert Dish_change_price(d1, 50) == Dish(name='Fried Rice', price=14.985, calories=900)
assert Dish_change_price(d3, -100) == Dish(name='Fish', price=0.0, calories=600)

Count = namedtuple('Count', 'letter number')

def letter_count(x:str, y:str) -> str:
    '''Takes a string and a number and return the counts of the string '''
    countlist = []
    for i in y:
        countlist.append(Count(i, x.count(i)))
    return countlist
assert letter_count('Treeview and elephant', 'eT a') == [Count(letter='e', number=5), Count(letter='T', number=1), Count(letter=' ', number=2), Count(letter='a', number=2)]
assert letter_count('Lee sin', 'e') == [Count(letter='e', number=2)]
