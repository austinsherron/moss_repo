def Dish_change_price(n:Dish, m:float) ->Dish:
    '''
    Takes in a dish and a number and changes the price according to
    the number (percentage)
    '''
    percentage=m/100
    return n._replace(price=n.price*percentage+n.price)

Count = namedtuple('Count', 'letter number')

def letter_count (n: str, x: str) ->list:
    '''
    Takes a word and letters and returns a list of tuples(Count)
    with the number of occurences of each letter.
    '''
    counts = []
    for i in range(len(x)):
        if x[i] in n:
            counts.append(Count(x[i], n.count(x[i])))
        else :
            counts.append(Count(x[i], 0))
            
    return counts
print(letter_count('Lean on me', 'eaox'))
