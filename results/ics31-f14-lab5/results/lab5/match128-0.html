Count = namedtuple('Count', 'letter number')

def find_letter(sentence: str, letter: str) -> Count:
    "Finds the number of times the first letter in count is in word"
    C1 = Count(letter, sentence.count(letter))
    return C1

print(find_letter('jason', 'a'))
print(find_letter('jason is so amazing and sexy', 'a'))
   


def letter_count(sentence: str, word: str) -> list:
    "Returns the counts of certain letters in the first parameter defined by the second parameter"
    result = []
    for i in range(len(word)):
        result.append(find_letter(sentence, word[i]))
    return result
