def Dish_str(x:Dish) -> str:
    ''' Return a string with the name price and calories'''

def Dish_change_price(x:Dish,y:float) -> Dish:
    ''' Takes in a Dish and  number and returns a Dish with the same parameters except for price change.'''
    return x._replace(price=x.price + y * float(.01) * x.price)

Count = namedtuple ('Count', 'letter number')

def count_letters (x: str, y: str) -> Count:
    ''' Takes a string and a single character and returns a single Count namedtuple. ''' 
    result = Count(y, x.count(y) )
    Count.letter = y
    Count.number = x.count(y)
    return result


assert count_letters ('hello', 'l') == Count(letter='l', number=2)


def letter_count (x:str,y:str) -> list:
    ''' Returns the counts of certain letters in the first parameter.'''
    result = []
    for i in range(len(x)):
        for g in range(len(y)):
            if g == i:
                result.append ( count_letters(x, y[g]) )
    return result

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3), Count(letter='c',number=1), Count(letter='d',number=0)]
