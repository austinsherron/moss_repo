def Restaurant_is_cheap(S:'Restaurant', N:float) -> bool:
    "Returns True if the average price of the Restaurant's menu is less than or equal to the number."
    Sum = 0
    for a in S.menu:
        Sum += a.price
    return Sum/len(S.menu) <= N

def Collection_select_cheap(C:[Restaurant], N:float) -> [Restaurant]:
    "Returns a list of restaurants with the average price that is cheaper than the input value"


    select_collection = []
    for r in C:
        Sum = 0

Count = namedtuple('Count', 'letter number')

def letter_count(string, chars) -> [Count]:
    "Returns a list of namedtuples with the counts of each letter in string and in chars"
    counted = []
    for c in chars:
        counted.append(Count(c, string.count(c)))
    return counted

assert letter_count('hello world', 'hlo') == [Count(letter = 'h', number = 1),
                                              Count(letter = 'l', number = 3),
                                              Count(letter = 'o', number = 2)]
