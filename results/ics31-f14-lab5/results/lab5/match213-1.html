Count = namedtuple('Count', 'letter number')
def letter_count(word:str, x:str) -> list:
    '''returns the counts of certain letters as a list of namedtuples'''
    
    result = []
    
    for i in x:
        count = 0
        for n in word:
            if n == i:
                count += 1
        result.append(Count(i,count))
    return result

assert letter_count('hello my house is here!', 'high') == [Count(letter='h', number=3), Count(letter='i', number=1), Count(letter='g', number=0), Count(letter='h', number=3)]
assert letter_count('this\n statement\t is three\n lines', '\n\te') == [Count(letter='\n', number=2), Count(letter='\t', number=1), Count(letter='e', number=5)]
