Count = namedtuple('Count', 'letter number')

def letter_count(a:str, b: str)->list:
    """takes two strings as parameters, then returns the counts of certain letters in the first parameter"""
    list1 =[]
    for x in b:
        if x in a:
           list1.append(Count(x, a.count(x)))
        else:
            list1.append(Count(x, 0))
    return list1
