def Dish_str(dish: Dish) -> str:
    '''Takes dish namedtuple and converts to string format'''
    result = (dish.name + " (" + str(dish.price) + '): ' + str(dish.calories) + " cal")
    return result

def Dish_same(dish1: Dish, dish2: Dish) -> bool:
    '''Returns boolean variable if the the two dish names
    and calories are the same'''
    return (dish1.name == dish2.name and dish1.calories == dish2.calories)
assert Dish_same(d1,d2) == False
assert Dish_same(d1,d1) == True

def Dishlist_average(DS: 'list of dishes') -> float:
    ''' Returns average price of a list of dishes'''
    prices = Dishlist_prices(DS)
    total = 0
    for i in prices:
        total += i

def letter_count(main: str,counters: str):
    '''Returns the number of counters characters in main'''
    Count = namedtuple('Count','letter number')
    countlst = []
    result = []
    for i in counters:
        new_count = Count(i,0)
        countlst.append(new_count)
    for i in countlst:
        counter = 0
        for m in main:
            if i.letter == m:
                counter += 1
        i = Count(i.letter, i.number + counter)
        result.append(i)
    return result
