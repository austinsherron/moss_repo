Count = namedtuple('Count', ' letter number')

def letter_count(target: str, keywords: str) -> list:
    """ return list of the counts of certain letters according to the target."""
    result = []
    for i in range(len(keywords)):
        result.append(Count(keywords[i],target.count(keywords[i])))
    return result


assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a',number=5), Count(letter='b',number=3),
