def Dishlist_average(dish_list: list) -> float:
    'Returns the average price of the dishes in the list'
    total_price = 0
    price_list = Dishlist_prices(dish_list)
    for i in price_list:
        total_price += i
    average_price = total_price/(len(price_list))
    return average_price

Count = namedtuple('Count', 'letter number')

def match_letter(string: str, letter: str) -> Count:
    ''' Takes a string and a letter and checks the number of times the letter apperas in the string and returns the letter
        and number of matches'''
    counter = 0
    for i in string:
        if i == letter:
            counter += 1
    return Count(letter,counter)

assert match_letter('The cabbage has baggage', 'a') == Count(letter='a', number=5)


assert match_letter('The cabbage has baggage', 'b') == Count(letter='b', number=3)

def letter_count(string_one: str, string_two: str) -> [Count]:
    ''' Takes a two strings and returns [Count] with the nunber of tines letters in string two appear in strin one '''
    count_list = []
    for i in string_two:
         count_list.append(match_letter(string_one, i))
    return count_list
    
assert letter_count('The cabbage has baggage','abcd')==[Count(letter='a', number=5), Count(letter='b', number=3), Count(letter='c', number=1), Count(letter='d', number=0)]
assert letter_count('University of California Irvine','n') ==[Count(letter='n', number=3)]
assert letter_count('l l',' ') == [Count(letter=' ', number=1)]
