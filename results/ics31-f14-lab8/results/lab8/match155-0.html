BandB = open('BedandBrekkiepoo.txt')
BandB_lines = BandB.readlines()

bedrooms_in_service = []
for lines in BandB_lines:
        command_lines = lines.split()
        command_lines[0] = command_lines[0].upper()
        if command_lines[0] == 'AB':
                bedrooms_in_service.append(command_lines[1])
        if command_lines[0] == 'BL':
                print('Number of bedrooms in service:',len(bedrooms_in_service))
                print('------------------------------------')
                for x in bedrooms_in_service:
                        print(x)
        if command_lines[0] == 'PL':
                print(lines[3:])

BandB.close()
