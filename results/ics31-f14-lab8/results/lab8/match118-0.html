"""Keeps track of the rooms that are available.

(for "add a new

BL
(for "bedroom list"). Print a list of the bedrooms currently

"""
bedrooms = {}

## True means the room is available.
## a dictionary item would be {Room: Availability}
##                             Key     Value
def addBedroom(room: str, avail=True):
    """Adds one bedroom with corresponding availability to the bedrooms dictionary.
    """
    bedrooms.update({room: avail})

def bedroomList(BD: dict) -> str:
    """Is called when cmd is BL. Returns str"""
    result = ''
    
    availableBedrooms = []
    for bedroom in bedrooms:
        if bedrooms[bedroom]: ## is True
            availableBedrooms.append(bedroom)  # bedroom is just the key
       
    result += 'Number of bedrooms in service:  {:3}\n'.format(len(availableBedrooms))
    result += '------------------------------------\n'
    for bedroom in availableBedrooms:
        result += bedroom + '\n'

    return result
    

def fileReader(filename: str):
    """Reads and inteprets a command file."""
    infile = open(filename, 'r')
    
    for line in infile:
        termlist = line.split() # line is now a list of words called termlist
        for i in range(len(termlist)):
            termlist[i] = termlist[i].lower()      # all cmds case insensitive

        cmd = termlist[0]  # is lowercase
        if cmd == '**':
            pass
        elif cmd == 'ab':
            room = termlist[1]
            addBedroom(room)
        elif cmd == 'pl':
            print(line[3:])
        elif cmd == 'bl':
            print(bedroomList(bedrooms))
