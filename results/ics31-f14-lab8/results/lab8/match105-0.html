def handle_commands(c: list, file: 'str') -> list:
    """ Accept and process commands"""
    while True:
        response = input_format(file.readlines())
        for line in response:
            wordlist = line.split(' ')
            for word in wordlist:
                if 'AB' == word:
                    r = line.strip('AB').replace(' ', '')
                    if int(r) >=1 and int(r) <=999:
                        c = Rooms_add_room(c,r)
                    else:
                        print('Sorry,', r, ',is not a room')
                elif 'BL' == word:
                    print('Number of bedrooms in service: {}\n------------------------------------'.format(Rooms_count(c)))
                    for room in c:
                        print(room)
                elif 'PL' == word:
                    print_line(line)
        break
### INPUT 
def astericks_comment(theinput: str):
    """Make astericks a comment function"""
    i = theinput.index('**')
    return theinput[:i]

def uppercase(string: str)-> str:
    '''Uppercase first two letters'''
    new = string[0:2].upper()
    return string.replace(string[0:2], new)

def input_format(thefile: [str]) -> [str]:
    '''Format the string of the file'''
    result = [ ]
    for line in thefile:
        line = uppercase(line.strip())
        if '**' in line[0:2]: 
            response = astericks_comment(line.strip())
        else:
            response = line.strip()
        result.append(response)
    return result

### DISPLAY
def print_line(line: str):
    '''print the line without the command and its whitespace'''
    print(line.strip('PL '))

### ROOMS 
def Rooms_new():
    """Starts a new collection of rooms"""
    return [ ]

def Rooms_add_room(c: [str], R:str) -> [str]:
    """Adds room to the collection"""
    c.append(R)
    return c

def Rooms_count(c: [str]) -> int:
    '''Count number of rooms in the list'''
    return len(c)
