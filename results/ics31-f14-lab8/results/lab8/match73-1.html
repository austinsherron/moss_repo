INFILE = "input_file.txt"

#**************************
#Commands
#**************************

#AB -- add bedroom
def AB(rooms:list, parameter:int):
    """Adds a bedroom to the BandB"""
    rooms.append(parameter)
#BL -- bedroom list
def BL(rooms:list):
    """Prints a list of currently available bedrooms"""
    result = "Number of bedrooms in service:{:3d}\n------------------------------------\n".format(len(rooms))
    for room in rooms:
        result += str(room) + "\n"
    
    print(result[:-1])
#PL -- print line
def PL(parameter:str):
    """Prints the text given as the parameter"""
    print(parameter)
#** -- comment, no function necessary

#**************************
#Processing/Display functions
#**************************

#commandHandler -- chooses correct command
def commandHandler(rooms:list, line:str):
    """Chooses and executes the correct command"""
    if line == "":
        return
    commandLine = line.split()
    command = commandLine[0].upper()
    parameter = " ".join(commandLine[1:])
    if command == "**":
        return
    elif command == "AB":
        AB(rooms, int(parameter))
    elif command == "BL":
        BL(rooms)
    elif command == "PL":
        PL(parameter)
    else:
        print("ERROR: Unrecognized command in input file")
    
#**************************
#Main program
#**************************

print("Output of program with given input file:\n")
rooms = []
commands = open(INFILE, "r")
for line in commands:
    commandHandler(rooms, line)
commands.close()
