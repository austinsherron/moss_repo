def BandB(s: str) -> None:
    """Main program
    """
    lines = s.split('\n')
    bedrooms = [ ]      # List of numbers (str)
    for text in lines:
        newline = text.strip()
        newline = newline[0:2].upper() + newline[2:]
        if newline[0:2] == 'AB':
            bedrooms.append(newline[2:].strip())
        if newline[0:2] == 'BL':
            print('Number of bedrooms in service:', len(bedrooms))
            print('------------------------------------')
            for n in bedrooms:
                print(n)
        if newline[0:2] == 'PL':
            print(newline[2:].strip())
    return
