bedroom_list = []

def add_bedroom(string :str):
    bedroom_list.append(str(string[-3:]))
    
def bed_list():
    print('Number of bedrooms in service: {}'.format(len(bedroom_list)))
    print('------------------------------------')
    for i in bedroom_list:
        print(i)
    #print(bedroom_list)

def print_line(string :str):
    print(string [3:])
    



def look_lines(string:str):
    string = string.lstrip()
    if 'AB' in string[0:2].upper():
        add_bedroom(string)
    elif 'BL' in string[0:2].upper():
        bed_list()
    elif 'PL' in string[0:2].upper():
        print_line(string)
    
        
for i in lists:
