menu = ''
new_list = []
def execute(c: list) -> list:
    """ Display menu, accept and process commands.
    """
    while True:
        response = input(menu)
        response2 = remove_blanks(response)
        if str.lower(response2[:2])==("ab"):
            c.append((response2[2:]))
        elif str.lower(response2)==("bl"):
            print('Number of bedrooms in service:  ' + str(len(c)))
            for i in c:
                print((i))
        elif str.lower(response[:2])==("pl"):
            print(response[3:])
        elif response[:2]=='**':
            continue
        else:
            print('Sorry not a valid response')

def remove_blanks(s: str) -> str:
    '''Removes all blanks in a string'''
    return s.replace(' ', '')
