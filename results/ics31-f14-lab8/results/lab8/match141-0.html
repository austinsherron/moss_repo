def bednb (i: str) ->None:
    bedlist = []
    text = i.split('\n')
    for i in range(len(text)):
        text[i] = text[i].strip()
        text[i] = text[i][0:2].upper() + text[i][2:]
        if text[i][0:2] == 'PL':
            print(text[i][2:].strip())
        text[i] = text[i].replace(' ', '')
        if text[i][0:2] == 'AB':
            bedlist.append(text[i][2:])
        elif text[i][0:2] == 'BL':
            print('Number of bedrooms in service:  ', len(bedlist), 
                  '\n------------------------------------')
            for b in bedlist:
                print(b)
