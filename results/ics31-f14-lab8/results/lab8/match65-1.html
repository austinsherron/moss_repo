#Stage 1


List_of_bedroom = []
def stage_one()-> None:
    infile = open('BandBI.py', 'r')
    file = infile.readlines()
    for i in file:
        new = i.strip()
        if new[0:2].upper() == ('AB'):
            AB(new)
        elif new[0:2].upper() == ('BL'):
            print('Number of bedrooms in service: ', len(List_of_bedroom))
            print('------------------------------------------')
            print(BL())
        elif new[0:2].upper() == ('PL'):
            print(new[3:])
        elif new[0:2] == ('**'):
            continue


def AB(x:str)->list:
    add_bed = x[2:].strip()
    return List_of_bedroom.append(add_bed)


def BL()->list:
    list = ''
    for i in List_of_bedroom:
        list += '\n' + i
    return list.strip()
