infile = open('bnb.txt','r')
data = infile.read()
lists=data.split('\n')
#print(lists)




bedroom_list = []

def add_bedroom(s:str):
    bedroom_list.append(str(s[-3:]))
    
def bed_list():
    print('Number of bedrooms in service: {}'.format(len(bedroom_list)))
    print('------------------------------------')
    for i in bedroom_list:
        print(i)
    #print(bedroom_list)

def print_line(s:str):
    print(s[3:])
    



def look_lines(s:str):
    s = s.lstrip()
    if 'AB' in s[0:2].upper():
        add_bedroom(s)
    elif 'BL' in s[0:2].upper():
        bed_list()
    elif 'PL' in s[0:2].upper():
        print_line(s)
    
        
for i in lists:
    #print(i)
    look_lines(i)
