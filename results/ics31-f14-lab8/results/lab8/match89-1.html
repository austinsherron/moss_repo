def read_infile()->[list]:
    'return a list of pairs of command and argument'
    file_name = input("Please input the name of the file containing your command: ")
    infile = open(file_name,'r')
    result = []    
    for line in infile:
        line = line.strip()
        command = line[:2]
        line = line.replace(command,'  ')
        argument = line.strip()
        result.append([command,argument])
    infile.close()
    return result
