infile = open('BandB1Sample.txt', 'r')
content = infile.read().split('\n')
noWS = []
for i in content:
    noWS.append(i.strip())
#print(noWS)




"""
AB
(for "add a new bedroom") followed by an integer room number (in the range 1â€“999).
    Add a new bedroom with the specified room number.
BL
(for "bedroom list"). Print a list of the bedrooms currently available.
    The input file may contain any number of these commands in any order;
    each BL command prints a list of available bedrooms based on what has been added as of that point. See the sample output below for the format of the printed bedroom list. For this stage, it doesn't matter what order the bedrooms appear in.
PL
(for "print line"), followed by any text. Simply print (or "echo") a line,
    copying the input (not counting the PL and leading whitespace) to the output.
    You'll find it useful in testing, and it's also a simple way to make the program's reports clearer or fancier.
**
Comment, followed by any text. Like comments in a program,
    comment lines don't have any effect on the program's behavior; they just serve as annotations in the command file.
"""

availableRooms = []

for line in noWS:
    #print(line)
    cmd = line[:2].upper()
    #print(cmd)
    if cmd == 'AB':
        s = line.split()
        #print(s)
        #x = line.replace(' ', '')
        #print(x)
        availableRooms.append(s[1])
        #print(availableRooms)
    elif cmd == 'BL':
        print('Number of rooms in service: {}'.format(len(availableRooms)))
        print('-'*32)
        for room in availableRooms:
            print(room)
    elif cmd == 'PL':
        splitLine = line.split()
        toPrint = ''
        for word in splitLine[1:]:
            toPrint += word + ' '
        print(toPrint)
