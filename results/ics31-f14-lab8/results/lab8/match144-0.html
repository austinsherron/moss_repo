def Main_program(file_name:str):
    '''Main program function. Opens/Reads/Closes file.'''
    f = open(file_name, 'r')
    instructions = f.readlines()
    f.close()
    Handle_commands(instructions)
    return

def Handle_commands(com_list:[str]):
    '''Follows instructions from input file and keeps track of reservations'''
    reservations = []
    for line in com_list:
        if line.split()[0].lower() == 'ab':
            reservations.append(int(line.split()[1]))
        elif line.split()[0].lower() == 'bl':
            print('Number of bedrooms in service: {:2}'.format(len(reservations)))
            print(36*'-')
            for br in reservations:
                print(br)
        elif line.split()[0].lower() == 'pl':
            print(' '.join(line.split()[1:]))
    return
