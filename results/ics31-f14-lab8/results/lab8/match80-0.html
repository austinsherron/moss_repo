##############Bedroom Managing System#################


infile = open(input('Please enter the input directory'))
outfile = open('Bed I.txt', 'w')
l = infile.readlines()
rooms = []

def menu():
    '''Main Menu

PL - print the lines

BL - print a blankline

AB - Add bedroom to the system

q - Abort and exit'''
    c == input('Please make selection')
    if c == 'PL':
        for i in l:
            outfile.write(i)
    if c == 'BL':
        outfile.write('\n')
    if c == 'AB':
        while:
            n = input('Input the room number')
            rooms.append(n)
            if n == '':
                break
        for r in rooms:
            outfile.write(r + '\n')
    if c == 'q'
        print('Good Bye!\n')
    
    else:
        print(IndexError)

menu()

outfile.close()
infile.close()
