COMMAND_FILE_NAME = 'BandB_command.txt'

Bedroom = namedtuple('Bedroom', 'number occupied')
Command = namedtuple('Command', 'command data')

bedroom_list = []

def print_banner(s:str, char: str = '*', indent: bool = True, num_of_char: int = 35) -> str:
    '''
    *********************************************
    PRINT_BANNER


    ---------------------------------------------
    Description:
    Prints a banner with a message

    PARAMETERS:
    s: str    -   message

    RETURNS:
    a string of the output
    *********************************************
    '''
    # make all whitespace a single space
    s = ' '.join( s.split() )

    # top of banner
    string = char * num_of_char + '\n'

    
    if indent:
        string += '\t'

    # crappy word wrap
    if len(s) > num_of_char:
        for i in range( len(s) ):
            if (i+1) % num_of_char == 0:
                s = s[:i+1] + '\n' + s[i+1:]
            
    string += s +'\n' + char * num_of_char
    print(string)
    return string


def str_to_Command( s: str) -> Command:
    '''
    *********************************************
    STR_TO_COMMAND
    ---------------------------------------------
    Description:
    takes a string and converts it to a command
    
    PARAMETERS:


    s: str - string to be converted

    RETURNS:
    Command
    *********************************************
    '''
    s = s.strip()


    return Command( s[:2].upper(), s[2:].strip())
    
def Command_str( c: Command ) -> str:
    '''
    *********************************************
    COMMANDS_STR
    ---------------------------------------------
    Description:
    takes a command and returns a string
    
    PARAMETERS:


    c: Command - command to be converted 

    RETURNS:
    str
    *********************************************
    '''
    result = ''
    result.append('Command = ' + c.command + '\n') 
    result.append('Data = ' + c.data)
    return result
    



def get_commands_from_file( file: str ) -> [Command]:
    '''
    *********************************************
    GET_COMMANDS_FROM_FILE
    ---------------------------------------------
    Description:
    returns a list of Commands from a file
    
    PARAMETERS:
    file: str - file name

    RETURNS:
    [Command]


    *********************************************
    '''
    command_file = open(COMMAND_FILE_NAME, 'r')
    command_lines = []
    
    # read all the commands
    for lines in command_file.readlines():
        command_lines.append( str_to_Command(lines) )
        
    command_file.close()
    return command_lines



def handle_commands( CL: [Command] ) -> None:
    '''
    *********************************************
    HANDLE_COMMANDS


    ---------------------------------------------
    Description:
    takes a list of commands and executes each one
    
    PARAMETERS:
    CL: [Command] - command list

    RETURNS:


    none
    *********************************************
    '''
    for c in CL:
        if c.command == 'AB':
            bedroom_list.append( int(c.data) )
        elif c.command == 'BL':
            size = len(bedroom_list)
            print('Number of bedrooms in service: ' + str( size ) )
            print('------------------------------------')
            for r in bedroom_list:
                print(r)
        elif c.command == 'PL':
            print(c.data)
        else:
            continue

handle_commands( get_commands_from_file( COMMAND_FILE_NAME))
