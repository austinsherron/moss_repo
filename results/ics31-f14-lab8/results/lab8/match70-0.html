def run_program(fname: 'name of file'):
    '''runs the B and B program identifying commands in a file'''
    file = open(fname)
    flist = file.readlines()
    Bedrooms = []
    for line in flist:
        split = line.split()
        for i in range(len(split)):
            if split[i].upper() == 'PL':
                print(' '.join(split[i+1:]))
            elif split[i].upper() == 'AB':
                Bedrooms.append(split[i+1])
            elif split[i].upper() == 'BL':
                print('Number of bedrooms in service:  ' + str(len(Bedrooms)) + '\n' +
                      '------------------------------------')
                for room in Bedrooms:
                    print(room)
    
            
run_program('BandBinput.txt')
