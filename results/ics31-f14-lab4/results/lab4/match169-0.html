def test_number(n:int, s:str) -> bool:
    "Returns True if the number has the property indicated by the string, and False if it doesn't."
    if n < 0 and s == 'negative':
        return True
    elif n > 0 and s == 'positive':
        return True
    elif n % 2 == 0 and s == 'even':             
        return True
    elif n % 2 != 0 and s == 'odd':
        return True
    else:
        return False  
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(s:str) -> str:


    '''Takes a string and returns a string containing all the
    characters in the parameter string that are not vowels.'''
    result = ''
    for i in s:
        if i not in 'aeiouAEIOU':
            result = result + i
    return result
assert nonvowels('!ba') == ('!b')
assert nonvowels('^&*tag') == ('^&*tg')
assert not nonvowels('*Hi') == ('i')
assert not nonvowels('Hello*') == ('eo')

def hide_vowels(s:str) -> str:


    '''takes a string and returns a string in which every vowel in the parameter
     is replaced with a hyphen ("-") and all other characters remain unchanged. '''
    new_str = ''
    for i in s:
        if is_vowel(i):
            new_str = new_str + '-'
        else:
            new_str = new_str + i
    return new_str
assert hide_vowels('cat') == 'c-t'
assert hide_vowels('DOG!') == 'D-G!'
assert hide_vowels('AeI') == '---'
assert not hide_vowels('COMPUTER@') == 'C-MPUTER-'

def Restaurant_change_price(R:Restaurant,P:int) -> Restaurant:
    '''Takes two arguments, a Restaurant object and a number, and returns a
       Restaurant object with the same contents as the parameter except that
       the price has been increased by the specified number.'''
    x = R._replace(price = R.price + P)
