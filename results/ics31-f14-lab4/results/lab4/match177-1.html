def test_number(x:int, y:str) -> bool:
    '''returns if the number entered is even, odd, negative, or positive as indicated'''
    if y == 'even':
        return (x%2 == 0)
    elif y == 'odd':
        return (x%2 == 1)
    elif y == 'positive':
        return (x >= 0)
    elif y == 'negative':
        return (x < 0)
    else:
        return False

def is_vowel(x:str) -> bool:
    ''' returns str if it is vowel '''
    vowel = ['a', 'A', 'e', 'E', 'i', 'I', 'o', 'O', 'u', 'U']
    return (x in vowel)

assert is_vowel('U') == True

def non_vowels(x:str) -> str:
    '''returns str if str is not a vowel'''
    result = ''
    for i in x:
        if is_vowel(i) == False:
            result = result + i
    return result

def consonants(z:str) -> str:
    '''returns str without the vowels'''
    new_word = ''
    for x in z:
        if is_vowel(x) == False and x.isalpha():
            new_word = new_word + x
    return new_word

assert consonants('canvas') == 'cnvs'
assert consonants('hallelujah') == 'hllljh'
assert not consonants('faun') == 'au'
assert not consonants('museum') == 'museum'

def select_letters(x:str, y:str) -> str:
    '''returns all the vowels in a str if specified character is v and consonants if c'''
    v_word = ''
    c_word = ''
    if x == 'v':
        for i in y:
            if is_vowel(i) == True:
                v_word = v_word + i
        return v_word
    elif x == 'c':
        c_word = y
        return consonants(c_word)
    else:
        return ''

assert select_letters('c', 'bye') == 'by'

def hide_vowels(word:str) -> str:
    '''returns string with all vowels replaced with a dash: '-' '''
    new_word = ''
    for i in word:
        if is_vowel(i) == False:
            new_word = new_word + i
        elif is_vowel(i) == True and i.isalpha():
            i = '-'
            new_word = new_word + i
    return new_word

assert hide_vowels('sayonara') == 's-y-n-r-'
assert hide_vowels('kitten') == 'k-tt-n'
assert not hide_vowels('hyung') == 'hyung'
assert not hide_vowels('unnie') == 'u--ie'

def average_price(rest:list) -> float:
    '''returns the average price of the best dishes at those restaurants'''
    avg = 0.0
    num = len(rest)
    for i in rest:
        avg = avg + i.price
    return (avg / num)
