        return(n<0)

    in the parameter string that are not vowels.'''

def consonants (s:str) -> str:
    '''Takes a string and returns a string containing all the letters in
    the parameter string that are not vowels.'''
    result=''
    for a in s:
        if not is_vowel(a):
            result = result +(a)
    return(result)

def hide_vowels (s:str) -> str:
    '''Takes a string and returns a string in which every vowel is
    replaced with a hyphen.'''
    result = ''
    for a in s:
        if is_vowel(a):
            result = result+'-'
        else:
            result = result+(a)
    return result 
assert(hide_vowels('Hello world.')=='H-ll- w-rld.')

def average_price (s:list) -> float:
    '''Takes a list of restaurants and returns the average price of
    those restaurants.'''
    x = 0.0

        x = (n.price + x)
