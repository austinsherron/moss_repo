def match_area_codes(area_code: list, phone: list) -> list:
    result = []
    for code in area_code:
        for num in phone:
            if code == num[1:4]:
                result.append(num)
    return result

import tkinter
window = tkinter.Tk()    
def create_rectangle_from_center(canvas:tkinter.Canvas, x: int, y: int, height: int, width: int):
    W = x - width/2
    X = y - height/2
    Y = x + width/2
    Z = y + height/2
    canvas.create_rectangle(W,X,Y,Z)
canvas = tkinter.Canvas(window, width=1000, height=1000)  
canvas.pack()
