def condition_test(n:int) -> bool:
    "Test number for even/odd"
    return n%2 == 0
assert condition_test(40) == True
assert condition_test(57) == False

def test_number(n:  int, condition : str):
    "Tests number for given condition(even/odd or positive/negative)"
    if condition == 'even' and condition_test(n) == True:
        return True

    elif condition == 'odd' and condition_test(n) != True:
        return True

    elif condition == 'positive' and n > 0:
        return True

    elif condition == 'negative' and n < 0:
        return True

    else:
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd') 
assert test_number(33, 'positive') 
assert not test_number(100, 'negative')

def matching_area_codes(ACL: list , PNL: list) -> str:
    "Returns the numbers in PNL that match the code given in ACL"
    matched_list = []
    for i in ACL:
        for a in PNL:
            if i == a[1:4]:
                matched_list.append(a)
    return [matched_list]

def is_vowel(char : str) -> bool:
    "Returns True if parameter is a vowel; else False"
    if char in ['a','e','i','o','u','A','E','I','O','U']:
        return True
    else:
        return False
assert is_vowel('A')
assert is_vowel('a')
assert not is_vowel('b')

def print_nonvowels(word : str) -> list:
    "Returns a list of characters from parameter that are not vowels"
    no_vowel = []
    for i in word:
        if i not in ['a','e','i','o','u','A','E','I','O','U']:
            no_vowel.append(i)
    return no_vowel
assert print_nonvowels('Hello!') == ['H', 'l', 'l', '!']
assert print_nonvowels('HELP!!') == ['H', 'L', 'P', '!', '!']

def nonvowels(word : str) -> str:
    "Returns parameter without vowels"
    result = ''
    no_vowel = print_nonvowels(word)
    for i in no_vowel:
            result += i
    return result


assert nonvowels('Hello!') == 'Hll!'

def consonants(word : str) -> str:
    "Returns parameter with only consonants"
    x = 'bcdfghjklmnpqrstvwxyz'
    y = x.upper()
    result = ''
    for i in word:
        if (i in x) or (i in y):
            result += i
    return result
assert consonants('BaNaNaS!') == 'BNNS'
assert consonants('Aooo...') == ''
assert not consonants('Black') == 'BLCK'

# Part d.5

def vowels(word : str) -> str:
    "Returns parameter with only vowels"
    result = ''
    vowels = 'aeiouAEIOU'
    for i in word:
        if i in vowels:
            result += i
    return result
assert vowels('Hello!') == 'eo'
assert vowels('') == ''

def select_letters(char : str, word : str) -> str:
    "Returns parameter with only vowels or only consonants depending on user input"
    if (char in 'v') or (char in 'V'):
        return vowels(word)
    elif (char in 'c') or (char in 'C'):
        return consonants(word)
    else:
        return ''
assert select_letters('v', 'Hello!') == 'eo'
assert select_letters('c', 'Hello!') == 'Hll'

# Part d.6

def hide_vowels(word : str) -> str:
    "Returns paramter with all vowels replaced with '-'"
    result = ''
    vowels = 'aeiouAEIOU'
    for i in word:
        if i in vowels:
            result = result + '-'
        else:
            result = result + i
    return result
assert hide_vowels('Hello!') == 'H-ll-!'
assert hide_vowels('Bananas') == 'B-n-n-s'
assert hide_vowels('') == ''

    "Returns sum of price fields in given list"
    price_list = []
