def test_number(n: int, n2: str) -> bool:
    '''checks if n is even/odd/positive/negative'''
    if n%2 == 0 and n2 == 'even':
        return True
    elif n%2 == 1 and n2 == 'odd':
        return True
    elif n > 0 and n2 == 'positive':
        return True
    elif n < 0 and n2 == 'negative':
        return True
    else:
        return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(n: str) -> str:
    '''return string with only nonvowel characters'''
    result = ''
    for i in n:
        if is_vowel(i)==False:
            result = result+i
    return result

def consonants(n:str) -> str:
    '''return string with only consonants'''
    result = ''
    for i in n:
        if is_vowel(i)==False and i in 'abcdefghijklmnopqrstuvwxyz':
            result = result+i
    return result
assert consonants('.abc.')=='bc'

def select_letters(n:str, m:str)->str:
    '''if n == v return a string with only vowels if n== c return a string with only consonants'''
    result = ''
    if n == 'v':
        for i in m:
            if is_vowel(i) == True:
                result = result + i
        return result
    elif n == 'c':
        return consonants(m)
    else:
        return ''

assert select_letters('v', 'facetiously')== 'aeiou'
assert select_letters('c', 'facetiously')== 'fctsly'

def hide_vowels(n:str)->str:
    '''replaces vowels with -'''
    result = ''
    for i in n:
        if is_vowel(i)==True:
            result = result + '-'
        else:
            result = result+i
    return result
print(hide_vowels('facetiously'))
