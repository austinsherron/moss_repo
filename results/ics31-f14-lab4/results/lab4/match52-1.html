def consonants (word:str)->str:
    # This function is to return all the consonants from a given word into a string
    result=''
    consonant = 'qwrtypsdfghjklzxcvbnm'
    for i in range(len(word)):
        for x in range(len(consonant)):
            if word[i].lower() == consonant[x]:   #.lower is used because all the upper case letter can be detected once all the changed in lower case
                result=result+word[i]
    return result

#These are some test cases

def vowels (word:str)->str:
    # This function is to return all the vowles from a given word into a string
    result=''
    consonant = 'aeiou'
    for i in range(len(word)):
        for x in range(len(consonant)):
            if word[i].lower() == consonant[x]:
                result=result+word[i]
    return result

    if requirement=='v':

def hide_vowels (word:str)->str:
    # This function is to hide all the vowel and change it to '-'
    result=''
    for i in range(len(word)):
        if is_vowel(word[i]):
            result=result+'-'
        else:
            result=result+word[i]
    return result
