def test_number(num: int, test: str) -> bool:
    ''' if the number has the property of test, then returns true, else false '''
    if test == 'even' and num%2 == 0:
        return True
    elif test == 'odd' and num%2 == 1:
        return True
    elif test == 'positive' and num >= 0:
        return True
    elif test == 'negative' and num < 0:
        return True
    else:
        return False

assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True

def hide_vowels (s:str) -> str:
    '''takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen '''
    c = ''
    for i in s:
        if i in vowels:
            c = c + '-'
        else:
            c = c + i
    return c
assert hide_vowels ('anteater') == '-nt--t-r'
