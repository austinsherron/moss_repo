def test_number(n:int, s:str) -> bool:
    '''evaluate if the input number as the property indicated by
the input string'''
    if s == 'even':
        return n%2 == 0
    elif s == 'odd':
        return n%2 != 0
    elif s == 'positive':
        return n > 0
    elif s == 'negative':
        return n < 0
    
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
print(test_number(14, 'even'))

def consonants(word:str) -> str:
    '''take the parameter and return all the consonants'''
    consonants = ['b','c','d','f','g','h','j','k','l','m','n','p','q','r','s','t','v','w','x','y','z']
    result = ''
    for i in word:
        if i in consonants:
            result = result + i
    return result

assert consonants('abcdefg') == 'bcdfg'
assert consonants('bayonetta') == 'byntt'
assert consonants('devil may cry') == 'dvlmycry'

def select_letters(ident:str, word:str) -> str:
    '''if first parameter is v, return vowels in second parameter; if 1st parameter
is c, return consonants in 2nd parameter; if 1st parameter is nether v nor c,
return empty str'''
    result = ''
    if ident == 'c':
        return consonants(word)
    elif ident == 'v':
        for i in word:
            if is_vowel(i) == True:
                result = result + i
        return result
    else:
        return result

assert select_letters('v', 'halo') == 'ao'
assert select_letters('c', 'halo') == 'hl'
assert select_letters('a', 'haol') == ''
print(select_letters('v', 'halo'))
print(select_letters('v', 'bayonetta'))

def hide_vowels(char:str) -> str:
    '''replace all vowels in parameter with -'''
    L = list(char)
    for i in range(len(L)):
        if is_vowel(L[i]) == True:
            L[i] = '-'
    return ''.join(L)

assert hide_vowels('bayonetta') == 'b-y-n-tt-'
assert hide_vowels('devil may cry') == 'd-v-l m-y cry'
print(hide_vowels('bayonetta'))
print(hide_vowels('devil may cry'))
print(hide_vowels('you may be able to understand the sentences even with all the vowels hidden'))

def alphabetical_names(L:list) -> list:
    '''return a list of names of restaurants in alphabetical order'''
    L_names = []
    for i in range(len(L)):
        L_names.append(sorted(L)[i].name)

def average_price(rest:list) -> float:
    '''returns the average price of restaurants'''
    L_price = []
    for i in range(len(rest)):
        L_price.append(rest[i].price)
        ave = sum(L_price) / len(rest)
    return ave
