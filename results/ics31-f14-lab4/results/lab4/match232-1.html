def test_number (number,no_type):
    '''Function takes a number and a type to return if it is true or false. (NUMBER, (odd/even/positive/negative).'''
    if no_type == 'even':
        return number%2 == 0
    elif no_type == 'odd':
        return number%2 > 0
    elif no_type == 'positive':
        return  number > 0
    elif no_type == 'negative':
        return number < 0

def nonvowels (word:str) -> str:
    '''Takes a string and returns characters that are not vowels'''
    result = ''
    for k in word:
        if is_vowel(k) == False:

def select_letters(choice:str,string_input:str) -> str:
    '''Input in choice v for vowels and c for consonants. The function will output only the selected type of letter.'''
    if choice == 'v':
        result = ''
        for hold in string_input:
            if is_vowel(hold) == True:
                result += hold
        return result

    elif choice == 'c':
        return nonvowels(string_input)

    else:
        result = ''
        return result

def hide_vowels (hide:str) -> str:
    '''Takes a string and returns a string where all vowels in the parameter
are replaced with a hyphen'''
    for k in hide:
        if is_vowel(k):
            hide = hide.replace(k,'-')
    return hide

import tkinter

def create_rectangle_from_center(x, y, height, width:int):
    x = x - (width/2)
    y = y - (height/2)
    x2 = x + (width/2)
    y2 = y + (height/2)
    return my_canvas.create_rectangle(x, y, x2, y2)
