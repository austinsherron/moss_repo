def test_number (x: int, y: str):
    if (x % 2 == 0) and (y == 'even'):
        return True
    elif (x % 2 == 1) and (y == 'odd'):
        return True
    elif (x > 0) and (y == 'positive'):
        return True
    elif (x < 0) and (y == 'negative'):
        return True
    else:
        return False
    
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def consonants (x: str) -> str:
    '''returns consonants in a given string'''
    result = ''
    for a in x:
        if a in ['b','c','d','f','g','h','j','k','l','m','n','p','q','r','s','t','v','q','x','y','z']:
            result += a
    return(result)

assert consonants('anteater') == 'nttr'
assert consonants('tests') == 'tsts'
assert consonants("children's") == "chldrns"

def hide_vowels (x: str) -> str:
    result = ''
    for i in range(len(x)):
        if x[i] in ['a', 'e', 'i', 'o', 'u']:
            result += ('-')
        else:
            result += (x[i])
    return result

def create_rectangle_from_center(x: int, y: int, h: int, w: int):
    import tkinter             
    my_window = tkinter.Tk()   

    my_canvas = tkinter.Canvas(my_window, width=500, height=500) 
    my_canvas.pack()
    
    x1 = x - (.5 * w)
    y1 = y - (.5 * h)
    x2 = x + (.5 * w)
    y2 = y + (.5 * h)
    
    my_canvas.create_rectangle(x1, y1, x2, y2)  

    tkinter.mainloop()
