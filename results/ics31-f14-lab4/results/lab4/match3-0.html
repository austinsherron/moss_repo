def test_number (num:int, term:str)-> bool:
    if term == 'even':
        if num % 2 == 0:
            return True
        else:
            return False
    if term == 'odd':
        if num % 2 != 0:
            return True
        else:
            return False
    if term == 'positive':
        if num > 0:
            return True
        else:
            return False
    if term == 'negative':
        if num < 0:
            return True
        else:
            return False

def consonants (word:str)->str:
    letters = ''
    for char in word:
        if char not in 'aeiouAEIOU':
            if char in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
                letters = letters + char
    return letters
