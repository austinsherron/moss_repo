def consonants(x:str)-> str:
        'return a string of consonants'
        consonant=['Q','W','R','T','Y','P','S','D','F','G',
                   'H','J','K','L','Z','X','C','V','B','N','M',
                   'q','w','r','t','y','p','s','d','f','g',
                   'h','j','k','l','z','x','c','v','b','n','m']
        consonants_list=[]
        for i in range(0,len(x)):
                if x[i] in consonant:
                        consonants_list.append(x[i])
        string=''
        for j in range(0,len(consonants_list)):
                string = string + consonants_list[j]
                        
        return string

def hide_vowels(x:str)->str:
        '''returns a string in which every vowel in the paramter is replaced
           with a hyphen (-). Other characters left unchanged'''
        vowels=['A','E','I','O','U','a','e','i','o','u']
        x=list(x) # converts our string to a list since strings are immutable
        string =''
        for i in range(0,len(x)):
                if x[i] in vowels:
                        x[i]='-' 
        for j in range(0,len(x)):
                string = string + x[j]
        return string

assert hide_vowels('pie')=='p--'
assert hide_vowels('')==''
assert hide_vowels('book')=='b--k'
assert hide_vowels('testing')=='t-st-ng'
assert hide_vowels('sky')=='sky'
