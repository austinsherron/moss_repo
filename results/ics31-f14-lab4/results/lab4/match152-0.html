def consonants(string: str) -> str:
    '''checks each character in the string and outputs a string with all the
        consonants'''
    cons = ''
    for i in string:
        if not is_vowel(i):
            if i in 'abcdefghijklmnopqrstuvwxyz':
                cons += i
    return cons

def select_letters(selection: str, string: str) -> str:
    '''gets a selection and a string from the user and if the selection is 'v'
        outputs a string with all the vowels and if it is 'c' outputs a string
        with all the consonants'''
    newString = ''
    if(selection.lower() == 'v'):
        for i in string:
            if(is_vowel(i)):
                newString += i
    elif(selection.lower() == 'c'):
        newString = consonants(string)
    return newString

def hide_vowels(string: str) -> str:
    ''' Replace all vowels with hyphen "-" and return string '''
    result = ''
    for i in string:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result

assert hide_vowels('Banana')
assert hide_vowels('Anteater') == '-nt--t-r'
