def consonants(x: str) -> str:
    '''Takes a string and returns a string containing all the letters in the parameter string that are not vowels'''
    result = ''
    for i in x:
        if i.isalpha() and not is_vowel(i):
            result += i
    return result
assert consonants('function') == 'fnctn'
assert consonants('rhythm') == 'rhythm'
assert consonants('hello') == 'hll'

def vowels(x: str)-> str:
    '''Takes a string and returns a string containing all the letters in the parameter string that are vowels'''
    result = ''
    for i in x:
        if i.isalpha() and is_vowel(i):
            result += i
    return result

def select_letters(x: str, y: str) -> str:
    '''Takes two parameters, both strings, and returns a string'''
    if x == 'c':
        return consonants(y)
    elif x == 'v':
        return vowels(y)
    if x != 'c' and x != 'v':
        return ''

def hide_vowels(x: str) -> str:
    '''takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters remain unchanged'''
    l=''
    for i in range(len(x)):
        if is_vowel(x[i]):
            l=l+'-'
        else:
            l=l+x[i]
    return l
assert hide_vowels('hello')=='h-ll-'
