def test_number(number:int,word:str )-> bool:
    '''returns True if number has the property indicated by the string'''
    if word == 'odd':
        return number % 2 == 1
          
    if word == 'even':
        return number % 2 == 0
            
    if word == 'positive':
        return number >= 0
         
    if word == 'negative':
        return number < 0
            
    return False

# consonants:  string:str
def consonants(string:str) -> str:
    '''prints all consonants in string'''
    con='bcdfghjklmnpqrstvwxyz'
    results=''
    for i in string:
        if i.lower() in con:
            results= results+ i
    return results
assert consonants('abcdefg') == 'bcdfg'
assert consonants('AbCdEfGhiJklmn') == 'bCdfGhJklmn'
assert consonants('\234AbCdefasdg') == 'bCdfsdg'

# select_letters: string1:str string2:str -> str
def select_letters(string1:str, string2:str) -> str:
    ''' If the first string is v, it will return all the vowels.
    If the string is c, it will return all the consonants. '''
    con='bcdfghjklmnpqrstvwxyz'
    vowels='aeiou'
    result=''
    if string1 == 'c':
        for i in string2:
            if i.lower() in con:
                result= result+ i
        return result
    if string1 == 'v':
        for i in string2:
            if i.lower() in vowels:
                result= result + i
        return result
assert select_letters('c','abcdefg') == 'bcdfg'
assert select_letters('v','abcdeFGHI') == 'aeI'

# hide_vowels:  string:str -> str
def hide_vowels(string: str) -> str:
    '''Returns a string with vowels in string replaced with hypens'''
    vowels='aeiou'
    result=''
    for i in string:
        if i.lower() in vowels:
            result= result + '-'
        else:
            result = result + i
    return result
assert hide_vowels('abcdei') == '-bcd--'
assert hide_vowels('ABCDIOULZ1') == '-BCD---LZ1'

# alphabetical_names  rest:list-> list
def alphabetical_names(rest: list) -> list:
    ''' Returns a list of names of the restaurants in alphabetical order'''
    new_list=[]
    q= sorted(rest)
    for i in q:
        new_list.append(i.name)
    return new_list
