def test_number(n: int, s: str) -> bool:
    "Inputs a number then checks if value is true or false based on string statements"
    if s == 'even':
       v = (n % 2 == 0)
    if s == 'odd':
       v = (n % 2 != 0)
    if s == 'positive':
        v = (n > 0)
    if s == 'negative':
        v = (n < 0)
    return v

def consonants(n:str) -> str:
    "Prints out the consonants within a string all in one string"
    result = ''
    lconsonant = 'bcdfghjklmnpqrstvwxyz'
    uconsonant = 'BCDFGHJKLMNPQRSTVWXYZ'
    for s in n:
        if (s in lconsonant) or (s in uconsonant):
            result += s
    return result
assert consonants('-IpkLO+RT') == 'pkLRT'
assert consonants('gh94uirfd') == 'ghrfd'
assert consonants('') == ''

def hide_vowels(n: str)->str:
    "Creates a new string list and hides vowels by replacing it with a hypen"
    result = ''
    for x in n:
        if is_vowel(x):
            result += '-'
        else:
            result += x
    return result
assert hide_vowels('Hello') == 'H-ll-'
