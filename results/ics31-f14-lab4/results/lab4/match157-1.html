def test_number(x: int, y: str)-> bool:
    """Tests a number to see if it is even, odd, negative, or positive"""
    if y == 'odd':
        return x % 2 != 0
    if y == 'even':
        return x % 2 == 0
    if y == 'positive':
        return x >= 0
    if y == 'negative':
        return x < 0

print(test_number(4, 'even'))
print(test_number(7, 'odd'))
print(test_number(-100, 'negative'))
print(test_number(-399, 'positive'))
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(x:str)->bool:
    vowel = ["a", "e", "i", "o", "u", "A", "E", "I", "O", "U"]
    if x in vowel:
        return True
    else:
        return False
    
print(is_vowel("E"))
print(is_vowel("y"))
assert(is_vowel("E")) == True
assert(is_vowel("y")) == False

def nonvowels(x:str)->str:
    '''returns all nonvowels in a string given as a string'''
    result = ""
    for i in range(len(x)):
        if is_vowel(x[i]) == False:
            result = result + x[i]
    return result

assert nonvowels("dada") == "dd"

def consonants(x:str)->str:
    '''Returns all consonants in a string given as a string'''
    conso = ["B", "C", "D", "F", "G", "H", "J", "K", "L", "M", "N", "P", "Q", "R", "S", "T", "V", "W", "X", "Y", "Z", "b", "c", "d", "f", "g", "h", "j", "k", "l", "m", "n", "p", "q", "r", "s", "t", "v", "w", "x", "y", "z"]
    res = ""
    for i in x:
        if i in conso:
            res = res + i
    return res
print(consonants("HdasFEdea"))
assert consonants("hH") == "hH"
assert not consonants("hH") == "hdsaH"

def hide_vowels(x: str)->str:
    '''Returns a string with all vowels replaced with a hypen'''
    result = ""
    for i in range(len(x)):
        if is_vowel(x[i]) == False:
            result = result + x[i]
        else:
            result = result + '-'
    return result
print(hide_vowels('facetious'))
print(hide_vowels('hide vowels'))
