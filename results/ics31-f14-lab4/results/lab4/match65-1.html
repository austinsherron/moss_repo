def nonvowels(a:str):
    'takes a str and returns a new str containning the non vowels'
    letters = ""
    for i in a:
        if not is_vowel(i):
              letters = letters + i

def consonants(n:str)->str:
    'takes a string and returns another string containning only consonants'
    x = ""
    for i in n:
        if i.isalpha() and not is_vowel(i):
            x = x + i


    return x
print(consonants('abc56284782746;;['))
    

def select_letters(a:str,b:str)->str:
    'takes a string and returns either vowels or consonants in a new string'
    ltr = ''
    if a == 'v':
        for i in b:
            if is_vowel(i):
                ltr = ltr + i
    if a == 'c':

    return ltr

def hide_vowels(a:str)->str:
    'takes a string and returns a string without vowels'
    word = ''
    for i in a:
        if is_vowel(i):
            word = word + '-'
        else:
            word = word + i
    return word
