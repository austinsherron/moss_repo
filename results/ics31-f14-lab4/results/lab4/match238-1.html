def test_number (x: int, y: str) -> bool:
    """tests a number"""
    if x%2 == 0 and y == 'even':
        return True
    elif x%2 != 0 and y == 'odd':
        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative':
        return True
    else:
        return False
assert test_number(16, 'even')
assert not test_number(102, 'odd')
assert test_number (33, 'positive')
assert not test_number (101, 'negative')

def is_vowel (x: str) -> bool:
    ''' Checks to see if character is a vowel. '''
    return (x in ['a','e','i','o','u', 'A', 'E', "I", "O", "U"])

consonantsLower = ['q', 'w', 'r', 't','y','p','s','d','f','g','h','j','k','l','z','x','c','v','b','n','m']
consonantsUpper = []
for i in range(len(consonantsLower)):

def consonants (x:str) -> str:
    ''' Returns consonants. '''
    result = ''
    
    for i in range(len(x)):
        if x[i] in consonantsLower or x[i] in consonantsUpper:
            result += x[i]
    return result

    "Replaces vowels with a hyphen"
