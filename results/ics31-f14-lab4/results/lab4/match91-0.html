def test_number( x: int, s: str ) -> bool:
    ''' Returns True if the number property corresponds to the string property '''
    if s == 'Even':
        return x%2 == 0
    elif s == 'Odd':
        return x%2 != 0
    elif s == 'Positive':
        return x >= 0
    else:
        s == 'Negative'
        return x < 0 

assert test_number(14, 'Even')
assert not test_number(100, 'Odd')
assert test_number(33, 'Positive')
assert not test_number(100, 'Negative')

def nonvowels(s: str) -> str:
    ''' Returns a string returning all characters in the parameter string that are NOT vowels '''
    result = ''
    for i in s:
        if is_vowel(i) == False:

    return result

def vowels( s: str ) -> str:
    ''' Returns a string containing all vowels in the parameter string '''
    result = ''
    for i in s:
        if i in ['a','e','o','u','A','E','I','O','U']:
            result += i
    return result
            
def select_letters( s1: str, s2: str ) -> str:
    ''' Returns vowels if s1 is v, consonants if s2 is c, and nothing if the first parameter is anything else '''
    result = ''
    for i in s2:
        if s1 == 'v':
            result += vowels(i)
        elif s1 == 'c':
            result += consonants(i)
        else:
            return result
    return result

def hide_vowels( s: str ) -> str:
    ''' Replaces vowels in a string with hyphens '''
    result = ''
    for i in s:
        if is_vowel( i ) == True:
            result += '-'
        else:
            result += i
    return result

print( hide_vowels( 'Alligator' ) )
print( hide_vowels( 'Badminton' ) )
print( hide_vowels( '?Str3b3rryFields' ) )
