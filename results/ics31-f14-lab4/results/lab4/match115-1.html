def consonants (p: str):
    'returns a string of consonants of a string disallowing number values'
    lowercase = 'bcdfghjklmnpqrstvwxyz'
    uppercase = 'BCDFGHJKLMNPQRSTVWXYZ'
    result = ''
    for n in p:
        if n in lowercase or n in uppercase:
            result = result + n
    return result

def select_letters (m: str, n: str):
    "returns vowels if 'v' or consonants if 'c' of a string"
    l = ''
    if m == 'v':
        for x in n:
            if is_vowel(x) == True:
                l = l + x
        return l
    elif m == 'c':
        return consonants(n)
