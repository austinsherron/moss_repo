def test_number (N: int, T: str) -> bool:
    '''takes as input a number and a string (one of 'even', 'odd', 'positive', or 'negative'and returns "True" or "False"'''
    if N%2 == 0 and T == 'even':
        return True

    elif N%2 == 0 and T == 'odd':
        return False

    elif N%2 == 1 and T == 'even':
        return False

    elif N%2 == 1 and T == 'odd':
        return True

    elif N>=0 and T == 'positive':
        return True

    elif N<0 and T == 'positive':
        return False

    elif N>=0 and T == 'negative':
        return False

    elif N<0 and T == 'negative':
        return True

    assert test_number(14, 'even') == 'true'
    assert not test_number(100, 'odd') == 'true'
    assert test_number(33, 'positive') == 'true'
    assert not test_number(100, 'negative') == 'true'

def hide_vowels (M: str) -> str:
    '''Takes a string and returns a string in which every vowel in the parameter is replaced by a hyphen.'''
    result_2 = ''
    for Q in M:
        if is_vowel(Q) == True:
            result_2 += '-'
