def test_number(n: int, s: str):
    '''Tests whether input is True or False'''
    if n > 0 and s == 'positive':
        return(n > 0)
    elif n > 0 and s == 'even':
        return(n % 2 == 0)
    elif n < 0 and s == 'even':
        return(n % 2 == 0)
    elif n < 0 and s == 'negative':
         return(n % 2 == 0)
    elif n < 0 and s == 'odd':
        return(n % 2 > 0)
    elif n < 0 and s == 'negative':
        return(n < 0)
    elif n > 0 and s == 'odd':
        return(n % 2 > 0)
    elif n > 0 and s == 'positive':
        return(n > 0)

def is_vowel(letter: str):
    '''takes a one-character-long

    is a vowel and False otherwise.'''
    vowels = ['a', 'A', 'e', 'E', 'i', 'I', 'o', 'O', 'u', 'U']

def consonants(cn: str):
    '''takes a string and returns a string containing
    all the letters in the parameter string that are not vowels'''


    consonant = ['b', 'B', 'c', 'C', 'd', 'D', 'f', 'F', 'g', 'G', 'h', 'H', 'j', 'J', 'k', 'l', 'L', 'm', 'M', 'n', 'N', 'p', 'P', 'q', 'Q', 'r', 'R', 's', 'S', 't', 'T', 'v', 'V', 'x', 'X', 'y', 'Y', 'z', 'Z']
    new_str = ''
    for i in cn:
        if i in consonant:
            new_str = new_str + i
    return(new_str)

def select_letters(c_or_v: str, word: str):
    """ takes two parameters, both strings, and returns a string. If
    the first parameter is 'v', it returns a string containing all
    the vowels in the second parameter; if the first parameter is
    'c', it returns a string containing all the consonants in the
    second parameter. If the first parameter is anything else, it
    returns the empty string. """
    new_word = ''
    if c_or_v == 'c':
        return consonants(word)
    if c_or_v == 'v':
        for i in word:
            if is_vowel(i):
                new_word = new_word + i 
        return(new_word)

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(word: str):
    """ takes a string and returns a string in which every vowel
    in the parameter is replaced with a hyphen ('-') and all other
    characters remain unchanged"""
    hyph = ''
    for i in word:
        if is_vowel(i):
            i = '-'
            hyph = hyph + i
        elif consonants(i):
            hyph = hyph + i
    return(hyph)
assert hide_vowels('Pickle') == 'P-ckl-'

def alphabetical_names(order: list):
    """takes a list of restaurants and returns a list of the names of
    all the restaurants in alphabetical order by name. """
    just_names = []
    alphabetical(order)
    for Restaurant in order:
        if Restaurant.name in Restaurant:
