def test_number (number:int, word:str)->bool:
    ''' Returns True if number has a property indicated by str
    amd False if it doesn't'''
    if number%2==0 and word == 'even':
        return True
    elif number%2!=0 and word == 'odd':
        return True
    elif number/-1==-number and word == 'positive':
        return True
    elif number/-1==number and word == 'negative':
        return True
    else:
        return False
    
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowel(word:str)->list:
    '''Takes a string and returns a string containing all the
    characters in the parameter string that are not vowels'''
    result=[]
    for c in word:
        if c not in 'aeiouAEIOU':
            result.append(c)
    return(result)

def hide_vowels(word:str)->str:
    '''Takes a string and returns a string in which every vowel in the
    parameter is replaced with a hyphen and all other characters remain unchanged'''
    new_word = ''
    for c in word:
        if c in 'aeiouAEIOU':
            new_word = new_word + '-'
        elif c not in 'aeiouAEIOU':
            new_word = new_word + c
    return new_word

print(hide_vowels('chicken'))

def alphabetical_names (R: list)->list:

def average_price(R:list)->float:

    of (the best dishes) of those restaurants'''
