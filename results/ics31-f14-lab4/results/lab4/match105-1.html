def test_number(n:int,s:str)-> bool:
    '''Return a number and string of either 'even', 'odd', 'positive,
       or 'negative' as True or False as appropriate'''

    if n%2 == 0 and n>0: 
        return s == 'even' or s == 'positive'
    elif n%2 == 0 and n<0: 
        return s == 'even' or s == 'negative'
    elif n%2 != 0 and n>0:
        return s == 'odd' or s == 'positive'
    elif n%2 != 0 and n<0:
        return s == 'odd' or s == 'negative'

            
assert test_number(14,'even') == True
assert not test_number (100,'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100,'negative') == True

def is_vowel(s:str)->bool:
    '''Given a one-character string, return True if character is a vowel and
       and False if not'''
    return s in 'aeiouAEIOU'

assert is_vowel('a') == True
assert is_vowel('b') == False
assert not is_vowel('U') == False
assert not is_vowel('X') == True

def nonvowels(x:str)->str:
    '''Returns all the characters in a string that are NOT vowels, given
       s string'''
    result = ''
    for i in x:
        if is_vowel(i) == False:
            result = result + i
    return result

assert nonvowels('Grace') == 'Grc'
assert nonvowels('Nick') == 'Nck'


assert nonvowels('Deadmau5') == 'Ddm5'
assert nonvowels('High5') == 'Hgh5'

def consonants(c:str)->str:
    '''Returns a string containing all the LETTERS in the string 
       that are not vowels'''
    r = ''
    L = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    for i in c:
        if i in L:
            r = r + i
    return r

assert consonants('Deadmau5') == 'Ddm'
assert consonants('High5') == 'Hgh'
