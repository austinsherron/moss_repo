def nonvowels(s:str)-> str:
    #Takes a string and returns a string containing all the characters in the parameter string that are not vowels.
    result = ''
    for i in s:
        if not is_vowel(i):
            result += i
    return result

def consonant(s:str)->str:
    #Takes a string and returns a string containing all the letters in the parameter string that are not vowels. 
    result = ''
    CONSONANTS = ('b','c','d','f','g','h','j','k,','l',
                  'm','n','p','q','r','s','t','v','w',
                  'x','y','z','B','C','D','F','G','H',
                  'J','K','L','M','N','P','Q','R','S',
                  'T','V','W','X','Y','Z')
    for i in s:
        if i in CONSONANTS:
            result += i
    return result

assert consonant('T3sTinG')== 'TsTnG'
assert consonant('I"m very upset about this.')== 'mvrypstbtths'

def select_letters(v:str, s:str)->str:
    #Returns consonants of s if v=='c' and returns vowels of s if v=='v'
    results = ''
    if v=='v':
        for i in s:
            if is_vowel(i):
                results += i
    if v=='c':
        for i in s:
            results += consonant(i)
    return results

def hide_vowels(s:str)-> str:
    #Replace every vowel in s with hyphen, other characters don't change
    hidden_vowels = ''
    for i in s:
        if not is_vowel(i):
            hidden_vowels += i
        else:
            hidden_vowels += '-'
    return hidden_vowels

assert hide_vowels('Learn more') == 'L--rn m-r-'
print("When you hide the vowels in the word 'Restaurant', it becomes:")
print(hide_vowels('Restaurant'),'\n')
