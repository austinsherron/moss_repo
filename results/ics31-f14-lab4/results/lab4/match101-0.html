def test_number(num: int, prop: str) ->bool:
    '''returns True if the number has the property indicated by the string,
    and False if it doesn't
    '''
    if prop == 'even':
        return num % 2 == 0
    elif prop == 'odd':
        return num % 2 != 0
    elif prop == 'positive':
        return num > 0
    elif prop == 'negative':
        return num < 0           

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels (nonvowels:str) -> str:
    '''returns a string containing all the characters in the parameter string that are not vowels
    '''
    result = ''
    for nv in nonvowels:
       if is_vowel(nv) == False:
          result = result + nv
    return result
assert nonvowels('abcdefg') == 'bcdfg'
assert nonvowels('aeiou') == ''
assert nonvowels('a,e,f..') == ',,f..'

def consonants (c:str) -> str:
    '''return a string containing all the letters in the parameter string that are not vowels
    '''
    o = ''
    for co in c:
       if co in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
          o = o + co
    return o
assert consonants('abcdefg') == 'bcdfg'


assert consonants('aeiou') == ''
assert consonants('a,e,f..') == 'f'

def select_letters (s:str,p:str) -> str:
    '''If the first parameter is 'v', it returns a string containing all the vowels in the second
    parameter; if the first parameter is 'c', it returns a string containing all the consonants in
    the second parameter.
    '''
    vow = ''
    con = ''
    if s == 'v':
        for r in p:
            if is_vowel(r):
                vow = vow + r
##          if r in 'aeiouAEIOU':
        return vow
    if s == 'c':
        for r in p:
            if not is_vowel(r):
                con = con + r
##         if r in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
        return con
    if s != 'v' or 'c':
        return ''

def hide_vowels (hv:str) -> str:


    ''' returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and
    all other characters remain unchanged
    '''
    hi=''
    for hid in hv:
        if is_vowel(hid):
            hi = hi + '-'
##       if hid in 'aeiouAEIOU':
        else:
            hi = hi+ hid
    return hi

    else:
        return result

def average_price(lst: 'list of restaurants') -> float:
    '''return the average price of (the best dishes at) those restaurants
    '''
    total = 0
    for r in lst:
        total = r.price + total
    ave_pri = total / len(lst)
    return ave_pri

import tkinter              

my_window = tkinter.Tk()    

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
my_canvas.pack()

def create_rectangle_from_center(x: 'x-value', y: 'y-value', h: 'height', w:'width'):
    '''creat a rectangle based on the x-value and y-value of the point
    in the center, and the height and width of the rectangle.
    '''
    my_canvas.create_rectangle(x - w * (1/2), y - h * (1/2), x + w * (1/2), y + (1/2), fill = 'gold')

create_rectangle_from_center(250, 250, 100, 100)
tkinter.mainloop()
