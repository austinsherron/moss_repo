def test_number (a: int, b: str) -> bool:
    '''returns boolean expression based on parameter of string'''
    if b == 'even':
        return a%2 == 0
    elif b == 'odd':
        return a%2 != 0
    elif b == 'positive':
        return a > 0
    elif b == 'negative':
        return a < 0

def nonvowels (a: str) -> str:
    '''return string with nonvowels from parameter string'''
    result = ''
    vowels = 'aeiouAEIOU'
    for b in a:
        if b not in vowels:
            result = result + b
    return result

def consonants (a: str) -> str:
    '''takes str and returns str with only consonants'''
    result = ''
    vowels = 'aeiouAEIOU'
    alphabet = 'abcdefghigklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    for c in a:
        if c in alphabet:
            for b in c:
                if b not in vowels:
                    result = result + b
    return result
assert consonants('abc123') == 'bc'
assert consonants('efg456') == 'fg'

def hide_vowels (a: str) -> str:
    '''takes str and returns str with vowels replaced with hyphens'''
    result = ''
    vowels = 'aeiouAEIOU'
    alphabet = 'abcdefghigklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    for b in a:
        if b in vowels:
            result = result + '-'
        else:
            result = result + b
    return result
assert hide_vowels('facetiously') == 'f-c-t---sly'
assert hide_vowels('onerous')== '-n-r--s'

from collections import namedtuple
Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')

def Restaurant_change_price (a: Restaurant, b: float) -> Restaurant:
    '''takes a restaurant object and a number and increases the restaurant object's price by that number'''
    a = a._replace(price = a.price + b)
    return a
