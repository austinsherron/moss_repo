def test_number(n: int, s: str) -> bool:
    '''Takes in a number and a string with the property even, odd, positive, or negative.
    If the number has the property indicated by the string the function returns True.
    Otherwise, it returns False'''
    if n % 2 == 0 and s == 'even':
        return True
    elif n % 2 == 0 and s == 'odd':
        return False
    elif n % 2 == 1 and s == 'odd':
        return True
    elif n % 2 == 1 and s == 'even':
        return False
    elif n == 1 and s == 'odd':
        return True
    elif n == 1 and s == 'even':
        return False
    elif n < 0 and s == 'positive':
        return False
    elif n > 0  and s == 'positive':
        return True
    elif n < 0 and s == 'negative':
        return True
    elif n > 0  and s == 'negative':
        return False

assert test_number(14, 'even')
assert test_number(-14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert test_number(-33, 'negative')
assert not test_number(100, 'negative')
assert test_number(0, 'odd') == False
assert test_number(1, 'odd')
assert not test_number(1, 'even')
print("Is '36' 'even'?", test_number(36, 'even'))
print("Is '-3' 'positive'?", test_number(-3, 'positive'))

def nonvowels(word: str) -> str:
    'Takes in a string and returns a string with all of the characters that are not vowels'
    result = '' # Empty string
    for letter in word:
        if is_vowel(letter) == False:
            result += letter # Adds nonvowels to empty string
    return result

def consonants(word: str) -> str:
    '''Takes in a string and returns a string containing all letters that are not vowels'''
    consonants_in_str = '' # Empty string
    for letters in word:
        if letters in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ': # All nonvowel letters
            consonants_in_str += letters #Adds nonvowel letters (consonants) to empty string

def select_letters(one_char: str, word: str) -> str:
    '''Takes in two strings and returns a string. If the first parameter is 'v', then the function
    returns a string with all of the vowels in the second parameter. If the first parameter is 'c',
    then the function returns a string containing containing all the consonants in the second parameter.
    Otherwise, the empty string is returned'''
    vowels_in_word = '' # Empty string
    if one_char == 'v':
        for letters in word:
            if is_vowel(letters) == True:
                vowels_in_word += letters
        return vowels_in_word
    elif one_char == 'c':
        return consonants(word)
    else:
        return vowels_in_word

print(select_letters('v', 'cookie'))
print(select_letters('c', 'AC/DC'))
assert select_letters('v', 'vanilla') == 'aia'
assert select_letters('c', 'chocolate') == 'chclt'
assert select_letters('d', 'strawberry') == ''

def hide_vowels(sentence: str) -> str:
    '''Takes in a string and replaces every vowel in the string with the hypen. It leaves all of the
    other characters unchanged'''
    for letter in sentence:
        if is_vowel(letter) == True:
            sentence = sentence.replace(letter, '-') #Replaces vowels with hypens
    return sentence

def average_price(restaurant_list: 'list of restaurant') -> float:
    '''Takes in a list of restaurants and returns the average price of the best dishes at those restaurants'''
    list_of_prices = [] # Empty list; all of the restaurant prices will be appended to the list
    sum_of_prices = 0 # Empty float; the prices of all the restaurants will be added up here
    for restaurant in restaurant_list:
        list_of_prices.append(restaurant.price) # Adding the price of the restaurants to the empty list
    for price in list_of_prices:
        sum_of_prices += price # Adding the values of the restaurant prices together
    average_price = sum_of_prices / len(restaurant_list) # Calculating the average price of all the restaurants
    return average_price

def create_rectangle_from_center(x_value, y_value, height, width):
    '''Creates a rectangle given the x and y coordinate of the center point of the rectangle
    and the height and width of the rectangle'''
    import tkinter
    
    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a canvas of 500x500 size to draw on
    my_canvas.pack()
    
    my_canvas.create_rectangle(x_value - (width / 2), y_value - (height / 2), x_value + (width / 2), y_value + (height / 2), fill = 'black') # Create the rectangle
    
    tkinter.mainloop()          # Combine all the elements and display the window

        
create_rectangle_from_center(200, 200, 100, 50)
