def test_number(number: int, string: str) -> bool:
    '''Returns True if number matches property indicated by string, false if not'''
    if number % 2 == 0 and string == 'even':
        return True
    elif number % 2 != 0 and string == 'odd':
        return True
    elif number > 0 and string == 'positive':
        return True
    elif number < 0 and string == 'negative':
        return True
    else:
        return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')


assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(letters : str) -> str:
    '''Returns characters in the string that are not vowels'''
    vowel = 'AEIOUaeiou'
    result = ""
    for x in letters:
        if x not in vowel:
            result = result + x
    return result

assert nonvowels('wetr') == 'wtr'

def consonants(letters : str) -> str:
    '''Returns a string containing letters in parameter string that aren't vowels'''
    vowel = 'AEIOUaeiou'
    result = ""
    alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZqwertyuiopasdfghjklzxcvbnm'
    for n in letters:
        if n in alphabet:
            if n not in vowel:
                result = result + n
    return result

assert consonants('tre8c') == 'trc'

def select_letters(x : str, word : str) -> str:
    '''If first parameter is 'v', return string containing all the vowels in second parameter. If first parameter is 'c', return a string containing all the consonants in the second parameter'''
    vowel = 'AEIOUaeiou'
    result = ""
    if x == 'v':
        for n in word:
            if n in vowel:
                result = result + n
        return result
    elif x == 'c':
        return consonants(word)
    else:
        return result


assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('a', 'awgaw') == ''

def hide_vowels(x: str) -> str:
    '''Returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters unchanged'''
    result = x
    vowel = 'AEIOUaeiou'
    for n in x:
        if n in vowel:
            result = result.replace(n, '-')
    return result

assert hide_vowels('abcde') == '-bcd-'
assert hide_vowels('ertyu') == '-rty-'
