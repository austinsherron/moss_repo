def test_number(a: int, b: str) -> bool:
    '''takes a number and a string and return True if the number has the
    property indicated by the string, else False'''
    if(b == 'even'):
        return (a % 2 == 0)
    if(b == 'odd'):
        return (a % 2 == 1)
    if(b == 'positive'):
        return(a > 0)
    if(b == 'negative'):
        return(a < 0)
    else:
        return False

assert test_number(15, 'odd')
assert not test_number(100, 'negative')
assert test_number(18, 'even')
assert test_number(1, 'positive')

def consonants(a: str) -> str:
    '''returns the string with only consonants (no vowels or symbols, numbers, etc.'''
    con = 'bcdfghjklmnpqrstvwxyz'
    CON = con.upper()
    result = ''
    for i in range(len(a)):
        for j in range(len(con)):
            if(a[i] == con[j] or a[i] == CON[j]):
                result += a[i]
    return result

assert consonants('@#$ abcd BT') == 'bcdBT'
assert consonants('aweoiunveoain') == 'wnvn'

def select_letters(a: str, b: str) -> str:
    '''returns a string with vowels only if 'v' is input, consonants only if \
    'c' is input'''
    vowel = ''
    if(a == 'v'):
        for i in b:
            if (is_vowel(i)):
                vowel += i
        return vowel
    if(a == 'c'):
        return consonants(b)
    return vowel

def hide_vowels(a: str) -> str:
    '''returns a string with '-' instead of vowels'''
    result = ''
    for i in a:
        if(is_vowel(i)):
            result += '-'
        else:
            result += i
    return result

assert hide_vowels('Aaron') == '--r-n'
assert hide_vowels('Tarang') == 'T-r-ng'
assert hide_vowels('I am taking ICS 31') == '- -m t-k-ng -CS 31'
