def test_number(num: int, s: str) -> bool:
    '''Returns true if num has the property indicated by string s'''
    if s == 'even':
        if num%2 == 0:
            return True
    elif s == 'odd':
        if num%2 == 1:
            return True
    elif s == 'positive':
        if num > 0:
            return True
    elif s == 'negative':
        if num < 0:
            return True
    else:
        return False
assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True

def nonvowels(s: str) -> str:
    '''Returns a string containing all the characters in the parameter that are not vowels'''
    result = ''
    for x in s:
        if not is_vowel(x):
            result = result + x
    return result

assert nonvowels('aeiou') == ''

def select_letters(s: str, word: str) -> str:

    answer = ''
    if s == 'v':
        for x in word:
            if is_vowel(x):
                answer = answer + x
    elif s == 'c':
        return consonant(word)
    else:
        return ''
    return answer

def hide_vowels(s: str) -> str:
    '''Returns a string where the vowels are replaced with hyphen'''
    words = ''
    for x in s:
        if is_vowel(x):
            words = words + '-'
        else:
            words = words + x
    return words


assert hide_vowels('Anteater #4') == '-nt--t-r #4'
assert hide_vowels('Irvine %^#') == '-rv-n- %^#'

import tkinter

def create_rectangle_from_center(x:int, y:int, h:int, w:int):
    '''Creates a rectangle from the center point and given height and width'''
    my_window = tkinter.Tk()    # Create the graphics window
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()            # Put the canvas into the window

    x1 = x - .5*w
    y1 = y - .5*h
    x2 = x + .5*w
    y2 = y + .5*h
    my_canvas.create_rectangle(x1, y1, x2, y2)

create_rectangle_from_center(200, 200, 100, 300)
tkinter.mainloop()
