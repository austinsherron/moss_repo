def test_number(n: int, test: str) -> bool:
    '''Return a bool based on if given number matches the string'''
    if test == 'even':
        return n % 2 == 0
    elif test == 'odd':
        return n % 2 != 0
    elif test == 'positive':
        return n > 0
    else:
        if test == 'negative':
            return n < 0
assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True

def is_vowel(letter: str) -> bool:
    '''Return True if the given letter is a vowel, false if otherwise'''
    
    vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
    return letter in vowels

def nonvowels(to_check: str) -> str:
    '''Take a string and return a string containing all the nonvowel
       characters in the parameter'''
    
    result = ''
    for char in to_check:
        if not is_vowel(char):
            result = result + char
    return result
assert nonvowels('there') == 'thr'
assert nonvowels('hi there') == 'h thr'

def consonants(s: str) -> str:
    '''Take a string, and return a string containing all
       letters in the parameter string that are not vowels'''
    
    result = ''
    
    for char in s:
        if (not is_vowel(char)) and char.isalpha():
            result = result + char
    return result
assert consonants('testString') == 'tstStrng'
assert consonants('!@#Test') == 'Tst'

def select_letters(v_or_c: str, test_me: str) -> str:
    '''If the first parametter is v, return a string
       containing all the vowels in the second parameter,
       if the first parameter is c, return a string
       containing all the consonants in the second parameter
       if the first parameter is anything else, return an empty string'''

    result = ''

    if v_or_c == 'v':
        for char in test_me:
            if is_vowel(char):
                result = result + char
    elif v_or_c == 'c':
        for char in test_me:
            if not is_vowel(char):
                result = result + char
    return result
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(s: str) -> str:
    '''Takes a string, and returns a string that replaces
       all the vowels with a hyphen'''
    result = ''
    for char in s:
        if is_vowel(char):
            result = result + '-'
        else:
            result = result + char
    return result
assert hide_vowels('Sleepy') == 'Sl--py'
assert hide_vowels('Nouns') == 'N--ns'

def sum_of_prices(R:'list of Restaurant')->float:
    '''takes a list of Restaurant and
    returns a sum of the prices as a float'''
    total=0
    for item in R:
        total=total+item.price
    return total

def average_price(R:'list of Restaurant')->float:
    '''takes a list of Restaurant and returns
    a float representing the average price'''
    return sum_of_prices(R)/len(R)
