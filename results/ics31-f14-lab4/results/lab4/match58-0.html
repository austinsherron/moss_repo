def test_number(x:int, y:str)->str:
    ''' Return if the set is true or false '''
    if x % 2 == 0 and y == 'even':
        return True
    elif x % 2 != 0 and y == 'odd':
        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative':
        return True
    else:
        return False

def is_vowel(a:str)->bool:
    ''' Return True if the letter is a vowel '''
    vowels = ['a', 'e', 'i', 'o', 'u']
    if a in vowels:
        return True
    else:
        return False

def nonvowels(a:str)->str:
    ''' Return nonvowels in a word '''
    result = ''

def consonants(a:str)->str:

def select_letters(a:str, b:str)->str:
    ''' Return vowels in a word if V is selected and print consonants if C is selected '''
    result = ''
    for x in b:
        if a == 'v' and is_vowel(x) and x.isalpha():
            result += x
        elif a == 'c' and not is_vowel(x) and x.isalpha():
            result += x
    return result

assert select_letters('c', '#Coding') == 'Cdng'

def hide_vowels(a:str)->str:
    ''' Replace vowels with a hyphen("-") and keep the rest unchange '''
    for x in a:
        if is_vowel(x):
            a = a.replace(x,'-')
    return a
print(hide_vowels('Coding'))
print(hide_vowels('Perfection'))
print(hide_vowels('Midterms'))

assert hide_vowels('Coding') == 'C-d-ng'
assert hide_vowels('Perfection') == 'P-rf-ct--n'
assert hide_vowels('Midterms') == 'M-dt-rms'

import tkinter

def create_rectangle_from_center(x:int, y:int, a:int, b:int):
    ''' Create a rectangle using the center point, height(a), and width(b) '''
    canvas.create_rectangle(x - (b / 2), y - (a / 2), x + (b / 2), y + (a / 2))
