def test_number(i:int, s:str) -> bool:
    ''' Takes a number, checks if it is even/odd, and returns the truth statement '''
    if s == "even":
        if i % 2 == 0:
            return True
        else:
            return False
    elif s == "odd":
        if i % 2 == 1:
            return True
        else:
            return False
    elif s == "positive":
        if i > 0:
            return True
        else:
            return False
    elif s == "negative":
        if i < 0:
            return True
        else:
            return False

assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True

def is_vowel(s:str) -> bool:
    ''' Checks char, returns True/False if char is a vowel '''
    s = s.upper()
    return (s == 'A' or s == 'E' or s == 'I' or s == 'O' or s == 'U')

def hide_vowels(s: str) -> str:
    ''' Takes in a string and returns string with vowels hidden as "-" '''
    i = 0
    string = ''
    while i < len(s):
        if is_vowel(s[i]):
            string = string + '-'
        else:
            string = string + s[i]
        i = i + 1
    return string

    ''' Creates a rectangle where user specifies starting coordinates,
        width, and height '''
