def test_number(num:int, arg:str) -> bool:
    if arg == 'even' or arg == 'Even':
        return (num % 2 == 0)
    if arg == 'odd' or arg == 'Odd':
        return (num % 2 != 0)
    if arg == 'positive' or arg == 'Positive':
        return (num/-1 == -num)
    if arg == 'negative' or arg == 'Negative':
        return (num/-1 == abs(num))

def is_a_consonant(letter:str) -> bool:
    consanants = ["b", "c", "d", "f", "g", "h", "j", "k", "l", "m", "n", "p", "q", "r", "s", "t", "v", "w", "x", "y", "z"]
    Uconsanants = ["B", "C", "D", "F", "G", "H", "J", "K", "L", "M", "N", "P", "Q", "R", "S", "T", "V", "W", "X", "Y", "Z"]
    if letter in consanants or letter in Uconsanants:
        return True
    
def is_a_vowel(letter:str) -> bool:
    vowels = ["a", "e", "i", "o", "u"]
    Uvowels = ["A", "E", "I", "O", "U"]
    if letter in vowels or letter in Uvowels:
        return True
    
def consonants(string:str) -> str:
    result = ""
    for i in range(len(string)):
        if is_a_consonant(string[i]) == True:
            result += string[i]
    return result

def vowels(string:str) -> str:
    result = ""
    for i in range(len(string)):
        if is_a_vowel(string[i]) == True:
            result += string[i]
    return result
