def test_number(value: int, string:str) -> bool:
    "This function takes a value and a string (True/False) and returns whether or not the number actually is true/false given the parameters"
    if string == "even":
        return value%2 == 0
    elif string == "odd":
        return not value%2 == 0
    elif string == "positive":
        return value >=0
    elif string == "negative":
        return value < 0

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(string:str)-> bool:
    "Returns True/False if a letter is a vowel or not."
    return (string in 'a,e,i,o,u' or string in 'A,E,I,O,U')

    "Returns a string of all of the consonants in a string"

        b,c,d,f,g,h,j,k,l,m,n,p,q,r,s,t,v,w,x,y,z''':
            result+=string[i]
    return result

def hide_vowels (string:str)->str:
    "Returns a string that replaces the vowels with hyphens"
    result = ''
    for i in range(len(string)):
        if is_vowel(string[i]) == True:
            result+='-'
        else:
            result+=string[i]
    return result
