def test_number(n: int, value: str) -> bool:
    'Takes a number and a string, tests to see if that number has characteristic of that string'
    if value == 'even' and n % 2 == 0:
        return(True)
    elif value == 'odd' and n % 2 != 0:
        return(True)
    elif value == 'positive' and n > 0:
        return(True)
    elif value == 'negative' and n < 0:
        return(True)
    else:
        return(False)


assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
