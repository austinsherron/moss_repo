def test_number (n: int, s:str) -> bool:
    if s == 'positive':
        if n > 0:
            return True
        else:
            return False 
        
    if s == 'negative':
        if n < 0:
            return True
        else:
            return False
        
    if s=='even':
        if n%2 ==0:
            return True
        else:
            return False
  
    if s=='odd':
        if n%2 != 0:
            return True
        else:
            return False
        


assert test_number(33, 'positive')
assert not test_number(100, 'negative')
assert test_number(14, 'even')
assert not test_number(100, 'odd')

def total_price(l:list)-> float:
    price_1 = []
    total = 0.00

    for restaurant in l:
        price_1.append(restaurant.price)
    for s in price_1:
        total = total + s
    return total 
    
def average_price(RL:list)->list:
    result = total_price(RL)/len(RL)
    return result
