def test_number(x: int, cond: str)->bool:
    if cond == 'even':
        if x%2 == 0:
            return True
        else:
            return False
    if cond == 'odd':
        if x%2 == 1:
            return True
        else:
            return False
    if cond == 'positive':
        if x >= 0:
            return True
        else:
            return False
    if cond == 'negative':
        if x < 0:
            return True
        else:
            return False
assert not test_number(15, 'even')
assert test_number(15, 'odd')
assert test_number(15, 'positive')
assert not test_number(15, 'negative')
