    'Returns boolean that verifies whether number is even, positiveis even, positive, odd, or, odd, or even.'
    if s == 'even':
        return (n%2) == 0

    elif s == 'odd':
        return (n%2)  != 0

    if s == 'negative':
        return n < 0

    elif s == 'positive':
        return n > 0

def matching_area_codes(n:list, t:list) -> list:
    'Returns a list of phone numbers with specified area codes'
    matching_codes = []
    for l in t:
        for h in n:
            if h == l[1:4]:
                matching_codes.append(l)
    return matching_codes
print (matching_area_codes(['949', '714'], ['(714)824-1234', '(419)312-8732', '(949)555-1234']))

def alphabetical_name(n) -> list:
    'Returns list of alphabetized field in specified list'
    restaurant_names = []

        restaurant_names.append(sorted(n)[x].name)
