    '''it returns the vowels in the string with a hyphen(-)'''
    result = ''

        if h in 'aioueAEIOU':
            s = s.replace(h,'-')
    result = s + result

def average_price(r: list) -> float:
    '''Takes a list of restaurants and returns the average price of
    the best dishes'''
    sumation = 0
    for i in range(len(r)):
        sumation = sumation + r[i].price
    avg = sumation / len(r)
    return avg

def create_rectangle_from_center(center_x: int, center_y: int, h: int, w: int):
    '''Takes the center of the rectangle (x,y), a height, and a width. Returns a
        rectangle drawn from center'''
    x1 = (center_x) - (w/2)
    x2 = (center_x) + (w/2)
    y1 = (center_y) - (h/2)
    y2 = (center_y) + (h/2)
    my_canvas.create_rectangle(x1,y1,x2,y2)


create_rectangle_from_center(700,700, 100,200)
create_rectangle_from_center(500,500,200,300)
