def test_number (a:int, b:str) -> bool:
    # return true if the number match the discription
    if a%2 == 0 and b == "even":
        return True
    elif a%2 != 0 and b == "odd":
        return True
    elif a < 0 and b == "negative":
        return True
    elif a > 0 and b == "positive":
        return True
    else:
        return False

assert test_number(14, "even")
assert not test_number(100, "odd")
assert test_number(33, "positive")
assert not test_number(100, "negative")

def hide_vowels(a:str) -> str:
    #replaces all vowels in the parameter with a hyphen
    result = ''
    for char in a:
        if is_vowel(char):
            result = result + '-'
        else:
            result = result + char
    return result
