def test_number(x:int, y:str)->bool:
    if y=="positive":
        if x > 0:
            return True
    if y=="odd":
        if x%2 == 1:
            return True
    if y=="negative":
        if x<0:
            return True
    if y=="even":
        if x%2==0:
            return True
    else:
        return False

    "returns letters that are not vowels"

def hide_vowels(x:str)->str:
    "returns vowels with (-)"
    result=''
    for i in range(len(x)):
        for g in x:
            if is_vowel(g):
                result=result+'-'
            else:
                result=result+g
        return result
print(hide_vowels('i like dogs'))

def average_price(restaurants:list)->float:
    'returns average price of those restaurants'
    sum=0
    for item in restaurants:
        sum = sum + item.price
    return (sum / len(restaurants))
