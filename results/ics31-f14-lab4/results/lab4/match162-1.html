        return True

def consonants(word:str) -> str:
    result = ''
    wordlist = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    for i in range(len(word)):
        if word[i] in wordlist:
            if is_vowel(word[i]) == False:
                result = result + word[i]
    return result
assert consonants('he!l') == 'hl'
assert consonants('jabs') == 'jbs'
assert consonants('!@3s') == 's'

def hide_vowels(word: str) -> str:
    ''' Takes a string abd returns another string will all vowels replaced by Hyphen("-") '''
    result = ''
    for i in range(len(word)):
        if (is_vowel(word[i]) == True):
            result = result + '-'
        else:
            result = result + word[i]
    return result
    
assert hide_vowels('hello') == 'h-ll-'
assert hide_vowels('parameter') == 'p-r-m-t-r'

def average_price(r_list: list)->float:
