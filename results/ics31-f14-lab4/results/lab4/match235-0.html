def test_number(x, y):
    if x % 2 == 0 and y == 'even':
        return True
    elif x % 2 != 0 and y == 'odd':
        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative':
        return True
    else:
        return False
    


assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(n: str) -> str:
    result = ''
    for a in n:
        if is_vowel(a) != True:
            result = result + a
    return result

def consonants (n: str) -> str:
    word = ''
    for a in n:
        if a in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':

def hide_vowels(x: str) -> str:
    p = ''
    for i in x:
        if is_vowel(i) == True:
            p = p + '-'
        elif is_vowel(i) != True:
            p = p + i
    return p
print(hide_vowels('Bat'))
print(hide_vowels('The ball went really high off the ground'))

def average_price(r: list) -> float:
    pq = []
    for t in range(len(r)):
        pq.append(r[t].price)
    return sum(pq) / len(r)
