    #function takes in a number and whether or not the number is even, odd, etc, and returns if it's true
    if (string == "even"):
        if (num % 2 == 0):
            return True
        else:
            return False
    if (string == "odd"):
        if (num % 2 == 1):
            return True
        else:
            return False
    if (string == "positive"):
        if (num > 0):
            return True


        else:
            return False
    if (string == "negative"):
        if (num < 0):
            return True
        else:
            return False

def nonvowels(string: str) -> str:
    #given a string, returns a string with all nonvowels in the string

def hide_vowels(string: str) -> str:
    #takes a string and returns a string where vowels are replaced with a hyphen


    result = ""
    for i in range(len(string)):
        if (is_vowel(string[i])):
            result += "-"
        else:
            result += string[i]
    return result

def sum_price(restaurants: list) -> float:
    #returns the sum of the prices of all restaursnts, given a list of restaurants
    sum = 0
    for i in restaurants:
        sum += i.price
    return sum

def average_price(restaurants: list) -> float:
    #returns the average price of all restaurants, given a list of restaurants
    return sum_price(restaurants)/len(restaurants)
