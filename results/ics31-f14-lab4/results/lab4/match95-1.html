def nonvowels(x:str)->str:
    '''takes a string and returns string without vowels'''
    result=''
    for i in x:
        if not is_vowel(i):
            result=result+i
    return result

assert nonvowels('anteater')=='nttr'
assert nonvowels('abcd1234')=='bcd1234'

def consonants(x:str)->str:
    '''takes a string and returns a string with only consonants'''
    result = ''
    for i in x:
        if i in 'bcdfghjklmnpqrstvwxyz':
            result=result+i
    return result

def select_letters(x:str,y:str)->str:
    ''' takes two parameters. If first parameter is 'v' then function will return
    a string of vowels from the second paramter. If first paramter is 'c' then
    function will return a string of consonants form the second parameter. If
    firt parameter is neither 'c' nor 'v' then function will return empty string'''
    result=''
    if x=='v':
        for i in y:
            if is_vowel(i):
                result=result+i
        return result
    elif x=='c':
        return consonants(y)
    else:
        return result
    
assert select_letters('v','anteater#%^1234') == 'aeae'
assert select_letters('c','anteater#$%^1234')=='nttr'
assert select_letters('d','weggf')==''

def hide_vowels(x:str)->str:
    ''' takes a string and replaces all vowels with a hyphen'''
    result=''
    for i in range(0,len(x)):
        if is_vowel(x[i]):
            result=result+'-'
        else:
            result=result+x[i]
    return result

def average_price(x:list)->float:
    '''takes a list of restaurants and return the average price of all
    restaurants'''
    totalprice=0
    for i in x:
        totalprice=totalprice+i.price
        return totalprice
    average=totalprice/len(x)
    return average
