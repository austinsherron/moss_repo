def test_number (n:int, s:str) -> bool:
    '''Attributes are true or false depending on the user input'''
    if s == "even":
        return n%2 == 0
    if s == "odd":
        return n%2 > 0


    if s == "positive":
        return n >= 0
    if s == "negative":
        return n < 0
