def test_number(x: int, s: str) -> bool:
    '''Returns True if int matches string description, False if not.'''


    if(x >= 0 and s == 'positive'):
        return True
    elif(x < 0 and s == 'negative'):
        return True 
    elif(x % 2 == 0 and s == 'even'):
        return True
    elif(x % 2 == 1 and s == 'odd'):
        return True
    else:
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

print(test_number(14, 'even'))
print(not test_number(100, 'odd'))
print(test_number(33, 'positive'))
print(not test_number(100, 'negative'))

def nonvowels(s: str) -> str:
    '''Returns string containing all nonvowels of parameter string'''
    result = ''
    for x in s:
        if not(is_vowel(x)):
            result = result + x
    return result

assert nonvowels('qwe?rty') == 'qw?rty'
assert not nonvowels('abcdef3g') == 'abcdfg'

print(nonvowels('qwe?rty'))
print(nonvowels('abcdef3g'))

def consonants(s:str) -> str:
    '''takes a string and returns a string containing all
       the consonants'''
    result = ""
    for x in s:
        if x.lower() in 'bcdfghjklmnpqrstvwxyz':
            result += x
    return result

assert consonants("aAsdTfGgh!*2werq") == 'sdTfGghwrq'
print(consonants("aAsdTfGgh!*2werq"))

def select_letters(s1: str, s2: str) -> str:
    '''if first parameter is 'v',
    Returns string containing all the vowels in the second parameter
    if first parameter is 'c',
    Returns string containing all the consonants in the second parameter'''
    result = ""
    for x in s2:
        if s1 is 'v' and is_vowel(x):
            result += x
        elif s1 is 'c' and consonants(x):
            result += x
    return result

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

print(select_letters('v', 'facetiously'))
print(select_letters('c', 'facetiously'))

def hide_vowels(s:str) -> str:
    '''hide a vowel and change it with (-)'''
    result = ""
    for x in s:
        if is_vowel(x):
            result += "-"
        else:
            result += x
    return result

assert hide_vowels('california') == 'c-l-f-rn--'
assert hide_vowels('facetiously') == 'f-c-t---sly'

print(hide_vowels('california'))
print(hide_vowels('facetiously'))

def alphabetical_names(l:'list of Restaurant') -> 'list of str':
    '''takes a list of restaurants and returns list of their names
       in alphabetical order'''
    result = []
    for x in l:
        result.append(x.name)
        
    return sorted(result)

def average_price(l:"list of Restaurant") -> float:
    '''Param: list of Restaurant
       Returns the average price of all Restaurants'''
    sum = 0
    for x in l:
        sum += x.price
    return sum / len(l)
