def test_number(num : int, string : str) -> bool:
    "takes in a number and string, then evaluates whether the given number is true or false to the given string"
    if (string == 'even'):  #even
        if (num % 2 == 0):
            return True
        else:
            return False

    elif (string == 'odd'): #odd
        if ((num - 1) % 2 == 0):
            return True
        else:
            return False

    elif (string == 'positive'): #positive
        if(num - num == 0):
            return True
        else:
            return False

    elif (string == 'negative'): #negative
        if(num - num == -(2*num)):
            return True
        else:
            return False

    else:
        print("string cannot be identified")
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

print(test_number(14, 'even'))
print(test_number(100, 'odd'))
print(test_number(33, 'positive'))
print(test_number(100, 'negative'))

def square_list(numbers : list) -> list:
    "takes in a list of numbers and prints the squared values in each line"
    l=[]
    for y in range(0, len(numbers)):
        sq = numbers[y] ** 2
        l.append(sq)
        print(sq)
    return l

assert square_list([2, 3, 4]) == [4, 9, 16]

    "takes in a word sring and returns the non vowel letters, including puncuation marks and other non vowel characters"

def consonants (word: str) -> str:
    "takes in a word string and returns only the consonant letters of the input word"
    result = ""
    for x in range(0, len(word)):
        if (word[x] in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'):
            result += word[x]
    return result

def select_letters(v_or_c: str, word: str) -> str:
    "takes in a letter (string) v or c and takes in a word string, then returns the word with only vowel or consononant letters based on v or c input"
    if (v_or_c == 'v'):
        vowel = ""
        for a in range(0, len(word)):
            if (word[a] in 'aeiouAEIOU'):
                vowel += word[a]
        return vowel

    elif (v_or_c == 'c'):
        consonant = ""
        for b in range(0, len(word)):
            if (word[b] in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'):
                consonant += word[b]
        return consonant

def hide_vowels (string: str) -> str:
    "takes in a string string and returns the string but with vowels replaced with '-'"
    result = ""
    for x in range(0, len(string)):
        if (string[x] not in 'aeiouAEIOU'):
            result += string[x]
        elif (string[x] in 'aeiouAEIOU'):
            result += '-'
    return result

def average_price(rest_list: list) -> float:
    total_price = 0
