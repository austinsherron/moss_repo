def test_number(number: int, string: str) -> bool:
    ''' Passes parameter with number to return True or False (i.e. even, odd, positive, negative) '''
    if string == 'even':
        if number % 2 == 0:
            return True
        else:
            return False
    if string == 'odd':
        if number % 2 != 0:
            return True
        else:
            return False
    if string == 'positive':
        if number >= 0:
            return True
        else:
            return False
    if string == 'negative':
        if number < 0:
            return True
        else:
            return False
    return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
assert test_number(223, 'odd')


assert not test_number(-20, 'positive')

def nonvowels(string: str) -> str:

def consonants(string : str) -> str :
    ''' Takes a string and returns a string of consonants'''
    consonants = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    result = ''
    for s in string :
        if s in consonants :
            result += s
    return result
assert consonants('Goodbye?') == 'Gdby'


assert consonants('12#$Hi') == 'H'
assert consonants('#12 is my num!!!') == 'smynm'

def hide_vowels(string: str) -> str:
    ''' returns string with vowels replaced with '-' '''
    result = ''
    for i in string:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result
assert hide_vowels('Jimmy fell into the creek') == 'J-mmy f-ll -nt- th- cr--k'
assert hide_vowels('Panda bears enjoy eating bamboo') == 'P-nd- b--rs -nj-y --t-ng b-mb--'
assert hide_vowels('#10 on Manchester United scored.') == '#10 -n M-nch-st-r -n-t-d sc-r-d.'

def add_restaurant_prices(rest_list: list) -> float:
    ''' Adds prices of all restaurants in a list '''
    total_price = 0.0
    for i in rest_list:
        total_price += i.price
    return total_price

def average_price(rest_list: list) -> float:
    ''' finds average price for the best dish in a list of Restaurants '''
    return add_restaurant_prices(rest_list) / len(rest_list)
