    if (num%2 == 0):
        if (string == "even"):
            return True
        if (string == "odd"):
            return False
    if (num>0):
        if (string == "positive"):
            return True
    if (num<0):
        if (string == "negative"):
            return True
    return False

def is_vowel(char: str)->bool:
    character = char.lower()
    return character=='a' or character == 'e' or character == 'i' or character == 'o' or character == 'u'

#(2)
def print_nonvowels(string: str):
    for i in string:
        if(not is_vowel(i)):
            print(i)

#(3)
def nonvowels(string: str)->str:
    result = ""
    for i in string:
        if(not is_vowel(i)):
            result = result + i
    return result

#(4)
def consonants(char: str)->str:
    ch = ""
    for i in char:
        if ord(i.lower()) > 97 and ord(i.lower()) < 123 and not is_vowel(i):
            ch = ch + i
    return ch

#(5)
def select_letters(one: str, two: str)->str:
    if(one=='v'):
        ch = ""
        for i in two:
            if(is_vowel(i)):
                ch = ch + i
        return ch
    if(one=='c'):
        return consonants(two)
    return ""

#(6)
def hide_vowels(one: str) -> str:
    for i in one:
        if(is_vowel(i)):
            one = one.replace(i, "-")
    return one
