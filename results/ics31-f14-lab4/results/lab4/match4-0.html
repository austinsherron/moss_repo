def is_consonant(s: str) -> bool:
    '''returns true if the input is a consonant'''
    return s in ['b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n',
                 'p', 'q', 'r', 's', 't', 'v', 'w', 'x', 'y', 'z', 'B',
                 'C', 'D', 'F', 'G', 'H', 'J', 'K', 'L', 'M', 'N', 'P',
                 'Q', 'R', 'S', 'T', 'V', 'W', 'X', 'Y', 'Z']

def consonants(s: str) -> str:
    '''returns all letters that are not vowels'''
    output = ''
    for character in s:
        if is_consonant(character):
            output = output + character
    return output

assert(consonants('apple')) == 'ppl'
assert(consonants('banana')) == 'bnn'
assert(consonants('penguins')) == 'pngns'

def select_letters(s: str, t: str) -> str:
    ''' takes two parameters, both strings, and returns a string. If the first parameter is 'v', it returns a string containing all the vowels in the second parameter; if the first parameter is 'c', it returns a string containing all the consonants in the second parameter. If the first parameter is anything else, it returns the empty string.'''
    if s == 'v':
        output = ''
        for character in t:
            if is_vowel(character):
                output = output + character
        return output
    elif s == 'c':
        return consonants(t)
    return ''

assert(select_letters('v', 'abcdefghijklmnopqrstuvwxyz')) == 'aeiou'
assert(select_letters('c', 'this is a string')) == 'thssstrng'

def hide_vowels(s: str) -> str:
    '''replaces all vowels with -'''
    output = ''
    for character in s:
        if is_vowel(character):
            output = output+ '-'
        else:
            output = output+character
    return output
assert(hide_vowels('This is a string')) == 'Th-s -s - str-ng'
assert(hide_vowels('Write a function called')) == 'Wr-t- - f-nct--n c-ll-d'

def average_price(rests:list) ->float:
    '''returns the average price of the restaurants'''
