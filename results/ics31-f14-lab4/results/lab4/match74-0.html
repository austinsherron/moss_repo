    if (s1 == 'odd' and x%2 == 1):

    if (s1 == 'positive' and x >= 0): #we count 0 as positive

    if (s1 == 'negative' and x < 0):

    else:
        return False

def is_vowel(s1: str):
    vowel = ['a', 'e', 'i', 'o', 'u','A', 'E', 'I', 'O', 'U']
    for i in vowel:
        if i == s1:
            return True
    else:
        return False

assert is_vowel('o')
assert not is_vowel('b')

def print_nonvowels(s1: str):

def nonvowels(s1: str) -> str:
    result = ""
    for i in s1:
        if is_vowel(i) == False:
            result = result + i
    return result

def consonants(s1: str) -> str:
    ''' Returns letters that are not vowels '''
    consonant = ['b', 'B', 'c', 'C', 'd', 'D', 'f', 'F', 'g', 'G', 'h', 'H', 'j', 'J', 'k', 'K', 'l', 'L', 'm', 'M', 'n', 'N', 'p', 'P', 'q', 'Q', 'r', 'R', 's', 'S', 't', 'T', 'v', 'V', 'w', 'W', 'x', 'X', 'y', 'Y', 'z', 'Z']
    result = ""
    for x in s1:
        for i in consonant:
            if x == i:
                result = result + x
    return result

assert(consonants('12,.lp3')) == 'lp'
print(consonants('12,.lp3'))

import tkinter              
def create_rectangle_from_center(x1: int, y1: int, height: int, width: int):
    my_window = tkinter.Tk()   
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
    my_canvas.pack()
    my_canvas.create_rectangle(x1-(width/2), y1-(height/2), x1+(width/2), y1+(height/2))
    tkinter.mainloop()
