def select_letters(first:str, second:str)->str:
    result = ''
    if first == 'v':
       for c in second:
           if is_vowel(c) == True:
               result += c.lstrip('0123456789`~!@#$%^&*()-_=+{}[]|\:;<,>.?/"''')
       return first, result
    if first == 'c':
        for c in second:
            if is_vowel(c) == False:
               result += c.lstrip('0123456789`~!@#$%^&*()-_=+{}[]|\:;<,>.?/"''')
        return first, result
    else:
        return ''

    overall_price = 0
    for x in range(len(food)):
        overall_price += food[x].price
        average_price = overall_price/len(food)
