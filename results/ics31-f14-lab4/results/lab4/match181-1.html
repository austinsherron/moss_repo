def test_number(n: int, value: str) -> bool:
    ''' Returns True if the number has the property indicated by the string,
        and False if it doesn't. '''
    return ((n % 2 == 0 and value == 'even') or (n % 2 != 0 and value == 'odd')
            or (n > 0 and value == 'positive') or (n < 0 and value == 'negative'))

assert not test_number(44, 'negative')

def match_area_codes(area_code: list, phone_number: list) -> list:

def is_vowel(v: str):
    ''' Takes a one-character-long string and returns
        True if that character is a vowel and False otherwise '''
    vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']

def nonvowels(value: str) -> str:


    ''' Takes a string and returns a string containing all the characters
        in the parameter string that are not vowels. '''
    result = ''
    for i in value:
        if not is_vowel(i):
            result += i
    return result

def consonants(value: str) -> str:
    ''' Takes a string and returns a string containing all the letters
       in the parameter string that are not vowels '''
    result = ''
    for i in value:
        if str.isalpha(i) and not is_vowel(i):
            result += i
    return result
assert consonants('anteater') == 'nttr'
assert consonants('value?') == 'vl'
print(consonants('anteater'))

def select_letters(letter_type: str, value: str) -> str:
    ''' If the first parameter is 'v', it returns a string containing all
       the vowels in the second parameter; if the first parameter is 'c',
       it returns a string containing all the consonants in the second
       parameter. If the first parameter is anything else, it returns
       the empty string'''
    result = ''
    if letter_type == 'c':
        result = consonants(value)
    elif letter_type == 'v':
        for i in value:
            if is_vowel(i):
                result += i
    return result
assert select_letters('c', 'animals') == 'nmls'
assert select_letters('v', 'animals') == 'aia'
assert select_letters('c', 'pokemon') == 'pkmn'
assert select_letters('v', 'pokemon') == 'oeo'
print(select_letters('c', 'animals'))
print(select_letters('v', 'animals'))
print(select_letters('c', 'pokemon'))
print(select_letters('v', 'pokemon'))

def hide_vowels(value: str) -> str:
    ''' Returns a string in which every vowel in the parameter is replaced
        with a hyphen ("-") and all other characters remain unchanged. '''
    result = ''
    for i in value:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result
assert hide_vowels('anteater') == '-nt--t-r'
assert hide_vowels('pokemon') == 'p-k-m-n'
print(hide_vowels('anteater'))
print(hide_vowels('Good morning'))

def Restaurant_change_price(rest: Restaurant, n: float) -> Restaurant:
    ''' Returns a Restaurant object with the same contents as the parameter
        except that the price has been increased by the specified number'''

def alphabetical_names(rests: list) -> list:
    ''' returns a list of the names of all the restaurants in alphabetical

def average_price(rests: list) -> float:
    ''' Takes a list of restaurants and returns the average price of
        (the best dishes at) those restaurants'''
    total = 0
    for i in rests:
        total += i.price
    return total / len(rests)
