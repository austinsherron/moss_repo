def select_letters(a:str, b:str)->str:
    '''If the first parameter is v, returns all the vowels in the second parameter
if the first parameter is c, returns all the consonants in the second parameter.
if the first parameter is anything else, return nothing'''


    result = ""
    if a == "v":
        for i in b:
            if i in ('aeiouAEIOU'):
                result=result+i
    elif a == "c":
        for i in b: 
            if i in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTUVWXYZ':
                result=result+i
    else:
        result=""
    return(result)

def hide_vowels(a:str)->str:
    '''takes a string and returns a string in which every vowel
   in the parameter is replaced with a hyphen'''
    vowel=""
    for i in a:
        if i in "aeiouAEIOU":
            vowel=vowel+"-"
        else:
            vowel=vowel+i
    return(vowel)

def average_price(ap:list)->float:
    '''takes a list of restaurant and returns the average price of those restaurant.'''
    total=0.00
    for i in ap:
        total = i.price+total
    return round(total / len(ap), 2)
