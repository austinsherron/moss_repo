def test_number(n: int, s: str) -> bool:
    '''Returns True or False based on the string's description of the number'''
    if (n % 2 == 0): # If n is even
        if(s == 'even'):
            return True
    if(n % 2 == 1): # If n is odd
        if(s == 'odd'):
            return True
    if(n > 0): # If n is positive
        if(s == 'positive'):
            return True
    if(n < 0): # If n is negative
        if(s == 'negative'):
            return True
    return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')

def is_vowel(s: str) -> bool:
    '''Returns True or False based on whether or not the letter is a vowel'''
    vowels = 'aeiouAEIOU' # Defining what a vowel is
    for a in vowels:
        if s == a: # Checks to see if s is a vowel
            return True
    return False
assert is_vowel('a') == True
assert is_vowel('b') == False
assert is_vowel('A') == True

def nonvowels(s: str) -> str:
    '''Returns all characters in the string that are not vowels'''

def consonants(s: str) -> str:
    '''Returns a string containing all letters in the parameter string that are not vowels'''
    consonants = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ' # Defining what a consonant is
    result = '' # Empty string
    for i in s:
        for j in consonants:


            if i == j: # Checks if i is a consonant
                result = result + i # Adds to the empty string
    return result
assert consonants('Anteater123') == 'nttr'
assert consonants('ICS 31') == 'CS'

def hide_vowels(s: str) -> str:
    '''Returns a string with all vowels changed to a hyphen'''
    result = '' # Empty String
    for i in s:
        if is_vowel(i): # Using the function in D1 to determine the vowels
            result = result + '-' # Adds a hyphen in place of the vowel
        else:
            result = result + i # Adds the original character without changing it
    return result

assert hide_vowels('I like apples') == '- l-k- -ppl-s'
