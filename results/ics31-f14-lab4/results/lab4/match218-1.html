def test_number(n: int, eonp: str) -> bool:
    if eonp == 'even':
        return n%2 == 0
    elif eonp == 'odd':
        return n%2 != 0
    elif eonp == 'positive':
        return n > 0
    elif eonp == 'negative':
        return n < 0

def is_vowel(c: str) -> bool:
    v = 'aeiouAEIOU'
    return c in v

def print_nonvowels(chars: str) -> str:
    str1=""
    for letter in chars:
        if not is_vowel(letter):
            str1=str1+letter

    return str1

def nonvowels(a: str) -> str:
    result = ''
    for i in a:
        if i not in 'aeiouAEIOU':
            result = result + i
    return result


assert nonvowels('doo') == 'd' #This should be true.
assert nonvowels('ooo') == '' # This should be true.
assert nonvowels('lala') == 'll' # This should be true.

def consonants(a: str) -> str:
    '''
    This is similar to nonvowels, but excludes special characters and numbers (more restrictions).
    '''
    result = ''
    for i in a:
        if i not in 'aeiouAEIOU1234567890~!@#$%^&*()_-+={[}]|\?/.>,<':
            result = result + i
    return result

assert consonants('ss@%0') == 'ss'
assert consonants('Tu$') == 'T'

def select_letters(a: str, b: str) -> str:
    result = ''
    if a == 'v':
        for i in b:
            if i in 'aeiouAEIOU':
                result = result + i
    elif a == 'c':
        for i in b:
            if i not in 'aeiouAEIOU':
                result = result + i
    elif a not in 'vc':
        return ''
    return result

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('r', 'dummy') == ''

def hide_vowels(a: str) -> str:
    result = ''
    for i in a:
        if i in 'aeiouAEIOU':
            i = '-'
        result = result + i
    return result

def average_price(l: list) -> float:
    result = []
    prices = 0
    for i in l:
        result.append(i.price)
        prices = prices + i.price
    return prices / len(result)
