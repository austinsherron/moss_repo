def consonants(n:str) -> str:
    '''Returns consonants(no other characters) of a string'''
    s = ''
    for i in range((len(n))):
        if (not is_vowel(n[i])) and (n[i].isalpha()):
            s += n[i]
    return s

assert consonants('computer#2') == 'cmptr'
assert consonants('b2') == 'b'
assert consonants('i') == ''
assert consonants('') == ''

def hide_vowels(n:str) -> str:
    '''Returns - in place of vowels'''
    for i in range(len(n)):
        if (is_vowel(n[i])):
            n = n.replace(n[i],'-')
    return n

def average_price (n:list)-> float:
    '''Returns the average price of the list'''
    list1 = []

    return sum(list1) / len(n)
