def nonvowels(in_str: str) -> str:
    result = ''
    for k in range(len(in_str)):
        if not is_vowel(in_str[k]):
            result = result + in_str[k]
    return result

def is_letter(in_str: str) -> str:
    return in_str in 'qwertyuiopasdfghjklzxcvbnmQWERTYUIOPASDFGHJKLZXCVBNM'
    
def consonants(in_str: str) -> str:
    in_str = nonvowels(in_str)
    result = ''
    for k in range(len(in_str)):
        if is_letter(in_str[k]):
            result = result + in_str[k]
    return result
assert consonants('non-vowels') == 'nnvwls'
assert consonants('blaze it 420!') == 'blzt'
assert consonants('I love the number 7') == 'lvthnmbr'

def vowels(in_str: str) -> str:
    result = ''
    for k in range(len(in_str)):


        if is_letter(in_str[k]) and is_vowel(in_str[k]):
            result = result + in_str[k]
    return result

def select_letters(vc: str, in_str: str) -> str:
    if vc == 'v':
        return vowels(in_str)
    if vc == 'c':
        return consonants(in_str)

print(select_letters('v', 'facetiously'))
print(select_letters('c', 'facetiously'))

def hide_vowels(in_str: str) -> str:
    result = ''
    for k in range(len(in_str)):
        if is_vowel(in_str[k]):
            result = result + '-'
        else:
            result = result + in_str[k]
    return result

    for r in in_RL:

    return sum_prices/len(in_RL)
