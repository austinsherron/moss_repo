def test_number(test_num: int, string: str) -> bool:
    " Returns True if the number has the indicated property. "
    if test_num % 2 == 0 and string == 'even':
        return True
    elif test_num % 2 != 0 and string == 'odd':
        return True
    elif test_num > 0 and string == 'positive':
        return True
    elif test_num < 0 and string == 'negative':
        return True
    else:
        return False
assert not test_number(100, 'odd')
assert test_number(14, 'even')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
print(test_number(14, 'even'))

def matching_area_codes(area_code: list, phone_num: list) -> str:
    " Returns a list of matching numbers. "
    num_list = []
    for d in area_code:
        for c in phone_num:
            if c[1:4] == d:
                num_list.append(c)
    return num_list

def nonvowels(n_v: str) -> str:
    " Returns non-vowels. "
    result = ''
    for i in n_v:
        if not is_vowel(i):
            result = result + i
    return result

def consonants(cons: str) -> str:
    " Returns non-vowels, not including spaces and non-characters. "
    result2 = ''
    for c in cons:
        if c in 'qwrtypsdfghjklzxcvbnmQWRTYPSDFGHJKLZXCVBNM':
            result2 = result2 + c
    return result2

def hide_vowels(hidden: str) -> str:
    " Replaces every vowel with a hyphen. "
    result4 = ''
    for h in hidden:
        if consonants(h):
            result4 = result4 + h
        else: 
            result4 = result4 + '-'
    return result4
assert hide_vowels('Hello') == 'H-ll-'

assert hide_vowels('Cookies') == 'C--k--s'

def average_price(Rest: list) -> float:
    " Returns average price of best dishes. "
    List3 = []

        List3.append(y.price)
        r = sum(List3) / len(List3)
