def test_number (n:int, t:str) -> bool:
    if t == 'even' and n % 2 == 0:
        return True
    elif t == 'odd' and n % 2 != 0:
        return True
    elif t == 'positive' and n > 0:
        return True
    elif t == 'negative' and n < 0:
        return True
    else:
        return False

assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel( s: str ) -> bool:
    '''Returns True if string is a vowel and False if otherwise'''
    vowels = 'aeiouAEIOU'
    return s in vowels

def nonvowels( s:str ) -> str:
    result = ''
    for x in s:
        if not is_vowel(x):
            result = result + x
    return result
assert nonvowels('orange') == 'rng'

'''Number 4'''
def consonants(s: str) -> str:
    '''Takes a string and returns a new string containing only consonants'''
    alphabet = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    c = ''
    for x in s:
        if x in alphabet and not is_vowel(x):
            c = c + x
    return c
assert consonants('Happy Birthday!') == 'HppyBrthdy'
assert consonants('computers') == 'cmptrs'

'''Number 5'''
def select_letters(l: str, w: str) -> str:
    '''Returns all the vowels in the word, w, if l is 'v'. If l is 'c', returns all the consonants in w'''
    s = ''
    if l == 'v':
        for x in w:
            if is_vowel(x):
                s = s+x
    elif l == 'c':
        for x in w:
            if not is_vowel(x):
                s = s+x
    return s
assert select_letters('v','facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('a', 'facetiously') == ''


'''Number 6'''
def hide_vowels(s:str) -> str:
    '''Takes a string and returns a string, replacing the vowels with hyphens'''
    for x in s:
        if is_vowel(x):
            s = s.replace(x, '-')
    return s
assert hide_vowels('Hello') == 'H-ll-'
assert hide_vowels('Hello, I am an Anteater.') == 'H-ll-, - -m -n -nt--t-r.'

def average_price (l:list) -> float:
    '''returns average prices of restaurants'''
    sum = 0
    for x in l:
        sum = sum + x.price
    return sum / len(l)
