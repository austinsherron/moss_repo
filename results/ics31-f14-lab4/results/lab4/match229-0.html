def nonvowels(x:str) -> str:
    ''' takes a string and returns a string containing all the characters that are not vowels'''
    nonVowels = ''
    for g in x:
        if is_vowel(g) == False:
            nonVowels += g #can also be written nonVowels = nonVowels + g
    return (nonVowels)

def hide_vowels(x: str) -> str:
    '''takes a string and returns a string with vowels replaced by a hyphen'''
    newStr = ''
    for i in x:
        if is_vowel(i):
            newStr += '-'
        else:
            newStr += i
    return newStr

    '''takes a list of restaurant and returns a list of the names in alphabetical order by NAME'''

def average_price(restaurants: list) -> float:
    '''returns the average price of the best dish at thoes restaurants'''
    sumOfRest = 0
    for i in restaurants:
        sumOfRest += i.price
    average = sumOfRest/len(restaurants)
