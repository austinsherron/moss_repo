def hide_vowels (n:str):
    '''Return a string in which every vowel in the parameter is replaced
       with a hyphen ('-') and other characters remain unchanged '''
    result = ''
    for i in n:
       if i in 'AaEeIiOoUu':
           result = result + '-'
       else:
           result = result + i
    return result
assert hide_vowels('I like pie') == '- l-k- p--'
