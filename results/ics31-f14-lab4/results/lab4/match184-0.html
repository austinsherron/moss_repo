def test_number (x:int, y:str) -> bool:
    ''' returns True if the number has the property
        indicated by the string, and False if it doesn't.'''
    if x%2 == 0 and y == 'even':
       return True
    elif x%2 != 0 and y == 'odd':
       return True
    elif x >= 0 and y == 'positive':
       return True
    elif x <= 0 and y == 'negative':
       return True
    else :
       return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

print(test_number(15, 'negative'))

def hide_vowels(x:str)->str:
    '''take a str and return a str which every vowels in it is
    replaced by "-", and all other characters remain the same.'''
    result=''
    for i in x:
        if is_vowel(i)==True:
            i = '-'
        else:
            i=i
        result = result+i
    return result
