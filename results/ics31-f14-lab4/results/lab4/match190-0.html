def test_number (n:int,s:str) -> str:
    '''Return True or False based on the property of the number'''
    if n % 2 == 0 and s == 'even':
        return True
    elif n % 2 != 0 and  s == 'odd':
        return True
    elif n > 0 and s == 'positive':
        return True
    elif n < 0 and s == 'negative':
        return True
    else:
        return False

def hide_vowels(s:str)-> str:
    '''return a string in which vowels are replaced by hyphens'''
    h = ''
    for x in s:
        if is_vowel(x) == True:
            h += '-'
        elif is_vowel(x) == False:
            h += x
    return h
