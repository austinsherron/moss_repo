def test_number(x: int, y: str) -> bool:


    if x%2 == 0 and y == 'even':
        return True
    elif x%2 == 0 and y == 'odd':
        return False
    elif x%2 != 0 and y =='odd':
        return True
    elif x%2 != 0 and y == 'even':
        return False
    elif x > 0 and y == 'positive':
        return True
    elif x > 0 and y == 'negative':
        return False
    elif x < 0 and y == 'negative':
        return True
    elif x < 0 and y == 'positive':
        return False
    
    assert test_number(14, 'even')
    assert not test_number(100, 'odd')
    assert test_number(33, 'positive')
    assert not test_number(100, 'negative')
print(test_number(4, 'even')) #prints True
print(test_number(3, 'negative')) #prints False

def select_letters(choice: str, obj: str) -> str:
    s = ""
    if choice == 'v':
        for i in obj:
            if i in 'aeiouAEIOU':
                s = s + i
        return(s)
    elif choice == 'c':
        for i in obj:
            if i not in 'aeiouAEIOU':
                s = s + i
        return(s)
    

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
