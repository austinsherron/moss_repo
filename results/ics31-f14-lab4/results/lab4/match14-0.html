def nonvowels(a:str) -> str: #Takes a string and returns a string containing all nonvowel chars
    result = ''
    for i in range(len(a)):
        if is_vowel(a[i]) == False:

def consonants(b:str) -> str: #Takes a string and returns consonant letters in a new string
    s = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    result = ''
    for i in range(len(b)):
        if (b[i] in s) == True:
            result = result + b[i]
    return result

assert consonants('123 !@_ Hello There') == 'HllThr'
assert consonants('alskdjf;;sal;clk') == 'lskdjfslclk'

def hide_vowels(x:str) -> str: #Input string, return string with vowels replaced with '-'


    result = ''
    for i in range(len(x)):

            result = result + '-'

from collections import namedtuple

Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')

def Restaurant_change_price(x:Restaurant, y:float) -> Restaurant: #Input a restaurant and float, output restaurant with price increased by inputted float
    d = y + x.price
    x = x._replace(price=d)
    return x
