def test_number(n: int, x: str) -> bool:
    if n % 2 == 0 and x == "even":
        return True
    if n % 2 == 1 and x == "odd":
        return True
    if n < 0 and x == "negative":
        return True
    if n > 0 and x == "positive":
        return True
    else:
        return False

def average_price(n: list) -> float:
    x = 0.00
    for i in n:
        x = x + i.price
    return x/len(n)
