def test_number (a: int, check: str) -> bool:
    if check == 'even':
        return a % 2 == 0
    elif check == 'odd':
        return a % 2 != 0
    elif check == 'positive':
        return a > 0
    else:
        return a < 0
assert test_number(14, 'even') == True
print ('14 is even: ',test_number(14, 'even'))
assert not test_number(100, 'odd') == True
print ('100 is odd: ',test_number(100, 'odd'))
assert test_number(33, 'positive') == True
print ('33 is positive: ',test_number(33, 'positive'))
assert not test_number(100, 'negative') == True
print ('100 is negative: ',test_number(100, 'negative'))

from collections import namedtuple

Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')      
def Restaurant_change_price (r: Restaurant, change: float) -> Restaurant:
    result = r._replace(price = r.price + change)

def create_rectangle_from_center(x:int,y:int,l:int,w:int)-> tkinter:
    '''Given the center the rectangle and length/width via ints, this function
    will produce a rectangle accordingly'''
    return my_canvas.create_rectangle(x-l/2,y-w/2,x+l/2,y+w/2)
