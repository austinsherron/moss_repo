    '''This function determine if the number entered has the property indicated by the string'''

        return n%2 ==0
    if s == 'odd':
        return n%2 == 1
    if s == 'positive':
        return n>0
    if s == 'negative':
        return n<0

                 ,'l','m','n','p','q','r','s','t'
                 ,'v','w','x','y','z','B','C','D'
                 ,'F','G','H','J','K','L','M','N'
                 ,'P','Q','R','S','T','V','W','X'
                 ,'Y','Z']

def select_letters(n: str, s: str) ->str:
    '''Return string based on n. If n is letter v then the function return a string
    with only vowel letter. If n is letter c then the string contains consonants or else
    the function return a blank line'''
    if n == 'v':
        result = ''
        for x in s:
            if is_vowel(x) == True:
                result+=str(x)
        return result
    if n == 'c':
        result = ''
        for x in s:
            if check_nonvowels(x) == True:
                result+=str(x)
        return result
    else:
        return ' '

def hide_vowels(s: str) -> str:
    '''Replace the vowels in a string with hyphen'''
    result = ''
    for x in s:
        if is_vowel(x) == True:
            result+='-'
        else:
            result+=str(x)
    return result
print(hide_vowels('making Money Online'))
assert hide_vowels('abcdefg') == '-bcd-fg'

import tkinter

my_canvas = tkinter.Canvas(my_window, width = 500, height = 500) #Creates a 500x500 canvas
my_canvas.pack() #Puts canvas into window

def create_rectangle_from_center (x: int, y: int, height: int, width: int):
    '''Determine the coordinates of the upper left and lower right corners of
    a rectangle using the center point with width and height'''
    my_canvas.create_rectangle(x-(width/2),y-(height/2),x+(width/2),y+(height/2), fill = 'blue')
