def test_number(n:int, s:str) -> bool:
    '''takes a number and returns a string'''
    if n % 2 == 0 and s == 'even':
        return True
    elif n % 2 == 1 and s == 'odd':
        return True
    elif n > 0 and s == 'positive':
        return True
    elif n < 0 and s == 'negative':
        return True
    else:
        return False

assert not test_number(100, 'negative')

def hide_vowels(x:str) -> str:
    '''hides vowels with a '-' '''
    g = list(x)
    for i in range(len(g)):
        if is_vowel(g[i]) == True:
            g[i] = '-'
    x = ''.join(g)
    return x
