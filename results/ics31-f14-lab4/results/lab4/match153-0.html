    '''Returns True if the number has the property indicated by the string, and False if it doesn't'''
    if n>0 and s=='positive':
        return True
    if n<0 and s=='negative':
        return True
    if n%2==0 and s=='even':
        return True
    if n%2!= 0 and s=='odd':
        return True
    else:
        return False

def consonants(n:str)-> str:
    ''' Return a string containing all the letters in the parameter string that are not vowels'''
    con=['q','w','r','t','y','p','s','d','f','g','h','j','k','l','z','x','c','v','b','n','m','Q','W','R','T','Y','P','S','D','F','G','H','J','K','L','Z','X','C','V','B','N','M']
    new_string=''
    for characters in n:
        if characters in con:
            new_string= new_string + characters
    return new_string

assert consonants('he@#$%llo')=='hll'
assert consonants('good*&^%night')=='gdnght'
assert consonants('HIGH*&^DEF')=='HGHDF'

    vowels='aeiouAEIOU'
    result=''
    for i in n:
        if i in con:
            result= result+ i
        if i in vowels:
            result= result+ '-'
