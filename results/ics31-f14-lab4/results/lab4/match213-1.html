def test_number (n:int, s:str) -> str:


    if n%2 == 0 and s == 'even':
        return 'true'
    elif n%2 != 0 and s == 'odd':
        return 'true'
    elif n > 0 and s == 'positive':
        return 'true'
    elif n < 0 and s == 'negative':
        return 'true'
    else:
        return 'false'

def hide_vowels(hvs:str) -> str:
    'takes a str and returns a str in which every vowel in the parameter is replaced with a hyphen ("-") all other characters unchanged'
    result = ''
    for ev in hvs:
        if ev in 'aeiouAEIOU':
