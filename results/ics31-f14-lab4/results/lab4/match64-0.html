        return True
    elif num<0 and s=='negative':
        return True

def nonvowels(char: str) -> str:
    "Take an inputted string, convert each character to upper case then returns a string of the nonvowel characters"
    result = ""
    for i in char:
        if not is_vowel(i):
            result += i
    return result

def consonants(char: str) -> str:
    "Takes an inputted string, converts each character to upper case then returns a string of the consonant characters"
    result = ""
    for i in char:
        if i.upper() in 'BCDFGHJKLMNPQRSTVWXZ':
            result += i
    return result

print(consonants('The Alphabet'))

def select_letters(mode: str, word: str) -> str:
    '''If mode is 'v' returns a string containing the vowels in word. If mode is 'c' returns a
    string containing the consonants in word. If mode is anything else, return empty string'''
    if mode == 'v':
        result = ""
        for i in word:
            if is_vowel(i):
                result += i
        return result
    if mode == 'c':
        return consonants(word)
    else:
        return ''

print(select_letters('v', 'Circus act'))
print(select_letters('c', 'Circus act'))

print(select_letters('anything', 'anything'))
print(select_letters('v', '17'))

def hide_vowels(sentence: str) -> str:
    "Take a string and replace all of the vowels with a hyphen"
    result = ''
    for i in sentence:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result
