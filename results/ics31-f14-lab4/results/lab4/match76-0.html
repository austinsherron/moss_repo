def is_vowel(s:str)->bool:
    ''' takes a one-character-long string and returns True if that character is a vowel and False otherwise. [A vowel, for our purposes here,
    is one of the letters a, e, i, o, or u (lower case or upper case).] '''
    vowels='aeiouAEIOU'
    for i in vowels:
        if i==s:
            return i==s

assert is_vowel('o')==True

def nonvowels(n:str)->str:


    '''takes a string and returns a string

conso = ['b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n', 'p', 'q','r','s','t','v','w','x','y','z']
def cons(c: str) -> bool:
    return c.lower() in conso
# You will need to plug in 1 string and it will return a string with only the consonants.
def consonants(z:str) -> str:
    #takes a string and returns a string containing all the letters in the
    #parameter string that are not vowels.
    result= ""
    for c in z:
        if cons(c):
            result += c
    return result
print(consonants('thgji23 t'))
assert consonants('Hey2334') == 'Hy'
assert consonants('You_There 32!') == 'YThr'

def hide_vowels(x: str) -> str:
    '''Makes all vowels as a -.'''
    result = ''
    for i in x:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result 
    
print(hide_vowels('Hello'))
assert hide_vowels('Hello') == 'H-ll-'
assert hide_vowels('Howwas') == 'H-ww-s'

def create_rectangle_from_center(x:int,y:int,w:int,h:int)->'rectangle':
    ''' takes four arguments: the x-value and y-value of the point in the center
    of the rectangle and the height and width of the rectangle.'''
    x0=x-(w)/2
    y0=y-(h)/2
    x1=w+x0
    y1=h+y0
    my_canvas.create_rectangle(x0,y0,x1,y1,fill='blue')
create_rectangle_from_center(500,500,100,100)
