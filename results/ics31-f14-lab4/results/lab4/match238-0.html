def test_number(num: int, string:str) -> bool:
    if string == 'even':
        if num % 2 == 0:
            return True
        else:
            return False
    if string == 'odd':
        if num % 2 != 0:
            return True
        else:
            return False
    if string == 'positive': 
        if num >= 0:
            return True
        else:
            return False
    if string == 'negative': 
         if num < 0:
            return True
         else:
            return False

assert not test_number(100, 'odd')

def nonvowels(string:str) -> str:
    #return a string of characters that aren't vowels from the variable string
    result = ''
    for i in range(len(string)):
        if(not is_vowel(string[i])):
            result += string[i] #result = result + string[i]
    return result
assert nonvowels('chicken') == 'chckn'
assert nonvowels('elephant') == 'lphnt'
assert nonvowels('3 ,') == '3 ,'

def hide_vowels(string:str):
    #replaces every vowel with a hyphen in string
    result = ''
    for i in range(len(string)):
        if is_vowel(string[i]):
            result += '-'
        else:
            result += string[i]
    return result


assert hide_vowels('chicken') == 'ch-ck-n'
assert hide_vowels('elephant') == '-l-ph-nt'
