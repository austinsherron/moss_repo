def test_number(number, s: str) -> bool :
    """"detect if number match the property"""
    if s=='positive' and number>0:
        return True
    elif s=='positive' and number<0:
        return False
    elif s=='negative' and number>0:
        return False
    elif s=='negative' and number<0:
        return True
    elif s=='even' and number%2==0:
        return True
    elif s=='even' and number%2==1:
        return False
    elif s=='odd' and number%2==0:
        return False
    else:
        return True

def consonants (s:str)->str:
    '''return a string that contains the nonvowel letters in the input'''
    result=''
    for i in s:
        if i not in ['a','e','i','o','u','A','E','I','O','U'] and ((ord(i)>=65 and ord(i)<=90) or (ord(i)>=97 and ord(i)<=122)):
            result+=i
    return result

def hide_vowels(s:str)->str:
    '''return a string with every vowel character hidden'''
    result = ''
    for i in s:
        if is_vowel(i):
            result+='-'
        else:
            result+=i
    return result
