def test_number(num:int, condition:str) -> bool:
    ''' Prints if the condition is satisfied by num'''
    if(condition == 'even' and (num % 2) == 0):
        return True
    elif(condition == 'odd' and (num % 2) == 1):
        return True
    elif(condition == 'positive' and num > 0):
        return True
    elif(condition == 'negative' and num < 0):
        return True

    return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(c: str) -> bool:
    ''' Returns True/False if c is a vowel'''
    return (c in 'aeiou')

print(is_vowel('a'))
print(is_vowel('d'))

def print_nonvowels(word:str) -> None:
    ''' Prints all the nonvowels in word'''
    for c in word:
        if not is_vowel(c):
            print(c)

print_nonvowels("loloa")

def nonvowels(word:str) -> str:
    '''Returns a string of all the nonvowels in word'''
    result = ''
    for c in word:
        if not is_vowel(c):
            result = result + c
    return result

print(nonvowels("onomatopeia"))
assert not nonvowels('a')
assert nonvowels('b')
assert nonvowels('?')

def consonants(word: str) -> str:
    ''' Returns a string of all the consonants in word'''
    result = ''
    for c in word:
        if (c in map(chr, range(97, 123))) and not is_vowel(c):
            result = result + c
    return result

assert consonants('pythonisdabes') == 'pythnsdbs'

def select_letters(condition:str, word:str) -> str:
    ''' Returns a string of characters in word that satisify the condition'''
    result_vowels = ''
    if condition == 'v':
        for c in word:
            if is_vowel(c):
                result_vowels = result_vowels + c
        return result_vowels
    if condition == 'c':
        return consonants(word)

print(select_letters('v','bicboi'))
print(select_letters('c','bicboi'))

def hide_vowels(word:str) -> str:
    result = ''
    for i in range(len(word)):
        if is_vowel(word[i]):
            result = result + '-'
        else:
            result = result + word[i]
    return result

def alphabetical_names(restaurants: list) -> list:
    '''Returns list of restaurant names in alphabetical order'''
    alphylist = []

def average_price(restaurants: list) -> float:
    '''Returns average price from list of restaurants'''
    sum = 0
    for x in range(0, len(restaurants)):
        sum = sum + restaurants[x].price
    return sum / len(restaurants)
