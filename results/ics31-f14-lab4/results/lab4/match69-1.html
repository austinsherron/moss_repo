   if o == 'positive':
         return n >= 0
   if o == 'negative':
         return n < 0

def consonants(con: str) -> str:
   '''Returns all letters that are not vowels'''
   result1 = ''
   for t in con:
      if t.lower() in 'bcdfghjklmnpqrstvwxyz':

   return result1

def select_letters(par1: str, par2: str) -> str:
   ''' Returns a string containing all the vowels in the second parameter or all the consonants in the second parameter'''
   empty_string= ''
   if 'v' == par1:
      for letter in par2:
         if letter in 'aeiouAEIOU':
            empty_string=empty_string + letter
      return empty_string
   elif 'c' == par1:
      for letter in par2:
         if letter not in 'aeiouAEIOU':
            empty_string=empty_string + letter
      return empty_string

def hide_vowels(value: str) -> str:
   ''' Returns a string in which every vowel in the parameter is replaced with a hyphen'''
   new_str=''
   for v in value:
      if v in 'aeiouAEIOU':
         new_str = new_str + '-'
      else:
         new_str = new_str + v
   return new_str

assert hide_vowels('cookie')

import tkinter
def create_rectangle_from_center(x, y, height, width):
   '''Creates a rectangle from the center using the parameters'''
   my_window = tkinter.Tk()
   my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
   my_canvas.pack()
   my_canvas.create_rectangle(x-(width/2), y-(height/2), x+(width/2), y+(height/2))

create_rectangle_from_center(300, 300, 400, 500)

tkinter.mainloop()
