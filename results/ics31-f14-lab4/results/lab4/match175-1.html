def test_number(n:float, y:str):
    if n % 2 == 0 and y == 'even':
        return True
    elif n % 2 == 0 and y == 'odd':
        return False
    elif n % 2 != 0 and y == 'odd':
        return  True
    elif n % 2 != 0 and y == 'even':
        return False
    elif n > 0 and y == 'positive':
        return True
    elif n > 0 and y == 'negative':
        return False
    elif n < 0 and y== 'negative':
        return True
    elif n < 0 and y == 'positive':
        return False

from collections import namedtuple

Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')

def Restaurant_change_price(r:Restaurant,n:float):
    new_price = r.price + n
    r = r._replace(price = new_price)
    return r
