def test_number (n: int, m: str) -> bool:
    '''Return True if the number has the property indicated by the string,
       and False if it doesn't'''
    if m == 'even':
        if n%2 == 0:
            return True
    elif m == 'odd':
        if n%2 == 1:
            return True
    elif m == 'positive':
        if n > 0:
            return True
    else:
        if n < 0:
            return True
    return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def select_letters (n: str, word: str) -> str:
    '''If 1st parameter is 'v', returns a string containing all vowles in the 2nd
    parameter. If 1st parameter is 'c', returns a string containing all the consonant
    in 2nd parameter, else, returns empty string'''
    result = ''
    if (n == 'v'):
        for i in range (len(word)):
            if is_vowel (word[i])==True and word[i].isdigit() ==False:
                result += word[i]
        return result
    elif(n == 'c'):
        return consonants(word)
    else:
        return result

assert select_letters('v', 'alphaomega32')== 'aaoea'
assert select_letters('c', 'selectfood') == 'slctfd'
assert select_letters('f', 'yolo') == ''

def hide_vowels (n:str) -> str:
    '''Takes a string and returns a string in which every vowel in the parameter
    is replaced with a hyphen and all other characters remain unchanged.'''
    result = ''
    for i in range (len(n)):
        if is_vowel (n[i]):
            result += '-'
        else:
            result += n[i]
    return result

assert hide_vowels ('My name is Shannon') == 'My n-m- -s Sh-nn-n'
assert hide_vowels ('Suyash Misra is my TA') == 'S-y-sh M-sr- -s my T-'
assert hide_vowels ('December 25 is Christmas') == 'D-c-mb-r 25 -s Chr-stm-s'

    '''Takes list of restaurants and returns a list of the names of all the restaurants
    in alphabetical order'''
    result = []
