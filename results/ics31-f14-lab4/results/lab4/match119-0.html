        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative':
        return True
    else:
        return False

def is_vowel(x:str) ->bool:
    '''Takes a letter and returns true if the letter is a vowel'''
    if x == 'a':
        return x == 'a'
    elif x == 'e':
        return x == 'e'
    elif x == 'i':
        return x == 'i'
    elif x == 'o':
        return x == 'o'
    elif x == 'u':
        return x == 'u'
    else:
        return False
assert is_vowel('a')
assert not is_vowel('b')
assert is_vowel('u')

def nonvowels(x:str) ->str:
    '''Takes a string and returns a string with all the characters of the parameter that are not vowels'''
    result = ''
    for i in x:
        if not is_vowel(i):
            result = result + i
    return result
assert nonvowels('Tigger!!!') == 'Tggr!!!'
assert nonvowels('Cody.') == 'Cdy.'

def consonants(x:str) ->str:
    '''Takes a string and returns string containing all consonants'''
    result = ''
    consonant_alphabet = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    for i in x:
        for t in consonant_alphabet:
            if i == t:
                result = result + i
    return result
assert consonants('Gina$') == 'Gn'
assert consonants('Kawaii desu') == 'Kwds'

def select_letters(x:str, y:str) ->str:
    '''If the first string is v it returns a string of vowels in the second parameter
    and if the first string is c it returns a string of the consonats in the second parameter
    if the first string is anything else it returns an empty string'''
    result = ''
    if x == 'v':
        result = vowels(y)
    elif x == 'c':
        result = consonants(y)
    else:
        result
    return result

assert select_letters('i', 'Taco') == ''
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(x:str) ->str:
    '''Takes a string and returns a string with hyphens in place of vowels'''
    result = ''


    for i in x:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result

def average_price(list_of_rests: list)-> float:
    '''takes a list of restaurants and returns the average of the prices'''
    list_of_prices = []
    for i in list_of_rests:
        list_of_prices.append(i.price)
