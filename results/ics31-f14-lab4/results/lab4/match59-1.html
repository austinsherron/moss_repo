def test_number (n: int, s:str) -> str:
    """takes number and string such as odd, even, positive, or negative
    and returns a True or False."""
    if s == 'odd':
        return n%2 != 0
    elif s == 'even':
        return n%2 == 0
    elif s == 'positive':
        return n > 0
    elif s == 'negative':
        return n < 0

assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def select_letters(s: str, t:str)-> str:
    """takes two parameters, both strings, and returns a
    string. If the first parameter is 'v', it returns a string
    containing all the vowels in the second parameter; if the first
    parameter is 'c', it returns a string containing all the consonants
    in the second parameter. If the first parameter is anything else, it
    returns the empty string."""
    result = ''
    if s == 'v':
        for i in t:
            if is_vowel(i) == True:
                result = result + i
        return result
    elif s == 'c':
        return consonants(t)
    elif s != 'v' or 'c':
        return result
print(select_letters('', 'qwertyuiop'))

    of (the best dishes at) those restaurants."""
