def test_number(n: int, s: str)->bool:
    '''Takes a number and a string and return True if it has the property
    indicated by the string, and False if it doesn't.'''
    if n%2 == 0 and s == 'even':
        return True
    elif n%2 == 1 and s == 'odd':
        return True
    elif n > 0 and s == 'positive':
        return True
    elif n < 0 and s == 'negative':
        return True
    else:
        return False


assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(s: str)->str:
    '''Return a string containing all the characters in the parameter string
    that are not vowels.'''
    result = '' 
    for i in s:
        if i not in 'aeiouAEIOU': #Added for lowercase and uppercase.
            result = result + i
    return result

def hide_vowels(s: str)->str:
    '''Return a string where every vowel is replaced with a hyphen and
    the characters are unchanged.'''
    hide =''
    for i in s:
        if is_vowel(i) == True:
            hide = hide + '-'
        else:
            hide = hide + i
    return hide
print (hide_vowels('Cupcakes are good.'))

assert hide_vowels('How are you?') == "H-w -r- y--?"
