def test_number(n:int, s:str)->bool:
    """Compare the two inputs and return True or False"""
    if s== 'even':
        return n%2 == 0
        
    elif s == 'odd':
        return n%2 != 0
        
    elif s == 'positive':
        return n>0
        
    elif s == 'negative':
        return n<0

def nonvowels(v:str)->str:
    """Take a string and return a string containing all the characters in the parameter that are not vowels"""

def consonants(c: str)->str:
    """Take a string and return a string containing all the letters in the parameter string that are consonants"""
    word=''
    for x in c:
        if not is_vowel(x) and x.isalpha():

    return word

def select_letter(p1: str, p2: str)->str:
    """Takes two parameters, both strings, and returns a string. If the first parameter is 'v', return a string containing
all the vowels in the second parameter. If the first parameter is 'c', return a string containing all the consonants in the
second parameter. If the first paraemeter is anything else, return the empty string."""
    result=''
    if p1== 'v':
        for x in p2:
            if is_vowel(x):
               result+=x
    elif p1== 'c':
        for x in p2:
            if not is_vowel(x):
               result+=x 
    return result

def hide_vowels(s: str)->str:
    """Take a string and return a string in which every vowel in the parameter is replaced with a hypen("-") and all other


characters remain unchanged"""
    hypen=''
    for x in s:
        if is_vowel(x):
            hypen+='-'
        else:
            hypen+=x
    return hypen

assert hide_vowels('Nayoung') == 'N-y--ng'
assert hide_vowels('Ch3mistry') == 'Ch3m-stry'
