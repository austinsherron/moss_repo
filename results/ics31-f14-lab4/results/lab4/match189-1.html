def test_number(i: int, qualifier: str) -> bool:
    if qualifier == 'even':
        return i % 2 == 0
    elif qualifier == 'odd':
        return i % 2 != 0
    elif qualifier == 'positive':
        return i > 0
    elif qualifier == 'negative':
        return i < 0
    else:
        return False

CONSONANTS = ['b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n', 'p', 'q', 'r', 's', 't', 'v', 'w', 'x', 'y', 'z']

def consonants(letters: str) -> str:
    '''Returns only the consonants in a given string.'''
    result = ""
    for letter in letters:
        if is_consonant(letter):
            result += letter
    return result

def vowels(letters: str) -> str:
    '''Returns only the vowels in a given string.'''
    result = ""
    for letter in letters:
        if is_vowel(letter):
            result += letter
    return result
assert vowels('abcdefghijklmnopqrstuvwxyz') == 'aeiou'
def select_letters(selector: str, letters: str) -> str:
    if selector == 'v':
        return vowels(letters)
    elif selector == 'c':
        return consonants(letters)
    else:
        return ""
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('potato', 'facetiously') == ''

def hide_vowels(letters: str) -> str:
    '''Replaces vowels in a string with hyphens'''
    result = ""
    for letter in letters:
        if is_vowel(letter):
            result += '-'
        else:
            result += letter
    return result

def average_price(restaurants:list) -> float:
    total_price = 0
    for restaurant in restaurants:
        total_price += restaurant.price
    return total_price / len(restaurants)
