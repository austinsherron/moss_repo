def test_number(n: int, s: str) -> bool:
    ''' Test that a number is even, odd, positive, or negative '''
    if s == 'even':
        return n%2 == 0
    elif s == 'odd':
        return n % 2 != 0
    elif s == 'positive':
        return n > 0
    elif s == 'negative':


        return n < 0
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
           
print(test_number(15, 'odd'))
print(test_number(-100, 'positive'))

    on list of area codes '''

def matching_area_code(x: list, n: list):
    ''' Takes a list of area codes and prints a list of phone numbers whose area code is
    on list of area codes '''
    result = []
    for number in n:
        if number[1:4] in x:
            result.append(number)
    return result

def is_vowel(x: str):
    ''' Takes a one-character-long string and returns True if that character
    is a vowel and False otherwise. '''
    return x in ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
assert is_vowel('a')
assert is_vowel('U')
assert not is_vowel('P')

# Part (d)(2)
def print_nonvowels(x: str):
    ''' Prints out all characters in string that are NOT vowels '''
    for i in x:
        if is_vowel(i) == False:
            print(i)

# Part (d)(3)
def nonvowels(x: str) -> str:
    ''' Returns a string contaning all the characters in the paramater that
    are not vowels '''
    result = ''
    for i in x:
        if is_vowel(i) == False:
            result += i
    return result
assert nonvowels('1341') == '1341'
assert nonvowels('bird') == 'brd'

def consonants(x: str) -> str:
    ''' Returns a string containing all the letters in the parameter string
    that are not vowels '''
    consonants = ['b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n', 'p',
                  'q', 'r', 's', 't', 'v', 'w', 'x', 'y', 'z', 'B', 'C', 'D','F',
                  'G', 'H', 'J', 'K', 'L', 'M', 'N', 'P',' Q', 'R', 'S', 'T',
                  'V', 'W', 'X', 'Y', 'Z']
    result = ''
    for i in x:
        if i in consonants:
            result += i
    return result
assert consonants('bear') == 'br'
assert consonants('???') == ''

# Part (d)(5)
def select_letters(x: str, y: str) -> str:
    ''' If first parameter is v, returns all vowels in 2nd parameter.
    If first parameter is c, returns all consonants in 2nd parameter. '''
    result = ''
    if x == 'v':
        for i in y:
            if is_vowel(i) == True:
                result += i
    if x == 'c':
        for i in y:
            result += consonants(i)
    return result
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(x: str) -> str:
    result = ''
    for i in x:
        if is_vowel(i) == False:
            result += i
        else:
            result += '-'
    return result
assert hide_vowels('apple') == '-ppl-'

def alphabetical_names(x: list) -> list:
    ''' Takes a list of restaurants and returns the list of names of restaurants
     in alphabetical order '''
    result = []
    R = sorted(x)

def average_price(x:list) -> float:
    ''' Takes list of restaurants and returns average price of dishes at


    those restaurants '''
    total = 0

        total += x[i].price
    average = total / len(x)
