def create_rectangle_from_center(x,y,h,w):
	'''
	x = x coord upper left corner
	y = y coord upper left corner
	z = length of side
	'''


	import tkinter         	 
	my_window = tkinter.Tk()    
	my_canvas = tkinter.Canvas(my_window, width=500, height=500)  
	my_canvas.pack()      	 
	my_canvas.create_rectangle(x - (w / 2), y - (h / 2), x + (w / 2), y + (h / 2), fill='')
	tkinter.mainloop()    	 

create_rectangle_from_center(300,200,100,200)
