def nonvowels(s:str):
    '''Takes a string and returns a string containing all the characters in the parameter string that are not vowels.'''
    non_vowels = ''
    for a in s:
        if not is_vowel(a):
            non_vowels += a
    return non_vowels
assert nonvowels('aoizxcvbnm') == 'zxcvbnm'
assert nonvowels('QWERTYUIOP') == 'QWRTYP'
assert nonvowels('QW123 @#aou') == 'QW123 @#'

def consonants(s:str):
    '''Takes a string and returns a string containing all the letters in the parameter string that are not vowels.'''
    non_vowels = ''
    for a in s:
        if not is_vowel(a) and a.isalpha():
            non_vowels += a
    return non_vowels
assert nonvowels('aoizxcvbnm') == 'zxcvbnm'
assert nonvowels('QWERTYUIOP') == 'QWRTYP'
assert nonvowels('QW123 @#aou') == 'QW123 @#'
print()

def vowels(s:str):
    '''Takes two parameters, both strings, and returns a string.'''
    vowel_letters = ''
    for a in s:
        if is_vowel(a):
            vowel_letters += a
    return vowel_letters
def select_letters(a:str, b:str)->str:
    if a == 'v':
        return vowels(b)
    elif a== 'c':
        return consonants(b)

def hide_vowels(s:str):
    '''Takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen and all other characters remain unchanged.'''
    hide = ''
    for a in s:
        if not is_vowel(a):
            hide += a
        else:
            hide += '-'
    return hide
assert hide_vowels('asdfgiou') == '-sdfg---'
assert hide_vowels('ERTYASD') == '-RTY-SD'
