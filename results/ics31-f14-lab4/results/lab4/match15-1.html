def consonants (n:str) -> str:
    letter = ''
    letters = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    for i in n:
        if i in letters:
            if is_vowel(i) == False:
                letter = letter + i
    return letter

def average_price(l:list) -> float:
    price_sum = 0
    for i in l:
        price_sum = price_sum + i.price

    return average
