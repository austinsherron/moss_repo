def consonants(s:str) -> str:
        "Return consonants from a string"
        consonants= 'bcdfghjklmnpqrstvxzwyBCDFGHJKLMNPQRSTVXZWY'
        consonants_only=''
        for letter in s:
                if letter in consonants:
                        consonants_only = consonants_only + letter
        return consonants_only
print(consonants('building12'))
print(consonants('office'))
assert consonants('money') == 'mny'
assert consonants('laptop') == 'lptp'
