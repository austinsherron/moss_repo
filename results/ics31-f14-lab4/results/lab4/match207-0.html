def test_number(x: int,attribution: str) -> bool:
    ''' Checks to see if the input number is what you say it is'''
    num_attr = ''
    if x%2 == 0:
        num_attr += 'even'
    else:
        num_attr += 'odd'

    if  x > 0:
        num_attr += 'positive'
    else:
        num_attr += 'negative'

    return attribution in num_attr  #Returns True or False

def matching_area_codes(area: list,phone: list) -> list:

def consonants(string: str) -> str:
    ''' Returns all consonants in the string '''
    result = ''
    for i in string:
        if not is_vowel(i) and i in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
            result += i
    return result

                result += i

def hide_vowels(sentence: str) -> str:
    """ Replaces vowels with '-' """
    temp = []
    result = ''
    for i in sentence:
        temp.append(i)
    
    for j in range(len(temp)):
        if temp[j] in 'aeiouAEIOU': # Replacing the vowels with a hyphen
            temp[j] = '-'
    
    for s in temp:
        result += s
    
    return result
    
assert hide_vowels('I am a student.') == '- -m - st-d-nt.'
hide_vowels('I am a student.')
print('\n')

def sum_price(Res_list: list) -> float:
    ''' Returns a sum of the Restaurants '''
    total = 0
    for i in Res_list:
        total += i.price
    
    return total

def average_price(Res_list: list) -> float:
    '''Returns the average price of the best dishes at the restaurants'''
    average = sum_price(Res_list)/len(Res_list)
