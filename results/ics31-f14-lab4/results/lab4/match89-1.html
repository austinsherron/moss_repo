def test_number (number: int, sign: str)-> bool:
    '''return number and tells whether its even, odd, positive, or negative'''
    if number % 2 == 0 and sign == even: 
        return True
    if number % 2 == 1 and sign == odd:
        return True
    if number < 0 and sign == negative:
        return True
    if number > 0 and sign == positive:
        return True
    else: 
        return False

def nonvowels(x: str) -> str:
    '''Return string containing all characters in parameter string that are not vowel'''
    result = ''
    for i in x:
        if not is_vowel(i):
            result = result + i
    return result

def hide_vowels(x: str)-> str:
    '''Returns a string with every vowel replaced by a hyphen'''
    result = ''
    for i in x:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result

assert (hide_vowels('computer') == 'c-mp-t-r')
assert (hide_vowels('t') == 't')

def create_rectangle_from_center (x: float, y: float, h: float, w: float)-> None:
    '''Draw a rectangle'''
    my_canvas.create_rectangle(x-w/2, y-h/2, x+w/2, y+h/2)
