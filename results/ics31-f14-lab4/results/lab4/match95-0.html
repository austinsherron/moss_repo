def test_number(num: int, num_type: str) -> bool:
    ''' Receives a number and and string, returns result whether number has
        the preperity indeicated by the string '''
    if num_type == 'even':
        return num % 2 == 0
    elif num_type == 'odd':
        return num % 2 == 1
    elif num_type == 'positive':
        return num > 0
    elif num_type == 'negative':
        return num < 0
    else:
        print("Invalid parameter.")

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(string: str) -> str:
    ''' Receives a string and returns all characters in parameter string that are not vowel '''
    nonvowels = ''
    for i in string:
        if is_vowel(i) == False:
            nonvowels = nonvowels + i
    return nonvowels

assert nonvowels('abcde123') == 'bcd123'
assert nonvowels('aeiou') == ''
assert nonvowels('xyz!?.') == 'xyz!?.'

def consonants(string: str) -> str:
    ''' Receives a string and returns a string containing all the letters in parameter string
        that are not vowel '''
    nonvowels = ''
    for i in string:
        if is_vowel(i) == False:
            if i.isalpha() == True:
                nonvowels = nonvowels + i
    return nonvowels

assert consonants('abcde123') == 'bcd'
assert consonants('aeiou')  == ''
assert consonants('xyz!?.') == 'xyz'

def select_letters(str1: str, str2: str) -> str:
    ''' Receives two strings, if first string is 'v', returns a string containing all the vowels
        in second string; if first string is c, returns a string containing all the consonants in
        the second parameter; if first string is anything else, returns empty string '''
    characters = ''
    if str1 == 'v':
        for i in str2:
            if is_vowel(i):
                characters = characters + i
    if str1 == 'c':
        characters = consonants(str2)
    return characters
        
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('b', 'facetiously') == ''

def hide_vowels(str1: str) -> str:
    ''' Receives a string and returns a string in which every vowel in parameter is replaced
        with '-' '''
    str2 = ''
    for i in str1:
        if is_vowel(i):
            str2 = str2 + '-'
        else:
            str2 = str2 + i
    return str2

assert hide_vowels('aeiou') == '-----'
assert hide_vowels('abcd!!!') == '-bcd!!!'
assert hide_vowels('123<>?') == '123<>?'

def average_price(restaurants: list) -> float:
    ''' Receives a list of restaurants and returns the average of price of those restaurants '''
    sum = 0
    for i in restaurants:
        sum = sum + i.price
    return sum / len(restaurants)

def create_rectangle_from_center(x: int, y: int, width: int, height: int) -> None:
    ''' Receives a x and y value of point in center of a rectangle and the hieght and width
        of the rectangle, and draw the rectangle '''
    x1 = x - width / 2
    x2 = x + width / 2
    y1 = y - height / 2
    y2 = y + height / 2
    my_canvas.create_rectangle(x1, y1, x2, y2)
    return
