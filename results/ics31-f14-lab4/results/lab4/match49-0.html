    '''Returns true if the number has the property indicated by the string and false if it doesn't.'''
    if(((a % 2 == 0) and (b == 'even')) or ((a % 2 == 1) and (b == 'odd'))):
        return True
    elif(((a >= 0) and (b == 'positive')) or ((a < 0) and (b == 'negative'))):
        return True
    else:
        return False

def match_area_codes(f: list, g: list) -> list:
    '''Returns a list of matching area code numbers to phone numbers'''
    return_list = []
    
    for i in range(len(f)):
        
        for j in range(len(g)):

def hide_vowels(o: str) -> str:
    '''Replaces all vowels with a hyphen in a string.'''
    result = ''
    for i in range(len(o)):
        if (not is_vowel(o[i])):
            result = result + o[i]
        elif(is_vowel(o[i])):

def average_price(y: list) -> float:
    '''Gets the average of all the prices of all the restaurants in a specific list.'''
    counter = 0
    total = 0.0
    for i in y:
        counter += 1
        total += i.price

    return total / counter
