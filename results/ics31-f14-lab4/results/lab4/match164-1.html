def test_number(x: int, y: str)->bool:
    if (y=='even' and x%2==0):
        return True
    elif (y=='odd' and x%2!=0):
        return True
    elif (y=='positive' and x >= 0):
        return True
    elif (y=='negative' and x < 0):
        return True
    else:
        return False

def hide_vowels(x: str) -> str:
    result = ''
    for item in range(0, len(x)):
        if (is_vowel(x[item]) == True):
            result = result + '-'
        else:
            result = result + x[item]
    return(result)
