def test_number(num: int, string: str)-> bool:
    '''returns if num meets string condition'''
    if(string == "even"):
        return num%2==0
    elif(string == "odd"):
        return num%2==1
    elif(string == "positive"):
        return num > -1
    elif(string =="negative"):
        return num < 0
    else:
        return "bad argument"

assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

print("2 is even:", test_number(2, "even"))

def nonvowels (word:str) -> str:
    '''prints all characters that are nonvowels in a string'''
    new_word=''

import string

def consonants (word:str) -> str:
    '''returns a string containing all of the consonants in word'''
    new_word = ('')
    for x in word:
        if x in string.ascii_letters:
            new_word += x
    new_word = nonvowels(new_word)
    return new_word
assert consonants('64 torches') == 'trchs'
assert consonants('!~@#$%^&*dfdgde') == 'dfdgd'

print (consonants('2 birds & 1 stone'))

def hide_vowels (word:str) -> str:
    '''returns a string where the vowels have been replaced by '-' '''
    new_word = ''
    for x in word:
        if is_vowel(x):
            new_word += '-'
        else:
            new_word += x


    return new_word
    
assert hide_vowels('Samantha') == 'S-m-nth-'
assert hide_vowels('Nick') == 'N-ck'

def average_price(rest_list:"list of restaurants")->float:
    '''returns average price of rests in rest_list'''
    total = 0
    for x in rest_list:
        total += x.price
    total /= len(rest_list)
    return total
