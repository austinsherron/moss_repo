def even_odd(a:int):
    if a % 2 == 0:
        return 'even'
    else:
        return 'odd'
    
def positive_negative(b:int):
    if b > 0:
        return 'positive'
    else:
        return 'negative'
    
def test_number(x:int, y:str) -> str:
    if x % 2 == 0 and y == 'even':
        return True
    if x % 2 != 0 and y == 'odd':
        return True
    if x > 0 and y == 'positive':
        return True
    if x < 0 and y == 'negative':
        return True
    else:
        return False
    
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def select_letters(v_or_c:str, word:str) -> str:
    LV = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
    result = ''
    if v_or_c == 'v':
        for x in word:
            if x in LV:
                result = result + x
    if v_or_c == 'c':

            if p in LC:
                result = result + p
    return result

def hide_vowels(a:str) -> str:
    LV = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
    for x in a:
        if x in LV:
            a = a.replace(x, '-')
    return a
print(hide_vowels('Hello'))
print(hide_vowels('Happy Halloween!'))
