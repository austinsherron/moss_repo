def test_number(num: int, string: str) -> bool:
    '''Return Ture if the number has the property indicated by the string,
    and False if it doesnt't'''
    if string == "even":
        result = (num%2 == 0)
    elif string == "odd":
        result = (num%2 != 0)
    elif string == 'positive':
        result = (num > 0)
    elif string == 'negative':
        result = (num < 0)
    return result

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
print("c1:")
print(test_number(-32, 'positive'))
print()

    '''takes one-character-long string and returns True if that character is a

def nonvowels(string: str) -> str:


    ''' takes a string and returns a string
    containing all the characters in the parameter string that are not vowels.'''
    result = ''
    for i in string:
        if not is_vowel(i):
            result += i
    return result

def consonants(string: str) -> str:
    '''Reterns a string of all consonants in given string'''
    result = ''
    for i in string:
        if i.lower() in 'bcdfghjklmnpqrstvwxyz':
            result += i
    return result

def hide_vowels(string: str) -> str:
    '''Returns a string with -s in place of vowels'''
    result = ''
    for i in string:
        if is_vowel(i):
            result += '-'
        else:
            result += i
    return result

def alphabetical_names(R: [Restaurant]) -> [str]:
    ''' takes a list of restaurants and returns a
    list of the names of all the restaurants in alphabetical order by name.'''

    '''Creates a rectangle centered on given coordinates
    with given height and width'''
