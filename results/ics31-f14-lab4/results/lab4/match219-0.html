    '''Takes a string S (even odd positive or negative) and checks if N matches S'''
    if S == 'even':
        return N % 2 == 0
    elif S == 'odd':
        return not N % 2 == 0
    elif S == 'positive':
        return N >= 0
    elif S == 'negative':


        return N < 0

def is_vowel(c:str) -> bool:
    '''Checks if c is a vowel'''
    return (c == 'a' or c == 'e' or c == 'i' or c == 'o' or c == 'u'or
            c == 'A' or c == 'E' or c == 'I' or c == 'O' or c == 'U')

assert not is_vowel('p')
assert is_vowel('U')

def nonvowels(f:str) -> str:
    '''Returns a string of all the nonvowels in f'''
    result = ''
    for i in f:
        if (not is_vowel(i)):
            result = result + i
    return result
assert nonvowels('append') == 'ppnd'
assert nonvowels('abced') != 'bced'
assert nonvowels('aeub38-+{i}s') == 'b38-+{}s'

def consonants(f: str) -> str:
    '''Returns a string of consonants in f'''
    result = ''
    L = 'qwrtypsdfghjklzxcvbnmQWRTYPSDFGHJKLZXCVBNM'
    for i in f:
        for x in L:
            if (i == x):
                result = result + i
    return result
assert consonants("-9=-3121294'';;uuiUAuipuiuiuX") == 'pX'
print(' ')

def select_letters(r:str, u:str) -> str:
    '''If r is v it returns a string of the vowels in u
    if r is c, returns the string of consonants in u'''
    result = ''
    if r == 'v':
        for i in u:
            if(is_vowel(i)):
                result = result + i
        return result
    elif r == 'c':
        return consonants(u)
assert select_letters('v', 'look') == 'oo'
print(' ')

def hide_vowels(s: str) -> str:
    '''Replace all the vowels in s with a hyphen'''
    result = ''
    for i in s:
        if (is_vowel(i)):
            result = result + '-'
        else:
            result = result + i
    return result
assert hide_vowels('Hello there') == 'H-ll- th-r-'
assert hide_vowels('*&3oooop*') == '*&3----p*'

def alphabetical_names(L: list) -> list:
