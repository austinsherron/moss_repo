def test_number(n: int, s:str) -> bool:
    ''' returns True if the number has the property indicated by the string,
    and false if it doesn't '''
    if s == 'even':
        return n%2 == 0
    elif s == 'odd':
        return n%2 != 0
    elif s == 'positive':
        return n>= 0
    elif s == 'negative':
        return n<0
    else:
        return ('None')
        

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def consonants (s: str) -> str:
    ''' returns a string containing all the letters in the parameter
    string that are not vowels '''
    result = ''
    for i in s:
        if not is_vowel(i):
            if i not in '0123456789`!?@#$%^&*':
                result = result + i
    return result

def hide_vowels(n: str) -> str:
    ''' Returns a sting in which every vowel in the parameter
    is replaced bya hyphen'''
    result = ''
    for i in n:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result
