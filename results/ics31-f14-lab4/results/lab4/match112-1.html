# The function returns True if the number has the property indicated by the string, and False if it doesn't.

# test_number n:number ->str c:catrgory -> str

def test_number (n:float,c:str)->bool:
   ''' Take the number and see if both number and category match,
if match then return true,else return false.'''
   if n > 0 and c == 'positive':
       return True
   elif n < 0 and c == 'negative':
       return True
   elif n%2 == 0 and c == 'even':
       return True
   elif n%2 != 0 and c == 'odd':
       return True
   else:
       return False
