def test_number(n: int, s: str) -> bool:
    if s == 'even':
        return n%2==0
    if s == 'odd':
        return n%2 >0
    if s =='positive':
        return n >= 0
    if s == 'negative':
        return n < 0

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def hide_vowels(value: str)-> str:
    word = ''
    for l in value:
        if is_vowel(l):
            word += '-'
        else:
            word += l
    return word
