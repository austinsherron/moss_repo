def nonvowels(s:str)->str:
    '''Returns a string containing all the characters in the parameter string that are not vowels'''
    p=''

        if s[i].lower() not in ['a', 'e', 'i', 'o', 'u']:

def consonants(s:str)->str:
    ''' Takes a string and returns a string containing all the letters in the parameter string that are not vowels'''
    p=''
    for i in range(len(s)):
        if s[i].isalnum()==True:
            if s[i].isnumeric()==False:
                if s[i].lower() not in ['a', 'e', 'i', 'o', 'u']:
                    p=p+s[i]
    return p



assert consonants('Suraj')=='Srj'
assert consonants('Lisa@123')=='Ls'

def select_letters(s1:str,s2:str)->str:
    ''' Takes two parameters, both strings, and returns a string. If the first parameter is 'v',
    it returns a string containing all the vowels in the second parameter; if the first parameter is 'c',
    it returns a string containing all the consonants in the second parameter'''
    c=''
    if(s1=='c'):
        for i in range(len(s2)):
            if s2[i].isalnum()==True:
                if s2[i].isnumeric()==False:
                    if s2[i].lower() not in ['a', 'e', 'i', 'o', 'u']:
                        c=c+s2[i]
        return c

    v=''
    if(s1=='v'):
        for i in range(len(s2)):
           if s2[i].isalnum()==True:
               if s2[i].isnumeric()==False:
                    if s2[i].lower()  in ['a', 'e', 'i', 'o', 'u']:
                        v=v+s2[i]
        return v

    else:
        return ''

def hide_vowels(s:str)->str:


    ''' Takes a string and returns a string in which every vowel in the parameter is replaced
    with a hyphen ("-") and all other characters remain unchanged'''
    v=''
    for i in range(len(s)):
        if s[i].lower()  not in ['a', 'e', 'i', 'o', 'u']:
            v=v+s[i]
        if s[i].lower()   in ['a', 'e', 'i', 'o', 'u']:
            v=v+'-'
    return v

assert hide_vowels('Suraj')=='S-r-j'
assert hide_vowels('Andrew')=='-ndr-w'
assert hide_vowels('Su#raj1234')=='S-#r-j1234'
