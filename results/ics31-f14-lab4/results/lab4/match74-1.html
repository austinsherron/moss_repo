def test_number(a: int, b: str):
    if b == "even" and  a%2 == 0:
         return(True)
    if b == "even" and a%2 != 0:
         return(False)
    if b == "odd" and a%2 == 0:
         return(False)
    if b =="odd" and a%2 != 0:
         return(True)
    if b == 'positive' and a>0:
         return(True)
    if b == 'positive' and a<0:
         return(False)
    if b == 'negatve' and a>0:
         return(False)
    if b == 'negative' and a<0:
         return(True)
       

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

    ret = 0
    for Rt in RL:

    return ret 

def average_price(RL:list):
    return average_price_helper(RL) / len(RL)
