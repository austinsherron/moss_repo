def test_number(n:int, q:str) -> str:
    'return true or false based on property'
    
    if q == 'even':
        if n%2==0:
            return True
        else:
            return False
    if q=='odd':
        if n%2!=0:
            return True
        else:
            return False
    if q=='positive':
        if n>0:
            return True
        else:
            return False
    if q=='negative':
        if n<0:
            return True
        else:
            return False
assert test_number(6,'even')== True
assert test_number(7,'even')== False
assert test_number(7,'positive')== True
assert test_number(-1,'positive')== False
assert test_number(-1,'negative')== True
assert test_number(9,'negative')== False
assert test_number(9,'odd')== True
assert test_number(10,'odd')== False

def matching_area_codes(area_codes:list, phone_numbers:list) -> None:
    'return list of phone numbers whose area code is on the list of area codes'
    c = []
    for n in area_codes:
        for i in phone_numbers:
            if n == i[1:4]:
                c.append(i)
    return c
print(matching_area_codes(['714','415'],['(415)414-4123','(312)312-3313','(714)123-5678']))

def nonvowels (string:str)->str:
    "returns string containing all characters in parameter that are not vowels"
    result = ""
    for s in string:
        if is_vowel(s)==False:
            result += s
    return result

def hide_vowels(string:str)->str:
    "return a string replacing every vowel in parameter string with a hyphen"
    vowel = 'aeiouAEIOU'
    for n in string:
        if n in vowel:
            string=string.replace(n,'-')
    return string
assert hide_vowels('facetiously')=='f-c-t---sly'
