def test_number(a:int, b:str)->bool:
    if a%2==0 and b=='even':
        return True
    if a>=0 and b == 'positive':
        return True
    if a<=0 and b == 'negative':
        return True
    if (a%2)!=0 and b == 'odd':
        return True
    else:
        return False

def is_vowel(s:str)->bool:
    '''return vowels true or false'''
    vowels='a','e','i','o','u','y','A','E','I','O','U','Y'
    for a in s:
        if a in vowels:
            return True
        return False

assert is_vowel('b')==False
assert is_vowel('E')==True

def hide_vowels(h:str)->str:
    '''return hiding vowels with - '''
    thehomie=''
    for i in range(len(h)):
        if is_vowel(h[i]):
            thehomie = thehomie+'-'
        else:
            thehomie = thehomie+h[i]
    return thehomie
assert hide_vowels('Luis')=='L--s'
