def test_number(n,s):
    ''' Returns True or False depending on if the number has
    the property indicated by the string'''
    if s=='even':
        if n%2==0:
            return True
        return False
    elif s=='odd':
        if n%2!=0:
            return True
        return False
    elif s=='positive':
        if n>0:
            return True
        return False
    elif s=='negative':
        if n<0:
            return True
        return False

def nonvowels(s:str) -> str:
    '''Returns a string containing all nonvowel characters'''
    result = ''
    for i in s:
        if not is_vowel(i):
            result+=i
    return result

def hide_vowels(w:str) -> str:
    '''Returns the word with the vowels replaced by hyphens'''
    result = ''
    for i in w:
        if is_vowel(i):
            result+='-'
        if consonants(i):
            result+=i
    return result
