def consonants (x:str) -> str:
    '''Return string containing all letters in parameter string that are not vowels'''
    letters = ''
    for i in x:
        if not is_vowel(i) and i.isalpha():

    return letters

    ''' Returns a string of consonants if first parameter is 'c' and vowels if first parameter is 'v' '''
    vowel = ''
    if parameter == 'v':
        for character in word:
            if is_vowel(character):
                vowel += character            
    elif parameter == 'c': # use elif b/c there is a condition of checking parameter
        return consonants(word)
    return vowel

def hide_vowels(word:str) -> str:
    ''' Returns a string that replaces vowels with '-' '''
    w = ''
    for letter in word:
        if is_vowel(letter):
            w += '-'
        else:
            w += letter
    return w
assert hide_vowels('android')
