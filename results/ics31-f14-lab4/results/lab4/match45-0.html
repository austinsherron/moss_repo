def test_number (n: float, s: str) -> bool:
    '''Takes a number and a string, and checks to see if the string correctly describes the number'''
    if s == 'even':
        return n % 2 == 0
    elif s == 'odd':
        return n%2 != 0
    elif s == 'positive':
        return n > 0
    elif s == 'negative':
        return n < 0

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

    '''Takes a letter and prints true if it's a vowel'''
    vowels = ('aeiouAEIOU')
    return ltr in vowels

def nonvowels (word: str) -> str:
    '''Takes a string and print out the characters in the string that are not
vowels as a single string'''
    result = ''
    for i in word:
        if is_vowel(i) == False:
            result = result + i
    return result

assert nonvowels('Bananas') == 'Bnns'
assert nonvowels('?!@$%a223') == '?!@$%223'

#d.4
def consonants (word: str) -> str:
    '''Takes a string and prints out the consonants in the string'''
    result = ''
    cons = 'qwrtypsdfghjklzxcvbnmQWRTYPSDFGHJKLZXCVBNM'
    for i in word:
        if i in cons:
            result = result + i
    return result

assert consonants ('adcq qrajkdeqcepoq e*&^%234039-0') == 'dcqqrjkdqcpq'
assert consonants ('Anteater') == 'nttr'

def hide_vowels(word: str) -> str:
    '''#Takes a string and returns the same string with the vowels replaced with hyphens'''
    for i in word:
        if is_vowel(i):
            word = word.replace(i, '-')
    return word

def average_price(r: list) -> float:
    '''Takes a list of restarants and returns the average price of the best dishes'''
    total = 0
    for i in range(len(r)):
        total = total + r[i].price
    return total/len(r)
