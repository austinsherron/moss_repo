def test_number(number: int, string: str) -> bool:
    " Returns True if the number has the property indicated by the string, and False if it doesn't"
    if string == 'positive':
        if number >0:
            return True
        else:
            return False
    elif string =='negative':
        if number<0:
            return True
        else:
            return False
    elif string =='even':
        if number%2 ==0:
            return True
        else:
            return False
    elif string =='odd':
        if number%2!=0:
            return True
        else:
            return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def hide_vowels(x: str) ->str:
    "Return a string in which every vowel in the parameter is replaced with a hyphen and all other characters remain unchanged"
    x = list(x)
    for i in range(0,len(x)):
        if is_vowel(x[i])==True:
            x[i]='-'
    x = ''.join(x)
    return x

import tkinter          

my_window = tkinter.Tk()

my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)
my_canvas.pack()

def create_rectangle_from_center(x:int, y:int, w:int, h:int):
    "Return a rectangle using x and y as the center"
    my_canvas.create_line(x-0.5*w, y-0.5*h, x+0.5*w, y-0.5*h)
    my_canvas.create_line(x+0.5*w, y-0.5*h, x+0.5*w, y+0.5*h)
    my_canvas.create_line(x+0.5*w, y+0.5*h, x-0.5*w, y+0.5*h)
    my_canvas.create_line(x-0.5*w, y+0.5*h, x-0.5*w, y-0.5*h)
    
create_rectangle_from_center(500, 300, 300, 200)


tkinter.mainloop()
