def consonants (x:str) -> str:
    " Returns string containing all letters that are not vowels/special char."
    c = ['b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n', 'p', 'q',
         'r', 's', 't', 'v', 'w', 'x', 'y', 'z', 'B', 'C', 'D', 'F', 'G',
         'H', 'J', 'K', 'L', 'M', 'N', 'P', 'Q', 'R', 'S', 'T', 'V', 'W',
         'X', 'Y', 'Z']
    con = ''
    for q in x:
        if is_vowel(q) == False and q in c:
            con+=str(q)
    return con

def hide_vowels (x:str) -> str:
    " Returns string where vowels are replaced with a hyphen "
