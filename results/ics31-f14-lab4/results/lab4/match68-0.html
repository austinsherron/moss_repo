        if n-abs(n)==0 and s=='positive':
            return (n-abs(n)==0)
        elif n-abs(n)<0 and s=='negative':
            return (n-abs(n)<0)
        elif n-abs(n)==0 and s=='negative':
            return (n-abs(n)<0)
        elif n-abs(n)<0 and s=='positive':
            return (n-abs(n)==0)
