def test_number(num:int, string:str) -> bool:
    '''Takes input number and string, returns true if number
       has property indicated by the string, false otherwise.'''
    if string == 'even':
        return num % 2 == 0
    elif string == 'odd':
        return num % 2 == 1
    elif string == 'positive':
        #Had to cover case if num == 0
        return num >= 0
    elif string == 'negative':
        return num < 0

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

print(test_number(100, 'odd'))

def matching_area_codes (area: list, nums: list) -> list:
    '''Takes a list of area codes and list of phone numbers. Returns a
    list of phone numbers matching the area code.'''
    match = []
    for phone in nums:
        for codes in area:
            if codes == phone[1:4]:
                match.append(phone)
    return match

def nonvowels(string: str) -> str:
    '''Takes in a string, returns a string of the nonvowels'''
    ans = ''
    for char in string:
        if not is_vowel(char):
            ans += char
    return ans

def select_letters(strA: str, strB: str) -> str:
    '''Takes two parameters of strings, if first is 'v', return all
    vowels of second perameter. If first is 'c' return consonants.
    If anything else, return empty string'''
    if strA == 'v':
        ans = ''
        for letter in strB:
            if is_vowel(letter):
                ans += letter
        return ans
    elif strA == 'c':
        return consonants(strB)
    else:
        return ''
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('b', 'facetiously') == ''

print(select_letters('c', 'facetiously'))

def hide_vowels (string: str) -> str:
    '''Takes in a string, returns a string where every vowel
    is replacd by "-"'''
    ans = ''
    for letter in string:
        if is_vowel(letter):
            ans += '-'
        else:
            ans += letter;
    return ans;
assert hide_vowels('Bumblebee') == 'B-mbl-b--'

from collections import namedtuple 

Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')
# Restaurant attributes: name, kind of food served, phone number,
#	best dish, price of that dish

def Restaurant_change_price(rest: Restaurant, p: float) -> Restaurant:
    ''' Takes in a Restaurant and a price float, returns a Restaurant
    with the price increase by second parameter.'''
    rest = rest._replace(price = rest.price + p)
    return rest

thai = Restaurant("Thai Dishes", "Thai", "334-4433", "Mee Krob", 12.50)
nobu = Restaurant("Nobu", "Japanese", "335-4433", "Natto Temaki", 5.50)

assert Restaurant_change_price(nobu, 4.50) == Restaurant("Nobu", "Japanese", "335-4433", "Natto Temaki", 10.00)

print(Restaurant_change_price(thai, 5.50))
print(Restaurant_change_price(nobu, 4.50))

def average_price (rest: list) -> float:
    '''Takes in a list and returns the average price of the restaurants.'''
    num = len(rest)
    totalPrice = 0

    for rests in rest:
        totalPrice += rests.price
    return totalPrice / num

def create_rectangle_from_center(x: float, y:float, l: float, w:float):
    xRect1 = x - (w/2)
    yRect1 = y - (l/2)
    xRect2 = x + (w/2)
    yRect2 = y + (l/2)
