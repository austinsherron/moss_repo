def test_number(num: int, s: str) -> bool:
    '''Return whether information given on number is True'''
    if num % 2 == 0 and s == 'even':
        return True
    elif num % 2 ==1 and s == 'odd':
        return True
    elif num > 0 and s == 'positive':
        return True
    elif num < 0 and s == 'negative':
        return True
    else:
        return False   
assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True
print(test_number(400, 'even'))
print(test_number(64, 'odd'))

def is_vowel(char: str) -> bool:

def nonvowels(characters: str) -> str:

def consonants(char: str) -> str:
    '''Return a string containing only consonants'''
    result = ''
    cons = 'BCDHFGHJKLMNPQRSTVWXYZbcdfghjklmnpqrstvwxyz'
    for n in char:
       if n in cons:
           result = result + n
    return result
assert consonants('character')=='chrctr'
assert consonants('950.abl')=='bl'

print(consonants('123aeioubcd'))

def hide_vowels(word: str) ->str:
    '''Return string with all vowels replaced with hyphens'''
    vowels = "AEIOUaeiou"
    for letter in word:
        if letter in vowels:
            word=word.replace(letter, "-")
    return word
print(hide_vowels('donald'))
print(hide_vowels('Happy Halloween'))
print(hide_vowels('Hello! How was your day today?'))
