def test_number(number: int, prop: str) -> bool:
    '''The function returns True if the number has the property indicated by the string, and False if it doesn't.'''
    if number % 2 == 0 and prop == 'even':
        return True
    elif number % 2 == 1 and prop == 'odd':
        return True
    elif number > 0 and prop == 'positive':
        return True
    elif number < 0 and prop == 'negative':
        return True
    else:
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def match_area_codes(area: list, codes: list) -> None:
    '''The function will return a list of the phone numbers whose area code is on the list of area codes.'''
    result = []
    for i in area:
        for x in codes:
            if i == x[1:4]:
                result.append(x)
    return result

def hide_vowels(word:str)->str:
    '''Takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters remain unchanged.'''
    new = ''
    for i in word:
        if is_vowel(i):
            new += '-'
        else:
            new += i
    return new

assert hide_vowels('The cat ate the dog') == 'Th- c-t -t- th- d-g'

def alphabetical_names(res: list) -> list:
    '''Takes a list of restaurants and returns a list of the names of all the restaurants in alphabetical order by name.'''
    name = []
    res2 = sorted(res) #this is to avoid side effects
