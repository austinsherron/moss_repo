def test_number(num: int, string : str) -> bool:
    """ Takes as input a number and a string, returns true if the number has the property
    indecated by the string, False if it doesn't"""
    if string == 'even':
        return (num % 2 == 0)
    elif string == 'odd':
        return (num % 2 == 1)
    elif string == 'positive':
        return ( num >= 0)
    elif string == 'negative':
        return ( num < 0 )

def consonants(string: str) -> str:
    '''Takes in a string and returns nonvowels with no special characters just
    letters '''
    result = ''
    for i in string:
        if  (64 < ord(i) < 91) or (96 < ord(i) < 123):
            result = result + i
    return nonvowels(result)

assert consonants("HIFIVE#$#$#") == 'HFV'
assert consonants("2343class#$#$#") == 'clss'
assert consonants("34#$$winner$%44") == 'wnnr'

def hide_vowels(string1: str) -> str:
    """Takes a string and returns a string in which every vowel is replaced with
    a hyphen"""
    result = ""
    for i in string1:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result

assert hide_vowels("Hello") == 'H-ll-'
assert hide_vowels("34Highfive") == '34H-ghf-v-'
assert hide_vowels("") == ""

def average_price(RL: 'list of restaurants') -> float:
    ''' Takes a list of restaurants and returns the average price'''
    total_sum = 0
    getcontext().prec = 4
    for i in RL:
        total_sum = total_sum + i.price
    average_price = Decimal(total_sum) / Decimal(len(RL))
    return average_price
