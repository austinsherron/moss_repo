def test_number(n=int,m=str) -> bool:
    '''Return True if the number has the property indicated by the string.'''
    if m == 'even':
        if n % 2 == 0:
            return True
        else:
            return False
    elif m == 'odd':
        if n % 2 != 0:
            return True
        else:
            return False
    elif m == 'positive':
        if n >= 0:
            return True
        else:
            return False
    elif m == 'negative':
        if n < 0:
            return True
        else:
            return False
assert test_number(14, 'even') == True
assert test_number(100, 'odd') == False
assert test_number(33, 'positive') == True
assert test_number(100, 'negative') == False

def nonvowels(n=str) -> str:
    '''Return the nonvowels in the string as a string.'''
    result = ''
    for i in n:
        if is_vowel(i) == False:
            result = result + i
    return result
assert nonvowels('Warren') == 'Wrrn'
assert nonvowels('Kevin') == 'Kvn'
assert nonvowels('John@117') == 'Jhn@117'

#(d.4)
def consonants(n=str) -> str:
    '''Return the consonants in the string as a string.'''
    lol = ''
    for i in n:
        if i.isalpha() == True and is_vowel(i) == False:
            lol = lol + i
    return lol
assert consonants('John@117') == 'Jhn'
assert consonants('Kurt@115!!!') == 'Krt'
assert consonants('K3v1n15ab0s5@gmail.com') == 'Kvnbsgmlcm'
    
#(d.5)
def select_letters(n=str,m=str) -> str:
    '''If 'v' return a string with all the vowels in the second parameter, if 'c' return a string with all the consonants in the second parameter.'''
    if n == 'v':
        e = ''
        for i in m:
            if is_vowel(i) == True:
                e = e + i
        return e
    elif n == 'c':
        return consonants(m)
    else:
        return ''
assert select_letters('c','K3v1n15ab0s5@gmail.com') == 'Kvnbsgmlcm'
assert select_letters('v', 'K3v1n15ab0s5@gmail.com') == 'aaio'
assert select_letters('rjektgbrjek','K3v1n15ab0s5@gmail.com') == ''
          
#(d.6)
def hide_vowels(n=str) -> str:
    '''Return string with ever vowel replaced with a '-'.'''
    result = ''
    for i in n:
        if is_vowel(i) == False:
            result = result + i
        if is_vowel(i) == True:
            result = result + '-'
    return result
assert hide_vowels('Kevin is the partner of Warren@gmail.com!') == 'K-v-n -s th- p-rtn-r -f W-rr-n@gm--l.c-m!'
assert hide_vowels('K3v1n15ab0s5@gmail.com') == 'K3v1n15-b0s5@gm--l.c-m'
