def test_number(x:int,word:str) -> bool:
    '''Returns True if the number has the property indicated by the string,
       and False if it doesn't'''
    if word == 'even' or word == 'odd':
        if x%2 == 0:
            if word == 'even':
                return True
            return False
        elif x%2 != 0:
            if word == 'odd':
                return True
            return False
    else:
        if x>0:
            if word == 'positive':
                return True
            return False
        elif x<0:
            if word == 'negative':
                return True
            return False

assert test_number(14, 'even') == True
assert test_number(15,'odd') == True
assert test_number(151,'even') == False
assert test_number(142,'positive') == True
assert test_number(-5, 'negative') == True
print(test_number(-600,'negative'))

def matching_area_codes(area_codes:list, phone_nums:list):
    '''Returns a list of the numbers with matching area codes'''
    plist = []
    for i in range(len(area_codes)):
        for n in range(len(phone_nums)):
            if phone_nums[n][1:4] == area_codes[i]:

def is_vowel(let:str) -> bool:
    '''Returns True if the character is a vowel and False otherwise'''
    vowels = ['a','e','i','o','u','A','E','I','O','U']
    if let in vowels:
        return True
    return False

def hide_vowels(s:str) -> str:


    '''Returns a string in which every vowel in the parameter is replaced
     with a hyphen and all other characters remain unchanged.'''
    result = ''
    for i in range(len(s)):
        if is_vowel(s[i]):
            result += '-'
        else:
            result += s[i]
    return result

assert hide_vowels('wonderful') == 'w-nd-rf-l'
assert hide_vowels('BEYONCE') == 'B-Y-NC-'
print(hide_vowels('I hate hot weather'))

def alphabetical_names(L:list) -> list:
    '''Takes a list of restaurants and returns a list of the names
       of all the restaurants in alphabetical order'''
