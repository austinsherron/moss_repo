def test_number (num:int, char:str):
    '''Tells whether or not the number has the property indicated by the string''' 
    if char == 'even' and num % 2 == 0:
        return True
    elif char == 'odd' and num % 2 == 1:
        return True
    elif char == 'positive' and num > 0:
        return True
    elif char == 'negative' and num < 0:
        return True
    else:
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert not test_number(100, 'negative')
assert test_number(33, 'positive')

def consonants (phrase: str) -> str:
    '''takes a string and returns string containing all the letters in the string that aren't vowels'''
    all_consonants = ('bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTUVWXYZ')
    consonant = ''
    for character in phrase:
        if character in all_consonants:
            consonant += (character)
    return consonant

assert consonants('sriracha') == 'srrch'

def select_letters (letter: str, select: str) -> str:
    '''returns all consonants or all vowels of the string (letter = 'c' or 'v')'''
    vowels = 'aeiouAEIOU'
    result_letters = ''
    if letter == 'v':
        for t in select:
            if t in vowels:
                result_letters += t
    elif letter == 'c':
        for t in select:
            result_letters += consonants(t)
        
    return result_letters

assert select_letters('c', 'sriracha') == 'srrch'
assert select_letters('v', 'sriracha') == 'iaa'

def hide_vowels (some: str) -> str:
    '''replaces vowels in string with hyphens'''
    all_consonants = ('bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTUVWXYZ')
    hyphen = '-'
    vowels = 'aeiouAEIOU'
    letter = ''
    for character in some:
        if character in vowels:
            letter += hyphen
        elif character in all_consonants:
            letter += character

    return letter

assert hide_vowels('natural color') == 'n-t-r-lc-l-r' 
assert hide_vowels('galaxy') == 'g-l-xy'
