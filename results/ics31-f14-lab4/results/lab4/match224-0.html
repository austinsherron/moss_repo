def test_number(n: int, s: str)-> bool:
  '''Returns True if the number has the property'''
  s = s.lower()
  if n % 2 == 0:
    if s == 'even':
      return True
  if n % 2 == 1:
    if s == 'odd':
      return True
  if n > 0:
    if s == 'positive':
      return True
  if n < 0:
    if s == 'negative':
      return True
  else:
    return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(s: str)-> bool:
  '''Takes a 1 character long string and returns True if it's a vowel'''
  vowels = ['a','e','i','o','u']
  return s.lower() in vowels

def hide_vowels(s: str)-> str:
  '''Returns a string but replaces all vowels with "-"'''
  result = ''
  for n in range(len(s)):
    if is_vowel(s[n]):
      result = result + '-'
    else:
      result = result + s[n]
  return result
assert hide_vowels('Wow!') == 'W-w!'

def alphabetical_names(l: list)-> list:
  '''Returns the names of the list in alphabetical order'''
  alphabetical(l)
  result = []
  for n in range(len(l)):
    result.append(l[n].name)
  return result

def average_price(l: list)-> float:
  '''Returns the average price of the best dishes of all restaurants in a list'''
  Sum = []
  for n in range(len(l)):
    Sum.append(l[n].price)
  avg = (sum(Sum) / (len(l)))
  return avg
