    '''Takes a string and returns all characters that are not vowels'''
    result = ' '

def hide_vowels(n:str) -> str:
    '''Takes a string and replaces the vowels in the string with hyphens'''
    result4 = ''
    for i in n:
        if is_vowel(i):
            result4 = result4 + '-'
        else:
            result4 = result4 + i
    return result4

    '''Returns average price of list of restaurants'''
