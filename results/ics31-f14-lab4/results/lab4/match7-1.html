    '''Given a str for the second paramater and a 'v' (for vowel) or 'c' for

    check if the given str is a vowel or consonant and return the Boolean value.
    If neither 'v' or 'c' is taken as the first argument the function will return
    an empty string.'''
    if check == 'v':
        result = ''
        for i in range(len(word)):
            if is_vowel(word[i]) == True and word[i].isdigit() == False:
                result = result + word[i]
        return result
    elif check == 'c':
        return consonant (word)
    else:
        return ''

def hide_vowels (phrase: str) -> str:
    '''Given a phrase/word, this function will replace all vowels with a '-'
    and return the results'''
    result = ''
    for i in range(len(phrase)):
        if phrase[i].lower() in 'aeiou':
            result =result + '-'
        else:
            result = result + phrase[i]
    return result
