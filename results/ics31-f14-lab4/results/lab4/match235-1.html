def test_number(i: int, s: str) -> bool:
    if i%2 == 0 and s == "even":
        return True
    elif i%2 != 0 and s == "even":
        return False
    elif i%2 == 0 and s == "odd":
        return False
    elif i%2 != 0 and s== "odd":
        return True
    elif i>=0 and s == "positive":
        return True
    elif i>=0 and s == "negative":
        return False
    elif i<0 and s=="negative":
        return True
    elif i<0 and s == "positive":
        return False




assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(vowel: str) -> str:
    result = ""

       if is_vowel(vowel[x])== False:
