    elif y == "positive" and abs(x) == x:

        return False

def nonvowels(p:str) -> str:
    "This function takes a string and returns a string containing all the characters in the parameter string that are not vowels"
    result = ''
    for i in p:
        if not is_vowel(i):
            result = result + i
    return result
assert nonvowels('bottle') == 'bttl'
assert nonvowels('12.hfe') == '12.hf'
print(nonvowels('computer'))

def consonants(p:str) -> str:
    "This function takes a string and returns a string containing all the letters in the parameter string that are not vowels"
    result = ''
    for i in p:


        if not is_vowel(i) and i.isalpha():
            result = result + i
    return result

assert consonants('2AB.sg') == 'Bsg'

def select_letters(p:str, k:str) -> str:
    "This function checks string p for v or c and reacts accordingly"
    result = ''
    if p == 'v':
        for temp in k:
            if is_vowel(temp):
                result = result + temp
        return result
    elif p == 'c':
        result = result + consonants(k)
        return result
    else:
        return ''

assert select_letters('v', 'laptop') == 'ao'
assert select_letters('c', 'laptop') == 'lptp'
assert select_letters('f', 'banana') == ''

def hide_vowels(x: str) -> str:
    "This turns all vowels in x into '-'"
    temp = ""
    for i in x:
        if is_vowel(i):
            temp = temp + "-"
        else:
            temp = temp + i
    return temp
assert hide_vowels("Hello") == "H-ll-"

print(hide_vowels("BANANA"))

def average_price(L:list) -> float:
    "Takes in a list of restaurants and returns the average price of them."
    x = 0.0
    for temp in L:
        x += temp.price
    return x / len(L)
