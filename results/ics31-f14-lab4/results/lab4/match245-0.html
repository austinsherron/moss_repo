def test_number(n: float,type: str):
    "Returns true/false statements."
    if type == 'even':
        if n%2 == 0:
            return 'True'
        else:
            return 'False'
    elif type == 'odd':
        if n%2 != 0:
            return 'True'
        else:
            return 'False'
    elif type == 'positive':
        if n >= 0:
            return 'True'
        else:
            return 'False'
    elif type == 'negative':
        if n < 0:
            return 'True'
        else:
            return 'False'
assert test_number(14, 'even') == 'True'
assert not test_number(100, 'odd') == 'True'
assert test_number(33, 'positive') == 'True'
assert not test_number(100, 'negative') == 'True'

def hide_vowels(x:str) ->str:
    "Returns a string in which every vowel is replaced with a hyphen."
    for i in range(len(x)):
        if is_vowel(x[i]) == True:
            x = x.replace(x[i],'-')
    return x

import tkinter              # Load the library; do this just once per program

my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window

def create_rectangle_from_center(x:int, y: int, height: int, width: int):
    "Creates a rectangle using the coordinates of its center and it's width and height"
    my_canvas.create_rectangle(x - width/2, y - height/2, x + width/2, y + height/2)

create_rectangle_from_center(150, 250, 400, 200)

tkinter.mainloop()          # Combine all the elements and display the window
