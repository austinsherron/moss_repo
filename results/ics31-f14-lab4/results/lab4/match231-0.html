def test_number (x: int, y: str) -> bool:
    ''' Checks the number to see if it has the definition given by the string'''
    if (x % 2 == 0 and y == 'even'):
        return True
    elif (x % 2 !=0 and y == 'odd'):
        return True
    elif (x > 0 and y == 'positive'):
        return True
    elif (x < 0 and y== 'negative'):
        return True
    else:
        return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(x: str) -> bool:

def nonvowels(value: str) -> str:
    '''returns the string of nonvowels from the string given in the function'''
    result=''
    for x in range(0, len(value)):
        if(is_vowel(value[x]) == False):
            result+=value[x]
    return result

def consonants(value: str) -> str:
    '''returns the string of consonants from the string given in the parameters'''
    consonants = ['b','B','c','C','d','D','f','F','g','G','h','H','j','J','k','K','l','L','m','M','n','N','p','P','q','Q','r','R','s','S','t','T','v','V','w','W','x','X','y','Y','z','Z']
    result = ''
    for x in range(0, len(value)):
        if(value[x] in consonants):
            result += value[x]
    return result
assert consonants('banana') == 'bnn'
assert consonants('tea pot') == 'tpt'

        return consonants(word)

            if(is_vowel(word[x])):
                result+=word[x]

        return result

def hide_vowels(value: str) -> str:
    '''takes a string and replaces the vowels with hyphens'''
    result = ''
    for x in range(0,len(value)):
        if(is_vowel(value[x])):
            result+= '-'
        else:
            result+=value[x]
    return result

assert hide_vowels("Arvind") == "-rv-nd"
assert hide_vowels("I love you") == "- l-v- y--"
