def test_number(x: int, s: str) -> bool:
    '''Returns True if the number has the property of the string'''
    if s == 'even':
        return x % 2 == 0
    elif s == 'odd':
        return x % 2 != 0            
    elif s == 'positive':
        return x >= 0        
    elif s == 'negative':
        return x < 0

assert not test_number(100, 'negative')

def matching_area_codes(x: list, y: list) -> list:
    '''Takes in a list of 3 digit area codes (as strings) and a list of phone numbers
    and print the phone numbers whose area code is on the list of area codes.'''

def consonants(x: str) -> str:
    '''Takes a string and returns a string containing all the letters in the paramter
    that are not vowels or other characters like numbers, spaces, and punctuation'''
    result = ''
    for i in x:
        if i in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
            result = result + i
    return result

def select_letters(x: str, y: str) -> str:
    '''Takes in two strings. Depending on the first string, the second string will be returned
    with only its vowels, consonants, or nothing at all'''
    result = ''
    if x == 'v':
        for i in y:
            if is_vowel(i):
                result = result + i
        return result
    elif x == 'c':
        return consonants(y)
    elif x == '':
        return result
        

assert select_letters('', 'Team Fortres 2') == ''
assert select_letters('v', 'Team Fortress 2') == 'eaoe'
assert select_letters('c', 'Team Fortress 2') == 'TmFrtrss'

def hide_vowels(x: str) -> str:
    '''Takes a string and returns the string with all the vowels replaced with "-" '''
    result = ''
    for i in x:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result

assert hide_vowels('Team Fortress 2') == 'T--m F-rtr-ss 2'
