def test_number (num: int, s1: str) -> bool:
    ''' return True if the number has the property indicated by the string '''
    if (s1 == 'even'):
        return (num % 2 == 0)
    if (s1 == 'odd'):
        return (num % 2 != 0)
    if (s1 == 'positive'):
        return (num > 0)
    if (s1 == 'negative'):
        return (num < 0)

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')


assert not test_number(100, 'negative')

def is_vowel (letter: str) -> bool:
    ''' return True if the character is a vowel '''
    vowels = 'aeiouAEIOU'
    return (letter in vowels)

def nonvowels (word: str) -> str:
    ''' return a string containing all the characters in the parameter string that are not vowels '''


    result = ''
    for char in word:
        if not is_vowel(char):
             result = result + char
    return result

assert nonvowels('anteater') == 'nttr'
assert nonvowels('California!') == 'Clfrn!'

#D4
def consonants (word: str) -> str:
    ''' return a string containing all the letters in the parameter string that are not vowels '''
    consonants = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    letters = ''
    for char in word:


        if char in consonants:
            letters = letters + char
    return letters

assert consonants('Scarsdale, New York') == 'ScrsdlNwYrk'
assert consonants('Hello, World!!!') == 'HllWrld'

def select_letters (s1: str, s2: str) -> str:
    ''' return a string containing all of the vowels or all of the consonants in the string '''
    string = ''
    if s1 == 'v':
        for char in s2:
            if is_vowel(char):
                string = string + char
    if s1 == 'c':
        string = consonants(s2)
    return string

assert select_letters('c', 'Zot! Zot! Zot!') == 'ZtZtZt'
assert select_letters('w', 'anteater') == ''
assert select_letters('v', 'facetiously') == 'aeiou'

#D6
def hide_vowels (string: str) -> str:
    ''' return a string in which every vowel in the parameter is replaced with a hyphen '''
    new_string = ''
    for char in string:
        if is_vowel(char):
            new_string = new_string + '-'
        else:
            new_string = new_string + char
    return new_string

assert hide_vowels('Python is awesome!') == 'Pyth-n -s -w-s-m-!'
assert hide_vowels('This sentence has no vowels.') == 'Th-s s-nt-nc- h-s n- v-w-ls.'
