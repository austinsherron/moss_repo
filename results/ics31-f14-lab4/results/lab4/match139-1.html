def test_number(num: int, string: str)-> bool:
    """Takes an integer and a string, returns True if integer has property
    listed by string"""
    if string == 'even' and num % 2 == 0:
        return True
    elif string == 'odd' and num % 2 != 0:
        return True
    elif string == 'positive' and num > 0:
        return True
    elif string == 'negative' and num < 0:
        return True
    return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(chara: str) -> bool:
    "Takes a one chara string and returns true if a vowel"
    return chara in 'aeiouAEIOU'

def nonvowels(a: str) -> str:
    "Takes a string and returns a string containing nonvowels"
    result = ''
    for i in a:
        if not is_vowel(i):
            result = result + i
    return result
assert nonvowels('word') == 'wrd'
assert nonvowels('HEL LO') == 'HL L'

#part d4
def consonants(a: str) -> str:
    'Takes a string and returns a string of the consonants'
    result = ''
    for i in a:
        if i.isalpha() and not is_vowel(i):
            result = result + i
    return result
assert consonants('word!#$') == 'wrd'
assert consonants('13DSJF9@') == 'DSJF'
assert consonants('Nevad!?a') == 'Nvd'

#part d5
def select_letters(a: str, b: str) -> str:
    """Takes two strings and returns all vowels if first string is v
    and returns all consonants if first string is c, otherwise returns
    empty string"""
    result = ''
    if a == 'v':
        for i in b:
            if is_vowel(i):
                result = result + i
        return result           
    elif a == 'c':
        return consonants(b)
    else:
        return ''
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('d', 'facetiously') == ''

#part d6
def hide_vowels(a: str) -> str:
    """Takes a string and returns a string with the vowels replaced by '-'"""
    text = ''
    for i in a:
        if is_vowel(i):
            text = text + '-'
        else:
            text = text + i
    return text
assert hide_vowels('word') == 'w-rd'
assert hide_vowels('facetiously') == 'f-c-t---sly'
assert hide_vowels('Hello') == 'H-ll-'
