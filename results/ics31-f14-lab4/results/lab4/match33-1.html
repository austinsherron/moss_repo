def test_number ( number : int, string :str)-> bool:
    ''' Returns true if number has property of string else returns false'''
    if string=="even":
        return number%2==0

    elif string=="odd":
        return number%2==1

    elif string =="positive":
        return number>0

    elif string =="negative":
        return number<0
    else:
        return false
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def hide_vowels(string : str) -> str :
    '''returns string with vowels replaced as - '''
    results = ''
    for x in string:
        if is_vowel(x):
            results+= '-'
        else:
            results +=x
    return results

assert hide_vowels('facetious') == 'f-c-t---s'
assert hide_vowels('I am happy') == '- -m h-ppy'
assert hide_vowels('this works well :)') == 'th-s w-rks w-ll :)'

def average_price(rests:[Restaurant])->float:
    '''Returns average price for list of Restaurant'''
    average=0
    for x in rests:
        average+=x.price
    average/=len(rests)
    return average
