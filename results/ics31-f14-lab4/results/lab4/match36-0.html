def test_number(x: int, y: str) -> bool:
    '''Returns True if number fits into category defined by str.'''
    if x%2 is 0 and y == 'even':
        return True
    elif x%2 is not 0 and y == 'odd':
        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative':
        return True
    else:
        return False

assert test_number(-33, 'positive') == False

def nonvowels(x: str) -> str:
    ''' Returns a string containing all the characters that are not vowels '''
    result = ""
    for i in x:
        if is_vowel(i) == False:
            result += i
    return result

def consonants(x: str) -> str:
    ''' Returns a string containing all the letters in the parameter that are not vowels '''
    spoopy = ""
    for i in x:
        if is_vowel(i) == False and i in 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLNMOPQRSTUVWXYZ':
            spoopy += i
    return spoopy

def hide_vowels(x: str) -> str:
    ''' Return every vowel in the string with a hyphen (-) '''
    
    for i in x:
        if is_vowel(i) == True:
            x = x.replace(i, "-")
    return x
