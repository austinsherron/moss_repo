        ''' create and return a string composed only of consonants created from the provided

def hide_vowels(word: str) -> str:
        ''' create a new string that replaces the vowels in a word or phrase with
        the character '-' and return it'''
        returnString = ''
        for i in word:
                if is_vowel(i):
                        returnString = returnString + '-'
                else:
                        returnString = returnString + i
        return returnString
print(hide_vowels('The car is in the garage.'))
assert hide_vowels('Piano Player') == 'P--n- Pl-y-r'
assert hide_vowels('aeiou') == '-----'
assert hide_vowels('Fly') == 'Fly'

def average_price(rlist: list) -> float:
        ''' return the average price of the best dishes of all restaurants in the
        provided list '''
        total = 0.0
        for i in rlist:
                total += i.price
        return total / len(rlist)
