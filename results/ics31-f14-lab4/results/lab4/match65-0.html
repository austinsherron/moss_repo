def test_number(n: int, s: str) -> bool:
    '''Take a number and a quality and return a bool'''
    if (s == 'even' and n%2 == 0) or (s=='odd' and n%2 != 0):
        return True
    elif (s== 'even' and n%2 != 0) or (s=='odd' and n%2 == 0):
        return False
    if (s == 'positive' and n>0) or (s=='negative' and n<0):
        return True
    elif (s=='positive' and n<0) or (s=='negative' and n>0):
        return False

print(test_number(5,'odd'))
print(test_number(8,'even'))

def hide_vowels(S: str) -> str:
    'Replace all vowels in a string with \'-\''
    result = ''
    for i in S:
        if is_vowel(i):
            result += '-'
        if not is_vowel(i):
            result += i
    return result

print(hide_vowels('The quick brown fox jumps over the lazy dog'))

def create_rectangle_from_center(x: int, y: int, h: int, w: int) -> None:
    'Create a rectangle from a center point, height, and width'
    x1 = x - w/2
    x2 = x + w/2
    y1 = y - h/2
    y2 = y + h/2
    
    my_canvas.create_rectangle(x1, y1, x2, y2)
