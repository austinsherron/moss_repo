def test_number(n: int, s: str):
    if n % 2 == 0 and s == 'even':
        return True
    elif n % 2 != 0 and s == 'odd':
        return True
    elif n > 0 and s == 'postive':
        return True
    elif n < 0 and s == 'negative':
        return True
    else:
        return False

assert not test_number(100, 'odd')
assert test_number(33, 'positive')
