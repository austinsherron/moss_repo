    if n%2==0 and s=='even':

        return True

def consonants(y:str)->str:
    result=''
    for x in y:
        if x not in 'aeiouAEIOU':
            if x not in  '0123456789+-*/.`!@#$%^&*()_+-=[]\;'':':
                result= result + x
   
    return result
