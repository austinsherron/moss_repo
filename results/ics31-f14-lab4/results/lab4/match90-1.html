def test_number(n: int, s: str) -> bool:
    """ Takes an input and string and returns True/False if the number
        has the property indicated by the string """
    if s == 'even':
        return n%2 == 0
    if s == 'odd':
        return n%2 != 0
    if s == 'positive':
        return n > 0
    if s == 'negative':
        return n < 0
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
print('50 is an even number.')
test_number(50, 'even')

def nonvowels(non: str) -> str:
    """ Takes a string and returns all the characters in the parameter
        string that are not vowels """
    new = ''
    for x in non:
        if not is_vowel(x):
            new += x
    return new
assert nonvowels('Wesley 21') == ('Wsly 21')
assert nonvowels('Congcong.10') == ('Cngcng.10')
print('The nonvowels in Comp Sci 33 are:', nonvowels('Comp Sci 33'))

def consonants(c: str) -> str:
    """ Takes a string and returns a string containing all the consonants
        in the parameter string """
    consonants = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    new = ''
    for x in c:
        if x in consonants:
            new += x
    return new
assert consonants('Biology 102') == 'Blgy'
assert consonants('Math.2A') == 'Mth'
print('The consonants in Biology 10D are:', consonants('Biology 10D'))

def select_letters(a: str, b: str) -> str:
    """ Depending on first parameter, returns a string containing
        either all the vowels or all the consonants in the second string """
    new = ''
    if a == 'v':
        for x in b:
            if is_vowel(x):
                new += x
        return new
    elif a == 'c':
        return consonants(b)
    else:
        return new
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('l', 'facetiously') == ''
print('All consonants in facetiously are:', select_letters('c', 'facetiously'))

def hide_vowels(hv: str) -> str:
    """ Takes a string and returns a string in which every vowel
        is replaced with a hyphen """
    new = ''
    for x in hv:
        if is_vowel(x):
            new += '-'
        else:
            new += x
    return new
print(hide_vowels('I love computer science.'))

def average_price(lst: list) -> float:
    """ Takes a list of restaurants and returns the average price of
        those restaurants """
    total = 0
    for r in lst:
        total += r.price
    return total / len(lst)
