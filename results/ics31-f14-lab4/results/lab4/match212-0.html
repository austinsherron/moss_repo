def test_number(a:int, b:str)->bool:
    '''This function returns True if the number has the property
         indicated by the string, and  False if it doesn't'''
    if b == 'even':
        return a%2 == 0
    elif b == 'odd':
        return a%2 != 0
    elif b == 'positive':
        return a >= 0
    elif b == 'negative':
        return a < 0

assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel(x:str)->bool:
    '''Takes a one-character-long string and returns True if that
       character is a vowel and False otherwise'''
    b = ["a", "e", "i", "o", "u","A", "E", "I", "O", "U"]
    return x in b

def hide_vowels(a:str)->str:
    '''Takes a string and returns a string in which every vowel in the parameter is replaced
       with a hyphen ("-") and all other characters remain unchanged'''

def alphabetical_names(a:list)->list:
    '''Takes a list of restaurants and returns a list of the names of all
