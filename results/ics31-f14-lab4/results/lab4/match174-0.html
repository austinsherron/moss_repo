def test_number(n: int, s: str) -> bool:
    '''Returns True or False whether or the input string matches the number input'''
    if s == 'even':
        return n % 2 == 0
    elif s == 'odd':
        return n % 2 != 0
    elif s == 'positive':
        return n >= 0
    elif s == 'negative':
        return n < 0
    
assert test_number(4, 'positive')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def matching_area_codes(l: list, n: list) -> list:
    '''Returns a list of the matching numbers'''
    list_areacodes = []
    for m in n:
        for o in l:
            if m[1:4] == o:
                list_areacodes.append(m)
    return list_areacodes

def nonvowels(s: str) ->str:
    '''Returns a string containing all the characters in the parameter string that are not vowels'''
    result = ''
    for x in s:
        if is_vowel(x) == False:
            result = result + x
    return result

assert nonvowels('elephant') == 'lphnt'
assert nonvowels('youtube') == 'ytb'
assert nonvowels('intel') == 'ntl'
assert nonvowels('123abc') == '123bc'

def consonants(c: str) -> str:
    '''Returns a string containing all the letters in the parameter string that are not vowels'''
    consonants_list = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    consonantword = ''
    for x in c:
        if x in consonants_list:
            consonantword = consonantword + x
    return consonantword

assert consonants('pizza') == 'pzz'
assert consonants('123abc') == 'bc'

def select_letters(a: str, b:str) -> str:
    '''Takes two parameters and returns a string'''
    if a == 'c':
        return consonants(b)
    elif a == 'v':
        vowelword = ''
        for x in b:
            if is_vowel(x):
                vowelword = vowelword + x
        return vowelword
    else:
        return ''

def hide_vowels(s: str) -> str:
    '''Returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters remain unchanged'''
    for x in s:
        if is_vowel(x):
            s = s.replace(x, '-') 
    return s

assert hide_vowels('restaurants') == 'r-st--r-nts'
assert hide_vowels('umbrella') == '-mbr-ll-'
assert hide_vowels('donuts') == 'd-n-ts'

def average_price(l:list) -> float:
    '''Returns the average price of (the best dishes at) those restaurants'''
    sum_price = 0
    for x in l:
        sum_price = (sum_price + x.price)
    return sum_price / len(l)
