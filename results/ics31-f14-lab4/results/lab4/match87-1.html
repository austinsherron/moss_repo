def test_number (num, num_str) -> str:
    """ Takes as input a number and a string (one of 'even', 'odd', 'positive',
    or 'negative'). The function returns True if the number has the property
    indicated by the string, and False if it doesn't"""
    if num % 2 == 0:
        if num_str == 'even':
            return True
        elif num_str == 'odd':
            return False
    else:
        if num_str == "odd":
            return True
        elif num_str == 'even':
            return False
    if num > 0:
        if num_str == 'positive':
            return True
        elif num_str == 'negative':
            return False
    if num < 0:
        if num_str == "negative":
            return True
        elif num_str == 'positive':
            return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(stringy: str) -> str:
    """ Takes a string and returns a string containing all the characters in
    the parameter string that are not vowels """

def select_letters(ind: str, WORD: str) -> str:
    """ Takes two parameters, both strings, and returns a string. If the first
    parameter is 'v', it returns a string containing all the vowels in the second
    parameter; if the first parameter is 'c', it returns a string containing all
    the consonants in the second parameter. If the first parameter is anything
    else, it returns the empty string """
    if ind == 'v':
        result3 = ''
        for n in WORD:
            if n in vowel_list:
                result3 = result3 + n
        return result3
    elif ind == 'c':
        return consonants(WORD)

assert select_letters('v', 'Hopping around the woods.') == 'oiaoueoo'
assert select_letters('c', 'Hippos flying in the sky.') == 'Hppsflyngnthsky'

def hide_vowels(happy: str) -> str:
    """ Takes a string and returns a string in which every vowel in the parameter
    is replaced with a hyphen ("-") and all other characters remain unchanged """
    result4 = ''
    for n in happy:
        if n in vowel_list:
            result4 = result4 + '-'
        else:
            result4 = result4 + n
    return result4

assert hide_vowels('Hello there.') == 'H-ll- th-r-.'

def average_price(RL: list) -> float:
    """ Takes a list of restaurants and returns the average price of (the best
    dishes at) those restaurants """
    total_price = 0
    for x in RL:
        total_price = total_price + x.price
    average_price = total_price / len(RL)
    return average_price

def create_rectangle_from_center(x: float, y: float, height: float, width: float):
    """ Create a rectangle by taking four arguments, the x and y valuesof the
    center of the rectangle, and the height and width of the rectangle """
    my_canvas.create_rectangle((x - (width / 2)), (y - (height / 2)), (x + (width / 2)), (y + (height / 2)))
