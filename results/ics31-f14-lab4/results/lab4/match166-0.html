def test_number(n: int, s: str) -> bool:
    ''' Returns True if the number has property indicated by string, False otherwise '''
    if s == 'even':
        return n%2 == 0
    if s == 'odd':
        return n%2 != 0
    if s == 'positive':
        return n >= 0
    if s == 'negative':
        return n < 0
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(phrase: str) -> str:
    ''' Returns a string with the characters in the parameter string that are not vowels '''
    consonant = 'bcdfghjklmnpqrstvwxyz'
    consonant = consonant + consonant.upper()
    result = ''
    for i in phrase:
        if i in consonant:
            result = result + i
    return result
assert nonvowels('uduhbicboi') == 'dhbcb'

print(nonvowels('lol! Omg! wow!'))

def consonants(phrase: str) -> str:
    ''' Returns a string containing all the letters in the parameter string that are not vowels '''

def select_letters(char: str, phrase: str) -> str:
    ''' Return all vowels in phrase if char is v, or all consonants if char is c '''
    result = ''
    if char == 'v':
        for i in phrase:
            if is_vowel(i):
                result = result + i
    if char == 'c':
        for i in phrase:
            if nonvowels(i):
                result = result + i
    return result

assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(phrase: str) -> str:
    ''' Returns a string in which every vowel in the parameter is replaced with a hyphen ("-") '''
    result = ''
    for i in phrase:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result
assert hide_vowels("This sentence is false.") == "Th-s s-nt-nc- -s f-ls-."
print(hide_vowels("After your testing shows that it's correct, try running it with a couple of sentences."))
print(hide_vowels("You may be able to understand the sentences even with all the vowels hidden."))

def create_rectangle_from_center(xc: int, yc: int, height: int, width: int) -> None:
    ''' Create a rectangle with center (xc, yc), and a given height and width '''
    import tkinter
    my_window = tkinter.Tk()   
    my_canvas = tkinter.Canvas(my_window, width=xc*2, height=yc*2)  
    my_canvas.pack()            
    
    my_canvas.create_rectangle(xc-width/2, yc-height/2, xc+width/2, yc+height/2)
    
    tkinter.mainloop()          

create_rectangle_from_center(250,250,100,50)
