def test_number(n:int, s:str):
    '''takes a integer and a str of 'even','odd','positive',and 'negative'
    and tells you if the two data types have the same attributes'''
    if n % 2 ==0 and s == 'even':
        return('True')
    elif n % 2 != 0 and s=='odd':
        return ('True')
    elif n > 0 and s== 'positive':
        return ('True')
    elif n < 0 and s== 'negative':
        return('True')
    else:
        return('False')



assert test_number(14, 'even') == 'True'
assert not test_number(100, 'odd') == 'True'

assert not test_number(100, 'negative') == 'True'

def hide_vowels(s:str)->str:
    '''takes a string and reuturns a string in which every vowel is replaced with a hyphen'''
    letters=''
    for char in s:
        if is_vowel(char)== True:
            letters = letters + '-'
        if is_vowel(char)==False:
            letters = letters + char 
    return letters
