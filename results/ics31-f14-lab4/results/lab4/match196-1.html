def nonvowels (x:str ) -> str:
    '''Takes a string and returns a string containing all the characters
    in the parameter string that are not vowels'''
    result = []
    for i in x:
        if is_vowel(i) != True:
            result.append(i)
    return result

letters = ['a','b','c','d','e','f','g','h','i','j','k','l','m',
           'n','o','p','q','r','s','t','u','v','w','x','y','z']
def consonants(x:str) -> str:
    ''' Takes a string, returns a string of all the letters in that
    parameter that are not vowels'''
    cons = ('')
    for let in x:
        if let in letters and let not in vowels:
            cons=(cons+let)
    return cons
assert(consonants('aeio?uwnfpoall'))== 'wnfpll'

def select_letters (x:str , y:str) -> str:
    '''Takes two strings and returns either vowels or consonants depending
    on whether the first parameter is 'c' or 'v' '''
    zz = ('')
    if x == 'c':
        return consonants(y)
    elif x=='v':
        for z in y:
            if z in vowels:
                zz = zz + z
        return zz
