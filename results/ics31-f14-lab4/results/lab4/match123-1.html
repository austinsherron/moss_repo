def test_number(num: int, attr: str) -> bool:
    ''' Returns a Boolean based on whether first parameter matches
    the description of the second '''
    if attr == 'even':
        return (num % 2 == 0)
    elif attr == 'odd':
        return (num % 2 != 0)
    elif attr == 'positive':
        return (num > 0)
    elif attr == 'negative':
        return (num < 0)
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

    in the list that starts with that character'''

        if str_list[x].startswith(character):

def match_area_code(area_code: list, num_list: list) -> str:
    ''' Prints out the elements of the second parameter that starts with
    elements of the first parameter '''

def is_vowel(character: str) -> bool:
    ''' Returns a Boolean based on whether or not the character is a vowel '''
    return character.lower() in ['a', 'e', 'i', 'o', 'u']

def nonvowels(word: str) -> str:
    ''' Returns a string of all the characters in the parameter that aren't vowels '''
    new_word = ''
    for x in range(len(word)):
        if is_vowel(word[x]) == False:
            new_word += word[x]
    return new_word
assert nonvowels('hello world!') == 'hll wrld!'
print(nonvowels('This is a test.'))

def vowels(word: str) -> str:
    ''' Returns a string containing all vowels in the parameter '''
    new_word = ''
    for x in range(len(word)):
      if word[x].isalpha() and is_vowel(word[x]) == True:
          new_word += word[x]
    return new_word

def select_letters(character: str, word: str) -> str:
    ''' Returns parts of the second parameter according to the value
    of the first parameter '''
    if character == 'v':
        return vowels(word)
    elif character == 'c':
        return consonants(word)
    else:
        return ''
assert select_letters('c', 'facetiously') == 'fctsly'
print(select_letters('v', 'facetiously'))

def hide_vowels(word: str) -> str:
    new_word = ''
    for x in range(len(word)):
        if is_vowel(word[x]) == True:
            new_word += '-'
        else:
            new_word += word[x]
    return new_word
assert hide_vowels('hello world') == 'h-ll- w-rld'
print(hide_vowels("This is a test."))

def average_price(rest_list: list) -> float:
    ''' Returns the average price of a list of restaurants specified by the first paramater '''
    total = 0
    for x in range(len(rest_list)):
        total += rest_list[x].price
    return total/len(rest_list)
