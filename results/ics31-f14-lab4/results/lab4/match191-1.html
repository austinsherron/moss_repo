def test_number (n: int, stat: str) -> bool:
    '''Accepts a number and a string and returns if they match'''
    if stat =='even':
        if n%2 == 0:
            return True
        else:
            return False
    elif stat == 'odd':
        if n%2 != 0:
            return True
        else:
            return False
    elif stat == 'positive':
        if n > 0:
            return True
        else:
            return False
    elif stat == 'negative':
        if n < 0:
            return True
        else:
            return False
        


assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(word: str) -> str:
    '''Returns a string without vowels'''
    result = ""
    for i in range(len(word)):
        if is_vowel(word[i]) == False:
            result = result + word[i]
    return result
assert nonvowels('cake!!') == 'ck!!'
assert nonvowels('documentary') == 'dcmntry'
assert nonvowels('Justin timberlake') == 'Jstn tmbrlk'

def consonants(word: str) -> str:
    '''Takes a string and returns only consonants'''
    result = nonvowels(word)
    result_2 = ""
    for i in range(len(result)):
        if result[i].lower() in ['b','c','d','f','g','h','j','k','l','m','n','p','q','r','s','t','v','w','x','y','z']:
            result_2 = result_2 + result[i]
    return result_2
assert consonants("cake!!") == "ck"
assert consonants("documentary!!!!") == "dcmntry"
assert consonants("COW") == "CW"
assert consonants("Justin timberlake123") == "Jstntmbrlk"

def select_letters(select: str, word: str) -> str:
    '''Takes a string and removes either consonants or vowels or returns an empty string'''
    vowels = ""
    if select == 'c':
        return consonants(word)
    elif select == 'v':
        for i in range(len(word)):
            if is_vowel(word[i]) == True:
                vowels = vowels + word[i]
        return vowels
    else:
        return ""
    
assert select_letters('c', 'super') == 'spr'
assert select_letters('v', 'super') == 'ue'
assert select_letters('f', 'dog') == ""

def hide_vowels(word: str) -> str:
    '''Takes a string and replaces all vowels with a hyphen'''
    lword = list(word)
    for i in range(len(word)):
        if is_vowel(word[i]) == True:
            lword[i] = '-'
        word = ''.join(lword)
    return word
assert hide_vowels('bob') == 'b-b'
assert hide_vowels('super car') == 's-p-r c-r'

def average_price(rest: list) -> float:
    '''Prints the average price from a list of restaurants'''
    sum_rest = 0
    num_rest = 1
    for i in rest:
        sum_rest = sum_rest + i.price
        num_rest = num_rest + 1
    return sum_rest/num_rest
print(average_price(RL))

import tkinter

def create_rectangle_from_center(x: int, y: int, height: int, width) -> None:
    '''prints a rectangle using coordinates for the center and height and width'''  
    my_canvas.create_rectangle(x - (width/2), y - (height/2), x + (width/2),
                               y + (height/2))
    return None


my_window = tkinter.Tk()    # Create the graphics window

my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
my_canvas.pack()            # Put the canvas into the window
