def test_number (n: int, m: str)-> bool:
    ''' Returns True if the number has property of string '''
    m == 'even' or 'odd' or 'positive' or 'negative'
    
    if m == 'positive':
        return n >= 0
      
    elif m == 'negative':
        return n <= 0
    elif m == 'even':
        return n%2 == 0
    elif m == 'odd':
        return n%2 != 0
   
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(string: str) -> str:
    ''' Returns nonvowel characters in a string '''
    result = ""
    for i in string:
        if is_vowel(i) == False:
            result += i
    return result
assert nonvowels('Cow') == "Cw"
assert not nonvowels('Alphabet') == "Aabe"

def consonants(strings: str) -> str:
    ''' Returns consonants from a string '''
    consonant = ""
    for i in strings:
        if i in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
            consonant += i
    return consonant
assert consonants('Cowabunga') == "Cwbng"
assert consonants('That was rad!') == "Thtwsrd"
assert not consonants('Totes McGoats') == "Totes McGoats"
assert not consonants('Hi!@#! Man%#%') == "H!@#! Mn%#%"

def select_letters(command: str, word: str) -> str:
    ''' Return a string from word depending on command '''
    command == 'v' or 'c'
    end_string = ''
    if command == 'v':
        for i in word:
            if is_vowel(i) == True:
                end_string += i
        return end_string
    elif command == 'c':
        end_string += consonants(word)
        return end_string

assert not select_letters('v','Swine') == 'Swn'

def hide_vowels(sentence: str) -> str:
    ''' Returns a sentence with the vowels hyphentated'''
    for i in sentence:
        if is_vowel(i) == True:
             sentence = sentence.replace(i, '-')
    return sentence
assert hide_vowels('Chicken feet is good') == 'Ch-ck-n f--t -s g--d'
assert hide_vowels('ICS is fun!') == '-CS -s f-n!'
assert not hide_vowels('Apples are sweet!') == 'A---e- a-r --ee-!'

def Restaurant_change_price(RestaurantObject: 'list of Restaurants', pricechange: float) -> list:
    ''' Returns a restaurant namedtuple with a new price '''
    RestaurantObject = RestaurantObject._replace(price = RestaurantObject.price + pricechange)
    return RestaurantObject

import tkinter
my_window = tkinter.Tk()
my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)  
my_canvas.pack()

def create_rectangle_from_center(x: int, y: int, h: int, w: int):
    ''' Draws a rectangle from the center of x,y and using the height and width '''
    my_canvas.create_rectangle((x - (w/2)),(y - (h/2)),(x + (w/2)),(y + (h/2)))
