def test_number(x:int, y:str) -> bool:
    '''takes as input a number and a string and returns True if the number has the property indicated by the string'''
    if x%2==0 and y=='even':
        return True
    elif x%2!=0 and y=='odd':
        return True
    elif x>=0 and y=='positive':
        return True
    elif x<=0 and y=='negative':
        return True
    else:
        return False
assert test_number(2, 'even')
assert test_number(3, 'odd')
assert not test_number(8, 'negative')
assert not test_number(-9, 'positive')

def is_vowel(s:str) -> bool:
    '''takes a one-character-long string and returns True if that character is a vowel and False otherwise'''
    return (s.lower() in ['a','e','i','o','u'])
assert is_vowel('a')
assert not is_vowel('b')
assert is_vowel('e')

def print_nonvowels(s:str) -> None:

def nonvowels(s:str) -> str:
    '''takes a string and returns a string containing all the characters in the parameter string that are not vowels'''
    result = ''
    for x in s:
        if not is_vowel(x):
            result += x
    return result
assert nonvowels('hello') == 'hll'
assert nonvowels('welcome') == 'wlcm'
assert nonvowels('computer') == 'cmptr'

def consonants(s:str) -> str:
    '''takes a string and returns a string containing all the letters in the parameter string that are not vowels'''
    result=''
    for x in s:
        if x.isalpha() and not is_vowel(x):
            result+=x
    return result
assert consonants('hi!') == 'h'
assert consonants('why?') == 'why'
assert consonants('as432p[[[]]') == 'sp'

#d5
def select_letters(s:str,x:str) -> str:
    '''takes two parameters, both strings, and returns a string containing all the vowels in the second parameter'''
    result=''
    if s == 'v':
        for i in x:
            if is_vowel(i):
                result+=i
        return result
    elif s == 'c':
        return consonants(x)
    else:
        return ''
assert select_letters('c','science') == 'scnc'


assert select_letters('v','enter') == 'ee'
assert select_letters('a', 'print') == ''

def hide_vowels(s:str) -> str:
    '''takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters remain unchanged'''
    for x in s:
        if is_vowel(x):
            s = s.replace(x,'-')
    return s
assert hide_vowels('backspace') == 'b-cksp-c-'
assert hide_vowels('vowels') == 'v-w-ls'
assert hide_vowels('mouse') == 'm--s-'

def alphabetical_names(l:list) -> list:
