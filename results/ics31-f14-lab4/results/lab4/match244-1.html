def test_number(num: int, str_type: str) -> bool:
    ''' Returns True if number has property indicated by string (one of 'even', 'odd', 'positive', 'negative) '''
    if str_type is 'even' and num % 2 is 0:
        return True
    elif str_type is 'odd' and num % 2 is not 0:
        return True
    elif str_type is 'positive' and num > 0:
        return True
    elif str_type is 'negative' and num < 0:       
        return True

    return False


assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(string: str) -> str:
    ''' Returns a string containing all the characters in the parameter that are not vowels '''
    result = ''
    for char in string:
        if not is_vowel(char):
            result += char
    return result

assert nonvowels('Professor') == 'Prfssr'

# Helper function for use in the consonants function
def is_letter(char: str) -> bool:
    ''' Checks to see if the character is an alphabet letter '''
    return (char >= 'a' and char <= 'z') or (char >= 'A' and char <= 'Z')

def consonants(string: str) -> str:
    ''' Returns a string containing all the letters in the parameter that are not vowels '''
    result = ''
    for char in string:
        if not is_vowel(char) and is_letter(char):
            result += char
    return result

assert consonants('L33TPkEr1337') == 'LTPkr'
assert consonants('Mary had 12 pies today!') == 'Mryhdpstdy'
assert consonants('4evr 4l0ne') == 'vrln'

# Helper function for use in the select_letters function
def vowels(string: str) -> str:
    ''' Returns a string containing all the vowels in the parameter'''
    result = ''
    for char in string:
        if is_vowel(char):
            result += char
    return result


def select_letters(str_type: str, string: str) -> str:
    ''' If first parameter is 'v', return string containing all vowels in second parameter.
        If first parameter is 'c', return string containing all consonants in second parameter.
        If first parameter anything else, return empty string. '''
    if str_type is 'v':
        return vowels(string)
    elif str_type is 'c':
        return consonants(string)
    else:
        return ''

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'
assert select_letters('v', 'xXxGarretxXx9812') == 'ae'
assert select_letters('c', 'xXxGarretxXx9812') == 'xXxGrrtxXx'

def hide_vowels(string: str) -> str:
    result = ''
    for char in string:
        if is_vowel(char):
            result += '-'
        else:
            result += char
    return result

Restaurant = namedtuple('Restaurant', 'name cuisine phone dish price')

    ''' Returns the Restaurant inside parameter, but with its price increased by specified number '''
    return r1._replace(price = r1.price + num)

print(r2)

print(Restaurant_change_price(r1, 1.75))

def alphabetical_names(rest_list: list) -> list:
    ''' Takes a list of restaurants and returns a list of names of
        all the restaurants in alphabetical order by name '''
    new_list = []

def average_price(rest_list: list) -> float:
    ''' Takes a list of restaurants and returns the average price
        of (the best dishes) at those restaurants '''
    sum = 0
    for rest in rest_list:
        sum += rest.price
    return sum / len(rest_list)

def create_rectangle_from_center(x: float, y: float, height: float, width: float, canvas: tkinter.Canvas) -> None:
