def consonants (s:str) -> str:
    '''Returns a string that only has consonants with no vowels or symbols'''
    con = 'bcdfghjklmnpqrstvwxyz'
    result = ''
    for i in s:
        if i.lower() in con:
            result = result + i 
    return result
assert consonants('drlong8') == 'drlng'
assert consonants('ucirvine50') == 'crvn'
assert consonants('Yessssss!') == 'Yssssss'

def hide_vowels (s:str) -> str:
    '''Places a hyphen in place of a vowel in any string'''
    result = ''
    for i in s:
        if is_vowel(i):
            result = result + '-'
        else:
            result = result + i
    return result
print(hide_vowels('The dog jumped over the fence'))
