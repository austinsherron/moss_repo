def test_number(x: int, y:str) -> bool:
    ''' Takes as input a number and a string and returns True if the number has
        the property indicated by the string and False if it doesn't '''
    if x%2 == 0 and y == 'even':
        return True
    elif x%2 != 0 and y == 'odd': 
        return True
    elif x > 0 and y == 'positive':
        return True
    elif x < 0 and y == 'negative': 
        return True
    return False
        
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def matching_area_codes(area_codes: list, phone_numbers: list) -> str:
    ''' Takes a list of telephone area codes and a list of phone numbers
        and returns a list of the matching numbers '''
    phone_list = []
    for three_numbers in area_codes:
        for digits in phone_numbers:
            if digits[1:4] == three_numbers:
                phone_list.append(digits[1:4])
    return(phone_list)

    is a vowel and False otherwise '''

def hide_vowels(word: str) -> str:
    ''' Takes a string and returns a string in which every vowel in the
        parameter is replaced with a hyphen (" - ") and all other
        characters remain unchanged '''
    result = ""
    for i in range(len(word)): 
        for letters in word:
            if (not is_vowel(letters)):
                result += str(letters)
            elif is_vowel(letters):
                result += str('-')
        return result


print(hide_vowels("I'm craving anchovies"))

def sum_of_best_dishes(list_of_Restaurants: list) -> float:
    ''' Takes a list of restaurants and returns the sum of the prices
        of (the best dishes at) those restaurants '''
    prices = []
    for restaurant in list_of_Restaurants:
            prices.append(restaurant.price)
            total_price = 0
            for price in prices:
                total_price = total_price + price
    return total_price

print(sum_of_best_dishes(RL))
print()



def average_price(list_of_Restaurants:list) -> float:
    ''' Takes a list of restaurants and returns the average price of
        (the best dishes at) those restaurants '''
    average = sum_of_best_dishes(list_of_Restaurants) / len(list_of_Restaurants)
    return(average)
