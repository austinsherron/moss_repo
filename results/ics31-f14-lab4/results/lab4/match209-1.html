        return True
    if y == 'positive' and x > 0:

def hide_vowels (x:str) ->str:
    '''Takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen ("-") and all other characters remain unchanged'''

        if is_vowel(y) == True:

def create_rectangle_from_center(m1,m2,width,length):
    '''Takes four arguments: the x-value and y-value of the point in the center of the rectangle and the height and width of the rectangle.'''
    import tkinter              
    my_window = tkinter.Tk()    
    my_canvas = tkinter.Canvas(my_window, width=1000, height=1000)  
    my_canvas.pack()            
    my_canvas.create_rectangle(m1-width/2, m2-length/2, m1+width/2, m2+length/2, fill='orange')    
    tkinter.mainloop()
create_rectangle_from_center(300,100,150,70)
