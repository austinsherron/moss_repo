def test_number(n: float, m: str) -> bool:
    '''Returns true or false if statement is correct'''
    if m=='even' or m=='odd':
        if n%2==0:
            if m=='even':
                return(True)
            else:
                return(False)
        elif n%2!=0:
            if m=="odd":
                return(True)
            else:
                return(False)
    if m=='positive' or m=='negative':
        if n > 0:
            if m=="positive":
                return(True)
            else:
                return(False)
        elif n < 0:
            if m=="negative":
                return(False)
            else:
                return(False)
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def hide_vowels(n:str)->str:
    ''' Returns vowels replaced by hyphens'''
    Vowel = 'aeiou'
    for y in n:
        if y in Vowel:
            n = n.replace(y, '-')
    return n
assert hide_vowels('apple') == '-ppl-'
