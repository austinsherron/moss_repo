        or negative, based on the given string and will return a boolean value'''

def matching_area_codes(codes: list, phones: list) -> list:
    '''Returns a list of phone numbers from the phone's list if it's area codes matches the
       codes in the codes list'''

# consonants     k: str   ->    str
def consonants (k: str) -> str:
    j = [",", "!", ".", "?", "#", "$", "'", "(", ")", "[", "]", "/",
         "-", "*", "+", "=", "^", " ", "%", "1", "2", "3", "4", "5", "6",
         "7", "8", "9", "0"]
    h = ''
    for i in k:
        if not is_vowel(i) and i not in j:
            h = h + i
    return h

assert consonants("How 'bout no.") == "Hwbtn"

print(consonants("How 'bout no."))

# select_letters:   m: str   o: str -> str
def select_letters (m: str, o: str) -> str:
    '''If m is 'v', we return a string of all the vowels in the given string o.
       If m is 'c', we return a string of all the consonants in the given string o.'''
    g = ''
    if m == 'v':
        for r in o:
            if is_vowel(r):
                g = g + r
    elif m == 'c':
        g = consonants(o)
    else:
        print('You did not specify vowel or consonant.')
    return g
assert select_letters('v', 'HippoMonkey') == 'iooe'
assert select_letters('b', 'Car') == ''
assert select_letters('c', 'Why not.')  == 'Whynt'

# hide_vowels:   f: str -> str
def hide_vowels (f: str) -> str:
    '''This function takes in a string and returns that string with the
       vowels replaced with a hyphen'''
    u = ''
    for w in f:
        if is_vowel(w):
            u = u + '-'
        else:
            u = u + w
    return u
assert hide_vowels('Random') == 'R-nd-m'

print(hide_vowels('I like to eat ice cream.'))

def get_sum (merp: list) -> float:
    '''This function returnss the total sum of all the prices in the list
       of restaurants'''
