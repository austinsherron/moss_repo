def test_number( n, s ):
    '''Takes a number and a string (even, odd, positive, or negative) and returns
    true if the number has the indicated property and false otherwise.'''
    if( s == 'even' and n % 2 == 0 ):
        return True
    elif( s == 'odd' and n % 2 == 1):
        return True
    elif( s == 'positive' and n > 0 ):
        return True
    elif( s == 'negative' and n < 0 ):
        return True
    return False

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')
print ('20 is an odd number:', test_number(20, 'odd'))

def nonvowels( word ):


    '''Takes a string and returns a string containing all characters in the parameter string that
    are not vowels.'''
    result = ""
    for character in word:
        if( not is_vowel(character) ):
            result += character
    return result

assert nonvowels('abcde') == 'bcd'
assert nonvowels('ABCDE') == 'BCD'
assert nonvowels('a1b2c3') == '1b2c3'

def consonants( word ):
    '''Takes a string and returns a string containing all the letters in the parameter string that
    are not vowels.'''
    result = ""
    for character in word:
        if( character.isalpha() and nonvowels(character) ):
            result += character
    return result

assert consonants('abcde') == 'bcd'
assert consonants('ABCDE') == 'BCD'
assert consonants('a1b2c3') == 'bc'

def select_letters( letter, word ):
    '''Takes two string parameters and returns a string. If the first parameter is 'v', it returns
    a string containing all the vowels in the second parameter. If the first parameter is 'c',
    it returns a string containing all the consonants in the second parameter.'''
    result = ""
    if( letter == 'v' ):
        for character in word:
            if( is_vowel( character ) ):
                result += character
    elif( letter == 'c' ):
        result = consonants( word )
    return result

assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels( word ):


    '''Takes a string and returns a string in which every vowel in the parameter is replaced with
    a hyphen ("-") and all other characters remain unchanged. '''
    result = ""
    for character in word:
        if( is_vowel(character)):
            result += '-'
        else:
            result += character
    return result

assert hide_vowels('facetiously') == 'f-c-t---sly'
print (hide_vowels('facetiously'))

def average_price( restaurants ):
    '''Takes a list of restaurants and returns the average price of (the best dishes at) those restaurants.'''
    total_price = 0
    count = 0
    for restaurant in restaurants:
        total_price += restaurant.price
        count += 1
    return float(total_price) / count

print ('Average price of the best dishes in the list RL:')
