def test_number(x: int, test: str)-> bool:
    '''
    Returns True if the number has the property indicated by the string, and False if it doesn't.
    '''
    if test == "even":
        return x % 2 == 0
    if test == "odd":
        return x % 2 == 1
    if test == "positive":
        return x > 0
    if test == "negative":
        return x < 0
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def nonvowels(word: str)-> str:
    '''
    Returns a string containing all the characters in the parameter string that are not vowels.
    '''
    result = ''
    for i in word:
        if not is_vowel(i):
            result += i
    return result
assert nonvowels('Test') == 'Tst'
assert nonvowels('') == ''
assert nonvowels('?') == '?'

def consonants(word: str) -> str:
    '''
    Returns a string containing all the letters in the parameter string that are not vowels.
    '''
    result = ''
    for i in word:
        if i in string.ascii_letters and i not in ['a','e','i','o','u']:
            result += i
    return result
assert consonants('Test') == 'Tst'
assert consonants('') == ''
assert consonants('?') == ''

    for i in word:
        if is_vowel(i):
            result += i
    return result
def select_letters(strtype: str, word: str) -> str:
    '''
    If the first parameter is 'v', it returns a string containing all the vowels in the second parameter;
    If the first parameter is 'c', it returns a string containing all the consonants in the second parameter.
    If the first parameter is anything else, it returns the empty string
    '''
    if strtype == 'v':
        return vowels(word)
    if strtype == 'c':
        return consonants(word)
    return ''
assert select_letters('v','aeibcde') == 'aeie'
assert select_letters('c', 'aeibcde') == 'bcd'
assert select_letters('','blahblah') == ''

def hide_vowels(word: str) -> str:
    '''
    Takes a string and returns a string in which every vowel in the parameter is replaced with a hyphen ("-")
    '''
    result = ''
    for i in range(len(word)):
        if is_vowel(word[i]):
            result += '-'
        else:
            result += word[i]
    return result
assert hide_vowels("Hello world of python.") == "H-ll- w-rld -f pyth-n."
assert hide_vowels("This is a test") == "Th-s -s - t-st"
assert hide_vowels("Test for this") == "T-st f-r th-s"

def average_price(restlist: list):
    '''
    Returns the average price from list restaurants.
    '''
    if len(restlist) == 0:
        return 0.0
    restsum = 0
    for rest in restlist:
       restsum += rest.price       
    return restsum / len(restlist)
