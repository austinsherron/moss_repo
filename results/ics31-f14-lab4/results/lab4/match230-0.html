def test_number(n:int, string:str) -> bool:
    "Tests to see if n matches the test string"
    if(string == 'even' and n % 2 == 0):
        return True
    elif(string == 'odd' and n % 2 == 1):
        return True
    elif(string == 'posative' and n >= 0):
        return True
    elif(string == 'negative' and n < 0):
        return True
    else:
        return False
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'posative')
assert not test_number(100, 'negative')

#part 3
def nonvowels(string:str) -> str:
    "returns a string of all nonvowels"
    lst = ''
    for char in string:
        if not is_vowel(char):
            lst += char
    return lst
assert nonvowels('testing testing 123') == 'tstng tstng 123'
assert nonvowels('aeiou') == ''
assert nonvowels('qwrtypsdfghjklzxcvbnm1234567890') == 'qwrtypsdfghjklzxcvbnm1234567890'
#part 4
def consonants(string:str) -> str:
    "Returns a string of all consonants"
    lst =''
    for char in string:
        if not is_vowel(char) and char.isalpha():
            lst += char
    return lst
assert consonants('testing testing 123') == 'tstngtstng'
assert consonants('aeiou') == ''
assert consonants('qwrtypsdfghjklzxcvbnm1234567890') == 'qwrtypsdfghjklzxcvbnm'
#part 5
def select_letters(operator:str, string:str) -> str:
    "returns a string that is changed based on the operator, returning vowels or consonants"
    if operator == 'v':
        lst = ''
        for char in string:
            if is_vowel(char):
                lst += char
        return lst
    elif operator == 'c':
        return consonants(string)
assert select_letters('v', 'testing testing 123') == 'eiei'
assert select_letters('c', 'testing testing 123') == 'tstngtstng'
#part 6
def hide_vowels(string: str) -> str:
    "changes all vowels in a string to -"
    lst =''
    for char in string:
        if(is_vowel(char)):
            lst += '-'
        else:
            lst += char
    return lst
assert hide_vowels('testing testing 123') == 't-st-ng t-st-ng 123'

def alphabetical_names(lst:[Restaurant]) ->[Restaurant]:
    "returns a list of the names in alphabetical order"
    restaurant_list = sorted(lst)
    name_list = []

def select_cheaper(lst:[Restaurant], price:float) -> [Restaurant]:
    "returns a list of restaurants that are cheaper than price"
    restaurant_list = []
    for restaurant in lst:
        if(restaurant.price < price):
            restaurant_list.append(restaurant)
    return restaurant_list

def average_price(lst:[Restaurant]) -> float:
    "returns the average price of the restaurants in lst"
    total_price = 0
    for restaurant in lst:
        total_price += restaurant.price
    return total_price / len(lst)
