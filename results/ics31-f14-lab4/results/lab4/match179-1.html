def test_number(n: int, s: str):
    '''Evaluates whether a number is positive, negative, even, or odd'''
    if s == 'even':
        return n%2 == 0
    elif s == 'odd':
        return n%2 != 0
    elif s == 'positive':
        if n == 0:
            return n == 0
        else:
            return n >= 0
    elif s == 'negative':
        return n<= 0

assert test_number(14, 'even') == True
assert not test_number(100, 'odd') == True
assert test_number(33, 'positive') == True
assert not test_number(100, 'negative') == True

def is_vowel(s: str):

def consonants(s: str) -> str:
    '''Returns a str of consonants in a given str without characters
        other than letters'''
    result = ''
    consonants_list = ['b', 'B', 'c', 'C', 'd', 'D', 'f', 'F', 'g', 'G', 'h',
                       'H', 'j', 'J', 'k', 'K', 'l', 'L', 'm', 'M', 'n', 'N',
                       'p', 'P', 'q', 'Q', 'r', 'R', 's', 'S', 't', 'T', 'v',
                       'V', 'w', 'W', 'x', 'X', 'y', 'Y', 'z', 'Z']
    for letter in s:
        if letter in consonants_list:
            result += letter
    return result

assert consonants('He Ate The Bread.') == 'HtThBrd'

def select_letters(command: str, string: str) -> str:
    '''Returns only consonants or vowels in the given str based on the
        command (either c or v)'''
    result_string = ''
    if command == 'c':
        return consonants(string)
    elif command == 'v':
        for letter in string:
            if is_vowel(letter) == True:
                result_string += letter
        return result_string
    else:
        return result_string

assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(string: str) -> str:
    '''Replace vowels in a given str with hyphens'''
    result_string = ''
    for letter in string:
        if is_vowel(letter) == True:
            result_string += '-'
        else:
            result_string += letter
    return result_string

assert hide_vowels('The boy ate the apple.') == 'Th- b-y -t- th- -ppl-.'

def sum_prices(list_of_rests: list) -> float:
    '''Takes a list of Restaurants and returns the sum of their prices'''
    total = 0
    for rest in list_of_rests:
        total += rest.price
    return total

def average_price(list_of_rests: list) -> float:
    '''Takes a list of Restaurants and returns the average price'''
    average = (sum_prices(list_of_rests) / len(list_of_rests))
