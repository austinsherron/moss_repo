def test_number(num:int, type1:str) -> bool:
    '''Takes a interger and a type and returns whether it's true or false'''
    if num%2 == 0:
        if type1 == 'even':
            return True
    elif num%2 == 0:
        if type1 == 'odd':
            return True
    elif num>0:
        if type1 == 'positive':
            return True
    elif num<0:
        if type1 == 'negative':
            return True
    else:
        return False
    
assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

import tkinter
my_window = tkinter.Tk()
def create_rectangle_from_center(x:float,y:float,height:float,width:float):
    '''Uses the height and width dimensions to create a circle with the origin at the specified points'''
    my_canvas = tkinter.Canvas(my_window, width=500, height=500)  # Create a 500x500 canvas to draw on
    my_canvas.pack()
    my_canvas.create_rectangle((x-(width/2)),(y-(height/2)),(x+(width/2)),(y+(height/2)))
    tkinter.mainloop()

create_rectangle_from_center(250,250,100,100)
