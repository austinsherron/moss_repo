def nonvowels(characters:str)->str:
    '''takes a string and returns all characters that are not a vowel in a string'''
    result = ''
    for i in characters:
        if is_vowel(i) == False:

def consonants(letters:str)->str:
    '''takes a string and returns only the consonants'''
    consonants = 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ'
    consonants_result = ''
    for i in letters:
        if i in consonants:
            consonants_result = consonants_result + i
    return consonants_result

VOWELS = 'aeiouAEIOU'

def select_letters(letter:str,word:str)->str:
    '''function select_letters takes in a word and separates them into either all vowels
    or all consonants'''
    vowel_result = ''
    if letter is 'v':
        for i in word:
           if i in VOWELS:
               vowel_result = vowel_result + i
        return vowel_result
    elif letter is 'c':
        for i in word:
            return consonants(word)
    else:
        return ''

print(select_letters('c', 'facetiously'))

assert select_letters('c', 'cl0cks') == 'clcks'

def average_price(AP:list)->float:
    '''function average_price takes in a list of restaurants and computes the average for the best dish of each restaurant'''
    sumList=[]
    for i in range(len(AP)):
        sumList.append(AP[i].price)
    return sum(sumList)/len(sumList)
