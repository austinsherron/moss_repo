def hide_vowels(s: str) -> str:
    result = ""
    for x in range(len(s)):
        if is_vowel(s[x]):
            result = result + '-'
        else:
            result = result + s[x]
    return result

def average_price(L: list) -> float:
    price_list = []
    for x in range(len(L)):
        price_list.append(L[x].price)
    n = 0
    for x in range(len(price_list)):
        n += price_list[x]
    return n/len(price_list)
