def test_number(num:int,condition:str)->bool:
    if (condition == 'positive'):
        if (num>0):
            return True
        else:
            return False

    elif (condition=='negative'):
        if(num<0):
            return True
        else:
            return False
    elif(condition=='even'):
        if(num%2==0):
            return True
        else:
            return False
    elif(condition=='odd'):
        if(num%2==1):
            return True
        else:
            return False

def vowels(string:str) -> str:
    '''returns a word/phrase with only vowels'''
    s = ''
    for i in string:
        if(i.isalpha()):
            if(is_vowel(i) == True):
                s += i
    return s

def consonants(string:str) -> str:
    '''returns a word/phrase with only consonants'''
    s = ''
    for i in string:
        if(i.isalpha()):
            if(is_vowel(i) == False):
                s += i
    return s
print(consonants('j3remi45_y'))
print()

def select_letters(condition, string: str) -> str:
    '''modular/able to choose to make string without vowels or only vowels'''
    if(condition == 'c'):
        return consonants(string)
    elif(condition == 'v'):
        return vowels(string)
assert select_letters('v', 'facetiously') == 'aeiou'
assert select_letters('c', 'facetiously') == 'fctsly'

def hide_vowels(string:str) -> str:
    '''returns a word/phrase with vowels replaced by "-"'''
    s = ''
    for i in string:
        if(is_vowel(i) == True):
            s += '-'
        else:
            s += i
    return s
print(hide_vowels('banana'))
print(hide_vowels('The lazy dog jumped over the blue cat squirrel'))

def average_price(RL: list) -> float:
    '''returns the average price of all restaurants in a list'''
    total = 0
    for i in RL:
        total += i.price
