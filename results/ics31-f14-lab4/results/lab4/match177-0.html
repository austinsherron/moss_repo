def test_number(n:int,s:str) -> bool:
    '''Takes a number and a string and returns True or False.'''
    if s == 'even':
        return (n % 2 == 0)
    elif s == 'odd':
        return (n % 2 != 0)
    elif s == 'positive':
        return (n > 0)
    elif s == 'negative':
        return (n < 0)
    

assert test_number(14, 'even')
assert not test_number(100, 'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def is_vowel (n:str) -> bool:

        return consonants(m)

            if i in vowel:

def hide_vowels (n:str) -> str:
    '''Takes a string and replaces all vowels with a hyphen'''
    vowel = ['A','a','E','e','I','i','O','o','U','u']
    result = n
    for x in n:
        if x in vowel:
            result = result.replace(x,'-')
    return result

assert hide_vowels('I am Bob') == ('- -m B-b')
