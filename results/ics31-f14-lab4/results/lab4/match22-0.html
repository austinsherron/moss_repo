def test_number(number, typeofnumber):

    if typeofnumber == 'even':
         return (number % 2) == 0
    elif typeofnumber == 'odd':
        return (number % 2) == 1
    elif typeofnumber == 'positive':
         return number >= 0
    else:
         return number < 0
      
assert test_number(14, 'even')
assert not test_number(100,'odd')
assert test_number(33, 'positive')
assert not test_number(100, 'negative')

def consonants(b: str) -> str:
   result = ''
   for x in b:
      if x in 'bcdfghjklmnpqrstvwxyzBCDFGHJKLMNPQRSTVWXYZ':
         result += x
      
   return result
assert consonants('dog!') == 'dg'
assert consonants('Cat?!!!') == 'Ct'

   word = ''
   if a == 'v':
      for x in b:
         if is_vowel(x):
            word += x
   if a == 'c':

def hide_vowels(a: str) -> str:
   word = ''
   for x in a:
     if is_vowel(x):
        word += '-'
     else:
         word += x
   return word
assert hide_vowels('I like to eat bananas') == '- l-k- t- --t b-n-n-s'
