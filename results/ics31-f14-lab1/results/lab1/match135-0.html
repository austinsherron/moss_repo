def factorial (n: int)-> int:
    ''' Compute n! (n factorial)'''
    if n<= 0:
        return 1


    else:
        return n * factorial(n - 1 )
assert factorial (0) == 1
assert factorial (5) ==120

print ("100! is", factorial(120))
print ("2! is", factorial(factorial(5)))
